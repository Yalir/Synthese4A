00504 gboolean
00505 purple_plugin_load(PurplePlugin *plugin)
00506 {
00507 #ifdef PURPLE_PLUGINS
00508         GList *dep_list = NULL;
00509         GList *l;
00510 
00511         g_return_val_if_fail(plugin != NULL, FALSE);
00512 
00513         if (purple_plugin_is_loaded(plugin))
00514                 return TRUE;
00515 
00516         if (purple_plugin_is_unloadable(plugin))
00517                 return FALSE;
00518 
00519         g_return_val_if_fail(plugin->error == NULL, FALSE);
00520 
00521         /*
00522          * Go through the list of the plugin's dependencies.
00523          *
00524          * First pass: Make sure all the plugins needed are probed.
00525          */
00526         for (l = plugin->info->dependencies; l != NULL; l = l->next)
00527         {
00528                 const char *dep_name = (const char *)l->data;
00529                 PurplePlugin *dep_plugin;
00530 
00531                 dep_plugin = purple_plugins_find_with_id(dep_name);
00532 
00533                 if (dep_plugin == NULL)
00534                 {
00535                         char *tmp;
00536 
00537                         tmp = g_strdup_printf(_("The required plugin %s was not found. "
00538                                                 "Please install this plugin and try again."),
00539                                               dep_name);
00540 
00541                         purple_notify_error(NULL, NULL,
00542                                           _("Unable to load the plugin"), tmp);
00543                         g_free(tmp);
00544 
00545                         g_list_free(dep_list);
00546 
00547                         return FALSE;
00548                 }
00549 
00550                 dep_list = g_list_append(dep_list, dep_plugin);
00551         }
00552 
00553         /* Second pass: load all the required plugins. */
00554         for (l = dep_list; l != NULL; l = l->next)
00555         {
00556                 PurplePlugin *dep_plugin = (PurplePlugin *)l->data;
00557 
00558                 if (!purple_plugin_is_loaded(dep_plugin))
00559                 {
00560                         if (!purple_plugin_load(dep_plugin))
00561                         {
00562                                 char *tmp;
00563 
00564                                 tmp = g_strdup_printf(_("The required plugin %s was unable to load."),
00565                                                       plugin->info->name);
00566 
00567                                 purple_notify_error(NULL, NULL,
00568                                                  _("Unable to load your plugin."), tmp);
00569                                 g_free(tmp);
00570 
00571                                 g_list_free(dep_list);
00572 
00573                                 return FALSE;
00574                         }
00575                 }
00576         }
00577 
00578         /* Third pass: note that other plugins are dependencies of this plugin.
00579          * This is done separately in case we had to bail out earlier. */
00580         for (l = dep_list; l != NULL; l = l->next)
00581         {
00582                 PurplePlugin *dep_plugin = (PurplePlugin *)l->data;
00583                 dep_plugin->dependent_plugins = g_list_prepend(dep_plugin->dependent_plugins, plugin->info->id);
00584         }
00585 
00586         g_list_free(dep_list);
00587 
00588         if (plugin->native_plugin)
00589         {
00590                 if (plugin->info != NULL && plugin->info->load != NULL)
00591                 {
00592                         if (!plugin->info->load(plugin))
00593                                 return FALSE;
00594                 }
00595         }
00596         else {
00597                 PurplePlugin *loader;
00598                 PurplePluginLoaderInfo *loader_info;
00599 
00600                 loader = find_loader_for_plugin(plugin);
00601 
00602                 if (loader == NULL)
00603                         return FALSE;
00604 
00605                 loader_info = PURPLE_PLUGIN_LOADER_INFO(loader);
00606 
00607                 if (loader_info->load != NULL)
00608                 {
00609                         if (!loader_info->load(plugin))
00610                                 return FALSE;

00616         plugin->loaded = TRUE;

00666                                 /* This isn't necessary. This has already been done when unloading dep_plugin. */
00667                                 plugin->dependent_plugins = g_list_delete_link(plugin->dependent_plugins, l);
00668 #endif
00669                         }
00670                 }
00671         }
00672 
00673         /* Remove this plugin from each dependency's dependent_plugins list. */
00674         for (l = plugin->info->dependencies; l != NULL; l = l->next)
00675         {
00676                 const char *dep_name = (const char *)l->data;
00677                 PurplePlugin *dependency;
00678 
00679                 dependency = purple_plugins_find_with_id(dep_name);
00680 
00681                 if (dependency != NULL)
00682                         dependency->dependent_plugins = g_list_remove(dependency->dependent_plugins, plugin->info->id);
00683                 else
00684                         purple_debug_error("plugins", "Unable to remove from dependency list for %s\n", dep_name);
00685         }
00686 
00687         if (plugin->native_plugin) {
00688                 if (plugin->info->unload && !plugin->info->unload(plugin))
00689                         return FALSE;
00690 
00691                 if (plugin->info->type == PURPLE_PLUGIN_PROTOCOL) {
00692                         PurplePluginProtocolInfo *prpl_info;
00693                         GList *l;
00694 
00695                         prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(plugin);
00696 
00697                         for (l = prpl_info->user_splits; l != NULL; l = l->next)
00698                                 purple_account_user_split_destroy(l->data);
00699 
00700                         for (l = prpl_info->protocol_options; l != NULL; l = l->next)
00701                                 purple_account_option_destroy(l->data);
00702 
00703                         if (prpl_info->user_splits != NULL) {
00704                                 g_list_free(prpl_info->user_splits);
00705                                 prpl_info->user_splits = NULL;
00706                         }
00707 
00708                         if (prpl_info->protocol_options != NULL) {
00709                                 g_list_free(prpl_info->protocol_options);
00710                                 prpl_info->protocol_options = NULL;
00711                         }
00712                 }
00713         } else {
00714                 PurplePlugin *loader;
00715                 PurplePluginLoaderInfo *loader_info;
00716 
00717                 loader = find_loader_for_plugin(plugin);
00718 
00719                 if (loader == NULL)
00720                         return FALSE;
00721 
00722                 loader_info = PURPLE_PLUGIN_LOADER_INFO(loader);
00723 
00724                 if (loader_info->unload && !loader_info->unload(plugin))
00725                         return FALSE;
00726         }
00727 
00728         /* cancel any pending dialogs the plugin has */
00729         purple_request_close_with_handle(plugin);
00730         purple_notify_close_with_handle(plugin);
00731 
00732         purple_signals_disconnect_by_handle(plugin);
00733         purple_plugin_ipc_unregister_all(plugin);
00734 
00735         loaded_plugins = g_list_remove(loaded_plugins, plugin);
00736         if ((plugin->info != NULL) && PURPLE_IS_PROTOCOL_PLUGIN(plugin))
00737                 protocol_plugins = g_list_remove(protocol_plugins, plugin);
00738         plugins_to_disable = g_list_remove(plugins_to_disable, plugin);
00739         plugin->loaded = FALSE;
00740 
00741         /* We wouldn't be anywhere near here if the plugin wasn't loaded, so
00742          * if plugin->error is set at all, it had to be from a previous
00743          * unload failure.  It's obviously okay now.
00744          */
00745         g_free(plugin->error);
00746         plugin->error = NULL;
00747 
00748         if (unload_cb != NULL)
00749                 unload_cb(plugin, unload_cb_data);
00750 
00751         purple_signal_emit(purple_plugins_get_handle(), "plugin-unload", plugin);
00752 
00753         purple_prefs_disconnect_by_handle(plugin);
00754 
00755         return TRUE;
00756 #else
00757         return TRUE;
00758 #endif /* PURPLE_PLUGINS */
00759 }

01294 {
01295 #ifdef PURPLE_PLUGINS
01296         GList *f, *files;
01297 
01298         g_return_if_fail(key != NULL);
01299 
01300         files = purple_prefs_get_path_list(key);
01301 
01302         for (f = files; f; f = f->next)
01303         {
01304                 char *filename;
01305                 char *basename;
01306                 PurplePlugin *plugin;
01307 
01308                 if (f->data == NULL)
01309                         continue;
01310 
01311                 filename = f->data;
01312 
01313                 /*
01314                  * We don't know if the filename uses Windows or Unix path

01317                  * of either.
01318                  */
01319                 basename = strrchr(filename, '/');
01320                 if ((basename == NULL) || (basename < strrchr(filename, '\\')))
01321                         basename = strrchr(filename, '\\');
01322                 if (basename != NULL)
01323                         basename++;
01324 
01325                 /* Strip the extension */
01326                 if (basename)
01327                         basename = purple_plugin_get_basename(basename);
01328 
01329                 if (((plugin = purple_plugins_find_with_filename(filename)) != NULL) ||
01330                                 (basename && (plugin = purple_plugins_find_with_basename(basename)) != NULL) ||
01331                                 ((plugin = purple_plugin_probe(filename)) != NULL))
01332                 {
01333                         purple_debug_info("plugins", "Loading saved plugin %s\n",
01334                                                         plugin->path);
01335                         purple_plugin_load(plugin);
01336                 }
01337                 else
01338                 {
01339                         purple_debug_error("plugins", "Unable to find saved plugin %s\n",
01340                                                          filename);
01341                 }
01342 
01343                 g_free(basename);
01344 
01345                 g_free(f->data);
01346         }
01347
