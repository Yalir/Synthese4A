00334 {
00335         PurplePlugin *prpl;
00336         PurplePluginProtocolInfo *prpl_info;
00337 
00338         if (purple_status_is_online(new_status) &&

00340                 purple_network_is_available())
00341         {
00342                 purple_account_connect(account);
00343                 return;
00344         }
00345 
00346         if (!purple_status_is_online(new_status))
00347         {
00348                 if (!purple_account_is_disconnected(account))
00349                         purple_account_disconnect(account);

00351                 else if (!purple_account_get_remember_password(account))
00352                         purple_account_set_password(account, NULL);
00353                 return;
00354         }
00355 
00356         if (purple_account_is_connecting(account))
00357                 /*
00358                  * We don't need to call the set_status PRPL function because
00359                  * the PRPL will take care of setting its status during the
00360                  * connection process.
00361                  */
00362                 return;
00363 
00364         prpl = purple_find_prpl(purple_account_get_protocol_id(account));
00365 
00366         if (prpl == NULL)
00367                 return;
00368 
00369         prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(prpl);
00370 
00371         if (!purple_account_is_disconnected(account) && prpl_info->set_status != NULL)
00372         {
