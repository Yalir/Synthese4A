00031 static void
00032 signed_on_cb(PurpleConnection *conn, void *data)
00033 {
00034         PurpleAccount *account = purple_connection_get_account(conn);
00035 
00036         if (purple_account_get_bool(account, PREF_NAME, FALSE)) {
00037                 if(purple_account_get_remember_password(account))
00038                         purple_debug_error("One Time Password",
00039                                            "Unable to enforce one time password for account %s (%s).\n"
00040                                            "Account is set to remember the password.\n",
00041                                            purple_account_get_username(account),
00042                                            purple_account_get_protocol_name(account));
00043                 else {
00044 
00045                         purple_debug_info("One Time Password", "Clearing password for account %s (%s).\n",
00046                                           purple_account_get_username(account),
00047                                           purple_account_get_protocol_name(account));
00048 
00049                         purple_account_set_password(account, NULL);
00050                         /* TODO: Do we need to somehow clear conn->password ? */
00051                 }
00052         }
00053 }

00055 static gboolean
00056 plugin_load(PurplePlugin *plugin)
00057 {
00058         PurplePlugin *prpl;
00059         PurplePluginProtocolInfo *prpl_info;
00060         PurpleAccountOption *option;
00061         GList *l;
00062 
00063         /* Register protocol preference. */
00064         for (l = purple_plugins_get_protocols(); l != NULL; l = l->next) {
00065                 prpl = (PurplePlugin *)l->data;
00066                 prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(prpl);
00067                 if (prpl_info != NULL && !(prpl_info->options & OPT_PROTO_NO_PASSWORD)) {
00068                         option = purple_account_option_bool_new(_("One Time Password"),
00069                                                                 PREF_NAME, FALSE);
00070                         prpl_info->protocol_options = g_list_append(prpl_info->protocol_options, option);
00071                 }
00072         }
00073 
00074         /* Register callback. */
00075         purple_signal_connect(purple_connections_get_handle(), "signed-on",
00076                               plugin, PURPLE_CALLBACK(signed_on_cb), NULL);
00077 
00078         return TRUE;
00079 }

00081 static gboolean
00082 plugin_unload(PurplePlugin *plugin)
00083 {
00084         PurplePlugin *prpl;
00085         PurplePluginProtocolInfo *prpl_info;
00086         PurpleAccountOption *option;
00087         GList *l, *options;
00088 
00089         /* Remove protocol preference. */
00090         for (l = purple_plugins_get_protocols(); l != NULL; l = l->next) {
00091                 prpl = (PurplePlugin *)l->data;
00092                 prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(prpl);
00093                 if (prpl_info != NULL && !(prpl_info->options & OPT_PROTO_NO_PASSWORD)) {
00094                         options = prpl_info->protocol_options;
00095                         while (options != NULL) {
00096                                 option = (PurpleAccountOption *) options->data;
00097                                 if (strcmp(PREF_NAME, purple_account_option_get_setting(option)) == 0) {
00098                                         prpl_info->protocol_options = g_list_delete_link(prpl_info->protocol_options, options);
00099                                         purple_account_option_destroy(option);
00100                                         break;
00101                                 }
00102                                 options = options->next;
00103                         }
00104                 }
00105         }
00106 
00107         /* Callback will be automagically unregistered */
00108 
00109         return TRUE;
00110 }
