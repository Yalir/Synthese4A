00048 int aim_cachecookie(OscarData *od, IcbmCookie *cookie)
00049 {
00050         IcbmCookie *newcook;
00051 
00052         if (!od || !cookie)
00053                 return -EINVAL;
00054 
00055         newcook = aim_checkcookie(od, cookie->cookie, cookie->type);
00056 
00057         if (newcook == cookie) {
00058                 newcook->addtime = time(NULL);
00059                 return 1;
00060         } else if (newcook)
00061                 aim_cookie_free(od, newcook);
00062 
00063         cookie->addtime = time(NULL);
00064 
00065         cookie->next = od->msgcookies;
00066         od->msgcookies = cookie;
00067 
00068         return 0;
00069 }

00081 IcbmCookie *aim_uncachecookie(OscarData *od, guint8 *cookie, int type)
00082 {
00083         IcbmCookie *cur, **prev;
00084 
00085         if (!cookie || !od->msgcookies)
00086                 return NULL;
00087 
00088         for (prev = &od->msgcookies; (cur = *prev); ) {
00089                 if ((cur->type == type) &&
00090                                 (memcmp(cur->cookie, cookie, 8) == 0)) {
00091                         *prev = cur->next;
00092                         return cur;
00093                 }
00094                 prev = &cur->next;
00095         }
00096 
00097         return NULL;
00098 }
