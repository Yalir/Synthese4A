00378 void
00379 jingle_parse(JabberStream *js, const char *from, JabberIqType type,
00380              const char *id, xmlnode *jingle)
00381 {
00382         const gchar *action;
00383         const gchar *sid;
00384         JingleActionType action_type;
00385         JingleSession *session;
00386 
00387         if (type != JABBER_IQ_SET) {
00388                 /* TODO: send iq error here */
00389                 return;
00390         }
00391 
00392         if (!(action = xmlnode_get_attrib(jingle, "action"))) {
00393                 /* TODO: send iq error here */
00394                 return;
00395         }
00396 
00397         action_type = jingle_get_action_type(action);
00398 
00399         purple_debug_info("jabber", "got Jingle package action = %s\n",
00400                           action);
00401 
00402         if (!(sid = xmlnode_get_attrib(jingle, "sid"))) {
00403                 /* send iq error here */
00404                 return;
00405         }
00406 
00407         if (!(session = jingle_session_find_by_sid(js, sid))
00408                         && strcmp(action, "session-initiate")) {
00409                 purple_debug_error("jingle", "jabber_jingle_session_parse couldn't find session\n");
00410                 /* send iq error here */
00411                 return;
00412         }
00413 
00414         if (action_type == JINGLE_SESSION_INITIATE) {
00415                 if (session) {
00416                         /* This should only happen if you start a session with yourself */
00417                         purple_debug_error("jingle", "Jingle session with "
00418                                         "id={%s} already exists\n", sid);
00419                         /* send iq error */
00420                         return;
00421                 } else {
00422                         char *own_jid = g_strdup_printf("%s@%s/%s", js->user->node,
00423                                         js->user->domain, js->user->resource);
00424                         session = jingle_session_create(js, sid, own_jid, from, FALSE);
00425                         g_free(own_jid);
00426                 }
00427         }
00428 
00429         jingle_actions[action_type].handler(session, jingle);
00430 }
