00066 /*------------------------------------------------------------------------
00067  * Return the MIME type matching the data file.
00068  *
00069  *  @param filename             The name of file
00070  *  @param buf                  The data
00071  *  @param buflen               The length of the data
00072  *  @return                             A MIME type string
00073  */
00074 const char* file_mime_type( const char* filename, const char* buf, int buflen )
00075 {
00076         unsigned int    i;
00077 
00078         /* check for matching magic headers */
00079         for ( i = 0; i < ARRAY_SIZE( mime_types ); i++ ) {
00080 
00081                 if ( buflen < mime_types[i].magic_len ) /* data is shorter than size of magic */
00082                         continue;
00083 
00084                 if ( memcmp( buf, mime_types[i].magic, mime_types[i].magic_len ) == 0 )
00085                         return mime_types[i].mime;
00086         }
00087 
00088         /* we did not find the MIME type, so return the default (application/octet-stream) */
00089         return MIME_TYPE_OCTETSTREAM;
00090 }

00148  * Start the file transfer.
00149  *
00150  *  @param xfer                 The file transfer object
00151  */
00152 static void mxit_xfer_start( PurpleXfer* xfer )
00153 {
00154         unsigned char*  buffer;
00155         int                             size;
00156         int                             wrote;
00157 
00158         purple_debug_info( MXIT_PLUGIN_ID, "mxit_xfer_start\n" );
00159 
00160         if ( purple_xfer_get_type( xfer ) == PURPLE_XFER_SEND ) {
00161                 /*
00162                  * the user wants to send a file to one of his contacts. we need to create
00163                  * a buffer and copy the file data into memory and then we can send it to
00164                  * the contact. we will send the whole file with one go.
00165                  */
00166                 buffer = g_malloc( xfer->bytes_remaining );
00167                 size = fread( buffer, xfer->bytes_remaining, 1, xfer->dest_fp );
00168 
00169                 wrote = purple_xfer_write( xfer, buffer, xfer->bytes_remaining );
00170                 if ( wrote > 0 )
00171                         purple_xfer_set_bytes_sent( xfer, wrote );
00172 
00173                 /* free the buffer */
00174                 g_free( buffer );
00175                 buffer = NULL;
00176         }
00177 }

00386  *  @param session              The MXit session object
00387  *  @param fileid               A unique ID that identifies this file

00389 static PurpleXfer* find_mxit_xfer( struct MXitSession* session, const char* fileid )
00390 {
00391         GList*          item    = NULL;
00392         PurpleXfer*     xfer    = NULL;
00393 
00394         item = purple_xfers_get_all();          /* list of all active transfers */
00395         while ( item ) {
00396                 xfer = item->data;
00397 
00398                 if ( xfer->account == session->acc ) {
00399                         /* transfer is associated with this MXit account */
00400                         struct mxitxfer* mx     = xfer->data;
00401 
00402                         /* does the fileid match? */
00403                         if ( ( mx ) && ( memcmp( mx->fileid, fileid, MXIT_CHUNK_FILEID_LEN ) == 0 ) )
00404                                 break;
00405                 }
00406 
00407                 item = g_list_next( item );
00408         }
00409 
00410         if ( item )
00411                 return item->data;
00412         else
00413                 return NULL;
