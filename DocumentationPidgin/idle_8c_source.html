00094 static gboolean no_away = FALSE;
00095 static gint time_until_next_idle_event;
00096 /*
00097  * This function should be called when you think your idle state
00098  * may have changed.  Maybe you're over the 10-minute mark and
00099  * Purple should start reporting idle time to the server.  Maybe
00100  * you've returned from being idle.  Maybe your auto-away message
00101  * should be set.
00102  *
00103  * There is no harm to calling this many many times, other than
00104  * it will be kinda slow.  This is called by a timer set when
00105  * Purple starts.  It is also called when you send an IM, a chat, etc.
00106  *
00107  * This function has 3 sections.
00108  * 1. Get your idle time.  It will query XScreenSaver or Windows
00109  *    or use the Purple idle time.  Whatever.
00110  * 2. Set or unset your auto-away message.
00111  * 3. Report your current idle time to the IM server.
00112  */
00113 
00114 static void
00115 check_idleness(void)
00116 {
00117         time_t time_idle;
00118         gboolean auto_away;
00119         const gchar *idle_reporting;
00120         gboolean report_idle = TRUE;
00121         gint away_seconds = 0;
00122         gint idle_recheck_interval = 0;
00123         gint idle_poll_seconds = purple_prefs_get_int("/purple/away/mins_before_away") * 60;
00124         purple_signal_emit(purple_blist_get_handle(), "update-idle");
00125 
00126         idle_reporting = purple_prefs_get_string("/purple/away/idle_reporting");
00127         auto_away = purple_prefs_get_bool("/purple/away/away_when_idle");
00128 
00129         if (purple_strequal(idle_reporting, "system") &&
00130                 (idle_ui_ops != NULL) && (idle_ui_ops->get_time_idle != NULL))
00131         {
00132                 /* Use system idle time (mouse or keyboard movement, etc.) */
00133                 time_idle = idle_ui_ops->get_time_idle();
00134                 idle_recheck_interval = 1;
00135         }
00136         else if (purple_strequal(idle_reporting, "purple"))
00137         {
00138                 /* Use 'Purple idle' */
00139                 time_idle = time(NULL) - last_active_time;
00140                 idle_recheck_interval = 0;
00141         }
00142         else
00143         {
00144                 /* Don't report idle time */
00145                 time_idle = 0;
00146                 report_idle = FALSE;
00147 
00148                 /* If we're not reporting idle, we can still do auto-away.
00149                  * First try "system" and if that isn't possible, use "purple" */
00150                 if (auto_away)
00151                 {
00152                         if ((idle_ui_ops != NULL) && (idle_ui_ops->get_time_idle != NULL))
00153                         {
00154                                 time_idle = idle_ui_ops->get_time_idle();
00155                                 idle_recheck_interval = 1;
00156                         }
00157                         else
00158                         {
00159                                 time_idle = time(NULL) - last_active_time;
00160                                 idle_recheck_interval = 0;
00161                         }
00162                 }
00163                 else
00164                 {
00165                         if (!no_away)
00166                         {
00167                                 no_away = TRUE;
00168                                 purple_savedstatus_set_idleaway(FALSE);
00169                         }
00170                         time_until_next_idle_event = 0;
00171                         return;
00172                 }
00173         }
00174 
00175         time_until_next_idle_event = idle_poll_seconds - time_idle;
00176         if (time_until_next_idle_event < 0)
00177         {
00178                 /* If we're already idle, check again as appropriate. */
00179                 time_until_next_idle_event = idle_recheck_interval;
00180         }
00181 
00182         if (auto_away || !no_away)
00183                 away_seconds = 60 * purple_prefs_get_int("/purple/away/mins_before_away");
00184 
00185         if (auto_away && time_idle > away_seconds)
00186         {
00187                 purple_savedstatus_set_idleaway(TRUE);
00188                 no_away = FALSE;
00189         }
00190         else if (purple_savedstatus_is_idleaway() && time_idle < away_seconds)
00191         {
00192                 purple_savedstatus_set_idleaway(FALSE);
00193                 if (time_until_next_idle_event == 0 || (away_seconds - time_idle) < time_until_next_idle_event)
00194                         time_until_next_idle_event = away_seconds - time_idle;
00195         }
00196 
00197         /* Idle reporting stuff */
00198         if (report_idle && (time_idle >= idle_poll_seconds))
00199         {
00200                 GList *l;
00201                 for (l = purple_connections_get_all(); l != NULL; l = l->next)
00202                 {
00203                         PurpleConnection *gc = l->data;
00204                         set_account_idle(purple_connection_get_account(gc), time_idle);
00205                 }
00206         }
00207         else if (!report_idle || (time_idle < idle_poll_seconds ))
00208         {
00209                 while (idled_accts != NULL)
00210                         set_account_unidle(idled_accts->data);
00211         }
00212 }

00215 /*
00216  * Check idle and set the timer to fire at the next idle-worth event
00217  */
00218 static gboolean
00219 check_idleness_timer(void)
00220 {
00221         check_idleness();
00222         if (time_until_next_idle_event == 0)
00223                 idle_timer = 0;
00224         else
00225         {
00226                 /* +1 for the boundary,
00227                  * +1 more for g_timeout_add_seconds rounding. */
00228                 idle_timer = purple_timeout_add_seconds(time_until_next_idle_event + 2, (GSourceFunc)check_idleness_timer, NULL);
00229         }
00230         return FALSE;
00231 }
