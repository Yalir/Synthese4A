00041 {

00047                 return -EINVAL;

00049         byte_stream_new(&bs, 2 + 2 + iconlen);

00100 {
00101         FlapConnection *conn;
00102         ByteStream bs;
00103         aim_snacid_t snacid;
00104 
00105         if (!od || !(conn = flap_connection_findbygroup(od, SNAC_FAMILY_BART)) || !bn || !strlen(bn) || !iconcsum || !iconcsumlen)
00106                 return -EINVAL;
00107 
00108         byte_stream_new(&bs, 1+strlen(bn) + 4 + 1+iconcsumlen);
00109 
00110         /* Buddy name */
00111         byte_stream_put8(&bs, strlen(bn));
00112         byte_stream_putstr(&bs, bn);
00113 
00114         /* Some numbers.  You like numbers, right? */
00115         byte_stream_put8(&bs, 0x01);
00116         byte_stream_put16(&bs, 0x0001);
00117         byte_stream_put8(&bs, iconcsumtype);
00118 
00119         /* Icon string */
00120         byte_stream_put8(&bs, iconcsumlen);
00121         byte_stream_putraw(&bs, iconcsum, iconcsumlen);
00122 
00123         snacid = aim_cachesnac(od, SNAC_FAMILY_BART, 0x0004, 0x0000, NULL, 0);
00124         flap_connection_send_snac(od, conn, SNAC_FAMILY_BART, 0x0004, snacid, &bs);
00125 
00126         byte_stream_destroy(&bs);
00127 
00128         return 0;
