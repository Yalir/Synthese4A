00355 /*
00356  * Step 1:
00357  * Isolate RTF keywords and send them to rtf_parse_keyword;
00358  * Push and pop state at the start and end of RTF groups;
00359  * Send text to rtf_dispatch_char for further processing.
00360  */
00361 static int
00362 rtf_parse(NMRtfContext *ctx)
00363 {
00364     int status;
00365     guchar ch;
00366     guchar hex_byte = 0;
00367     int hex_count = 2;
00368         int len;
00369 
00370         if (ctx->input == NULL)
00371                 return NMRTF_OK;
00372 
00373     while (rtf_get_char(ctx, &ch) == NMRTF_OK) {
00374         if (ctx->depth < 0)
00375             return NMRTF_STACK_UNDERFLOW;
00376 
00377                 /* if we're parsing binary data, handle it directly */
00378         if (ctx->ris == NMRTF_STATE_BIN) {
00379             if ((status = rtf_dispatch_char(ctx, ch)) != NMRTF_OK)
00380                 return status;
00381         } else {
00382             switch (ch) {
00383                                 case '{':
00384                                         if (ctx->depth > NMRTF_MAX_DEPTH)
00385                                                 return NMRTF_STACK_OVERFLOW;
00386                     rtf_flush_data(ctx);
00387                                         if ((status = rtf_push_state(ctx)) != NMRTF_OK)
00388                                                 return status;
00389                                         break;
00390                                 case '}':
00391                                         rtf_flush_data(ctx);
00392 
00393                                         /* for some reason there is always an unwanted '\par' at the end */
00394                                         if (ctx->rds == NMRTF_STATE_NORMAL) {
00395                                                 len = ctx->output->len;
00396                                                 if (ctx->output->str[len-1] == '\n')
00397                                                         ctx->output = g_string_truncate(ctx->output, len-1);
00398                                         }
00399 
00400                                         if ((status = rtf_pop_state(ctx)) != NMRTF_OK)
00401                                                 return status;
00402 
00403                                         if (ctx->depth < 0)
00404                                                 return NMRTF_STACK_OVERFLOW;
00405                                         break;
00406                                 case '\\':
00407                                         if ((status = rtf_parse_keyword(ctx)) != NMRTF_OK)
00408                                                 return status;
00409                                         break;
00410                                 case 0x0d:
00411                                 case 0x0a:          /*  cr and lf are noise characters... */
00412                                         break;
00413                                 default:
00414                                         if (ctx->ris == NMRTF_STATE_NORMAL) {
00415                                                 if ((status = rtf_dispatch_char(ctx, ch)) != NMRTF_OK)
00416                                                         return status;
00417                                         } else {               /* parsing a hex encoded character */
00418                                                 if (ctx->ris != NMRTF_STATE_HEX)
00419                                                         return NMRTF_ASSERTION;
00420 
00421                                                 hex_byte = hex_byte << 4;
00422                                                 if (isdigit(ch))
00423                                                         hex_byte += (char) ch - '0';
00424                                                 else {
00425                                                         if (islower(ch)) {
00426                                                                 if (ch < 'a' || ch > 'f')
00427                                                                         return NMRTF_INVALID_HEX;
00428                                                                 hex_byte += (char) ch - 'a' + 10;
00429                                                         } else {
00430                                                                 if (ch < 'A' || ch > 'F')
00431                                                                         return NMRTF_INVALID_HEX;
00432                                                                 hex_byte += (char) ch - 'A' + 10;
00433                                                         }
00434                                                 }
00435                                                 hex_count--;
00436                                                 if (hex_count == 0) {
00437                                                         if ((status = rtf_dispatch_char(ctx, hex_byte)) != NMRTF_OK)
00438                                                                 return status;
00439                                                         hex_count = 2;
00440                                                         hex_byte = 0;
00441                                                         ctx->ris = NMRTF_STATE_NORMAL;
00442                                                 }
00443                                         }
00444                                         break;
00445             }
00446         }
00447     }
00448     if (ctx->depth < 0)
00449         return NMRTF_STACK_OVERFLOW;
00450     if (ctx->depth > 0)
00451         return NMRTF_UNMATCHED_BRACE;
00452     return NMRTF_OK;
00453 }

00458 static int
00459 rtf_push_state(NMRtfContext *ctx)
00460 {
00461     NMRtfStateSave *save = g_new0(NMRtfStateSave, 1);

00471 /*
00472  * Restore the state at the top of the stack
00473  */
00474 static int
00475 rtf_pop_state(NMRtfContext *ctx)
00476 {
00477     NMRtfStateSave *save_old;
00478         GSList *link_old;
00479 
00480     if (ctx->saved == NULL)
00481         return NMRTF_STACK_UNDERFLOW;
00482 
00483         save_old = ctx->saved->data;
00484     ctx->chp = save_old->chp;
00485     ctx->rds = save_old->rds;
00486     ctx->ris = save_old->ris;
00487     (ctx->depth)--;
00488 
00489     g_free(save_old);
00490         link_old = ctx->saved;
00491         ctx->saved = g_slist_remove_link(ctx->saved, link_old);
00492         g_slist_free_1(link_old);
00493     return NMRTF_OK;
00494 }

00496 /*
00497  * Step 2:
00498  * Get a control word (and its associated value) and
00499  * dispatch the control.
00500  */
00501 static int
00502 rtf_parse_keyword(NMRtfContext *ctx)
00503 {
00504         int status = NMRTF_OK;
00505     guchar ch;
00506     gboolean param_set = FALSE;
00507     gboolean is_neg = FALSE;
00508     int param = 0;
00509     char keyword[30];
00510     char parameter[20];
00511         int i;
00512 
00513     keyword[0] = '\0';
00514     parameter[0] = '\0';
00515     if ((status = rtf_get_char(ctx, &ch)) != NMRTF_OK)
00516         return status;
00517 
00518     if (!isalpha(ch)) {
00519                 /* a control symbol; no delimiter. */
00520         keyword[0] = (char) ch;
00521         keyword[1] = '\0';
00522         return rtf_dispatch_control(ctx, keyword, 0, param_set);
00523     }
00524 
00525         /* parse keyword */
00526         for (i = 0; isalpha(ch) && (i < sizeof(keyword) - 1); rtf_get_char(ctx, &ch)) {
00527                 keyword[i] = (char) ch;
00528                 i++;
00529         }
00530         keyword[i] = '\0';
00531 
00532         /* check for '-' indicated a negative parameter value  */
00533     if (ch == '-') {
00534         is_neg = TRUE;
00535         if ((status = rtf_get_char(ctx, &ch)) != NMRTF_OK)
00536             return status;
00537     }
00538 
00539         /* check for numerical param */
00540     if (isdigit(ch)) {
00541 
00542         param_set = TRUE;
00543                 for (i = 0; isdigit(ch) && (i < sizeof(parameter) - 1); rtf_get_char(ctx, &ch)) {
00544                         parameter[i] = (char) ch;
00545                         i++;
00546                 }
00547                 parameter[i] = '\0';
00548 
00549         ctx->param = param = atoi(parameter);
00550         if (is_neg)
00551             ctx->param = param = -param;
00552     }
00553 
00554         /* space after control is optional, put character back if it is not a space */
00555     if (ch != ' ')
00556         rtf_unget_char(ctx, ch);
00557 
00558     return rtf_dispatch_control(ctx, keyword, param, param_set);
00559 }

00564 static int

00694 /*
00695  * Step 3.
00696  * Search the table for keyword and evaluate it appropriately.
00697  *
00698  * Inputs:
00699  * keyword:   The RTF control to evaluate.
00700  * param:     The parameter of the RTF control.
00701  * param_set: TRUE if the control had a parameter; (that is, if param is valid)
00702  *            FALSE if it did not.
00703  */
00704 static int
00705 rtf_dispatch_control(NMRtfContext *ctx, char *keyword, int param, gboolean param_set)
00706 {
00707     int idx;
00708 
00709     for (idx = 0; idx < table_size; idx++) {
00710         if (strcmp(keyword, rtf_symbols[idx].keyword) == 0)
00711             break;
00712         }
00713 
00714     if (idx == table_size)  {
00715         if (ctx->skip_unknown)
00716             ctx->rds = NMRTF_STATE_SKIP;
00717         ctx->skip_unknown = FALSE;
00718         return NMRTF_OK;
00719     }
00720 
00721     /* found it! use kwd_type and action to determine what to do with it. */
00722     ctx->skip_unknown = FALSE;
00723     switch (rtf_symbols[idx].kwd_type) {
00724                 case NMRTF_KWD_PROP:
00725                         if (rtf_symbols[idx].pass_default || !param_set)
00726                                 param = rtf_symbols[idx].default_val;
00727                         return rtf_apply_property(ctx, rtf_symbols[idx].action, param);
00728                 case NMRTF_KWD_CHAR:
00729                         return rtf_dispatch_char(ctx, rtf_symbols[idx].action);
00730                 case NMRTF_KWD_DEST:
00731                         return rtf_change_destination(ctx, rtf_symbols[idx].action);
00732                 case NMRTF_KWD_SPEC:
00733                         return rtf_dispatch_special(ctx, rtf_symbols[idx].action);
00734                 default:
00735                         return NMRTF_BAD_TABLE;
00736     }
00737     return NMRTF_BAD_TABLE;
00738 }
