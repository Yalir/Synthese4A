01050

01111 static void
01112 do_transfer(PurpleXfer *xfer)
01113 {
01114         PurpleXferUiOps *ui_ops;
01115         guchar *buffer = NULL;
01116         gssize r = 0;
01117 
01118         ui_ops = purple_xfer_get_ui_ops(xfer);
01119 
01120         if (xfer->type == PURPLE_XFER_RECEIVE) {
01121                 r = purple_xfer_read(xfer, &buffer);
01122                 if (r > 0) {
01123                         size_t wc;
01124                         if (ui_ops && ui_ops->ui_write)
01125                                 wc = ui_ops->ui_write(xfer, buffer, r);
01126                         else
01127                                 wc = fwrite(buffer, 1, r, xfer->dest_fp);
01128 
01129                         if (wc != r) {
01130                                 purple_debug_error("filetransfer", "Unable to write whole buffer.\n");
01131                                 purple_xfer_cancel_local(xfer);
01132                                 g_free(buffer);
01133                                 return;
01134                         }
01135 
01136                         if ((purple_xfer_get_size(xfer) > 0) &&
01137                                 ((purple_xfer_get_bytes_sent(xfer)+r) >= purple_xfer_get_size(xfer)))
01138                                 purple_xfer_set_completed(xfer, TRUE);
01139                 } else if(r < 0) {
01140                         purple_xfer_cancel_remote(xfer);
01141                         g_free(buffer);
01142                         return;
01143                 }
01144         } else if (xfer->type == PURPLE_XFER_SEND) {
01145                 size_t result = 0;
01146                 size_t s = MIN(purple_xfer_get_bytes_remaining(xfer), xfer->current_buffer_size);
01147                 PurpleXferPrivData *priv = g_hash_table_lookup(xfers_data, xfer);
01148                 gboolean read = TRUE;
01149 
01150                 /* this is so the prpl can keep the connection open
01151                    if it needs to for some odd reason. */
01152                 if (s == 0) {
01153                         if (xfer->watcher) {
01154                                 purple_input_remove(xfer->watcher);
01155                                 xfer->watcher = 0;
01156                         }
01157                         return;
01158                 }
01159 
01160                 if (priv->buffer) {
01161                         if (priv->buffer->len < s) {
01162                                 s -= priv->buffer->len;
01163                                 read = TRUE;
01164                         } else {
01165                                 read = FALSE;
01166                         }
01167                 }
01168 
01169                 if (read) {
01170                         if (ui_ops && ui_ops->ui_read) {
01171                                 gssize tmp = ui_ops->ui_read(xfer, &buffer, s);
01172                                 if (tmp == 0) {
01173                                         /*
01174                                          * The UI claimed it was ready, but didn't have any data for
01175                                          * us...  It will call purple_xfer_ui_ready when ready, which
01176                                          * sets back up this watcher.
01177                                          */
01178                                         if (xfer->watcher != 0) {
01179                                                 purple_input_remove(xfer->watcher);
01180                                                 xfer->watcher = 0;
01181                                         }
01182 
01183                                         /* Need to indicate the prpl is still ready... */
01184                                         priv->ready |= PURPLE_XFER_READY_PRPL;
01185 
01186                                         g_return_if_reached();
01187                                 } else if (tmp < 0) {
01188                                         purple_debug_error("filetransfer", "Unable to read whole buffer.\n");
01189                                         purple_xfer_cancel_local(xfer);
01190                                         return;
01191                                 }
01192 
01193                                 result = tmp;
01194                         } else {
01195                                 buffer = g_malloc(s);
01196                                 result = fread(buffer, 1, s, xfer->dest_fp);
01197                                 if (result != s) {
01198                                         purple_debug_error("filetransfer", "Unable to read whole buffer.\n");
01199                                         purple_xfer_cancel_local(xfer);
01200                                         g_free(buffer);
01201                                         return;
01202                                 }
01203                         }
01204                 }
01205 
01206                 if (priv->buffer) {
01207                         g_byte_array_append(priv->buffer, buffer, result);
01208                         g_free(buffer);
01209                         buffer = priv->buffer->data;
01210                         result = priv->buffer->len;
01211                 }
01212 
01213                 r = purple_xfer_write(xfer, buffer, result);
01214 
01215                 if (r == -1) {
01216                         purple_xfer_cancel_remote(xfer);
01217                         if (!priv->buffer)
01218                                 /* We don't free buffer if priv->buffer is set, because in
01219                                    that case buffer doesn't belong to us. */
01220                                 g_free(buffer);
01221                         return;
01222                 } else if (r == result) {
01223                         /*
01224                          * We managed to write the entire buffer.  This means our
01225                          * network is fast and our buffer is too small, so make it
01226                          * bigger.
01227                          */
01228                         purple_xfer_increase_buffer_size(xfer);
01229                 } else {
01230                         if (ui_ops && ui_ops->data_not_sent)
01231                                 ui_ops->data_not_sent(xfer, buffer + r, result - r);
01232                 }
01233 
01234                 if (priv->buffer) {
01235                         /*
01236                          * Remove what we wrote
01237                          * If we wrote the whole buffer the byte array will be empty
01238                          * Otherwise we'll keep what wasn't sent for next time.
01239                          */
01240                         buffer = NULL;
01241                         g_byte_array_remove_range(priv->buffer, 0, r);
01242                 }
01243         }
01244 
01245         if (r > 0) {
01246                 if (purple_xfer_get_size(xfer) > 0)
01247                         xfer->bytes_remaining -= r;
