00163 /*------------------------------------------------------------------------
00164  * Extract an ASN.1 formatted length field from the data.
00165  *
00166  *  @param data                         The source data
00167  *  @param size                         The extracted length
00168  *  @return                                     The number of bytes extracted
00169  */
00170 static unsigned int asn_getlength( const char* data, int* size )
00171 {
00172         unsigned int    len             = 0;
00173         unsigned char   bytes;
00174         unsigned char   byte;
00175         int                             i;
00176 
00177         /* first byte specifies the number of bytes in the length */
00178         bytes = ( data[0] & ~0x80 );
00179         if ( bytes > sizeof( unsigned int ) ) {
00180                 /* file too big! */
00181                 return -1;
00182         }
00183         data++;
00184 
00185         /* parse out the actual length */
00186         for ( i = 0; i < bytes; i++ ) {
00187                 byte = data[i];
00188                 len <<= 8;
00189                 len += byte;
00190         }
00191 
00192         *size = len;
00193         return bytes + 1;
00194 }

00197 /*------------------------------------------------------------------------
00198  * Extract an ASN.1 formatted UTF-8 string field from the data.
00199  *
00200  *  @param data                         The source data
00201  *  @param type                         Expected type of string
00202  *  @param utf8                         The extracted string.  Must be deallocated by caller.
00203  *  @return                                     The number of bytes extracted
00204  */
00205 static int asn_getUtf8( const char* data, char type, char** utf8 )
00206 {
00207         int             len;
00208 
00209         /* validate the field type [1 byte] */
00210         if ( data[0] != type ) {
00211                 /* this is not a utf-8 string! */
00212                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid UTF-8 encoded string in ASN data (0x%02X)\n", (unsigned char) data[0] );
00213                 return -1;
00214         }
00215 
00216         len = data[1];                                          /* length field [1 bytes] */
00217         *utf8 = g_malloc( len + 1 );
00218         memcpy( *utf8, &data[2], len );         /* data field */
00219         (*utf8)[len] = '\0';
00220 
00221         return ( len + 2 );
00222 }

00242 /*------------------------------------------------------------------------
00243  * Split the message into smaller messages and send them one at a time
00244  * to pidgin to be displayed on the UI
00245  *
00246  *  @param mx                           The received message object
00247  */
00248 static void mxit_show_split_message( struct RXMsgData* mx )
00249 {
00250         GString*                msg             = NULL;
00251         char*                   ch              = NULL;
00252         int                             pos             = 0;
00253         int                             start   = 0;
00254         int                             l_nl    = 0;
00255         int                             l_sp    = 0;
00256         int                             l_gt    = 0;
00257         int                             stop    = 0;
00258         int                             tags    = 0;
00259         gboolean                intag   = FALSE;
00260 
00261         /*
00262          * awful hack to work around the awful hack in pidgin to work around GtkIMHtml's
00263          * inefficient rendering of messages with lots of formatting changes.
00264          * (reference: see the function pidgin_conv_write_conv() in gtkconv.c) the issue
00265          * is that when you have more than 100 '<' characters in the message passed to
00266          * pidgin, none of the markup (including links) are rendered and thus just dump
00267          * all the text as is to the conversation window. this message dump is very
00268          * confusing and makes it totally unusable. to work around this we will count
00269          * the amount of tags and if its more than the pidgin threshold, we will just
00270          * break the message up into smaller parts and send them separately to pidgin.
00271          * to the user it will look like multiple messages, but at least he will be able
00272          * to use and understand it.
00273          */
00274 
00275         ch = mx->msg->str;
00276         pos = start;
00277         while ( ch[pos] ) {
00278 
00279                 if ( ch[pos] == '<' ) {
00280                         tags++;
00281                         intag = TRUE;
00282                 }
00283                 else if ( ch[pos] == '\n' ) {
00284                         l_nl = pos;
00285                 }
00286                 else if ( ch[pos] == '>' ) {
00287                         l_gt = pos;
00288                         intag = FALSE;
00289                 }
00290                 else if ( ch[pos] == ' ' ) {
00291                         /* ignore spaces inside tags */
00292                         if ( !intag )
00293                                 l_sp = pos;
00294                 }
00295                 else if ( ( ch[pos] == 'w' ) && ( pos + 4 < mx->msg->len ) && ( memcmp( &ch[pos], "www.", 4 ) == 0 ) ) {
00296                         tags += 2;
00297                 }
00298                 else if ( ( ch[pos] == 'h' ) && ( pos + 8 < mx->msg->len ) && ( memcmp( &ch[pos], "http://", 7 ) == 0 ) ) {
00299                         tags += 2;
00300                 }
00301 
00302                 if ( tags > MXIT_MAX_MSG_TAGS ) {
00303                         /* we have reached the maximum amount of tags pidgin (gtk) can handle per message.
00304                            so its time to send what we have and then start building a new message */
00305 
00306                         /* now find the right place to break the message */
00307                         if ( l_nl > start ) {
00308                                 /* break at last '\n' char */
00309                                 stop = l_nl;
00310                                 ch[stop] = '\0';
00311                                 msg = g_string_new( &ch[start] );
00312                                 ch[stop] = '\n';
00313                         }
00314                         else if ( l_sp > start ) {
00315                                 /* break at last ' ' char */
00316                                 stop = l_sp;
00317                                 ch[stop] = '\0';
00318                                 msg = g_string_new( &ch[start] );
00319                                 ch[stop] = ' ';
00320                         }
00321                         else {
00322                                 /* break at the last '>' char */
00323                                 char t;
00324                                 stop = l_gt + 1;
00325                                 t = ch[stop];
00326                                 ch[stop] = '\0';
00327                                 msg = g_string_new( &ch[start] );
00328                                 ch[stop] = t;
00329                                 stop--;
00330                         }
00331 
00332                         /* push message to pidgin */
00333                         serv_got_im( mx->session->con, mx->from, msg->str, mx->flags, mx->timestamp );
00334                         g_string_free( msg, TRUE );
00335                         msg = NULL;
00336 
00337                         /* next part need this flag set */
00338                         mx->flags |= PURPLE_MESSAGE_RAW;
00339 
00340                         tags = 0;
00341                         start = stop + 1;
00342                         pos = start;
00343                 }
00344                 else
00345                         pos++;
00346         }
00347 
00348         if ( start != pos ) {
00349                 /* send the last part of the message */
00350 
00351                 /* build the string */
00352                 ch[pos] = '\0';
00353                 msg = g_string_new( &ch[start] );
00354                 ch[pos] = '\n';
00355 
00356                 /* push message to pidgin */
00357                 serv_got_im( mx->session->con, mx->from, msg->str, mx->flags, mx->timestamp );
00358                 g_string_free( msg, TRUE );
00359                 msg = NULL;
00360         }
00361 }

00364 /*------------------------------------------------------------------------
00365  * Insert custom emoticons and inline images into the message (if there
00366  * are any), then give the message to the UI to display to the user.
00367  *
00368  *  @param mx                           The received message object
00369  */
00370 void mxit_show_message( struct RXMsgData* mx )
00371 {
00372         char*                           pos;
00373         int                                     start;
00374         unsigned int            end;
00375         int                                     emo_ofs;
00376         char*                           ii;
00377         char                            tag[64];
00378         int*                            img_id;
00379 
00380         if ( mx->got_img ) {
00381                 /* search and replace all emoticon tags with proper image tags */
00382 
00383                 while ( ( pos = strstr( mx->msg->str, MXIT_II_TAG ) ) != NULL ) {
00384                         start = pos - mx->msg->str;                                     /* offset at which MXIT_II_TAG starts */
00385                         emo_ofs = start + strlen( MXIT_II_TAG );        /* offset at which EMO's ID starts */
00386                         end = emo_ofs + 1;                                                      /* offset at which MXIT_II_TAG ends */
00387 
00388                         while ( ( end < mx->msg->len ) && ( mx->msg->str[end] != '>' ) )
00389                                 end++;
00390 
00391                         if ( end == mx->msg->len )                      /* end of emoticon tag not found */
00392                                 break;
00393 
00394                         ii = g_strndup(&mx->msg->str[emo_ofs], end - emo_ofs);
00395 
00396                         /* remove inline image tag */
00397                         g_string_erase( mx->msg, start, ( end - start ) + 1 );
00398 
00399                         /* find the image entry */
00400                         img_id = (int*) g_hash_table_lookup( mx->session->iimages, ii );
00401                         if ( !img_id ) {
00402                                 /* inline image not found, so we will just skip it */
00403                                 purple_debug_error( MXIT_PLUGIN_ID, "inline image NOT found (%s)\n", ii );
00404                         }
00405                         else {
00406                                 /* insert img tag */
00407                                 g_snprintf( tag, sizeof( tag ), "<img id=\"%i\">", *img_id );
00408                                 g_string_insert( mx->msg, start, tag );
00409                         }
00410 
00411                         g_free(ii);
00412                 }
00413         }
00414 
00415 #ifdef MXIT_DEBUG_MARKUP
00416         purple_debug_info( MXIT_PLUGIN_ID, "Markup RX (converted): '%s'\n", mx->msg->str );
00417 #endif
00418 
00419         if ( mx->processed ) {
00420                 /* this message has already been taken care of, so just ignore it here */
00421         }
00422         else if ( mx->chatid < 0 ) {
00423                 /* normal chat message */
00424                 mxit_show_split_message( mx );
00425         }
00426         else {
00427                 /* this is a multimx message */
00428                 serv_got_chat_in( mx->session->con, mx->chatid, mx->from, mx->flags, mx->msg->str, mx->timestamp);
00429         }
00430 
00431         /* freeup resource */
00432         free_markupdata( mx );
00433 }

00437  * Extract the custom emoticon ID from the message.
00438  *
00439  *  @param message                      The input data
00440  *  @param emid                         The extracted emoticon ID
00441  */
00442 static void parse_emoticon_str( const char* message, char* emid )
00443 {
00444         int             i;
00445 
00446         for ( i = 0; ( message[i] != '\0' && message[i] != '}' && i < MXIT_MAX_EMO_ID ); i++ ) {
00447                 emid[i] = message[i];
00448         }
00449 
00450         if ( message[i] == '\0' ) {
00451                 /* end of message reached, ignore the tag */
00452                 emid[0] = '\0';
00453         }
00454         else if ( i == MXIT_MAX_EMO_ID ) {
00455                 /* invalid tag length, ignore the tag */
00456                 emid[0] = '\0';
00457         }
00458         else
00459                 emid[i] = '\0';
00460 }

00463 /*------------------------------------------------------------------------
00464  * Callback function invoked when a custom emoticon request to the WAP site completes.
00465  *
00466  *  @param url_data
00467  *  @param user_data            The Markup message object
00468  *  @param url_text                     The data returned from the WAP site
00469  *  @param len                          The length of the data returned
00470  *  @param error_message        Descriptive error message
00471  */
00472 static void emoticon_returned( PurpleUtilFetchUrlData* url_data, gpointer user_data, const gchar* url_text, gsize len, const gchar* error_message )
00473 {
00474         struct RXMsgData*       mx                      = (struct RXMsgData*) user_data;
00475         const char*                     data            = url_text;
00476         unsigned int            pos                     = 0;
00477         char                            emo[16];
00478         int                                     id;
00479         char*                           str;
00480         int                                     em_size         = 0;
00481         char*                           em_data         = NULL;
00482         char*                           em_id           = NULL;
00483         int*                            intptr          = NULL;
00484         int                                     res;
00485 
00486 #ifdef  MXIT_DEBUG_EMO
00487         purple_debug_info( MXIT_PLUGIN_ID, "emoticon_returned\n" );
00488 #endif
00489 
00490         if ( !url_text ) {
00491                 /* no reply from the WAP site */
00492                 purple_debug_error( MXIT_PLUGIN_ID, "Error contacting the MXit WAP site. Please try again later (emoticon).\n" );
00493                 goto done;
00494         }
00495 
00496 #ifdef  MXIT_DEBUG_EMO
00497         hex_dump( data, len );
00498 #endif
00499 
00500         /* parse out the emoticon */
00501         pos = 0;
00502 
00503         /* validate the binary data received from the wapsite */
00504         if ( memcmp( MXIT_FRAME_MAGIC, &data[pos], strlen( MXIT_FRAME_MAGIC ) ) != 0 ) {
00505                 /* bad data, magic constant is wrong */
00506                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad magic)\n" );
00507                 goto done;
00508         }
00509         pos += strlen( MXIT_FRAME_MAGIC );
00510 
00511         /* validate the image frame desc byte */
00512         if ( data[pos] != '\x6F' ) {
00513                 /* bad frame desc */
00514                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad frame desc)\n" );
00515                 goto done;
00516         }
00517         pos++;
00518 
00519         /* get the data length */
00520         res = asn_getlength( &data[pos], &em_size );
00521         if ( res <= 0 ) {
00522                 /* bad frame length */
00523                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad frame length)\n" );
00524                 goto done;
00525         }
00526         pos += res;
00527 #ifdef  MXIT_DEBUG_EMO
00528         purple_debug_info( MXIT_PLUGIN_ID, "read the length '%i'\n", em_size );
00529 #endif
00530 
00531         /* utf-8 (emoticon name) */
00532         res = asn_getUtf8( &data[pos], 0x0C, &str );
00533         if ( res <= 0 ) {
00534                 /* bad utf-8 string */
00535                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad name string)\n" );
00536                 goto done;
00537         }
00538         pos += res;
00539 #ifdef  MXIT_DEBUG_EMO
00540         purple_debug_info( MXIT_PLUGIN_ID, "read the string '%s'\n", str );
00541 #endif
00542         g_free( str );
00543         str = NULL;
00544 
00545         /* utf-8 (emoticon shortcut) */
00546         res = asn_getUtf8( &data[pos], 0x81, &str );
00547         if ( res <= 0 ) {
00548                 /* bad utf-8 string */
00549                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad shortcut string)\n" );
00550                 goto done;
00551         }
00552         pos += res;
00553 #ifdef  MXIT_DEBUG_EMO
00554         purple_debug_info( MXIT_PLUGIN_ID, "read the string '%s'\n", str );
00555 #endif
00556         em_id = str;
00557 
00558         /* validate the image data type */
00559         if ( data[pos] != '\x82' ) {
00560                 /* bad frame desc */
00561                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad data type)\n" );
00562                 g_free( em_id );
00563                 goto done;
00564         }
00565         pos++;
00566 
00567         /* get the data length */
00568         res = asn_getlength( &data[pos], &em_size );
00569         if ( res <= 0 ) {
00570                 /* bad frame length */
00571                 purple_debug_error( MXIT_PLUGIN_ID, "Invalid emoticon received from wapsite (bad data length)\n" );
00572                 g_free( em_id );
00573                 goto done;
00574         }
00575         pos += res;
00576 #ifdef  MXIT_DEBUG_EMO
00577         purple_debug_info( MXIT_PLUGIN_ID, "read the length '%i'\n", em_size );
00578 #endif
00579 
00580         if ( g_hash_table_lookup( mx->session->iimages, em_id ) ) {
00581                 /* emoticon found in the table, so ignore this one */
00582                 goto done;
00583         }
00584 
00585         /* make a copy of the data */
00586         em_data = g_malloc( em_size );
00587         memcpy( em_data, &data[pos], em_size );
00588 
00589         /* strip the mxit markup tags from the emoticon id */
00590         if ( ( em_id[0] == '.' ) && ( em_id[1] == '{' ) ) {
00591                 parse_emoticon_str( &em_id[2], emo );
00592                 strcpy( em_id, emo );
00593         }
00594 
00595         /* we now have the emoticon, store it in the imagestore */
00596         id = purple_imgstore_add_with_id( em_data, em_size, NULL );
00597 
00598         /* map the mxit emoticon id to purple image id */
00599         intptr = g_malloc( sizeof( int ) );
00600         *intptr = id;
00601         g_hash_table_insert( mx->session->iimages, em_id, intptr );
00602 
00603         mx->flags |= PURPLE_MESSAGE_IMAGES;
00604 done:
00605         mx->img_count--;
00606         if ( ( mx->img_count == 0 ) && ( mx->converted ) ) {
00607                 /*
00608                  * this was the last outstanding emoticon for this message,
00609                  * so we can now display it to the user.
00610                  */
00611                 mxit_show_message( mx );
00612         }
00613 }

00665 /*------------------------------------------------------------------------
00666  * Extract the nickname from a chatroom message and display it nicely in
00667  * libPurple-style (HTML) markup.
00668  *
00669  *  @param mx                           The received message data object
00670  *  @param message                      The message text
00671  *  @return                                     The length of the message to skip
00672  */
00673 static int mxit_extract_chatroom_nick( struct RXMsgData* mx, char* message, int len, int msgflags )
00674 {
00675         int             i;
00676 
00677         if ( message[0] == '<' ) {
00678                 /*
00679                  * The message MIGHT contains an embedded nickname.  But we can't
00680                  * be sure unless we find the end-of-nickname sequence: (>\n)
00681                  * Search for it....
00682                  */
00683                 gboolean        found   = FALSE;
00684 
00685                 for ( i = 1; i < len; i++ ) {
00686                         if ( ( message[i] == '\n' ) && ( message[i-1] == '>' ) ) {
00687                                 found = TRUE;
00688                                 message[i-1] = '\0';    /* loose the '>' */
00689                                 i++;                                    /* and skip the new-line */
00690                                 break;
00691                         }
00692                 }
00693 
00694                 if ( found ) {
00695                         gchar*          nickname;
00696 
00697                         /*
00698                          * The message definitely had an embedded nickname - generate a marked-up
00699                          * message to be displayed.
00700                          */
00701                         nickname = g_markup_escape_text( &message[1], -1 );
00702 
00703                         /* Remove any MXit escaping from nickname ("\X" --> "X") */
00704                         if ( msgflags & CP_MSG_MARKUP ) {
00705                                 int     nicklen = strlen( nickname );
00706                                 int     j, k;
00707 
00708                                 for ( j = 0, k = 0; j < nicklen; j++ ) {
00709                                         if ( nickname[j] == '\\' )
00710                                                 j++;
00711 
00712                                         nickname[k] = nickname[j];
00713                                         k++;
00714                                 }
00715 
00716                                 nickname[k] = '\0';             /* terminate string */
00717                         }
00718 
00719                         /* add nickname within some BOLD markup to the new converted message */
00720                         g_string_append_printf( mx->msg, "<b>%s:</b> ", nickname );
00721 
00722                         /* free up the resources */
00723                         g_free( nickname );
00724 
00725                         return i;
00726                 }
00727         }
00728 
00729         return 0;

00734 /*------------------------------------------------------------------------
00735  * Convert a message containing MXit protocol markup to libPurple-style (HTML) markup.
00736  *
00737  *  @param mx                           The received message data object
00738  *  @param message                      The message text
00739  *  @param len                          The length of the message
00740  */
00741 void mxit_parse_markup( struct RXMsgData* mx, char* message, int len, short msgtype, int msgflags )
00742 {
00743         char            tmpstr1[128];
00744         char*           ch;
00745         int                     i                       = 0;
00746 
00747         /* tags */
00748         gboolean        tag_bold        = FALSE;
00749         gboolean        tag_under       = FALSE;
00750         gboolean        tag_italic      = FALSE;
00751         int                     font_size       = 0;
00752 
00753 #ifdef MXIT_DEBUG_MARKUP
00754         purple_debug_info( MXIT_PLUGIN_ID, "Markup RX (original): '%s'\n", message );
00755 #endif
00756 
00757 
00758         /*
00759          * supported MXit markup:
00760          * '*'                  bold
00761          * '_'                  underline
00762          * '/'                  italics
00763          * '$'                  highlight text
00764          * '.+'                 inc font size
00765          * '.-'                 dec font size
00766          * '#XXXXXX'    foreground color
00767          * '.{XX}'              custom emoticon
00768          * '\'                  escape the following character
00769          * '::'                 MXit commands
00770          */
00771 
00772 
00773         if ( is_mxit_chatroom_contact( mx->session, mx->from ) ) {
00774                 /* chatroom message, so we need to extract and skip the sender's nickname
00775                  * which is embedded inside the message */
00776                 i = mxit_extract_chatroom_nick( mx, message, len, msgflags );
00777         }
00778 
00779         /* run through the message and check for custom emoticons and markup */
00780         for ( ; i < len; i++ ) {
00781                 switch ( message[i] ) {
00782 
00783 
00784                         /* mxit markup parsing */
00785                         case '*' :
00786                                         if ( !( msgflags & CP_MSG_MARKUP ) ) {
00787                                                 g_string_append_c( mx->msg, message[i] );
00788                                                 break;
00789                                         }
00790 
00791                                         /* bold markup */
00792                                         if ( !tag_bold )
00793                                                 g_string_append( mx->msg, "<b>" );
00794                                         else
00795                                                 g_string_append( mx->msg, "</b>" );
00796                                         tag_bold = !tag_bold;
00797                                         break;
00798                         case '_' :
00799                                         if ( !( msgflags & CP_MSG_MARKUP ) ) {
00800                                                 g_string_append_c( mx->msg, message[i] );
00801                                                 break;
00802                                         }
00803 
00804                                         /* underscore markup */
00805                                         if ( !tag_under )
00806                                                 g_string_append( mx->msg, "<u>" );
00807                                         else
00808                                                 g_string_append( mx->msg, "</u>" );
00809                                         tag_under = !tag_under;
00810                                         break;
00811                         case '/' :
00812                                         if ( !( msgflags & CP_MSG_MARKUP ) ) {
00813                                                 g_string_append_c( mx->msg, message[i] );
00814                                                 break;
00815                                         }
00816 
00817                                         /* italics markup */
00818                                         if ( !tag_italic )
00819                                                 g_string_append( mx->msg, "<i>" );
00820                                         else
00821                                                 g_string_append( mx->msg, "</i>" );
00822                                         tag_italic = !tag_italic;
00823                                         break;
00824                         case '$' :
00825                                         if ( !( msgflags & CP_MSG_MARKUP ) ) {
00826                                                 g_string_append_c( mx->msg, message[i] );
00827                                                 break;
00828                                         }
00829                                         else if ( i + 1 >= len ) {
00830                                                 /* message too short for complete link */
00831                                                 g_string_append_c( mx->msg, '$' );
00832                                                 break;
00833                                         }
00834 
00835                                         /* find the end tag */
00836                                         ch = strstr( &message[i + 1], "$" );
00837                                         if ( ch ) {
00838                                                 /* end found */
00839                                                 *ch = '\0';
00840                                                 mxit_add_html_link( mx, &message[i + 1], FALSE, &message[i + 1] );
00841                                                 *ch = '$';
00842                                                 i += ( ch - &message[i + 1] ) + 1;
00843                                         }
00844                                         else {
00845                                                 g_string_append_c( mx->msg, message[i] );
00846                                         }
00847                                         /* highlight text */
00848                                         break;
00849                         case '#' :
00850                                         if ( !( msgflags & CP_MSG_MARKUP ) ) {
00851                                                 g_string_append_c( mx->msg, message[i] );
00852                                                 break;
00853                                         }
00854                                         else if ( i + COLORCODE_LEN >= len ) {
00855                                                 /* message too short for complete colour code */
00856                                                 g_string_append_c( mx->msg, '#' );
00857                                                 break;
00858                                         }
00859 
00860                                         /* foreground (text) color */
00861                                         memcpy( tmpstr1, &message[i + 1], COLORCODE_LEN );
00862                                         tmpstr1[ COLORCODE_LEN ] = '\0';                        /* terminate string */
00863                                         if ( strcmp( tmpstr1, "??????" ) == 0 ) {
00864                                                 /* need to reset the font */
00865                                                 g_string_append( mx->msg, "</font>" );
00866                                                 i += COLORCODE_LEN;
00867                                         }
00868                                         else if ( strspn( tmpstr1, "0123456789abcdefABCDEF") == COLORCODE_LEN ) {
00869                                                 /* definitely a numeric colour code */
00870                                                 g_string_append_printf( mx->msg, "<font color=\"#%s\">", tmpstr1 );
00871                                                 i += COLORCODE_LEN;
00872                                         }
00873                                         else {
00874                                                 /* not valid colour markup */
00875                                                 g_string_append_c( mx->msg, '#' );
00876                                         }
00877                                         break;
00878                         case '.' :
00879                                         if ( i + 1 >= len ) {
00880                                                 /* message too short */
00881                                                 g_string_append_c( mx->msg, '.' );
00882                                                 break;
00883                                         }
00884 
00885                                         if ( ( msgflags & CP_MSG_EMOTICON ) && ( message[i+1] == '{' ) ) {
00886                                                 /* custom emoticon */
00887                                                 if ( i + 2 >= len ) {
00888                                                         /* message too short */
00889                                                         g_string_append_c( mx->msg, '.' );
00890                                                         break;
00891                                                 }
00892 
00893                                                 parse_emoticon_str( &message[i+2], tmpstr1 );
00894                                                 if ( tmpstr1[0] != '\0' ) {
00895                                                         mx->got_img = TRUE;
00896 
00897                                                         if ( g_hash_table_lookup( mx->session->iimages, tmpstr1 ) ) {
00898                                                                 /* emoticon found in the cache, so we do not have to request it from the WAPsite */
00899                                                         }
00900                                                         else {
00901                                                                 /* request emoticon from the WAPsite */
00902                                                                 mx->img_count++;
00903                                                                 emoticon_request( mx, tmpstr1 );
00904                                                         }
00905 
00906                                                         g_string_append_printf( mx->msg, MXIT_II_TAG"%s>", tmpstr1 );
00907                                                         i += strlen( tmpstr1 ) + 2;
00908                                                 }
00909                                                 else
00910                                                         g_string_append_c( mx->msg, '.' );
00911                                         }
00912                                         else if ( ( msgflags & CP_MSG_MARKUP ) && ( message[i+1] == '+' ) ) {
00913                                                 /* increment text size */
00914                                                 font_size++;
00915                                                 g_string_append_printf( mx->msg, "<font size=\"%+i\">", font_size );
00916                                                 i++;
00917                                         }
00918                                         else if ( ( msgflags & CP_MSG_MARKUP ) && ( message[i+1] == '-' ) ) {
00919                                                 /* decrement text size */
00920                                                 font_size--;
00921                                                 g_string_append_printf( mx->msg, "<font size=\"%+i\">", font_size );
00922                                                 i++;
00923                                         }
00924                                         else
00925                                                 g_string_append_c( mx->msg, '.' );
00926 
00927                                         break;
00928                         case '\\' :
00929                                         if ( i + 1 >= len ) {
00930                                                 /* message too short for an escaped character */
00931                                                 g_string_append_c( mx->msg, '\\' );
00932                                         }
00933                                         else {
00934                                                 /* ignore the next character, because its been escaped */
00935                                                 g_string_append_c( mx->msg, message[i + 1] );
00936                                                 i++;
00937                                         }
00938                                         break;
00939 
00940 
00941                         /* command parsing */
00942                         case ':' :
00943                                         if ( i + 1 >= len ) {
00944                                                 /* message too short */
00945                                                 g_string_append_c( mx->msg, ':' );
00946                                                 break;
00947                                         }
00948 
00949                                         if ( message[i+1] == '@' ) {
00950                                                 /* this is a vibe! */
00951                                                 int             size;
00952 
00953                                                 if ( i + 2 >= len ) {
00954                                                         /* message too short */
00955                                                         g_string_append_c( mx->msg, message[i] );
00956                                                         break;
00957                                                 }
00958 
00959                                                 size = mxit_parse_vibe( mx, &message[i] );
00960                                                 if ( size == 0 )
00961                                                         g_string_append_c( mx->msg, message[i] );
00962                                                 else
00963                                                         i += size;
00964                                         }
00965                                         else if ( msgtype != CP_MSGTYPE_COMMAND ) {
00966                                                 /* this is not a command message */
00967                                                 g_string_append_c( mx->msg, message[i] );
00968                                         }
00969                                         else if ( message[i+1] == ':' ) {
00970                                                 /* parse out the command */
00971                                                 int             size;
00972 
00973                                                 size = mxit_parse_command( mx, &message[i] );
00974                                                 if ( size == 0 )
00975                                                         g_string_append_c( mx->msg, ':' );
00976                                                 else
00977                                                         i += size;
00978                                         }
00979                                         else {
00980                                                 g_string_append_c( mx->msg, ':' );
00981                                         }
00982                                         break;
00983 
00984 
00985                         /* these aren't MXit markup, but are interpreted by libPurple */
00986                         case '<' :
00987                                         g_string_append( mx->msg, "&lt;" );
00988                                         break;
00989                         case '>' :
00990                                         g_string_append( mx->msg, "&gt;" );
00991                                         break;
00992                         case '&' :
00993                                         g_string_append( mx->msg, "&amp;" );
00994                                         break;
00995                         case '"' :
00996                                         g_string_append( mx->msg, "&quot;" );
00997                                         break;
00998 
00999                         default :
01000                                         /* text */
01001                                         g_string_append_c( mx->msg, message[i] );
01002                                         break;
01003                 }
01004         }
01005 
01006         if ( msgflags & CP_MSG_FAREWELL ) {
01007                 /* this is a farewell message */
01008                 g_string_prepend( mx->msg, "<font color=\""MXIT_FAREWELL_MSG_COLOR"\"><i>" );
01009                 g_string_append( mx->msg, "</i></font>" );
01010         }
01011 }

01044 /*------------------------------------------------------------------------
01045  * Convert libpurple (HTML) markup to MXit protocol markup (for sending to MXit).
01046  * Any MXit markup codes in the original message also need to be escaped.
01047  *
01048  *  @param message                      The message text containing libPurple (HTML) markup
01049  *  @return                                     The message text containing MXit markup
01050  */
01051 char* mxit_convert_markup_tx( const char* message, int* msgtype )
01052 {
01053         GString*                        mx;
01054         struct tag*                     tag;
01055         GList*                          entry;
01056         GList*                          tagstack        = NULL;
01057         char*                           reply;
01058         char                            color[8];
01059         int                                     len                     = strlen ( message );
01060         int                                     i;
01061 
01062 #ifdef MXIT_DEBUG_MARKUP
01063         purple_debug_info( MXIT_PLUGIN_ID, "Markup TX (original): '%s'\n", message );
01064 #endif
01065 
01066         /*
01067          * libPurple uses the following HTML markup codes:
01068          *   Bold:                      <b>...</b>
01069          *   Italics:           <i>...</i>
01070          *   Underline:         <u>...</u>
01071          *   Strikethrough:     <s>...</s>                                      (NO MXIT SUPPORT)
01072          *   Font size:         <font size="">...</font>
01073          *   Font type:         <font face="">...</font>        (NO MXIT SUPPORT)
01074          *   Font colour:       <font color=#">...</font>
01075          *   Links:                     <a href="">...</a>
01076          *   Newline:           <br>
01077          *   Inline image:  <IMG ID="">
01078          * The following characters are also encoded:
01079          *   &amp;  &quot;  &lt;  &gt;
01080          */
01081 
01082         /* new message data */
01083         mx = g_string_sized_new( len );
01084 
01085         /* run through the message and check for HTML markup */
01086         for ( i = 0; i < len; i++ ) {
01087 
01088                 switch ( message[i] ) {
01089                         case '<' :
01090                                 if ( purple_str_has_prefix( &message[i], "<b>" ) || purple_str_has_prefix( &message[i], "</b>" ) ) {
01091                                         /* bold */
01092                                         g_string_append_c( mx, '*' );
01093                                 }
01094                                 else if ( purple_str_has_prefix( &message[i], "<i>" ) || purple_str_has_prefix( &message[i], "</i>" ) ) {
01095                                         /* italics */
01096                                         g_string_append_c( mx, '/' );
01097                                 }
01098                                 else if ( purple_str_has_prefix( &message[i], "<u>" ) || purple_str_has_prefix( &message[i], "</u>" ) ) {
01099                                         /* underline */
01100                                         g_string_append_c( mx, '_' );
01101                                 }
01102                                 else if ( purple_str_has_prefix( &message[i], "<br>" ) ) {
01103                                         /* newline */
01104                                         g_string_append_c( mx, '\n' );
01105                                 }
01106                                 else if ( purple_str_has_prefix( &message[i], "<font size=" ) ) {
01107                                         /* font size */
01108                                         int fontsize;
01109 
01110                                         tag = g_new0( struct tag, 1 );
01111                                         tag->type = MXIT_TAG_SIZE;
01112                                         tagstack = g_list_prepend( tagstack, tag );
01113                                         // TODO: implement size control
01114                                         if ( sscanf( &message[i+12], "%i", &fontsize ) ) {
01115                                                 purple_debug_info( MXIT_PLUGIN_ID, "Font size set to %i\n", fontsize );
01116                                         }
01117                                 }
01118                                 else if ( purple_str_has_prefix( &message[i], "<font color=" ) ) {
01119                                         /* font colour */
01120                                         tag = g_new0( struct tag, 1 );
01121                                         tag->type = MXIT_TAG_COLOR;
01122                                         tagstack = g_list_append( tagstack, tag );
01123                                         memset( color, 0x00, sizeof( color ) );
01124                                         memcpy( color, &message[i + 13], 7 );
01125                                         g_string_append( mx, color );
01126                                 }
01127                                 else if ( purple_str_has_prefix( &message[i], "</font>" ) ) {
01128                                         /* end of font tag */
01129                                         entry = g_list_last( tagstack );
01130                                         if ( entry ) {
01131                                                 tag = entry->data;
01132                                                 if ( tag->type == MXIT_TAG_COLOR ) {
01133                                                         /* font color reset */
01134                                                         g_string_append( mx, "#??????" );
01135                                                 }
01136                                                 else if ( tag->type == MXIT_TAG_SIZE ) {
01137                                                         /* font size */
01138                                                         // TODO: implement size control
01139                                                 }
01140                                                 tagstack = g_list_remove( tagstack, tag );
01141                                                 g_free( tag );
01142                                         }
01143                                 }
01144                                 else if ( purple_str_has_prefix( &message[i], "<IMG ID=" ) ) {
01145                                         /* inline image */
01146                                         int imgid;
01147 
01148                                         if ( sscanf( &message[i+9], "%i", &imgid ) ) {
01149                                                 inline_image_add( mx, imgid );
01150                                                 *msgtype = CP_MSGTYPE_COMMAND;          /* inline image must be sent as a MXit command */
01151                                         }
01152                                 }
01153 
01154                                 /* skip to end of tag ('>') */
01155                                 for ( i++; ( i < len ) && ( message[i] != '>' ) ; i++ );
01156 
01157                                 break;
01158 
01159                         case '*' :      /* MXit bold */
01160                         case '_' :      /* MXit underline */
01161                         case '/' :      /* MXit italic */
01162                         case '#' :      /* MXit font color */
01163                         case '$' :      /* MXit highlight text */
01164                         case '\\' :     /* MXit escape backslash */
01165                                 g_string_append( mx, "\\" );                            /* escape character */
01166                                 g_string_append_c( mx, message[i] );            /* character to escape */
01167                                 break;
01168 
01169                         case '.' : /* might be a MXit font size change, or custom emoticon */
01170                                 if ( i + 1 < len ) {
01171                                         if ( ( message[i+1] == '+' ) || ( message[i+1] == '-' ) )
01172                                                 g_string_append( mx, "\\." );           /* escape "." */
01173                                         else
01174                                                 g_string_append_c( mx, '.' );
01175                                 }
01176                                 else
01177                                         g_string_append_c( mx, '.' );
01178                                 break;
01179 
01180                         default:
01181                                 g_string_append_c( mx, message[i] );
01182                                 break;
01183                 }
01184         }
01185 
01186         /* unescape HTML entities to their literal characters (reference: "libpurple/utils.h") */
01187         reply = purple_unescape_html( mx->str );
01188 
01189         g_string_free( mx, TRUE );
01190 
01191 #ifdef MXIT_DEBUG_MARKUP
01192         purple_debug_info( MXIT_PLUGIN_ID, "Markup TX (converted): '%s'\n", reply );
01193 #endif
01194 
01195         return reply;
01196 }
