00033 /*
00034  * This determines whether debug info should be written to the
00035  * console or not.
00036  *
00037  * It doesn't make sense to make this a normal Purple preference
00038  * because it's a command line option.  This will always be FALSE,
00039  * unless the user explicitly started the UI with the -d flag.
00040  * It doesn't matter what this value was the last time Purple was
00041  * started, so it doesn't make sense to save it in prefs.
00042  */
00043 static gboolean debug_enabled = FALSE;

00053 static void
00054 purple_debug_vargs(PurpleDebugLevel level, const char *category,
00055                                  const char *format, va_list args)
00056 {
00057         PurpleDebugUiOps *ops;
00058         char *arg_s = NULL;
00059 
00060         g_return_if_fail(level != PURPLE_DEBUG_ALL);
00061         g_return_if_fail(format != NULL);
00062 
00063         ops = purple_debug_get_ui_ops();
00064 
00065         if (!debug_enabled && ((ops == NULL) || (ops->print == NULL) ||
00066                         (ops->is_enabled && !ops->is_enabled(level, category))))
00067                 return;
00068 
00069         arg_s = g_strdup_vprintf(format, args);
00070 
00071         if (debug_enabled) {
00072                 gchar *ts_s;
00073                 const char *mdate;
00074                 time_t mtime = time(NULL);
00075 
00076 
00077                 mdate = purple_utf8_strftime("%H:%M:%S", localtime(&mtime));
00078                 ts_s = g_strdup_printf("(%s) ", mdate);
00079 
00080                 if (category == NULL)
00081                         g_print("%s%s", ts_s, arg_s);
00082                 else
00083                         g_print("%s%s: %s", ts_s, category, arg_s);
00084 
00085                 g_free(ts_s);
00086         }
00087 
00088         if (ops != NULL && ops->print != NULL)
00089                 ops->print(level, category, arg_s);
00090 
00091         g_free(arg_s);
00092 }
