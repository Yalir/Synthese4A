00076         PurpleConnection *gc = user_data;
00077         YahooData *yd = gc->proto_data;
00078 
00079         yd->url_datas = g_slist_remove(yd->url_datas, url_data);
00080 
00081         if (len == 0) {
00082                 purple_debug_info("yahoo", "No Aliases to process.%s%s\n",
00083                                                   error_message ? " Error:" : "", error_message ? error_message : "");
00084         } else {
00085                 gchar *full_name, *nick_name;
00086                 const char *yid, *id, *fn, *ln, *nn, *alias, *mn;
00087                 const char *hp, *wp, *mo;
00088                 YahooFriend *f;
00089                 PurpleBuddy *b;
00090                 xmlnode *item, *contacts;
00091                 PurpleAccount *account;
00092 
00093                 account = purple_connection_get_account(gc);
00094                 /* Put our web response into a xmlnode for easy management */
00095                 contacts = xmlnode_from_str(url_text, -1);
00096 
00097                 if (contacts == NULL) {
00098                         purple_debug_error("yahoo", "Badly formed Alias XML\n");
00099                         return;
00100                 }
00101                 purple_debug_info("yahoo", "Fetched %" G_GSIZE_FORMAT
00102                                 " bytes of alias data\n", len);
00103 
00104                 /* Loop around and around and around until we have gone through all the received aliases  */
00105                 for(item = xmlnode_get_child(contacts, "ct"); item; item = xmlnode_get_next_twin(item)) {
00106                         /* Yahoo replies with two types of contact (ct) record, we are only interested in the alias ones */
00107                         if ((yid = xmlnode_get_attrib(item, "yi"))) {
00108                                 YahooPersonalDetails *ypd = NULL;
00109                                 /* Grab all the bits of information we can */
00110                                 fn = xmlnode_get_attrib(item, "fn");
00111                                 ln = xmlnode_get_attrib(item, "ln");
00112                                 nn = xmlnode_get_attrib(item, "nn");
00113                                 mn = xmlnode_get_attrib(item, "mn");
00114                                 id = xmlnode_get_attrib(item, "id");
00115 
00116                                 hp = xmlnode_get_attrib(item, "hp");
00117                                 wp = xmlnode_get_attrib(item, "wp");
00118                                 mo = xmlnode_get_attrib(item, "mo");
00119 
00120                                 full_name = nick_name = NULL;
00121                                 alias = NULL;
00122 
00123                                 /* Yahoo stores first and last names separately, lets put them together into a full name */
00124                                 if (yd->jp)
00125                                         full_name = g_strstrip(g_strdup_printf("%s %s", (ln != NULL ? ln : "") , (fn != NULL ? fn : "")));
00126                                 else
00127                                         full_name = g_strstrip(g_strdup_printf("%s %s", (fn != NULL ? fn : "") , (ln != NULL ? ln : "")));
00128                                 nick_name = (nn != NULL ? g_strstrip(g_strdup(nn)) : NULL);
00129 
00130                                 if (nick_name != NULL)
00131                                         alias = nick_name;   /* If we have a nickname from Yahoo, let's use it */
00132                                 else if (strlen(full_name) != 0)
00133                                         alias = full_name;  /* If no Yahoo nickname, we can use the full_name created above */
00134 
00135                                 /*  Find the local buddy that matches */
00136                                 f = yahoo_friend_find(gc, yid);
00137                                 b = purple_find_buddy(account, yid);
00138 
00139                                 /*  If we don't find a matching buddy, ignore the alias !!  */
00140                                 if (f != NULL && b != NULL) {
00141                                         const char *buddy_alias = purple_buddy_get_alias(b);
00142                                         yahoo_friend_set_alias_id(f, id);
00143 
00144                                         /* Finally, if we received an alias, we better update the buddy list */
00145                                         if (alias != NULL) {
00146                                                 serv_got_alias(gc, yid, alias);
00147                                                 purple_debug_info("yahoo", "Fetched alias '%s' (%s)\n", alias, id);
00148                                         } else if (buddy_alias && *buddy_alias && !g_str_equal(buddy_alias, yid)) {
00149                                         /* Or if we have an alias that Yahoo doesn't, send it up */
00150                                                 yahoo_update_alias(gc, yid, buddy_alias);
00151                                                 purple_debug_info("yahoo", "Sent updated alias '%s'\n", buddy_alias);
00152                                         }
00153                                 }
00154 
00155                                 if (f != NULL)
00156                                         ypd = &f->ypd;
00157                                 else {
00158                                         /* May be the alias is for the account? */
00159                                         const char *yidn = purple_normalize(account, yid);
00160                                         if (purple_strequal(yidn, purple_connection_get_display_name(gc))) {
00161                                                 ypd = &yd->ypd;
00162                                         }
00163                                 }
00164 
00165                                 if (ypd) {
00166                                         yahoo_personal_details_reset(ypd, TRUE);
00167                                         ypd->id = g_strdup(id);
00168                                         ypd->names.first = g_strdup(fn);
00169                                         ypd->names.middle = g_strdup(mn);
00170                                         ypd->names.last = g_strdup(ln);
00171                                         ypd->names.nick = g_strdup(nn);
00172 
00173                                         ypd->phone.work = g_strdup(wp);
00174                                         ypd->phone.home = g_strdup(hp);
00175                                         ypd->phone.mobile = g_strdup(mo);
00176                                 }
00177 
00178                                 g_free(full_name);
00179                                 g_free(nick_name);
00180                         }
00181                 }
00182                 xmlnode_free(contacts);
00183         }

00190         const char *url;
00191         gchar *request, *webpage, *webaddress;
00192         PurpleUtilFetchUrlData *url_data;
00193 
00194         /* use whole URL if using HTTP Proxy */
00195         gboolean use_whole_url = yahoo_account_use_http_proxy(gc);
00196 
00197         /*  Build all the info to make the web request */
00198         url = yd->jp ? YAHOOJP_ALIAS_FETCH_URL : YAHOO_ALIAS_FETCH_URL;
00199         purple_url_parse(url, &webaddress, NULL, &webpage, NULL, NULL);
00200         request = g_strdup_printf("GET %s%s/%s HTTP/1.1\r\n"
00201                                  "User-Agent: " YAHOO_CLIENT_USERAGENT_ALIAS "\r\n"
00202                                  "Cookie: T=%s; Y=%s\r\n"
00203                                  "Host: %s\r\n"
00204                                  "Cache-Control: no-cache\r\n\r\n",
00205                                   use_whole_url ? "http://" : "", use_whole_url ? webaddress : "", webpage,
00206                                   yd->cookie_t, yd->cookie_y,
00207                                   webaddress);
00208 
00209         /* We have a URL and some header information, let's connect and get some aliases  */
00210         url_data = purple_util_fetch_url_request_len_with_account(purple_connection_get_account(gc),
00211                                 url, use_whole_url, NULL, TRUE, request, FALSE, -1,
00212                                 yahoo_fetch_aliases_cb, gc);
00213         if (url_data != NULL)
00214                 yd->url_datas = g_slist_prepend(yd->url_datas, url_data);
00215

00217         g_free(webpage);
00218         g_free(request);

00226 yahoo_update_alias_cb(PurpleUtilFetchUrlData *url_data, gpointer user_data, const gchar *url_text, size_t len, const gchar *error_message)
00227 {
00228         xmlnode *node, *result;
00229         struct callback_data *cb = user_data;
00230         PurpleConnection *gc = cb->gc;
00231         YahooData *yd;
00232 
00233         yd = gc->proto_data;
00234         yd->url_datas = g_slist_remove(yd->url_datas, url_data);
00235 
00236         if (len == 0 || error_message != NULL) {
00237                 purple_debug_info("yahoo", "Error updating alias for %s: %s\n",
00238                                   cb->who,
00239                                   error_message ? error_message : "");
00240                 g_free(cb->who);
00241                 g_free(cb->id);
00242                 g_free(cb);
00243                 return;
00244         }
00245 
00246         result = xmlnode_from_str(url_text, -1);
00247 
00248         if (result == NULL) {
00249                 purple_debug_error("yahoo", "Alias update for %s failed: Badly formed response\n",
00250                                    cb->who);
00251                 g_free(cb->who);
00252                 g_free(cb->id);
00253                 g_free(cb);
00254                 return;
00255         }
00256 
00257         if ((node = xmlnode_get_child(result, "ct"))) {
00258                 if (cb->id == NULL) {
00259                         const char *new_id = xmlnode_get_attrib(node, "id");
00260                         if (new_id != NULL) {
00261                                 /* We now have an addressbook id for the friend; we should save it */
00262                                 YahooFriend *f = yahoo_friend_find(cb->gc, cb->who);
00263 
00264                                 purple_debug_info("yahoo", "Alias creation for %s succeeded\n", cb->who);
00265 
00266                                 if (f)
00267                                         yahoo_friend_set_alias_id(f, new_id);
00268                                 else
00269                                         purple_debug_error("yahoo", "Missing YahooFriend. Unable to store new addressbook id.\n");
00270                         } else
00271                                 purple_debug_error("yahoo", "Missing new addressbook id in add response for %s (weird).\n",
00272                                                    cb->who);
00273                 } else {
00274                         if (g_ascii_strncasecmp(xmlnode_get_attrib(node, "id"), cb->id, strlen(cb->id))==0)
00275                                 purple_debug_info("yahoo", "Alias update for %s succeeded\n", cb->who);
00276                         else
00277                                 purple_debug_error("yahoo", "Alias update for %s failed (Contact record return mismatch)\n",
00278                                                    cb->who);
00279                 }
00280         } else
00281                 purple_debug_info("yahoo", "Alias update for %s failed (No contact record returned)\n", cb->who);
00282 
00283         g_free(cb->who);
00284         g_free(cb->id);
00285         g_free(cb);
00286         xmlnode_free(result);

00604 static gboolean
00605 parse_contact_details(YahooData *yd, const char *who, const char *xml)
00606 {
00607         xmlnode *node, *nd;
00608         YahooFriend *f;
00609         char *yid;
00610 
00611         node = xmlnode_from_str(xml, -1);
00612         if (!node) {
00613                 purple_debug_info("yahoo", "Received malformed XML for contact details from '%s':\n%s\n",
00614                                 who, xml);
00615                 return FALSE;
00616         }
00617 
00618         nd = xmlnode_get_child(node, "yi");
00619         if (!nd || !(yid = xmlnode_get_data(nd))) {
00620                 xmlnode_free(node);
00621                 return FALSE;
00622         }
00623 
00624         if (!purple_strequal(yid, who)) {
00625                 /* The user may not want to set the contact details about folks in the buddylist
00626                    to what some random dude might have sent. So it would be good if we popped
00627                    up a prompt requiring the user to confirm the details before we set them.
00628                    However, someone could send details about hundreds of users at the same time,
00629                    which would make things really bad. So for now, until we have a better way of
00630                    dealing with this, ignore this details. */
00631                 purple_debug_info("yahoo", "Ignoring contact details sent by %s about %s\n",
00632                                 who, yid);
00633                 g_free(yid);
00634                 xmlnode_free(node);
00635                 return FALSE;
00636         }
00637 
00638         f = yahoo_friend_find(yd->gc, yid);
00639         if (!f) {
00640                 g_free(yid);
00641                 xmlnode_free(node);
00642                 return FALSE;
00643         } else {
00644                 int i;
00645                 YahooPersonalDetails *ypd = &f->ypd;
00646                 char *alias = NULL;
00647                 struct {
00648                         char *id;
00649                         char **field;
00650                 } details[] = {
00651                         {"fn", &ypd->names.first},
00652                         {"mn", &ypd->names.middle},
00653                         {"ln", &ypd->names.last},
00654                         {"nn", &ypd->names.nick},
00655                         {"wp", &ypd->phone.work},
00656                         {"hp", &ypd->phone.home},
00657                         {"mo", &ypd->phone.mobile},
00658                         {NULL, NULL}
00659                 };
00660 
00661                 yahoo_personal_details_reset(ypd, FALSE);
00662 
00663                 for (i = 0; details[i].id; i++) {
00664                         nd = xmlnode_get_child(node, details[i].id);
00665                         *details[i].field = nd ? xmlnode_get_data(nd) : NULL;
00666                 }
00667 
00668                 if (ypd->names.nick)
00669                         alias = ypd->names.nick;
00670                 else if (ypd->names.first || ypd->names.last) {
00671                         alias = g_strstrip(g_strdup_printf("%s %s",
00672                                                 ypd->names.first ? ypd->names.first : "",
00673                                                 ypd->names.last ? ypd->names.last : ""));
00674                 }
00675 
00676                 if (alias) {
00677                         serv_got_alias(yd->gc, yid, alias);
00678                         if (alias != ypd->names.nick)
00679                                 g_free(alias);
00680                 }
00681         }
00682 
00683         xmlnode_free(node);
00684         g_free(yid);
00685         return TRUE;
00686 }
