00333                 const gchar *src = xmlnode_get_attrib(child, "src");
00334

00336                         const gchar *cid = src + 4;

00338                         /* if we haven't "fetched" this yet... */
00339                         if (!g_hash_table_lookup(table, cid)) {
00340                                 /* take a copy of the cid and let the SmileyRef own it... */
00341                                 gchar *temp_cid = g_strdup(cid);
00342                                 JabberSmileyRef *ref = g_new0(JabberSmileyRef, 1);
00343                                 const gchar *alt = xmlnode_get_attrib(child, "alt");
00344                                 ref->cid = temp_cid;
00345                                 /* if there is no "alt" string, use the cid...
00346                                  include the entire src, eg. "cid:.." to avoid linkification */
00347                                 if (alt && alt[0] != '\0') {
00348                                         /* workaround for when "alt" is set to the value of the
00349                                          CID (which Jabbim seems to do), to avoid it showing up
00350                                                  as an mailto: link */
00351                                         if (purple_email_is_valid(alt)) {
00352                                                 ref->alt = g_strdup_printf("smiley:%s", alt);
00353                                         } else {
00354                                                 ref->alt = g_strdup(alt);
00355                                         }
00356                                 } else {
00357                                         ref->alt = g_strdup(src);
00358                                 }
00359                                 g_hash_table_insert(table, temp_cid, ref);
00360                         }
00361                 }
00362         }
00363 
00364         for (child = message->child ; child ; child = child->next) {
00365                 jabber_message_get_refs_from_xmlnode_internal(child, table);

00398         int pos = 0;
00399         GString *out = g_string_new(NULL);

00401         while (pos < len) {
00402                 /* this is a bit cludgy, maybe there is a better way to do this...
00403                   we need to find all <img> tags within the XHTML and replace those
00404                         tags with the value of their "alt" attributes */
00405                 if (g_str_has_prefix(&(markup[pos]), "<img")) {
00406                         xmlnode *img = NULL;
00407                         int pos2 = pos;
00408                         const gchar *src;
00409

00415                                         pos2 += 5;
00416                                         break;
00417                                 }
00418                         }

00421                           it the parsed string will be until the end of the input string,
00422                           in which case xmlnode_from_str will bail out and return NULL,

00426                         src = xmlnode_get_attrib(img, "src");
00427 
00428                         if (g_str_has_prefix(src, "cid:")) {
00429                                 const gchar *alt = xmlnode_get_attrib(img, "alt");

00437                                                 g_free(safe_alt);

00439                                                 out = g_string_append(out, alt);
00440                                         }
00441                                 } else {
00442                                         out = g_string_append(out, src);
00443                                 }
00444                                 pos += pos2 - pos;

00446                                 out = g_string_append_c(out, markup[pos]);

00848 static gchar *
00849 jabber_message_get_smileyfied_xhtml(const gchar *xhtml, const GList *smileys)

00884                                 g_free(escaped);
00885                         }
00886                 }

00888                         /* there was no smiley here, just copy one byte */
00889                         result = g_string_append_c(result, xhtml[pos]);
00890                         pos++;

01000 void jabber_message_send(JabberMessage *jm)
01001 {
01002         xmlnode *message, *child;
01003         const char *type = NULL;
01004 
01005         message = xmlnode_new("message");
01006

01014                         break;
01015                 case JABBER_MESSAGE_HEADLINE:
01016                         type = "headline";
01017                         break;

01019                         type = "groupchat";

01037

01039                 child = xmlnode_new_child(message, "thread");

01041         }

01045         {
01046                 case JM_STATE_ACTIVE:
01047                         child = xmlnode_new_child(message, "active");

01050                         child = xmlnode_new_child(message, "composing");
01051                         break;
01052                 case JM_STATE_PAUSED:
01053                         child = xmlnode_new_child(message, "paused");
01054                         break;
01055                 case JM_STATE_INACTIVE:
01056                         child = xmlnode_new_child(message, "inactive");
01057                         break;
01058                 case JM_STATE_GONE:
01059                         child = xmlnode_new_child(message, "gone");
01060                         break;
01061                 case JM_STATE_NONE:
01062                         /* yep, nothing */
01063                         break;
01064         }

01066                 xmlnode_set_namespace(child, "http://jabber.org/protocol/chatstates");
01067

01070                 xmlnode_insert_data(child, jm->subject, -1);

01075                 xmlnode_insert_data(child, jm->body, -1);

01080                         xmlnode_insert_child(message, child);

01083                                         "XHTML translation/validation failed, returning: %s\n",

01093 /*
01094  * Compare the XHTML and plain strings passed in for "equality". Any HTML markup
01095  * other than <br/> (matches a newline) in the XHTML will cause this to return
01096  * FALSE.
01097  */
01098 static gboolean
01099 jabber_xhtml_plain_equal(const char *xhtml_escaped,
01100                          const char *plain)
01101 {
01102         int i = 0;
01103         int j = 0;
01104         gboolean ret;
01105         char *xhtml = purple_unescape_html(xhtml_escaped);
01106 
01107         while (xhtml[i] && plain[j]) {
01108                 if (xhtml[i] == plain[j]) {
01109                         i += 1;
01110                         j += 1;
01111                         continue;
01112                 }
01113 
01114                 if (plain[j] == '\n' && !strncmp(xhtml+i, "<br/>", 5)) {
01115                         i += 5;
01116                         j += 1;
01117                         continue;
01118                 }
01119 
01120                 g_free(xhtml);
01121                 return FALSE;
01122         }
01123 
01124         /* Are we at the end of both strings? */
01125         ret = (xhtml[i] == plain[j]) && (xhtml[i] == '\0');
01126         g_free(xhtml);
01127         return ret;
01128 }
