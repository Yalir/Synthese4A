00323                              gsize size)
00324 {
00325         JabberIBBSessionState state = jabber_ibb_session_get_state(sess);
00326 
00327         purple_debug_info("jabber", "sending data block of %" G_GSIZE_FORMAT " bytes on IBB stream\n",

00329 
00330         if (state != JABBER_IBB_SESSION_OPENED) {
00331                 purple_debug_error("jabber",

00340                 char *base64 = purple_base64_encode(data, size);
00341                 char seq[10];
00342                 g_snprintf(seq, sizeof(seq), "%u", jabber_ibb_session_get_send_seq(sess));
00343 
00344                 xmlnode_set_attrib(set->node, "to", jabber_ibb_session_get_who(sess));
00345                 xmlnode_set_namespace(data_element, NS_IBB);
00346                 xmlnode_set_attrib(data_element, "sid", jabber_ibb_session_get_sid(sess));
00347                 xmlnode_set_attrib(data_element, "seq", seq);
00348                 xmlnode_insert_data(data_element, base64, -1);
00349 
00350                 xmlnode_insert_child(set->node, data_element);
00351 
00352                 purple_debug_info("jabber",
00353                         "IBB: setting send <iq/> callback for session %p %s\n", sess,
00354                         sess->sid);
00355                 jabber_iq_set_callback(set, jabber_ibb_session_send_acknowledge_cb, sess);
00356                 sess->last_iq_id = g_strdup(xmlnode_get_attrib(set->node, "id"));
00357                 purple_debug_info("jabber", "IBB: set sess->last_iq_id: %s\n",
00358                         sess->last_iq_id);
00359                 jabber_iq_send(set);
00360 
00361                 g_free(base64);

00384 void
00385 jabber_ibb_parse(JabberStream *js, const char *who, JabberIqType type,
00386                  const char *id, xmlnode *child)
00387 {
00388         const char *name = child->name;
00389         gboolean data  = g_str_equal(name, "data");
00390         gboolean close = g_str_equal(name, "close");
00391         gboolean open  = g_str_equal(name, "open");
00392         const gchar *sid = (data || close) ?
00393                 xmlnode_get_attrib(child, "sid") : NULL;
00394         JabberIBBSession *sess =
00395                 sid ? g_hash_table_lookup(jabber_ibb_sessions, sid) : NULL;
00396 
00397         if (sess) {
00398 
00399                 if (strcmp(who, jabber_ibb_session_get_who(sess)) != 0) {
00400                         /* the iq comes from a different JID than the remote JID of the
00401                           session, ignore it */
00402                         purple_debug_error("jabber",
00403                                 "Got IBB iq from wrong JID, ignoring\n");
00404                 } else if (data) {
00405                         const gchar *seq_attr = xmlnode_get_attrib(child, "seq");
00406                         guint16 seq = (seq_attr ? atoi(seq_attr) : 0);
00407 
00408                         /* reject the data, and set the session in error if we get an
00409                           out-of-order packet */
00410                         if (seq_attr && seq == jabber_ibb_session_get_recv_seq(sess)) {
00411                                 /* sequence # is the expected... */
00412                                 JabberIq *result = jabber_iq_new(js, JABBER_IQ_RESULT);
00413 
00414                                 jabber_iq_set_id(result, id);
00415                                 xmlnode_set_attrib(result->node, "to", who);
00416 
00417                                 if (sess->data_received_cb) {
00418                                         gchar *base64 = xmlnode_get_data(child);
00419                                         gsize size;
00420                                         gpointer rawdata = purple_base64_decode(base64, &size);
00421 
00422                                         g_free(base64);
00423 
00424                                         if (rawdata) {
00425                                                 purple_debug_info("jabber",
00426                                                         "got %" G_GSIZE_FORMAT " bytes of data on IBB stream\n",
00427                                                         size);
00428                                                 /* we accept other clients to send up to block-size
00429                                                  of _unencoded_ data, since there's been some confusions
00430                                                  regarding the interpretation of this attribute
00431                                                  (including previous versions of libpurple) */
00432                                                 if (size > jabber_ibb_session_get_block_size(sess)) {
00433                                                         purple_debug_error("jabber",
00434                                                                 "IBB: received a too large packet\n");
00435                                                         if (sess->error_cb)
00436                                                                 sess->error_cb(sess);
00437                                                         g_free(rawdata);
00438                                                         return;
00439                                                 } else {
00440                                                         purple_debug_info("jabber",
00441                                                                 "calling IBB callback for received data\n");
00442                                                         sess->data_received_cb(sess, rawdata, size);
00443                                                 }
00444                                                 g_free(rawdata);
00445                                         } else {
00446                                                 purple_debug_error("jabber",
00447                                                         "IBB: invalid BASE64 data received\n");
00448                                                 if (sess->error_cb)
00449                                                         sess->error_cb(sess);
00450                                                 return;
00451 
00452                                         }
00453                                 }
00454 
00455                                 (sess->recv_seq)++;
00456                                 jabber_iq_send(result);
00457 
00458                         } else {
00459                                 purple_debug_error("jabber",
00460                                         "Received an out-of-order/invalid IBB packet\n");
00461                                 sess->state = JABBER_IBB_SESSION_ERROR;
00462 
00463                                 if (sess->error_cb) {
00464                                         sess->error_cb(sess);
00465                                 }
00466                         }
00467                 } else if (close) {
00468                         sess->state = JABBER_IBB_SESSION_CLOSED;
00469                         purple_debug_info("jabber", "IBB: received close\n");
00470 
00471                         if (sess->closed_cb) {
00472                                 purple_debug_info("jabber", "IBB: calling closed handler\n");
00473                                 sess->closed_cb(sess);
00474                         }
00475 
00476                 } else {
00477                         /* this should never happen */
00478                         purple_debug_error("jabber", "Received bogus iq for IBB session\n");
00479                 }
00480         } else if (open) {
00481                 JabberIq *result;
00482                 const GList *iterator;
00483 
00484                 /* run all open handlers registered until one returns true */
00485                 for (iterator = open_handlers ; iterator ;
00486                          iterator = g_list_next(iterator)) {
00487                         JabberIBBOpenHandler *handler = iterator->data;
00488 
00489                         if (handler(js, who, id, child)) {
00490                                 result = jabber_iq_new(js, JABBER_IQ_RESULT);
00491                                 xmlnode_set_attrib(result->node, "to", who);
00492                                 jabber_iq_set_id(result, id);
00493                                 jabber_iq_send(result);
00494                                 return;
00495                         }
00496                 }
00497                 /* no open callback returned success, reject */
00498                 jabber_ibb_send_error_response(js, who, id);
00499         } else {
00500                 /* send error reply */
00501                 jabber_ibb_send_error_response(js, who, id);
00502         }
00503 }
