00122 static void jabber_mood_cb(JabberStream *js, const char *from, xmlnode *items) {
00123         /* it doesn't make sense to have more than one item here, so let's just pick the first one */
00124         xmlnode *item = xmlnode_get_child(items, "item");
00125         const char *newmood = NULL;
00126         char *moodtext = NULL;
00127         JabberBuddy *buddy = jabber_buddy_find(js, from, FALSE);
00128         xmlnode *moodinfo, *mood;
00129         /* ignore the mood of people not on our buddy list */
00130         if (!buddy || !item)
00131                 return;
00132 
00133         mood = xmlnode_get_child_with_namespace(item, "mood", "http://jabber.org/protocol/mood");
00134         if (!mood)
00135                 return;
00136         for (moodinfo = mood->child; moodinfo; moodinfo = moodinfo->next) {
00137                 if (moodinfo->type == XMLNODE_TYPE_TAG) {
00138                         if (!strcmp(moodinfo->name, "text")) {
00139                                 if (!moodtext) /* only pick the first one */
00140                                         moodtext = xmlnode_get_data(moodinfo);
00141                         } else {
00142                                 int i;
00143                                 for (i = 0; moods[i].mood; ++i) {
00144                                         /* verify that the mood is known (valid) */
00145                                         if (!strcmp(moodinfo->name, moods[i].mood)) {
00146                                                 newmood = moods[i].mood;
00147                                                 break;
00148                                         }
00149                                 }
00150                         }
00151                         if (newmood != NULL && moodtext != NULL)
00152                            break;
00153                 }
00154                 if (newmood != NULL && moodtext != NULL)
00155                    break;
00156         }
00157         if (newmood != NULL) {
00158                 purple_prpl_got_user_status(js->gc->account, from, "mood",
00159                                 PURPLE_MOOD_NAME, newmood,
00160                                 PURPLE_MOOD_COMMENT, moodtext,
00161                                 NULL);
00162         } else {
00163                 purple_prpl_got_user_status_deactive(js->gc->account, from, "mood");
00164         }
00165         g_free(moodtext);
00166 }

00169         jabber_add_feature("http://jabber.org/protocol/mood", jabber_pep_namespace_only_when_pep_enabled_cb);
00170         jabber_pep_register_handler("http://jabber.org/protocol/mood", jabber_mood_cb);

00173 void jabber_mood_set(JabberStream *js, const char *mood, const char *text) {
00174         xmlnode *publish, *moodnode;
00175 
00176         publish = xmlnode_new("publish");
00177         xmlnode_set_attrib(publish,"node","http://jabber.org/protocol/mood");
00178         moodnode = xmlnode_new_child(xmlnode_new_child(publish, "item"), "mood");
00179         xmlnode_set_namespace(moodnode, "http://jabber.org/protocol/mood");
00180         if (mood && *mood) {
00181                 /* if mood is NULL, set an empty mood node, meaning: unset mood */
00182             xmlnode_new_child(moodnode, mood);
00183         }
00184 
00185         if (text && *text) {
00186                 xmlnode *textnode = xmlnode_new_child(moodnode, "text");
00187                 xmlnode_insert_data(textnode, text, -1);
00188         }
00189 
00190         jabber_pep_publish(js, publish);
00191         /* publish is freed by jabber_pep_publish -> jabber_iq_send -> jabber_iq_free
00192            (yay for well-defined memory management rules) */
00193 }
