00093 static gchar *generate_error_message(xmlnode *resp, const char *url)
00094 {
00095         xmlnode *text;
00096         xmlnode *status_code_node;
00097         gchar *status_code;
00098         gboolean have_error_code = TRUE;
00099         gchar *err = NULL;
00100         gchar *details = NULL;
00101 
00102         status_code_node = xmlnode_get_child(resp, "statusCode");
00103         if (status_code_node) {
00104                 /* We can get 200 OK here if the server omitted something we think it shouldn't have (see #12783).
00105                  * No point in showing the "Ok" string to the user.
00106                  */
00107                 if ((status_code = xmlnode_get_data_unescaped(status_code_node)) && strcmp(status_code, "200") == 0) {
00108                         have_error_code = FALSE;
00109                 }
00110         }
00111         if (have_error_code && resp && (text = xmlnode_get_child(resp, "statusText"))) {
00112                 details = xmlnode_get_data(text);
00113         }
00114 
00115         if (details && *details) {
00116                 err = g_strdup_printf(_("Received unexpected response from %s: %s"), url, details);
00117         } else {
00118                 err = g_strdup_printf(_("Received unexpected response from %s"), url);
00119         }
00120 
00121         g_free(details);
00122         return err;
00123 }

00129 static gchar *hmac_sha256(const char *key, const char *message)
00130 {
00131         PurpleCipherContext *context;
00132         guchar digest[32];
00133 
00134         context = purple_cipher_context_new_by_name("hmac", NULL);
00135         purple_cipher_context_set_option(context, "hash", "sha256");
00136         purple_cipher_context_set_key(context, (guchar *)key);
00137         purple_cipher_context_append(context, (guchar *)message, strlen(message));
00138         purple_cipher_context_digest(context, sizeof(digest), digest, NULL);
00139         purple_cipher_context_destroy(context);
00140 
00141         return purple_base64_encode(digest, sizeof(digest));
00142 }

00358 static void send_start_oscar_session(OscarData *od, const char *token, const char *session_key, time_t hosttime)
00359 {
00360         char *query_string, *signature, *url;
00361         PurpleAccount *account = purple_connection_get_account(od->gc);
00362         const gchar *encryption_type = purple_account_get_string(account, "encryption", OSCAR_DEFAULT_ENCRYPTION);
00363 
00364         /*
00365          * Construct the GET parameters.  0x00000611 is the distid given to
00366          * us by AOL for use as the default libpurple distid.
00367          */
00368         query_string = g_strdup_printf("a=%s"
00369                         "&distId=%d"
00370                         "&f=xml"
00371                         "&k=%s"
00372                         "&ts=%" PURPLE_TIME_T_MODIFIER
00373                         "&useTLS=%d",
00374                         purple_url_encode(token),
00375                         oscar_get_ui_info_int(od->icq ? "prpl-icq-distid" : "prpl-aim-distid", 0x00000611),
00376                         get_client_key(od),
00377                         hosttime,
00378                         strcmp(encryption_type, OSCAR_NO_ENCRYPTION) != 0 ? 1 : 0);
00379         signature = generate_signature("GET", get_start_oscar_session_url(od),
00380                         query_string, session_key);
00381         url = g_strdup_printf("%s?%s&sig_sha256=%s", get_start_oscar_session_url(od),
00382                         query_string, signature);
00383         g_free(query_string);
00384         g_free(signature);
00385 
00386         /* Make the request */
00387         od->url_data = purple_util_fetch_url_request_len_with_account(account,
00388                         url, TRUE, NULL, FALSE, NULL, FALSE, -1,
00389                         start_oscar_session_cb, od);
00390         g_free(url);
00391 }

00616          * limit for AIM passwords, too, but we really only need to do
00617          * this for ICQ because older ICQ clients let you enter a password
00618          * as long as you wanted and then they truncated it silently.
00619          *
00620          * And we can truncate based on the number of bytes and not the
00621          * number of characters because passwords for AIM and ICQ are
00622          * supposed to be plain ASCII (I don't know if this has always been
00623          * the case, though).
00624          */
00625         tmp = purple_connection_get_password(gc);
00626         password_len = strlen(tmp);
00627         password = g_strndup(tmp, od->icq ? MIN(password_len, MAXICQPASSLEN) : password_len);
00628 
00629         /* Construct the body of the HTTP POST request */
00630         body = g_string_new("");
00631         g_string_append_printf(body, "devId=%s", get_client_key(od));
00632         g_string_append_printf(body, "&f=xml");
00633         g_string_append_printf(body, "&pwd=%s", purple_url_encode(password));
00634         g_string_append_printf(body, "&s=%s", purple_url_encode(username));
00635         g_free(password);
00636 
00637         /* Construct an HTTP POST request */
00638         request = g_string_new("POST /auth/clientLogin HTTP/1.0\r\n"
00639                         "Connection: close\r\n"
00640                         "Accept: */*\r\n");
00641 
00642         /* Tack on the body */
00643         g_string_append_printf(request, "Content-Type: application/x-www-form-urlencoded; charset=UTF-8\r\n");
00644         g_string_append_printf(request, "Content-Length: %" G_GSIZE_FORMAT "\r\n\r\n", body->len);
00645         g_string_append_len(request, body->str, body->len);
00646         g_string_free(body, TRUE);
00647 
00648         /* Send the POST request  */
00649         od->url_data = purple_util_fetch_url_request_len_with_account(
00650                         purple_connection_get_account(gc), get_client_login_url(od),
00651                         TRUE, NULL, FALSE, request->str, FALSE, -1,
00652                         client_login_cb, od);
00653         g_string_free(request, TRUE);
00654 }
