00040 /*------------------------------------------------------------------------
00041  * Add ISO10126 Padding to the data.
00042  *
00043  *  @param data         The data to pad.
00044  */
00045 static void padding_add( GString* data )
00046 {
00047         unsigned int blocks = ( data->len / 16 ) + 1;
00048         unsigned int padding = ( blocks * 16 ) - data->len;
00049 
00050         g_string_set_size( data, blocks * 16 );
00051         data->str[data->len - 1] = padding;
00052 }

00057  *

00060 static void padding_remove( GString* data )
00061 {
00062         unsigned int padding;
00063 
00064         if ( data->len == 0 )
00065                 return;
00066 
00067         padding = data->str[data->len - 1];
00068         g_string_truncate( data, data->len - padding );
00069 }

00072 /*------------------------------------------------------------------------
00073  * Generate the Transport-Layer crypto key.
00074  *  (Note: this function is not-thread safe)
00075  *
00076  *  @param session      The MXit Session object
00077  *      @return                 The transport-layer crypto key.
00078  */
00079 static char* transport_layer_key( struct MXitSession* session )
00080 {
00081         static char     key[16 + 1];
00082         const char*     password                = purple_account_get_password( session->acc );
00083         int                     passlen                 = strlen( password );
00084 
00085         /* initialize with initial key */
00086         g_strlcpy( key, INITIAL_KEY, sizeof( key ) );
00087 
00088         /* client key (8 bytes) */
00089         memcpy( key, session->clientkey, strlen( session->clientkey ) );
00090 
00091         /* add last 8 characters of the PIN (no padding if less characters) */
00092         if ( passlen <= 8 )
00093                 memcpy( key + 8, password, passlen );
00094         else
00095                 memcpy( key + 8, password + ( passlen - 8 ), 8 );
00096 
00097         return key;
00098 }

00101 /*------------------------------------------------------------------------
00102  * Encrypt the user's cleartext password using the AES 128-bit (ECB)
00103  *  encryption algorithm.
00104  *
00105  *  @param session      The MXit session object
00106  *  @return                     The encrypted & encoded password.  Must be g_free'd when no longer needed.
00107  */
00108 char* mxit_encrypt_password( struct MXitSession* session )
00109 {
00110         char            key[16 + 1];
00111         char            exkey[512];
00112         GString*        pass                    = NULL;
00113         GString*        encrypted               = NULL;
00114         char*           base64;
00115         int                     i;
00116 
00117         purple_debug_info( MXIT_PLUGIN_ID, "mxit_encrypt_password\n" );
00118 
00119         /* build the AES encryption key */
00120         g_strlcpy( key, INITIAL_KEY, sizeof( key ) );
00121         memcpy( key, session->clientkey, strlen( session->clientkey ) );
00122         ExpandKey( (unsigned char*) key, (unsigned char*) exkey );
00123 
00124         /* build the secret data to be encrypted: SECRET_HEADER + password */
00125         pass = g_string_new( SECRET_HEADER );
00126         g_string_append( pass, purple_account_get_password( session->acc) );
00127         padding_add( pass );            /* add ISO10126 padding */
00128 
00129         /* now encrypt the secret. we encrypt each block separately (ECB mode) */
00130         encrypted = g_string_sized_new( pass->len );
00131         for ( i = 0; i < pass->len; i += 16 ) {
00132                 char    block[16];
00133 
00134                 Encrypt( (unsigned char*) pass->str + i, (unsigned char*) exkey, (unsigned char*) block );
00135                 g_string_append_len( encrypted, block, 16 );
00136         }
00137 
00138         /* now base64 encode the encrypted password */
00139         base64 = purple_base64_encode( (unsigned char*) encrypted->str, encrypted->len );
00140         g_string_free( encrypted, TRUE );
00141 
00142         g_string_free( pass, TRUE );
00143 
00144         return base64;
00145 }

00148 /*------------------------------------------------------------------------
00149  * Decrypt a message using transport-layer encryption.
00150  *
00151  *  @param session      The MXit session object
00152  *      @param message  The encrypted message data (is base64-encoded).
00153  *  @return                     The decrypted message.  Must be g_free'd when no longer needed.
00154  */
00155 char* mxit_decrypt_message( struct MXitSession* session, char* message )
00156 {
00157         guchar*         raw_message;
00158         gsize           raw_len;
00159         char            exkey[512];
00160         GString*        decoded         = NULL;
00161         int                     i;
00162 
00163         /* remove optional header: <mxitencrypted ver="5.2"/> */
00164         if ( strncmp( message, ENCRYPT_HEADER, strlen( ENCRYPT_HEADER ) ) == 0 )
00165                 message += strlen( ENCRYPT_HEADER );
00166 
00167         /* base64 decode the message */
00168         raw_message = purple_base64_decode( message, &raw_len );
00169 
00170         /* AES-encrypted data is always blocks of 16 bytes */
00171         if ( ( raw_len == 0 ) || ( raw_len % 16 != 0 ) )
00172                 return NULL;
00173 
00174         /* build the AES key */
00175         ExpandKey( (unsigned char*) transport_layer_key( session ), (unsigned char*) exkey );
00176 
00177         /* AES decrypt each block */
00178         decoded = g_string_sized_new( raw_len );
00179         for ( i = 0; i < raw_len; i += 16 ) {
00180                 char    block[16];
00181 
00182                 Decrypt( (unsigned char*) raw_message + i, (unsigned char*) exkey, (unsigned char*) block );
00183                 g_string_append_len( decoded, block, 16 );
00184         }
00185         g_free( raw_message );
00186 
00187         /* check that the decrypted message starts with header: <mxit/> */
00188         if ( strncmp( decoded->str, SECRET_HEADER, strlen( SECRET_HEADER ) != 0 ) ) {
00189                 g_string_free( decoded, TRUE );
00190                 return NULL;                    /* message could not be decrypted */
00191         }
00192 
00193         /* remove ISO10126 padding */
00194         padding_remove( decoded );
00195 
00196         /* remove encryption header */
00197         g_string_erase( decoded, 0, strlen( SECRET_HEADER ) );
00198 
00199         return g_string_free( decoded, FALSE );
00200 }
