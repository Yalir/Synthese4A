00259 void purple_roomlist_room_add_field(PurpleRoomlist *list, PurpleRoomlistRoom *room, gconstpointer field)
00260 {
00261         PurpleRoomlistField *f;
00262 
00263         g_return_if_fail(list != NULL);
00264         g_return_if_fail(room != NULL);
00265         g_return_if_fail(list->fields != NULL);
00266 
00267         /* If this is the first call for this room, grab the first field in
00268          * the Roomlist's fields.  Otherwise, grab the field that is one
00269          * more than the number of fields already present for the room.
00270          * (This works because g_list_nth_data() is zero-indexed and
00271          * g_list_length() is one-indexed.) */
00272         if (!room->fields)
00273                 f = list->fields->data;
00274         else
00275                 f = g_list_nth_data(list->fields, g_list_length(room->fields));
00276 
00277         g_return_if_fail(f != NULL);
00278 
00279         switch(f->type) {
00280                 case PURPLE_ROOMLIST_FIELD_STRING:
00281                         room->fields = g_list_append(room->fields, g_strdup(field));
00282                         break;
00283                 case PURPLE_ROOMLIST_FIELD_BOOL:
00284                 case PURPLE_ROOMLIST_FIELD_INT:
00285                         room->fields = g_list_append(room->fields, GINT_TO_POINTER(field));
00286                         break;
00287         }
00288 }
