00035 static void
00036 buddy_typing_cb(PurpleAccount *acct, const char *name, void *data) {
00037   PurpleConversation *gconv;
00038 
00039   if(purple_prefs_get_bool(PREF_STATUS) &&
00040      ! purple_status_is_available(purple_account_get_active_status(acct))) {
00041     purple_debug_info("psychic", "not available, doing nothing\n");
00042     return;
00043   }
00044 
00045   if(purple_prefs_get_bool(PREF_BUDDIES) &&
00046      ! purple_find_buddy(acct, name)) {
00047     purple_debug_info("psychic", "not in blist, doing nothing\n");
00048     return;
00049   }
00050 
00051   if(FALSE == purple_privacy_check(acct, name)) {
00052     purple_debug_info("psychic", "user %s is blocked\n", name);
00053     return;
00054   }
00055 
00056   gconv = purple_find_conversation_with_account(PURPLE_CONV_TYPE_IM, name, acct);
00057   if(! gconv) {
00058     purple_debug_info("psychic", "no previous conversation exists\n");
00059     gconv = purple_conversation_new(PURPLE_CONV_TYPE_IM, acct, name);
00060 
00061     if(purple_prefs_get_bool(PREF_RAISE)) {
00062       purple_conversation_present(gconv);
00063     }
00064 
00065     if(purple_prefs_get_bool(PREF_NOTICE)) {
00066 
00067       /* This is a quote from Star Wars.  You should probably not
00068          translate it literally.  If you can't find a fitting cultural
00069          reference in your language, consider translating something
00070          like this instead: "You feel a new message coming." */
00071       purple_conversation_write(gconv, NULL,
00072                               _("You feel a disturbance in the force..."),
00073                               PURPLE_MESSAGE_SYSTEM | PURPLE_MESSAGE_NO_LOG | PURPLE_MESSAGE_ACTIVE_ONLY,
00074                               time(NULL));
00075     }
00076 
00077     /* Necessary because we may be creating a new conversation window. */
00078     purple_conv_im_set_typing_state(PURPLE_CONV_IM(gconv), PURPLE_TYPING);
00079   }
00080 }
