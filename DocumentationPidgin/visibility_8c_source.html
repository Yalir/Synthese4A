00057 static void
00058 visibility_cb(PurpleBlistNode *node, gpointer whatever)
00059 {
00060         PurpleBuddy *buddy = PURPLE_BUDDY(node);
00061         const char* bname = purple_buddy_get_name(buddy);
00062         OscarData *od = purple_connection_get_protocol_data(purple_account_get_connection(purple_buddy_get_account(buddy)));
00063         guint16 list_type = get_buddy_list_type(od);
00064 
00065         if (!is_buddy_on_list(od, bname)) {
00066                 aim_ssi_add_to_private_list(od, bname, list_type);
00067         } else {
00068                 aim_ssi_del_from_private_list(od, bname, list_type);
00069         }
00070 }

00072 PurpleMenuAction *
00073 create_visibility_menu_item(OscarData *od, const char *bname)
00074 {
00075         PurpleAccount *account = purple_connection_get_account(od->gc);
00076         gboolean invisible = purple_account_is_status_active(account, OSCAR_STATUS_ID_INVISIBLE);
00077         gboolean on_list = is_buddy_on_list(od, bname);
00078         const gchar *label;
00079 
00080         if (invisible) {
00081                 label = on_list ? _(DONT_APPEAR_ONLINE) : _(APPEAR_ONLINE);
00082         } else {
00083                 label = on_list ? _(DONT_APPEAR_OFFLINE) : _(APPEAR_OFFLINE);
00084         }
00085         return purple_menu_action_new(label, PURPLE_CALLBACK(visibility_cb), NULL, NULL);
00086 }

00088 static void
00089 show_private_list(PurplePluginAction *action, guint16 list_type, const gchar *title, const gchar *list_description, const gchar *menu_action_name)
00090 {
00091         PurpleConnection *gc = (PurpleConnection *) action->context;
00092         OscarData *od = purple_connection_get_protocol_data(gc);
00093         PurpleAccount *account = purple_connection_get_account(gc);
00094         GSList *buddies, *filtered_buddies, *cur;
00095         gchar *text, *secondary;
00096 
00097         buddies = purple_find_buddies(account, NULL);
00098         filtered_buddies = NULL;
00099         for (cur = buddies; cur != NULL; cur = cur->next) {
00100                 PurpleBuddy *buddy;
00101                 const gchar *bname;
00102 
00103                 buddy = cur->data;
00104                 bname = purple_buddy_get_name(buddy);
00105                 if (aim_ssi_itemlist_finditem(od->ssi.local, NULL, bname, list_type)) {
00106                         filtered_buddies = g_slist_prepend(filtered_buddies, buddy);
00107                 }
00108         }
00109 
00110         g_slist_free(buddies);
00111 
00112         filtered_buddies = g_slist_reverse(filtered_buddies);
00113         text = oscar_format_buddies(filtered_buddies, _("you have no buddies on this list"));
00114         g_slist_free(filtered_buddies);
00115 
00116         secondary = g_strdup_printf(_("You can add a buddy to this list "
00117                                         "by right-clicking on them and "
00118                                         "selecting \"%s\""), menu_action_name);
00119         purple_notify_formatted(gc, title, list_description, secondary, text, NULL, NULL);
00120         g_free(secondary);
00121         g_free(text);
00122 }
