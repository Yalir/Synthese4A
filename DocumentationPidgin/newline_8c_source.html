00030 static gboolean
00031 addnewline_msg_cb(PurpleAccount *account, char *sender, char **message,
00032                                          PurpleConversation *conv, int *flags, void *data)
00033 {
00034         if (((purple_conversation_get_type(conv) == PURPLE_CONV_TYPE_IM) &&
00035                  !purple_prefs_get_bool("/plugins/core/newline/im")) ||
00036                 ((purple_conversation_get_type(conv) == PURPLE_CONV_TYPE_CHAT) &&
00037                  !purple_prefs_get_bool("/plugins/core/newline/chat")))
00038                 return FALSE;
00039 
00040         if (g_ascii_strncasecmp(*message, "/me ", strlen("/me "))) {
00041                 char *tmp = g_strdup_printf("<br/>%s", *message);
00042                 g_free(*message);
00043                 *message = tmp;
00044         }
00045 
00046         return FALSE;
00047 }
