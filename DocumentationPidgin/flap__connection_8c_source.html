00120 /*
00121  * Attempt to calculate what our new current average would be if we
00122  * were to send a SNAC in this rateclass at the given time.
00123  */
00124 static guint32
00125 rateclass_get_new_current(FlapConnection *conn, struct rateclass *rateclass, struct timeval *now)
00126 {
00127         unsigned long timediff; /* In milliseconds */
00128         guint32 current;
00129 
00130         /* This formula is documented at http://dev.aol.com/aim/oscar/#RATELIMIT */
00131         timediff = (now->tv_sec - rateclass->last.tv_sec) * 1000 + (now->tv_usec - rateclass->last.tv_usec) / 1000;
00132         current = ((rateclass->current * (rateclass->windowsize - 1)) + timediff) / rateclass->windowsize;
00133 
00134         return MIN(current, rateclass->max);
00135 }

00137 /*
00138  * Attempt to send the contents of a given queue
00139  *
00140  * @return TRUE if the queue was completely emptied or was initially
00141  *         empty; FALSE if rate limiting prevented it from being
00142  *         emptied.
00143  */
00144 static gboolean flap_connection_send_snac_queue(FlapConnection *conn, struct timeval now, GQueue *queue)
00145 {
00146         while (!g_queue_is_empty(queue))
00147         {
00148                 QueuedSnac *queued_snac;
00149                 struct rateclass *rateclass;
00150 
00151                 queued_snac = g_queue_peek_head(queue);
00152 
00153                 rateclass = flap_connection_get_rateclass(conn, queued_snac->family, queued_snac->subtype);
00154                 if (rateclass != NULL)
00155                 {
00156                         guint32 new_current;
00157 
00158                         new_current = rateclass_get_new_current(conn, rateclass, &now);
00159 
00160                         if (rateclass->dropping_snacs || new_current <= rateclass->alert)
00161                                 /* Not ready to send this SNAC yet--keep waiting. */
00162                                 return FALSE;
00163 
00164                         rateclass->current = new_current;
00165                         rateclass->last.tv_sec = now.tv_sec;
00166                         rateclass->last.tv_usec = now.tv_usec;
00167                 }
00168 
00169                 flap_connection_send(conn, queued_snac->frame);
00170                 g_free(queued_snac);
00171                 g_queue_pop_head(queue);
00172         }
00173 
00174         /* We emptied the queue */
00175         return TRUE;
00176 }

00178 static gboolean flap_connection_send_queued(gpointer data)
00179 {
00180         FlapConnection *conn;
00181         struct timeval now;
00182 
00183         conn = data;
00184         gettimeofday(&now, NULL);
00185

00187                                           (conn->queued_snacs ? conn->queued_snacs->length : 0),
00188                                           (conn->queued_lowpriority_snacs ? conn->queued_lowpriority_snacs->length : 0),
00189                                           conn);
00190         if (!conn->queued_snacs || flap_connection_send_snac_queue(conn, now, conn->queued_snacs)) {
00191                 if (!conn->queued_lowpriority_snacs || flap_connection_send_snac_queue(conn, now, conn->queued_lowpriority_snacs)) {
00192                         /* Both queues emptied. */
00193                         conn->queued_timeout = 0;
00194                         return FALSE;
00195                 }
00196         }
00197

00266 
00267                 if (high_priority) {

00269                                 conn->queued_snacs = g_queue_new();
00270                         g_queue_push_tail(conn->queued_snacs, queued_snac);
00271                 } else {
00272                         if (!conn->queued_lowpriority_snacs)
00273                                 conn->queued_lowpriority_snacs = g_queue_new();
00274                         g_queue_push_tail(conn->queued_lowpriority_snacs, queued_snac);
00275                 }
00276 
00277                 if (conn->queued_timeout == 0)
00278                         conn->queued_timeout = purple_timeout_add(500, flap_connection_send_queued, conn);
00279 
00280                 return;
00281         }
00282

00284 }

00735 static void
00736 parse_snac(OscarData *od, FlapConnection *conn, FlapFrame *frame)
00737 {
00738         aim_module_t *cur;
00739         aim_modsnac_t snac;
00740 
00741         if (byte_stream_bytes_left(&frame->data) < 10)
00742                 return;
00743 
00744         snac.family = byte_stream_get16(&frame->data);
00745         snac.subtype = byte_stream_get16(&frame->data);
00746         snac.flags = byte_stream_get16(&frame->data);
00747         snac.id = byte_stream_get32(&frame->data);
00748 
00749         /* SNAC flags are apparently uniform across all SNACs, so we handle them here */
00750         if (snac.flags & 0x0001) {
00751                 /*
00752                  * This means the SNAC will be followed by another SNAC with
00753                  * related information.  We don't need to do anything about
00754                  * this here.
00755                  */
00756         }
00757         if (snac.flags & 0x8000) {
00758                 /*
00759                  * This packet contains the version of the family that this SNAC is
00760                  * in.  You get this when your SSI module is version 2 or higher.
00761                  * For now we have no need for this, but you could always save
00762                  * it as a part of aim_modnsac_t, or something.  The format is...
00763                  * 2 byte length of total mini-header (which is 6 bytes), then TLV
00764                  * of  type 0x0001, length 0x0002, value is the 2 byte version
00765                  * number
00766                  */
00767                 byte_stream_advance(&frame->data, byte_stream_get16(&frame->data));
00768         }
00769 
00770         for (cur = (aim_module_t *)od->modlistv; cur; cur = cur->next) {
00771 
00772                 if (!(cur->flags & AIM_MODFLAG_MULTIFAMILY) &&
00773                                 (cur->family != snac.family))
00774                         continue;

00882 static void
00883 flap_connection_recv(FlapConnection *conn)
00884 {
00885         gpointer buf;
00886         gsize buflen;
00887         gssize read;
00888 
00889         /* Read data until we run out of data and break out of the loop */
00890         while (TRUE)
00891         {
00892                 /* Start reading a new FLAP */
00893                 if (conn->buffer_incoming.data.data == NULL)
00894                 {
00895                         buf = conn->header + conn->header_received;
00896                         buflen = 6 - conn->header_received;
00897 
00898                         /* Read the first 6 bytes (the FLAP header) */
00899                         if (conn->gsc)
00900                                 read = purple_ssl_read(conn->gsc, buf, buflen);
00901                         else
00902                                 read = recv(conn->fd, buf, buflen, 0);
00903 
00904                         /* Check if the FLAP server closed the connection */
00905                         if (read == 0)
00906                         {
00907                                 flap_connection_schedule_destroy(conn,
00908                                                 OSCAR_DISCONNECT_REMOTE_CLOSED, NULL);
00909                                 break;
00910                         }
00911 
00912                         /* If there was an error then close the connection */
00913                         if (read < 0)
00914                         {
00915                                 if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00916                                         /* No worries */
00917                                         break;
00918 
00919                                 /* Error! */
00920                                 flap_connection_schedule_destroy(conn,
00921                                                 OSCAR_DISCONNECT_LOST_CONNECTION, g_strerror(errno));
00922                                 break;
00923                         }
00924                         conn->od->gc->last_received = time(NULL);
00925 
00926                         /* If we don't even have a complete FLAP header then do nothing */
00927                         conn->header_received += read;
00928                         if (conn->header_received < 6)
00929                                 break;
00930 
00931                         /* All FLAP frames must start with the byte 0x2a */
00932                         if (aimutil_get8(&conn->header[0]) != 0x2a)
00933                         {
00934                                 flap_connection_schedule_destroy(conn,
00935                                                 OSCAR_DISCONNECT_INVALID_DATA, NULL);
00936                                 break;
00937                         }
00938 
00939                         /* Initialize a new temporary FlapFrame for incoming data */
00940                         conn->buffer_incoming.channel = aimutil_get8(&conn->header[1]);
00941                         conn->buffer_incoming.seqnum = aimutil_get16(&conn->header[2]);
00942                         conn->buffer_incoming.data.len = aimutil_get16(&conn->header[4]);
00943                         conn->buffer_incoming.data.data = g_new(guint8, conn->buffer_incoming.data.len);
00944                         conn->buffer_incoming.data.offset = 0;
00945                 }
00946 
00947                 buflen = conn->buffer_incoming.data.len - conn->buffer_incoming.data.offset;
00948                 if (buflen)
00949                 {
00950                         buf = &conn->buffer_incoming.data.data[conn->buffer_incoming.data.offset];
00951                         /* Read data into the temporary FlapFrame until it is complete */
00952                         if (conn->gsc)
00953                                 read = purple_ssl_read(conn->gsc, buf, buflen);
00954                         else
00955                                 read = recv(conn->fd, buf, buflen, 0);
00956 
00957                         /* Check if the FLAP server closed the connection */
00958                         if (read == 0)
00959                         {
00960                                 flap_connection_schedule_destroy(conn,
00961                                                 OSCAR_DISCONNECT_REMOTE_CLOSED, NULL);
00962                                 break;
00963                         }
00964 
00965                         if (read < 0)
00966                         {
00967                                 if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00968                                         /* No worries */
00969                                         break;
00970 
00971                                 /* Error! */
00972                                 flap_connection_schedule_destroy(conn,
00973                                                 OSCAR_DISCONNECT_LOST_CONNECTION, g_strerror(errno));
00974                                 break;
00975                         }
00976 
00977                         conn->buffer_incoming.data.offset += read;
00978                         if (conn->buffer_incoming.data.offset < conn->buffer_incoming.data.len)
00979                                 /* Waiting for more data to arrive */
00980                                 break;
00981                 }
00982 
00983                 /* We have a complete FLAP!  Handle it and continue reading */
00984                 byte_stream_rewind(&conn->buffer_incoming.data);
00985                 parse_flap(conn->od, conn, &conn->buffer_incoming);
00986                 conn->lastactivity = time(NULL);
00987 
00988                 g_free(conn->buffer_incoming.data.data);
00989                 conn->buffer_incoming.data.data = NULL;
00990 
00991                 conn->header_received = 0;
00992         }
00993 }

01063 static void
01064 flap_connection_send_byte_stream(ByteStream *bs, FlapConnection *conn, size_t count)
01065 {
01066         if (conn == NULL)
01067                 return;
01068 
01069         /* Make sure we don't send past the end of the bs */
01070         if (count > byte_stream_bytes_left(bs))
01071                 count = byte_stream_bytes_left(bs); /* truncate to remaining space */
01072 
01073         if (count == 0)
01074                 return;
01075 
01076         /* Add everything to our outgoing buffer */
01077         purple_circ_buffer_append(conn->buffer_outgoing, bs->data, count);
01078 
01079         /* If we haven't already started writing stuff, then start the cycle */
01080         if (conn->watcher_outgoing == 0)
01081         {
01082                 if (conn->gsc) {
01083                         conn->watcher_outgoing = purple_input_add(conn->gsc->fd,
01084                                         PURPLE_INPUT_WRITE, send_cb, conn);
01085                         send_cb(conn, -1, 0);
01086                 } else if (conn->fd >= 0) {
01087                         conn->watcher_outgoing = purple_input_add(conn->fd,
01088                                         PURPLE_INPUT_WRITE, send_cb, conn);
