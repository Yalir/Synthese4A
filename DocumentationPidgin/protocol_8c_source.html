00387                 /* we are not connected so ignore all packets to be send */

00414         else {
00415                 /* http connection */
00416

00418                         /* multimedia packets must be send with a HTTP POST */

00423                 }
00424         }

00427         session->last_tx = mxit_now_milli();

00431          * the server on this request

00441  * TX Step 1: Create a new Tx packet and queue it for sending.
00442  *
00443  *  @param session              The MXit session object
00444  *  @param data                 The packet data (payload)
00445  *  @param datalen              The length of the packet data
00446  *  @param cmd                  The MXit command for this packet
00447  */

00449 {

00451         char                            header[256];
00452         int                                     hlen;
00453 
00454         /* create a packet for sending */

00471         }
00472

00477                 packet->headerlen += sprintf( packet->header, "ln=%i%c", ( datalen + hlen ), CP_REC_TERM );             /* packet length */

00479

00481         memcpy( packet->header + packet->headerlen, header, hlen );

00488 
00489

00514 /*------------------------------------------------------------------------
00515  * Manage the packet send queue (send next packet, timeout's, etc).
00516  *
00517  *  @param session              The MXit session object
00518  */
00519 static void mxit_manage_queue( struct MXitSession* session )
00520 {
00521         struct tx_packet*       packet          = NULL;
00522         gint64                          now                     = mxit_now_milli();
00523 
00524         if ( !( session->flags & MXIT_FLAG_CONNECTED ) ) {
00525                 /* we are not connected, so ignore the queue */
00526                 return;
00527         }
00528         else if ( session->outack > 0 ) {
00529                 /* we are still waiting for an outstanding ACK from the MXit server */
00530                 if ( session->last_tx <= mxit_now_milli() - ( MXIT_ACK_TIMEOUT * 1000 ) ) {
00531                         /* ack timeout! so we close the connection here */
00532                         purple_debug_info( MXIT_PLUGIN_ID, "mxit_manage_queue: Timeout awaiting ACK for command '%i'\n", session->outack );
00533                         purple_connection_error( session->con, _( "Timeout while waiting for a response from the MXit server." ) );
00534                 }
00535                 return;
00536         }
00537 
00538         /*
00539          * the mxit server has flood detection and it prevents you from sending messages to fast.
00540          * this is a self defense mechanism, a very annoying feature. so the client must ensure that
00541          * it does not send messages too fast otherwise mxit will ignore the user for 30 seconds.
00542          * this is what we are trying to avoid here..
00543          */
00544         if ( session->q_fast_timer_id == 0 ) {
00545                 /* the fast timer has not been set yet */
00546                 if ( session->last_tx > ( now - MXIT_TX_DELAY ) ) {
00547                         /* we need to wait a little before sending the next packet, so schedule a wakeup call */
00548                         gint64 tdiff = now - ( session->last_tx );
00549                         guint delay = ( MXIT_TX_DELAY - tdiff ) + 9;
00550                         if ( delay <= 0 )
00551                                 delay = MXIT_TX_DELAY;
00552                         session->q_fast_timer_id = purple_timeout_add( delay, mxit_manage_queue_fast, session );
00553                 }
00554                 else {
00555                         /* get the next packet from the queue to send */
00556                         packet = pop_tx_packet( session );
00557                         if ( packet != NULL ) {
00558                                 /* there was a packet waiting to be sent to the server, now is the time to do something about it */
00559 
00560                                 /* send the packet to MXit server */
00561                                 mxit_send_packet( session, packet );
00562                         }
00563                 }
00564         }
00565 }

00602  * Callback to manage HTTP server polling (HTTP connections ONLY)
00603  *
00604  *  @param session              The MXit session object
00605  */
00606 gboolean mxit_manage_polling( gpointer user_data )
00607 {
00608         struct MXitSession* session             = (struct MXitSession*) user_data;
00609         gboolean                        poll            = FALSE;
00610         gint64                          now                     = mxit_now_milli();
00611         int                                     polldiff;
00612         gint64                          rxdiff;
00613 
00614         if ( !( session->flags & MXIT_FLAG_LOGGEDIN ) ) {
00615                 /* we only poll if we are actually logged in */
00616                 return TRUE;
00617         }
00618 
00619         /* calculate the time differences */
00620         rxdiff = now - session->last_rx;
00621         polldiff = now - session->http_last_poll;
00622 
00623         if ( rxdiff < MXIT_HTTP_POLL_MIN ) {
00624                 /* we received some reply a few moments ago, so reset the poll interval */
00625                 session->http_interval = MXIT_HTTP_POLL_MIN;
00626         }
00627         else if ( session->http_last_poll < ( now - session->http_interval ) ) {
00628                 /* time to poll again */
00629                 poll = TRUE;
00630 
00631                 /* back-off some more with the polling */

01513 /*------------------------------------------------------------------------
01514  * Process a received message packet.
01515  *
01516  *  @param session              The MXit session object
01517  *  @param records              The packet's data records
01518  *  @param rcount               The number of data records
01519  */
01520 static void mxit_parse_cmd_message( struct MXitSession* session, struct record** records, int rcount )
01521 {
01522         struct RXMsgData*       mx                      = NULL;
01523         char*                           message         = NULL;
01524         char*                           sender          = NULL;
01525         int                                     msglen          = 0;
01526         int                                     msgflags        = 0;
01527         int                                     msgtype         = 0;
01528 
01529         if ( ( rcount == 1 ) || ( records[0]->fcount < 2 ) || ( records[1]->fcount == 0 ) || ( records[1]->fields[0]->len == 0 ) ) {
01530                 /* packet contains no message or an empty message */
01531                 return;
01532         }
01533 
01534         message = records[1]->fields[0]->data;
01535         msglen = strlen( message );
01536 
01537         /* strip off dummy domain */
01538         sender = records[0]->fields[0]->data;
01539         mxit_strip_domain( sender );
01540 
01541 #ifdef  DEBUG_PROTOCOL
01542         purple_debug_info( MXIT_PLUGIN_ID, "Message received from '%s'\n", sender );
01543 #endif
01544 
01545         /* decode message flags (if any) */
01546         if ( records[0]->fcount >= 5 )
01547                 msgflags = atoi( records[0]->fields[4]->data );
01548         msgtype = atoi( records[0]->fields[2]->data );
01549 
01550         if ( msgflags & CP_MSG_PWD_ENCRYPTED ) {
01551                 /* this is a password encrypted message. we do not currently support those so ignore it */
01552                 PurpleBuddy*    buddy;
01553                 const char*             name;
01554                 char                    msg[128];
01555 
01556                 buddy = purple_find_buddy( session->acc, sender );
01557                 if ( buddy )
01558                         name = purple_buddy_get_alias( buddy );
01559                 else
01560                         name = sender;
01561                 g_snprintf( msg, sizeof( msg ), _( "%s sent you an encrypted message, but it is not supported on this client." ), name );
01562                 mxit_popup( PURPLE_NOTIFY_MSG_WARNING, _( "Message Error" ), msg );
01563                 return;
01564         }
01565         else if ( msgflags & CP_MSG_TL_ENCRYPTED ) {
01566                 /* this is a transport-layer encrypted message. */
01567                 message = mxit_decrypt_message( session, message );
01568                 if ( !message ) {
01569                         /* could not be decrypted */
01570                         serv_got_im( session->con, sender, _( "An encrypted message was received which could not be decrypted." ), PURPLE_MESSAGE_ERROR, time( NULL ) );
01571                         return;
01572                 }
01573         }
01574 
01575         if ( msgflags & CP_MSG_NOTIFY_DELIVERY ) {
01576                 /* delivery notification is requested */
01577                 if ( records[0]->fcount >= 4 )
01578                         mxit_send_msgevent( session, sender, records[0]->fields[3]->data, CP_MSGEVENT_DELIVERED );
01579         }
01580 
01581         /* create and initialise new markup struct */
01582         mx = g_new0( struct RXMsgData, 1 );
01583         mx->msg = g_string_sized_new( msglen );
01584         mx->session = session;
01585         mx->from = g_strdup( sender );
01586         mx->timestamp = atoi( records[0]->fields[1]->data );
01587         mx->got_img = FALSE;
01588         mx->chatid = -1;
01589         mx->img_count = 0;
01590 
01591         /* update list of active chats */
01592         if ( !find_active_chat( session->active_chats, mx->from ) ) {
01593                 session->active_chats = g_list_append( session->active_chats, g_strdup( mx->from ) );
01594         }
01595 
01596         if ( is_multimx_contact( session, mx->from ) ) {
01597                 /* this is a MultiMx chatroom message */
01598                 multimx_message_received( mx, message, msglen, msgtype, msgflags );
01599         }
01600         else {
01601                 mxit_parse_markup( mx, message, msglen, msgtype, msgflags );
01602         }
01603 
01604         /* we are now done parsing the message */
01605         mx->converted = TRUE;
01606         if ( mx->img_count == 0 ) {
01607                 /* we have all the data we need for this message to be displayed now. */
01608                 mxit_show_message( mx );
01609         }
01610         else {
01611                 /* this means there are still images outstanding for this message and
01612                  * still need to wait for them before we can display the message.
01613                  * so the image received callback function will eventually display
01614                  * the message. */
01615         }
01616 
01617         /* cleanup */
01618         if ( msgflags & CP_MSG_TL_ENCRYPTED )
01619                 g_free( message );
01620 }

02613 /*------------------------------------------------------------------------
02614  * Parse the received byte stream into a proper client protocol packet.
02615  *
02616  *  @param session              The MXit session object
02617  *  @return                             Success (0) or Failure (!0)
02618  */
02619 int mxit_parse_packet( struct MXitSession* session )
02620 {
02621         struct rx_packet        packet;
02622         struct record*          rec;
02623         struct field*           field;
02624         gboolean                        pbreak;
02625         unsigned int            i;
02626         int                                     res     = 0;
02627 
02628 #ifdef  DEBUG_PROTOCOL
02629         purple_debug_info( MXIT_PLUGIN_ID, "Received packet (%i bytes)\n", session->rx_i );
02630         dump_bytes( session, session->rx_dbuf, session->rx_i );
02631 #endif
02632 
02633         i = 0;
02634         while ( i < session->rx_i ) {
02635 
02636                 /* create first record and field */
02637                 rec = NULL;
02638                 field = NULL;
02639                 memset( &packet, 0x00, sizeof( struct rx_packet ) );
02640                 rec = add_record( &packet );
02641                 pbreak = FALSE;
02642 
02643                 /* break up the received packet into fields and records for easy parsing */
02644                 while ( ( i < session->rx_i ) && ( !pbreak ) ) {
02645 
02646                         switch ( session->rx_dbuf[i] ) {
02647                                 case CP_SOCK_REC_TERM :
02648                                                 /* new record */
02649                                                 if ( packet.rcount == 1 ) {
02650                                                         /* packet command */
02651                                                         packet.cmd = atoi( packet.records[0]->fields[0]->data );
02652                                                 }
02653                                                 else if ( packet.rcount == 2 ) {
02654                                                         /* special case: binary multimedia packets should not be parsed here */
02655                                                         if ( packet.cmd == CP_CMD_MEDIA ) {
02656                                                                 /* add the chunked to new record */
02657                                                                 rec = add_record( &packet );
02658                                                                 field = add_field( rec );
02659                                                                 field->data = &session->rx_dbuf[i + 1];
02660                                                                 field->len = session->rx_i - i;
02661                                                                 /* now skip the binary data */
02662                                                                 res = get_chunk_len( field->data );
02663                                                                 /* determine if we have more packets */
02664                                                                 if ( res + 6 + i < session->rx_i ) {
02665                                                                         /* we have more than one packet in this stream */
02666                                                                         i += res + 6;
02667                                                                         pbreak = TRUE;
02668                                                                 }
02669                                                                 else {
02670                                                                         i = session->rx_i;
02671                                                                 }
02672                                                         }
02673                                                 }
02674                                                 else if ( !field ) {
02675                                                         field = add_field( rec );
02676                                                         field->data = &session->rx_dbuf[i];
02677                                                 }
02678                                                 session->rx_dbuf[i] = '\0';
02679                                                 rec = add_record( &packet );
02680                                                 field = NULL;
02681 
02682                                                 break;
02683                                 case CP_FLD_TERM :
02684                                                 /* new field */
02685                                                 session->rx_dbuf[i] = '\0';
02686                                                 if ( !field ) {
02687                                                         field = add_field( rec );
02688                                                         field->data = &session->rx_dbuf[i];
02689                                                 }
02690                                                 field = NULL;
02691                                                 break;
02692                                 case CP_PKT_TERM :
02693                                                 /* packet is done! */
02694                                                 session->rx_dbuf[i] = '\0';
02695                                                 pbreak = TRUE;
02696                                                 break;
02697                                 default :
02698                                                 /* skip non special characters */
02699                                                 if ( !field ) {
02700                                                         field = add_field( rec );
02701                                                         field->data = &session->rx_dbuf[i];
02702                                                 }
02703                                                 field->len++;
02704                                                 break;
02705                         }
02706 
02707                         i++;
02708                 }
02709 
02710                 if ( packet.rcount < 2 ) {
02711                         /* bad packet */
02712                         purple_connection_error( session->con, _( "Invalid packet received from MXit." ) );
02713                         free_rx_packet( &packet );
02714                         continue;
02715                 }
02716 
02717                 session->rx_dbuf[session->rx_i] = '\0';
02718                 packet.errcode = atoi( packet.records[1]->fields[0]->data );
02719 
02720                 purple_debug_info( MXIT_PLUGIN_ID, "Packet received CMD:%i (%i)\n", packet.cmd, packet.errcode );
02721 #ifdef  DEBUG_PROTOCOL
02722                 /* debug */
02723                 dump_packet( &packet );
02724 #endif
02725 
02726                 /* reset the out ack */
02727                 if ( session->outack == packet.cmd ) {
02728                         /* outstanding ack received from mxit server */
02729                         session->outack = 0;
02730                 }
02731 
02732                 /* check packet status */
02733                 if ( packet.errcode != MXIT_ERRCODE_SUCCESS ) {
02734                         /* error reply! */
02735                         if ( ( packet.records[1]->fcount > 1 ) && ( packet.records[1]->fields[1]->data ) )
02736                                 packet.errmsg = packet.records[1]->fields[1]->data;
02737                         else
02738                                 packet.errmsg = NULL;
02739 
02740                         res = process_error_response( session, &packet );
02741                 }
02742                 else {
02743                         /* success reply! */
02744                         res = process_success_response( session, &packet );
02745                 }
02746 
02747                 /* free up the packet resources */
02748                 free_rx_packet( &packet );
02749         }
02750 
02751         if ( session->outack == 0 )
02752                         mxit_manage_queue( session );
02753 
02754         return res;
02755 }

02758 /*------------------------------------------------------------------------
02759  * Callback when data is received from the MXit server.
02760  *
02761  *  @param user_data            The MXit session object
02762  *  @param source                       The file-descriptor on which data was received
02763  *  @param cond                         Condition which caused the callback (PURPLE_INPUT_READ)
02764  */
02765 void mxit_cb_rx( gpointer user_data, gint source, PurpleInputCondition cond )
02766 {
02767         struct MXitSession*     session         = (struct MXitSession*) user_data;
02768         char                            ch;
02769         int                                     res;
02770         int                                     len;
02771 
02772         if ( session->rx_state == RX_STATE_RLEN ) {
02773                 /* we are reading in the packet length */
02774                 len = read( session->fd, &ch, 1 );
02775                 if ( len < 0 ) {
02776                         /* connection error */
02777                         purple_connection_error( session->con, _( "A connection error occurred to MXit. (read stage 0x01)" ) );
02778                         return;
02779                 }
02780                 else if ( len == 0 ) {
02781                         /* connection closed */
02782                         purple_connection_error( session->con, _( "A connection error occurred to MXit. (read stage 0x02)" ) );
02783                         return;
02784                 }
02785                 else {
02786                         /* byte read */
02787                         if ( ch == CP_REC_TERM ) {
02788                                 /* the end of the length record found */
02789                                 session->rx_lbuf[session->rx_i] = '\0';
02790                                 session->rx_res = atoi( &session->rx_lbuf[3] );
02791                                 if ( session->rx_res > CP_MAX_PACKET ) {
02792                                         purple_connection_error( session->con, _( "A connection error occurred to MXit. (read stage 0x03)" ) );
02793                                 }
02794                                 session->rx_state = RX_STATE_DATA;
02795                                 session->rx_i = 0;
02796                         }
02797                         else {
02798                                 /* still part of the packet length record */
02799                                 session->rx_lbuf[session->rx_i] = ch;
02800                                 session->rx_i++;
02801                                 if ( session->rx_i >= sizeof( session->rx_lbuf ) ) {
02802                                         /* malformed packet length record (too long) */
02803                                         purple_connection_error( session->con, _( "A connection error occurred to MXit. (read stage 0x04)" ) );
02804                                         return;
02805                                 }
02806                         }
02807                 }
02808         }
02809         else if ( session->rx_state == RX_STATE_DATA ) {
02810                 /* we are reading in the packet data */
02811                 len = read( session->fd, &session->rx_dbuf[session->rx_i], session->rx_res );
02812                 if ( len < 0 ) {
02813                         /* connection error */
02814                         purple_connection_error( session->con, _( "A connection error occurred to MXit. (read stage 0x05)" ) );
02815                         return;
02816                 }
02817                 else if ( len == 0 ) {
02818                         /* connection closed */
02819                         purple_connection_error( session->con, _( "A connection error occurred to MXit. (read stage 0x06)" ) );
02820                         return;
02821                 }
02822                 else {
02823                         /* data read */
02824                         session->rx_i += len;
02825                         session->rx_res -= len;
02826 
02827                         if ( session->rx_res == 0 ) {
02828                                 /* ok, so now we have read in the whole packet */
02829                                 session->rx_state = RX_STATE_PROC;
02830                         }
02831                 }
02832         }
02833 
02834         if ( session->rx_state == RX_STATE_PROC ) {
02835                 /* we have a full packet, which we now need to process */
02836                 res = mxit_parse_packet( session );
02837 
02838                 if ( res == 0 ) {
02839                         /* we are still logged in */
02840                         session->rx_state = RX_STATE_RLEN;
02841                         session->rx_res = 0;
02842                         session->rx_i = 0;
02843                 }
02844         }
02845 }
