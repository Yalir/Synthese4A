00745         PurpleStatusType *status_type;

00747         g_return_if_fail(status != NULL);
00748 
00749         if (!active && purple_status_is_exclusive(status))
00750         {
00751                 purple_debug_error("status",
00752                                    "Cannot deactivate an exclusive status (%s).\n",
00753                                    purple_status_get_id(status));
00754                 return;
00755         }
00756 
00757         if (status->active != active)
00758         {
00759                 changed = TRUE;
00760         }
00761 
00762         status->active = active;
00763

00766         while (l != NULL)
00767         {
00768                 const gchar *id;
00769                 PurpleValue *value;
00770 
00771                 id = l->data;
00772                 l = l->next;
00773                 value = purple_status_get_attr_value(status, id);
00774                 if (value == NULL)
00775                 {
00776                         purple_debug_warning("status", "The attribute \"%s\" on the status \"%s\" is "
00777                                                            "not supported.\n", id, status->type->name);
00778                         /* Skip over the data and move on to the next attribute */
00779                         l = l->next;
00780                         continue;
00781                 }
00782

00784

00790                                 continue;
00791                         purple_status_set_attr_string(status, id, string_data);
00792                         changed = TRUE;
00793                 }
00794                 else if (value->type == PURPLE_TYPE_INT)
00795                 {
00796                         int int_data = GPOINTER_TO_INT(l->data);
00797                         l = l->next;
00798                         if (int_data == value->data.int_data)
00799                                 continue;
00800                         purple_status_set_attr_int(status, id, int_data);
00801                         changed = TRUE;
00802                 }
00803                 else if (value->type == PURPLE_TYPE_BOOLEAN)
00804                 {
00805                         gboolean boolean_data = GPOINTER_TO_INT(l->data);
00806                         l = l->next;
00807                         if (boolean_data == value->data.boolean_data)
00808                                 continue;
00809                         purple_status_set_attr_boolean(status, id, boolean_data);
00810                         changed = TRUE;
00811                 }
00812                 else
00813                 {
00814                         /* We don't know what the data is--skip over it */
00815                         l = l->next;
00816                 }
00817         }

00820         status_type = purple_status_get_type(status);

00823                 PurpleStatusAttr *attr;
00824 
00825                 attr = l->data;
00826                 l = l->next;
00827 
00828                 if (!g_list_find_custom(specified_attr_ids, attr->id, (GCompareFunc)strcmp)) {

00831                         if (default_value->type == PURPLE_TYPE_STRING) {
00832                                 const char *cur = purple_status_get_attr_string(status, attr->id);
00833                                 const char *def = purple_value_get_string(default_value);
00834                                 if ((cur == NULL && def == NULL)
00835                                     || (cur != NULL && def != NULL
00836                                         && !strcmp(cur, def))) {
00837                                         continue;
00838                                 }
00839 
00840                                 purple_status_set_attr_string(status, attr->id, def);
00841                         } else if (default_value->type == PURPLE_TYPE_INT) {
00842                                 int cur = purple_status_get_attr_int(status, attr->id);
00843                                 int def = purple_value_get_int(default_value);
00844                                 if (cur == def)
00845                                         continue;
00846 
00847                                 purple_status_set_attr_int(status, attr->id, def);
00848                         } else if (default_value->type == PURPLE_TYPE_BOOLEAN) {
00849                                 gboolean cur = purple_status_get_attr_boolean(status, attr->id);
00850                                 gboolean def = purple_value_get_boolean(default_value);
00851                                 if (cur == def)
00852                                         continue;
00853 
00854                                 purple_status_set_attr_boolean(status, attr->id, def);
00855                         }
00856                         changed = TRUE;
00857                 }
00858         }
