00077 /*------------------------------------------------------------------------
00078  * Callback function invoked when an inline image request to a web site completes.
00079  *
00080  *  @param url_data
00081  *  @param user_data            The Markup message object
00082  *  @param url_text                     The data returned from the WAP site
00083  *  @param len                          The length of the data returned
00084  *  @param error_message        Descriptive error message
00085  */
00086 static void mxit_cb_ii_returned(PurpleUtilFetchUrlData* url_data, gpointer user_data, const gchar* url_text, gsize len, const gchar* error_message)
00087 {
00088         struct ii_url_request*  iireq           = (struct ii_url_request*) user_data;
00089         int*                                    intptr          = NULL;
00090         int                                             id;
00091 
00092 #ifdef  MXIT_DEBUG_COMMANDS
00093         purple_debug_info(MXIT_PLUGIN_ID, "Inline Image returned from %s\n", iireq->url);
00094 #endif
00095 
00096         if (!url_text) {
00097                 /* no reply from the WAP site */
00098                 purple_debug_error(MXIT_PLUGIN_ID, "Error downloading Inline Image from %s.\n", iireq->url);
00099                 goto done;
00100         }
00101 
00102         /* lets first see if we don't have the inline image already in cache */
00103         if (g_hash_table_lookup(iireq->mx->session->iimages, iireq->url)) {
00104                 /* inline image found in the cache, so we just ignore this reply */
00105                 goto done;
00106         }
00107 
00108         /* we now have the inline image, store a copy in the imagestore */
00109         id = purple_imgstore_add_with_id(g_memdup(url_text, len), len, NULL);
00110 
00111         /* map the inline image id to purple image id */
00112         intptr = g_malloc(sizeof(int));
00113         *intptr = id;
00114         g_hash_table_insert(iireq->mx->session->iimages, iireq->url, intptr);
00115 
00116         iireq->mx->flags |= PURPLE_MESSAGE_IMAGES;
00117 
00118 done:
00119         iireq->mx->img_count--;
00120         if ((iireq->mx->img_count == 0) && (iireq->mx->converted)) {
00121                 /*
00122                  * this was the last outstanding emoticon for this message,
00123                  * so we can now display it to the user.
00124                  */
00125                 mxit_show_message(iireq->mx);
00126         }
00127 
00128         g_free(iireq);
00129 }

00132 /*------------------------------------------------------------------------
00133  * Return the command identifier of this MXit Command.
00134  *
00135  *  @param cmd                  The MXit command <key,value> map
00136  *  @return                             The MXit command identifier
00137  */
00138 static MXitCommandType command_type(GHashTable* hash)
00139 {
00140         char* op;
00141         char* type;
00142 
00143         op = g_hash_table_lookup(hash, "op");
00144         if (op) {
00145                 if ( strcmp(op, "cmd") == 0 ) {
00146                         type = g_hash_table_lookup(hash, "type");
00147                         if (type == NULL)                                                               /* no command provided */
00148                                 return MXIT_CMD_UNKNOWN;
00149                         else if (strcmp(type, "clear") == 0)                    /* clear */
00150                                 return MXIT_CMD_CLEAR;
00151                         else if (strcmp(type, "sendsms") == 0)                  /* send an SMS */
00152                                 return MXIT_CMD_SENDSMS;
00153                         else if (strcmp(type, "reply") == 0)                    /* list of options */
00154                                 return MXIT_CMD_REPLY;
00155                         else if (strcmp(type, "platreq") == 0)                  /* platform request */
00156                                 return MXIT_CMD_PLATREQ;
00157                         else if (strcmp(type, "selc") == 0)                             /* select contact */
00158                                 return MXIT_CMD_SELECTCONTACT;
00159                 }
00160                 else if (strcmp(op, "img") == 0)                                        /* inline image */
00161                         return MXIT_CMD_IMAGE;
00162                 else if (strcmp(op, "csc") == 0)                                        /* chat-screen config */
00163                         return MXIT_CMD_SCREENCONFIG;
00164                 else if (strcmp(op, "csi") == 0)                                        /* chat-screen info */
00165                         return MXIT_CMD_SCREENINFO;
00166                 else if (strcmp(op, "is") == 0)                                         /* image-strip */
00167                         return MXIT_CMD_IMAGESTRIP;
00168                 else if (strcmp(op, "tbl") == 0)                                        /* table */
00169                         return MXIT_CMD_TABLE;
00170         }
00171 
00172         return MXIT_CMD_UNKNOWN;
00173 }

00176 /*------------------------------------------------------------------------
00177  * Tokenize a MXit Command string into a <key,value> map.
00178  *
00179  *  @param cmd                  The MXit command string
00180  *  @return                             The <key,value> hash-map, or NULL on error.
00181  */
00182 static GHashTable* command_tokenize(char* cmd)
00183 {
00184         GHashTable* hash        = NULL;
00185         gchar**         parts;
00186         int                     i               = 0;
00187 
00188 #ifdef MXIT_DEBUG_COMMANDS
00189         purple_debug_info(MXIT_PLUGIN_ID, "command: '%s'\n", cmd);
00190 #endif
00191 
00192         /* explode the command into parts */
00193         parts = g_strsplit(cmd, "|", 0);
00194 
00195         hash = g_hash_table_new_full(g_str_hash, g_str_equal, g_free, g_free);
00196 
00197         /* now break part into a key & value */
00198         while (parts[i] != NULL) {
00199                 char* value;
00200 
00201                 value = strchr(parts[i], '=');          /* find start of value */
00202                 if (value != NULL) {
00203                         *value = '\0';
00204                         value++;
00205                 }
00206 
00207 #ifdef MXIT_DEBUG_COMMANDS
00208                 purple_debug_info(MXIT_PLUGIN_ID, "  key='%s' value='%s'\n", parts[i], value);
00209 #endif
00210 
00211                 g_hash_table_insert(hash, g_strdup(parts[i]), g_strdup(value));
00212 
00213                 i++;
00214         }
00215 
00216         g_strfreev(parts);
00217 
00218         return hash;
00219 }

00557 /*------------------------------------------------------------------------
00558  * Process a received MXit Command message.
00559  *
00560  *  @param mx                           The received message data object
00561  *  @param message                      The message text
00562  *  @return                                     The length of the command
00563  */
00564 int mxit_parse_command(struct RXMsgData* mx, char* message)
00565 {
00566         GHashTable* hash        = NULL;
00567         char*           start;
00568         char*           end;
00569 
00570         /* ensure that this is really a command */
00571         if ( ( message[0] != ':' ) || ( message[1] != ':' ) ) {
00572                 /* this is not a command */
00573                 return 0;
00574         }
00575 
00576         start = message + 2;
00577         end = strstr(start, ":");
