00052 send_keepalive(gpointer data)
00053 {
00054         PurpleConnection *gc = data;
00055         PurplePluginProtocolInfo *prpl_info;
00056 
00057         /* Only send keep-alives if we haven't heard from the
00058          * server in a while.
00059          */
00060         if ((time(NULL) - gc->last_received) < KEEPALIVE_INTERVAL)
00061                 return TRUE;
00062 
00063         prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(gc->prpl);
00064         if (prpl_info->keepalive)
00065                 prpl_info->keepalive(gc);
00066 
00067         return TRUE;

00076                 prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(gc->prpl);

00079                 return;

00083                 purple_debug_info("connection", "Activating keepalive.\n");

00087         {
00088                 purple_debug_info("connection", "Deactivating keepalive.\n");

00557          */

00559                 purple_debug_error("connection",
00560                         "purple_connection_error_reason: reason %u isn't a "

00564 
00565         if (description == NULL) {
00566                 purple_debug_error("connection", "purple_connection_error_reason called with NULL description\n");
00567                 description = _("Unknown error");
00568         }
00569 
00570         /* If we've already got one error, we don't need any more */
00571         if (gc->disconnect_timeout > 0)
00572                 return;
00573 
00574         gc->wants_to_die = purple_connection_error_is_fatal (reason);
00575 
00576         purple_debug_info("connection", "Connection error on %p (reason: %u description: %s)\n",
00577                           gc, reason, description);
00578 
00579         ops = purple_connections_get_ui_ops();
00580 
00581         if (ops != NULL)
00582         {
00583                 if (ops->report_disconnect_reason != NULL)
00584                         ops->report_disconnect_reason (gc, reason, description);
