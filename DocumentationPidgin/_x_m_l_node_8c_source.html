00297 void
00298 xmlnode_free(xmlnode *node)
00299 {
00300         xmlnode *x, *y;
00301 
00302         g_return_if_fail(node != NULL);
00303 
00304         /* if we're part of a tree, remove ourselves from the tree first */
00305         if(NULL != node->parent) {
00306                 if(node->parent->child == node) {
00307                         node->parent->child = node->next;
00308                         if (node->parent->lastchild == node)
00309                                 node->parent->lastchild = node->next;
00310                 } else {
00311                         xmlnode *prev = node->parent->child;
00312                         while(prev && prev->next != node) {
00313                                 prev = prev->next;
00314                         }
00315                         if(prev) {
00316                                 prev->next = node->next;
00317                                 if (node->parent->lastchild == node)
00318                                         node->parent->lastchild = prev;
00319                         }
00320                 }
00321         }
00322 
00323         /* now free our children */
00324         x = node->child;
00325         while(x) {
00326                 y = x->next;
00327                 xmlnode_free(x);
00328                 x = y;
00329         }
00330 
00331         /* now dispose of ourselves */
00332         g_free(node->name);
00333         g_free(node->data);
00334         g_free(node->xmlns);
00335         g_free(node->prefix);
00336 
00337         if(node->namespace_map)
00338                 g_hash_table_destroy(node->namespace_map);
00339 
00340         PURPLE_DBUS_UNREGISTER_POINTER(node);
00341         g_free(node);
00342 }

00430 static char *
00431 xmlnode_to_str_helper(const xmlnode *node, int *len, gboolean formatting, int depth)
00432 {
00433         GString *text = g_string_new("");
00434         const char *prefix;
00435         const xmlnode *c;
00436         char *node_name, *esc, *esc2, *tab = NULL;
00437         gboolean need_end = FALSE, pretty = formatting;
00438 
00439         g_return_val_if_fail(node != NULL, NULL);
00440 
00441         if(pretty && depth) {
00442                 tab = g_strnfill(depth, '\t');
00443                 text = g_string_append(text, tab);
00444         }
00445 
00446         node_name = g_markup_escape_text(node->name, -1);
00447         prefix = xmlnode_get_prefix(node);
00448 
00449         if (prefix) {
00450                 g_string_append_printf(text, "<%s:%s", prefix, node_name);
00451         } else {
00452                 g_string_append_printf(text, "<%s", node_name);
00453         }
00454 
00455         if (node->namespace_map) {
00456                 g_hash_table_foreach(node->namespace_map,
00457                         (GHFunc)xmlnode_to_str_foreach_append_ns, text);
00458         } else if (node->xmlns) {
00459                 if(!node->parent || !purple_strequal(node->xmlns, node->parent->xmlns))
00460                 {
00461                         char *xmlns = g_markup_escape_text(node->xmlns, -1);
00462                         g_string_append_printf(text, " xmlns='%s'", xmlns);
00463                         g_free(xmlns);
00464                 }
00465         }
00466         for(c = node->child; c; c = c->next)
00467         {
00468                 if(c->type == XMLNODE_TYPE_ATTRIB) {
00469                         const char *aprefix = xmlnode_get_prefix(c);
00470                         esc = g_markup_escape_text(c->name, -1);
00471                         esc2 = g_markup_escape_text(c->data, -1);
00472                         if (aprefix) {
00473                                 g_string_append_printf(text, " %s:%s='%s'", aprefix, esc, esc2);
00474                         } else {
00475                                 g_string_append_printf(text, " %s='%s'", esc, esc2);
00476                         }
00477                         g_free(esc);
00478                         g_free(esc2);
00479                 } else if(c->type == XMLNODE_TYPE_TAG || c->type == XMLNODE_TYPE_DATA) {
00480                         if(c->type == XMLNODE_TYPE_DATA)
00481                                 pretty = FALSE;
00482                         need_end = TRUE;
00483                 }
00484         }
00485 
00486         if(need_end) {
00487                 g_string_append_printf(text, ">%s", pretty ? NEWLINE_S : "");
00488 
00489                 for(c = node->child; c; c = c->next)
00490                 {
00491                         if(c->type == XMLNODE_TYPE_TAG) {
00492                                 int esc_len;
00493                                 esc = xmlnode_to_str_helper(c, &esc_len, pretty, depth+1);
00494                                 text = g_string_append_len(text, esc, esc_len);
00495                                 g_free(esc);
00496                         } else if(c->type == XMLNODE_TYPE_DATA && c->data_sz > 0) {
00497                                 esc = g_markup_escape_text(c->data, c->data_sz);
00498                                 text = g_string_append(text, esc);
00499                                 g_free(esc);
00500                         }
00501                 }
00502 
00503                 if(tab && pretty)
00504                         text = g_string_append(text, tab);
00505                 if (prefix) {
00506                         g_string_append_printf(text, "</%s:%s>%s", prefix, node_name, formatting ? NEWLINE_S : "");
00507                 } else {
00508                         g_string_append_printf(text, "</%s>%s", node_name, formatting ? NEWLINE_S : "");
00509                 }
00510         } else {
00511                 g_string_append_printf(text, "/>%s", formatting ? NEWLINE_S : "");
00512         }
00513 
00514         g_free(node_name);
00515
