01499 static void
01500 s5_canread_again(gpointer data, gint source, PurpleInputCondition cond)
01501 {
01502         guchar *dest, *buf;
01503         PurpleProxyConnectData *connect_data = data;
01504         int len;
01505 
01506         if (connect_data->read_buffer == NULL) {
01507                 connect_data->read_buf_len = 5;
01508                 connect_data->read_buffer = g_malloc(connect_data->read_buf_len);
01509                 connect_data->read_len = 0;
01510         }
01511 
01512         dest = connect_data->read_buffer + connect_data->read_len;
01513         buf = connect_data->read_buffer;
01514 
01515         len = read(connect_data->fd, dest, (connect_data->read_buf_len - connect_data->read_len));
01516 
01517         if (len == 0)
01518         {
01519                 purple_proxy_connect_data_disconnect(connect_data,
01520                                 _("Server closed the connection"));
01521                 return;
01522         }
01523 
01524         if (len < 0)
01525         {
01526                 if (errno == EAGAIN)
01527                         /* No worries */
01528                         return;
01529 
01530                 /* Error! */
01531                 purple_proxy_connect_data_disconnect_formatted(connect_data,
01532                                 _("Lost connection with server: %s"), g_strerror(errno));
01533                 return;
01534         }
01535 
01536         connect_data->read_len += len;
01537 
01538         if(connect_data->read_len < 4)
01539                 return;
01540 
01541         if ((buf[0] != 0x05) || (buf[1] != 0x00)) {
01542                 if ((buf[0] == 0x05) && (buf[1] < 0x09)) {
01543                         purple_debug_error("socks5 proxy", "%s", socks5errors[buf[1]]);
01544                         purple_proxy_connect_data_disconnect(connect_data,
01545                                         socks5errors[buf[1]]);

01547                         purple_debug_error("socks5 proxy", "Bad data.\n");

01549                                         _("Received invalid data on connection with server"));
01550                 }
01551                 return;
01552         }
01553 
01554         /* Skip past BND.ADDR */
01555         switch(buf[3]) {
01556                 case 0x01: /* the address is a version-4 IP address, with a length of 4 octets */
01557                         if(!s5_ensure_buffer_length(connect_data, 4 + 4))
01558                                 return;
01559                         buf += 4 + 4;
01560                         break;
01561                 case 0x03: /* the address field contains a fully-qualified domain name.  The first
01562                                           octet of the address field contains the number of octets of name that
01563                                           follow, there is no terminating NUL octet. */
01564                         if(!s5_ensure_buffer_length(connect_data, 4 + 1))
01565                                 return;
01566                         buf += 4;
01567                         if(!s5_ensure_buffer_length(connect_data, 4 + 1 + buf[0]))
01568                                 return;
01569                         buf += buf[0] + 1;
01570                         break;
01571                 case 0x04: /* the address is a version-6 IP address, with a length of 16 octets */
01572                         if(!s5_ensure_buffer_length(connect_data, 4 + 16))
01573                                 return;
01574                         buf += 4 + 16;
01575                         break;
01576                 default:
01577                         purple_debug_error("socks5 proxy", "Invalid ATYP received (0x%X)\n", buf[3]);
01578                         purple_proxy_connect_data_disconnect(connect_data,
01579                                         _("Received invalid data on connection with server"));
01580                         return;
01581         }
01582 
01583         /* Skip past BND.PORT */
01584         if(!s5_ensure_buffer_length(connect_data, (buf - connect_data->read_buffer) + 2))
01585                 return;
01586 
01587         purple_proxy_connect_data_connected(connect_data);
01588 }

01716 /*
01717  * Return how many bytes we processed
01718  * -1 means we've shouldn't keep reading from the buffer
01719  */
01720 static gssize
01721 s5_parse_chap_msg(PurpleProxyConnectData *connect_data)
01722 {
01723         guchar *buf, *cmdbuf = connect_data->read_buffer;
01724         int len, navas, currentav;
01725 
01726         purple_debug_misc("socks5 proxy", "Reading CHAP message: %x\n", *cmdbuf);
01727 
01728         if (*cmdbuf != 0x01) {
01729                 purple_proxy_connect_data_disconnect(connect_data,
01730                                 _("Received invalid data on connection with server"));
01731                 return -1;
01732         }
01733         cmdbuf++;
01734 
01735         navas = *cmdbuf;
01736 
01737         purple_debug_misc("socks5 proxy", "Expecting %d attribute(s).\n", navas);
01738 
01739         cmdbuf++;
01740 
01741         for (currentav = 0; currentav < navas; currentav++) {
01742 
01743                 len = connect_data->read_len - (cmdbuf - connect_data->read_buffer);
01744                 /* We don't have enough data to even know how long the next attribute is,
01745                  * or we don't have the full length of the next attribute. */
01746                 if (len < 2 || len < (cmdbuf[1] + 2)) {
01747                         /* Clear out the attributes that have been read - decrease the attribute count */
01748                         connect_data->read_buffer[1] = navas - currentav;
01749                         /* Move the unprocessed data into the first attribute position */
01750                         memmove((connect_data->read_buffer + 2), cmdbuf, len);
01751                         /* Decrease the read count accordingly */
01752                         connect_data->read_len = len + 2;
01753 
01754                         purple_debug_info("socks5 proxy", "Need more data to retrieve attribute %d.\n", currentav);
01755 
01756                         return -1;
01757                 }
01758 
01759                 buf = cmdbuf + 2;
01760 
01761                 if (cmdbuf[1] == 0) {
01762                         purple_debug_error("socks5 proxy", "Attribute %x Value length of 0; ignoring.\n", cmdbuf[0]);
01763                         cmdbuf = buf;
01764                         continue;
01765                 }
01766 
01767                 switch (cmdbuf[0]) {
01768                         case 0x00:
01769                                 purple_debug_info("socks5 proxy", "Received STATUS of %x\n", buf[0]);
01770                                 /* Did auth work? */
01771                                 if (buf[0] == 0x00) {
01772                                         purple_input_remove(connect_data->inpa);
01773                                         connect_data->inpa = 0;
01774                                         g_free(connect_data->read_buffer);
01775                                         connect_data->read_buffer = NULL;
01776                                         /* Success */
01777                                         s5_sendconnect(connect_data, connect_data->fd);
01778                                 } else {
01779                                         /* Failure */
01780                                         purple_debug_warning("proxy",
01781                                                 "socks5 CHAP authentication "
01782                                                 "failed.  Disconnecting...");
01783                                         purple_proxy_connect_data_disconnect(connect_data,
01784                                                         _("Authentication failed"));
01785                                 }
01786                                 return -1;
01787                         case 0x01:
01788                                 /* We've already validated that cmdbuf[1] is sane. */
01789                                 purple_debug_info("socks5 proxy", "Received TEXT-MESSAGE of '%.*s'\n", (int) cmdbuf[1], buf);
01790                                 break;
01791                         case 0x03:
01792                                 purple_debug_info("socks5 proxy", "Received CHALLENGE\n");
01793                                 /* Server wants our credentials */
01794 
01795                                 connect_data->write_buf_len = 16 + 4;
01796                                 connect_data->write_buffer = g_malloc(connect_data->write_buf_len);
01797                                 connect_data->written_len = 0;
01798

01801                                         connect_data->write_buffer + 4);

01804                                 connect_data->write_buffer[1] = 0x01;
01805                                 connect_data->write_buffer[2] = 0x04;
01806                                 connect_data->write_buffer[3] = 0x10;
01807 
01808                                 purple_input_remove(connect_data->inpa);
01809                                 g_free(connect_data->read_buffer);
01810                                 connect_data->read_buffer = NULL;
01811 
01812                                 connect_data->read_cb = s5_readchap;
01813 
01814                                 connect_data->inpa = purple_input_add(connect_data->fd,
01815                                         PURPLE_INPUT_WRITE, proxy_do_write, connect_data);
01816 
01817                                 proxy_do_write(connect_data, connect_data->fd, PURPLE_INPUT_WRITE);
01818                                 return -1;
01819                         case 0x11:
01820                                 purple_debug_info("socks5 proxy", "Received ALGORIGTHMS of %x\n", buf[0]);
01821                                 /* Server wants to select an algorithm */
01822                                 if (buf[0] != 0x85) {
01823                                         /* Only currently support HMAC-MD5 */
01824                                         purple_debug_warning("proxy",
01825                                                 "Server tried to select an "
01826                                                 "algorithm that we did not advertise "
01827                                                 "as supporting.  This is a violation "
01828                                                 "of the socks5 CHAP specification.  "
01829                                                 "Disconnecting...");
01830                                         purple_proxy_connect_data_disconnect(connect_data,
01831                                                         _("Received invalid data on connection with server"));
01832                                         return -1;
01833                                 }
01834                                 break;
01835                         default:
01836                                 purple_debug_info("socks5 proxy", "Received unused command %x, length=%d\n", cmdbuf[0], cmdbuf[1]);
01837                 }
01838                 cmdbuf = buf + cmdbuf[1];
01839         }
01840 
01841         return (cmdbuf - connect_data->read_buffer);
01842 }

01885         connect_data->read_len += len;
01886 
01887         /* We may have read more than one message into the buffer, we need to make sure to process them all */
01888         while (1) {
01889 
01890                 /* We need more to be able to read this message */
01891                 if (connect_data->read_len < 2)
01892                         return;
01893 
01894                 msg_ret = s5_parse_chap_msg(connect_data);
01895 
01896                 if (msg_ret < 0)
01897                         return;
01898 
01899                 /* See if we have another message already in the buffer */
01900                 if ((len = connect_data->read_len - msg_ret) > 0) {
01901 
01902                         /* Move on to the next message */
01903                         memmove(connect_data->read_buffer, connect_data->read_buffer + msg_ret, len);
01904                         /* Decrease the read count accordingly */
01905                         connect_data->read_len = len;
01906 
01907                         /* Try to read the message that connect_data->read_buffer now points to */
01908                         continue;
01909                 }
01910 
01911                 break;
01912         }
01913 
01914         /* Fell through.  We ran out of CHAP events to process, but haven't
01915          * succeeded or failed authentication - there may be more to come.
01916          * If this is the case, come straight back here. */
01917 
01918         purple_debug_info("socks5 proxy", "Waiting for another message from which to read CHAP info.\n");
01919 
01920         /* We've processed all the available attributes, so get ready for a whole new message */
01921         g_free(connect_data->read_buffer);
01922         connect_data->read_buffer = NULL;
01923 }

01925 static void
01926 s5_canread(gpointer data, gint source, PurpleInputCondition cond)
01927 {
01928         PurpleProxyConnectData *connect_data = data;
01929         int len;
01930 
01931         if (connect_data->read_buffer == NULL) {
01932                 connect_data->read_buf_len = 2;
01933                 connect_data->read_buffer = g_malloc(connect_data->read_buf_len);
01934                 connect_data->read_len = 0;
01935         }
01936 
01937         purple_debug_info("socks5 proxy", "Able to read.\n");
01938 
01939         len = read(connect_data->fd, connect_data->read_buffer + connect_data->read_len,
01940                 connect_data->read_buf_len - connect_data->read_len);
01941 
01942         if (len == 0)
01943         {
01944                 purple_proxy_connect_data_disconnect(connect_data,
01945                                 _("Server closed the connection"));
01946                 return;
01947         }
01948 
01949         if (len < 0)
01950         {
01951                 if (errno == EAGAIN)
01952                         /* No worries */
01953                         return;
01954 
01955                 /* Error! */
01956                 purple_proxy_connect_data_disconnect_formatted(connect_data,
01957                                 _("Lost connection with server: %s"), g_strerror(errno));
01958                 return;
01959         }
01960 
01961         connect_data->read_len += len;
01962         if (connect_data->read_len < 2)
01963                 return;
01964 
01965         purple_input_remove(connect_data->inpa);
01966         connect_data->inpa = 0;
01967 
01968         if ((connect_data->read_buffer[0] != 0x05) || (connect_data->read_buffer[1] == 0xff)) {
01969                 purple_proxy_connect_data_disconnect(connect_data,
01970                                 _("Received invalid data on connection with server"));
01971                 return;
01972         }
01973 
01974         if (connect_data->read_buffer[1] == 0x02) {
01975                 size_t i, j;
01976                 const char *u, *p;
01977 
01978                 u = purple_proxy_info_get_username(connect_data->gpi);
01979                 p = purple_proxy_info_get_password(connect_data->gpi);
01980 
01981                 i = (u == NULL) ? 0 : strlen(u);
01982                 j = (p == NULL) ? 0 : strlen(p);
01983 
01984                 connect_data->write_buf_len = 1 + 1 + i + 1 + j;
01985                 connect_data->write_buffer = g_malloc(connect_data->write_buf_len);
01986                 connect_data->written_len = 0;
01987 
01988                 connect_data->write_buffer[0] = 0x01;   /* version 1 */
01989                 connect_data->write_buffer[1] = i;
01990                 if (u != NULL)
01991                         memcpy(connect_data->write_buffer + 2, u, i);
01992                 connect_data->write_buffer[2 + i] = j;
01993                 if (p != NULL)
01994                         memcpy(connect_data->write_buffer + 2 + i + 1, p, j);
01995 
01996                 g_free(connect_data->read_buffer);
01997                 connect_data->read_buffer = NULL;
01998 
01999                 connect_data->read_cb = s5_readauth;
02000 
02001                 connect_data->inpa = purple_input_add(connect_data->fd, PURPLE_INPUT_WRITE,
02002                         proxy_do_write, connect_data);
02003 
02004                 proxy_do_write(connect_data, connect_data->fd, PURPLE_INPUT_WRITE);
02005 
02006                 return;
02007         } else if (connect_data->read_buffer[1] == 0x03) {
02008                 size_t userlen;
02009                 userlen = strlen(purple_proxy_info_get_username(connect_data->gpi));
02010 
02011                 connect_data->write_buf_len = 7 + userlen;
02012                 connect_data->write_buffer = g_malloc(connect_data->write_buf_len);
02013                 connect_data->written_len = 0;
02014 
02015                 connect_data->write_buffer[0] = 0x01;
02016                 connect_data->write_buffer[1] = 0x02;
02017                 connect_data->write_buffer[2] = 0x11;
02018                 connect_data->write_buffer[3] = 0x01;
02019                 connect_data->write_buffer[4] = 0x85;
02020                 connect_data->write_buffer[5] = 0x02;
02021                 connect_data->write_buffer[6] = userlen;
02022                 memcpy(connect_data->write_buffer + 7,
02023                         purple_proxy_info_get_username(connect_data->gpi), userlen);
02024 
02025                 g_free(connect_data->read_buffer);
02026                 connect_data->read_buffer = NULL;
02027 
02028                 connect_data->read_cb = s5_readchap;
02029 
02030                 connect_data->inpa = purple_input_add(connect_data->fd, PURPLE_INPUT_WRITE,
02031                         proxy_do_write, connect_data);
02032 
02033                 proxy_do_write(connect_data, connect_data->fd, PURPLE_INPUT_WRITE);
02034 
02035                 return;
02036         } else {
02037                 g_free(connect_data->read_buffer);
02038                 connect_data->read_buffer = NULL;
02039 
02040                 s5_sendconnect(connect_data, connect_data->fd);
02041         }
