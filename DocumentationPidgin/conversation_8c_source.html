00929                                                 const char *message, PurpleMessageFlags flags,
00930                                                 time_t mtime)
00931 {
00932         PurplePluginProtocolInfo *prpl_info = NULL;
00933         PurpleConnection *gc = NULL;
00934         PurpleAccount *account;
00935         PurpleConversationUiOps *ops;
00936         const char *alias;
00937         char *displayed = NULL;
00938         PurpleBuddy *b;

00940         PurpleConversationType type;
00941         /* int logging_font_options = 0; */

00944         g_return_if_fail(message != NULL);
00945 
00946         ops = purple_conversation_get_ui_ops(conv);
00947 
00948         account = purple_conversation_get_account(conv);
00949         type = purple_conversation_get_type(conv);
00950 
00951         if (account != NULL)
00952                 gc = purple_account_get_connection(account);
00953 
00954         if (purple_conversation_get_type(conv) == PURPLE_CONV_TYPE_CHAT &&
00955                 (gc != NULL && !g_slist_find(gc->buddy_chats, conv)))
00956                 return;
00957 
00958         if (purple_conversation_get_type(conv) == PURPLE_CONV_TYPE_IM &&
00959                 !g_list_find(purple_get_conversations(), conv))
00960                 return;
00961 
00962         displayed = g_strdup(message);
00963 
00964         if (who == NULL || *who == '\0')
00965                 who = purple_conversation_get_name(conv);
00966         alias = who;
00967 
00968         plugin_return =
00969                 GPOINTER_TO_INT(purple_signal_emit_return_1(
00970                         purple_conversations_get_handle(),
00971                         (type == PURPLE_CONV_TYPE_IM ? "writing-im-msg" : "writing-chat-msg"),
00972                         account, who, &displayed, conv, flags));
00973 
00974         if (displayed == NULL)
00975                 return;
00976 
00977         if (plugin_return) {
00978                 g_free(displayed);
00979                 return;
00980         }
00981 
00982         if (account != NULL) {
00983                 prpl_info = PURPLE_PLUGIN_PROTOCOL_INFO(purple_find_prpl(purple_account_get_protocol_id(account)));
00984 
00985                 if (purple_conversation_get_type(conv) == PURPLE_CONV_TYPE_IM ||
00986                         !(prpl_info->options & OPT_PROTO_UNIQUE_CHATNAME)) {
00987 
00988                         if (flags & PURPLE_MESSAGE_SEND) {
00989                                 b = purple_find_buddy(account,
00990                                                         purple_account_get_username(account));
00991 
00992                                 if (purple_account_get_alias(account) != NULL)
00993                                         alias = account->alias;
00994                                 else if (b != NULL && !purple_strequal(purple_buddy_get_name(b), purple_buddy_get_contact_alias(b)))
00995                                         alias = purple_buddy_get_contact_alias(b);
00996                                 else if (purple_connection_get_display_name(gc) != NULL)
00997                                         alias = purple_connection_get_display_name(gc);
00998                                 else
00999                                         alias = purple_account_get_username(account);
01000                         }
01001                         else
01002                         {
01003                                 b = purple_find_buddy(account, who);
01004 
01005                                 if (b != NULL)
01006                                         alias = purple_buddy_get_contact_alias(b);
01007                         }
01008                 }
01009         }
01010

01012                 GList *log;
