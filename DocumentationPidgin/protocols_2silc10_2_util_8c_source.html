00325 void silcpurple_show_public_key(SilcPurple sg,
00326                               const char *name, SilcPublicKey public_key,
00327                               GCallback callback, void *context)
00328 {
00329         SilcPublicKeyIdentifier ident;
00330         SilcPKCS pkcs;
00331         char *fingerprint, *babbleprint;
00332         unsigned char *pk;
00333         SilcUInt32 pk_len, key_len = 0;
00334         GString *s;
00335         char *buf;
00336 
00337         ident = silc_pkcs_decode_identifier(public_key->identifier);
00338         if (!ident)
00339                 return;
00340 
00341         pk = silc_pkcs_public_key_encode(public_key, &pk_len);
00342         fingerprint = silc_hash_fingerprint(NULL, pk, pk_len);
00343         babbleprint = silc_hash_babbleprint(NULL, pk, pk_len);
00344 
00345         if (silc_pkcs_alloc((unsigned char *)public_key->name, &pkcs)) {
00346                 key_len = silc_pkcs_public_key_set(pkcs, public_key);
00347                 silc_pkcs_free(pkcs);
00348         }
00349 
00350         s = g_string_new("");
00351         if (ident->realname)
00352                 /* Hint for translators: Please check the tabulator width here and in
00353                    the next strings (short strings: 2 tabs, longer strings 1 tab,
00354                    sum: 3 tabs or 24 characters) */
00355                 g_string_append_printf(s, _("Real Name: \t%s\n"), ident->realname);
00356         if (ident->username)
00357                 g_string_append_printf(s, _("User Name: \t%s\n"), ident->username);
00358         if (ident->email)
00359                 g_string_append_printf(s, _("Email: \t\t%s\n"), ident->email);
00360         if (ident->host)
00361                 g_string_append_printf(s, _("Host Name: \t%s\n"), ident->host);
00362         if (ident->org)
00363                 g_string_append_printf(s, _("Organization: \t%s\n"), ident->org);
00364         if (ident->country)
00365                 g_string_append_printf(s, _("Country: \t%s\n"), ident->country);
00366         g_string_append_printf(s, _("Algorithm: \t%s\n"), public_key->name);
00367         g_string_append_printf(s, _("Key Length: \t%d bits\n"), (int)key_len);
00368         g_string_append_printf(s, "\n");
00369         g_string_append_printf(s, _("Public Key Fingerprint:\n%s\n\n"), fingerprint);
00370         g_string_append_printf(s, _("Public Key Babbleprint:\n%s"), babbleprint);
00371 
00372         buf = g_string_free(s, FALSE);
00373 
00374         purple_request_action(sg->gc, _("Public Key Information"),
00375                             _("Public Key Information"),
00376                             buf, 0, purple_connection_get_account(sg->gc),
00377                                 NULL, NULL, context, 1, _("Close"), callback);
00378 
00379         g_free(buf);
00380         silc_free(fingerprint);
00381         silc_free(babbleprint);
00382         silc_free(pk);
00383         silc_pkcs_free_identifier(ident);
00384 }

00642 /* Checks if message has images, and assembles MIME message if it has.
00643    If only one image is present, creates simple MIME image message.  If
00644    there are multiple images and/or text with images multipart MIME
00645    message is created. */

00647 SilcDList silcpurple_image_message(const char *msg, SilcUInt32 *mflags)
00648 {
00649         SilcMime mime = NULL, p;
00650         SilcDList list, parts = NULL;
00651         const char *start, *end, *last;
00652         GData *attribs;
00653         char *type;
00654         gboolean images = FALSE;
00655 
00656         last = msg;
00657         while (last && *last && purple_markup_find_tag("img", last, &start,
00658                                                      &end, &attribs)) {
00659                 PurpleStoredImage *image = NULL;
00660                 const char *id;
00661 
00662                 /* Check if there is text before image */
00663                 if (start - last) {
00664                         char *text, *tmp;
00665                         p = silc_mime_alloc();
00666 
00667                         /* Add content type */
00668                         silc_mime_add_field(p, "Content-Type",
00669                                             "text/plain; charset=utf-8");
00670 
00671                         tmp = g_strndup(last, start - last);
00672                         text = purple_unescape_html(tmp);
00673                         g_free(tmp);
00674                         /* Add text */
00675                         silc_mime_add_data(p, (unsigned char *)text, strlen(text));
00676                         g_free(text);
00677 
00678                         if (!parts)
00679                                 parts = silc_dlist_init();
00680                         silc_dlist_add(parts, p);
00681                 }
00682 
00683                 id = g_datalist_get_data(&attribs, "id");
00684                 if (id && (image = purple_imgstore_find_by_id(atoi(id)))) {
00685                         unsigned long imglen = purple_imgstore_get_size(image);
00686                         gconstpointer img = purple_imgstore_get_data(image);
00687 
00688                         p = silc_mime_alloc();
00689 
00690                         /* Add content type */
00691                         type = silcpurple_file2mime(purple_imgstore_get_filename(image));
00692                         if (!type) {
00693                                 g_datalist_clear(&attribs);
00694                                 last = end + 1;
00695                                 continue;
00696                         }
00697                         silc_mime_add_field(p, "Content-Type", type);
00698                         silc_free(type);
00699 
00700                         /* Add content transfer encoding */
00701                         silc_mime_add_field(p, "Content-Transfer-Encoding", "binary");
00702 
00703                         /* Add image data */
00704                         silc_mime_add_data(p, img, imglen);
00705 
00706                         if (!parts)
00707                                 parts = silc_dlist_init();
00708                         silc_dlist_add(parts, p);
00709                         images = TRUE;
00710                 }
00711 
00712                 g_datalist_clear(&attribs);
00713 
00714                 /* Continue after tag */
00715                 last = end + 1;
00716         }
00717 
00718         /* Check for text after the image(s) */
00719         if (images && last && *last) {
00720                 char *tmp = purple_unescape_html(last);
00721                 p = silc_mime_alloc();
00722 
00723                 /* Add content type */
00724                 silc_mime_add_field(p, "Content-Type",
00725                                     "text/plain; charset=utf-8");
00726 
00727                 /* Add text */
00728                 silc_mime_add_data(p, (unsigned char *)tmp, strlen(tmp));
00729                 g_free(tmp);
00730 
00731                 if (!parts)
00732                         parts = silc_dlist_init();
00733                 silc_dlist_add(parts, p);
00734         }
00735 
00736         /* If there weren't any images, don't return anything. */
00737         if (!images) {
00738                 if (parts)
00739                         silc_dlist_uninit(parts);
00740                 return NULL;
00741         }
00742 
00743         if (silc_dlist_count(parts) > 1) {
00744                 /* Multipart MIME message */
00745                 char b[32];
00746                 mime = silc_mime_alloc();
00747                 silc_mime_add_field(mime, "MIME-Version", "1.0");
00748                 g_snprintf(b, sizeof(b), "b%4X%4X",

00750                            silc_dlist_count(parts));

00752                 silc_dlist_start(parts);
00753                 while ((p = silc_dlist_get(parts)) != SILC_LIST_END)
00754                         silc_mime_add_multipart(mime, p);
00755         } else {
00756                 /* Simple MIME message */
00757                 silc_dlist_start(parts);
00758                 mime = silc_dlist_get(parts);
00759                 silc_mime_add_field(mime, "MIME-Version", "1.0");
00760         }
00761

00763         *mflags |= SILC_MESSAGE_FLAG_DATA;
00764

00767 
00768         silc_dlist_uninit(parts);
00769 
00770         /* Added multiparts gets freed here */
00771         silc_mime_free(mime);
