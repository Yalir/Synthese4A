00124 /* Returns -1 if a is a higher priority resource than b, or is
00125  * "more available" than b.  0 if they're the same, and 1 if b is
00126  * higher priority/more available than a.
00127  */
00128 static gint resource_compare_cb(gconstpointer a, gconstpointer b)
00129 {
00130         const JabberBuddyResource *jbra = a;
00131         const JabberBuddyResource *jbrb = b;
00132         JabberBuddyState state_a, state_b;
00133 
00134         if (jbra->priority != jbrb->priority)
00135                 return jbra->priority > jbrb->priority ? -1 : 1;
00136 
00137         /* Fold the states for easier comparison */
00138         /* TODO: Differentiate online/chat and away/dnd? */
00139         switch (jbra->state) {
00140                 case JABBER_BUDDY_STATE_ONLINE:
00141                 case JABBER_BUDDY_STATE_CHAT:
00142                         state_a = JABBER_BUDDY_STATE_ONLINE;
00143                         break;
00144                 case JABBER_BUDDY_STATE_AWAY:
00145                 case JABBER_BUDDY_STATE_DND:
00146                         state_a = JABBER_BUDDY_STATE_AWAY;
00147                         break;
00148                 case JABBER_BUDDY_STATE_XA:
00149                         state_a = JABBER_BUDDY_STATE_XA;
00150                         break;
00151                 case JABBER_BUDDY_STATE_UNAVAILABLE:
00152                         state_a = JABBER_BUDDY_STATE_UNAVAILABLE;
00153                         break;
00154                 default:
00155                         state_a = JABBER_BUDDY_STATE_UNKNOWN;
00156                         break;
00157         }
00158 
00159         switch (jbrb->state) {
00160                 case JABBER_BUDDY_STATE_ONLINE:
00161                 case JABBER_BUDDY_STATE_CHAT:
00162                         state_b = JABBER_BUDDY_STATE_ONLINE;
00163                         break;
00164                 case JABBER_BUDDY_STATE_AWAY:
00165                 case JABBER_BUDDY_STATE_DND:
00166                         state_b = JABBER_BUDDY_STATE_AWAY;
00167                         break;
00168                 case JABBER_BUDDY_STATE_XA:
00169                         state_b = JABBER_BUDDY_STATE_XA;
00170                         break;
00171                 case JABBER_BUDDY_STATE_UNAVAILABLE:
00172                         state_b = JABBER_BUDDY_STATE_UNAVAILABLE;
00173                         break;
00174                 default:
00175                         state_b = JABBER_BUDDY_STATE_UNKNOWN;
00176                         break;
00177         }
00178 
00179         if (state_a == state_b) {
00180                 if (jbra->idle == jbrb->idle)
00181                         return 0;
00182                 else if ((jbra->idle && !jbrb->idle) ||
00183                                 (jbra->idle && jbrb->idle && jbra->idle < jbrb->idle))
00184                         return 1;
00185                 else
00186                         return -1;
00187         }
00188 
00189         if (state_a == JABBER_BUDDY_STATE_ONLINE)
00190                 return -1;
00191         else if (state_a == JABBER_BUDDY_STATE_AWAY &&
00192                                 (state_b == JABBER_BUDDY_STATE_XA ||
00193                                  state_b == JABBER_BUDDY_STATE_UNAVAILABLE ||
00194                                  state_b == JABBER_BUDDY_STATE_UNKNOWN))
00195                 return -1;
00196         else if (state_a == JABBER_BUDDY_STATE_XA &&
00197                                 (state_b == JABBER_BUDDY_STATE_UNAVAILABLE ||
00198                                  state_b == JABBER_BUDDY_STATE_UNKNOWN))
00199                 return -1;
00200         else if (state_a == JABBER_BUDDY_STATE_UNAVAILABLE &&
00201                                 state_b == JABBER_BUDDY_STATE_UNKNOWN)
00202                 return -1;
00203 
00204         return 1;
00205 }

00389 /*
00390  * Insert a tag node into an xmlnode tree, recursively inserting parent tag
00391  * nodes as necessary
00392  *
00393  * Returns pointer to inserted node
00394  *
00395  * Note to hackers: this code is designed to be re-entrant (it's recursive--it
00396  * calls itself), so don't put any "static"s in here!
00397  */
00398 static xmlnode *insert_tag_to_parent_tag(xmlnode *start, const char *parent_tag, const char *new_tag)
00399 {
00400         xmlnode *x = NULL;
00401 
00402         /*
00403          * If the parent tag wasn't specified, see if we can get it
00404          * from the vCard template struct.
00405          */
00406         if(parent_tag == NULL) {
00407                 const struct vcard_template *vc_tp = vcard_template_data;
00408 
00409                 while(vc_tp->label != NULL) {
00410                         if(strcmp(vc_tp->tag, new_tag) == 0) {
00411                                 parent_tag = vc_tp->ptag;
00412                                 break;
00413                         }
00414                         ++vc_tp;
00415                 }
00416         }
00417 
00418         /*
00419          * If we have a parent tag...
00420          */
00421         if(parent_tag != NULL ) {
00422                 /*
00423                  * Try to get the parent node for a tag
00424                  */
00425                 if((x = xmlnode_get_child(start, parent_tag)) == NULL) {
00426                         /*
00427                          * Descend?
00428                          */
00429                         char *grand_parent = g_strdup(parent_tag);
00430                         char *parent;
00431 
00432                         if((parent = strrchr(grand_parent, '/')) != NULL) {
00433                                 *(parent++) = '\0';
00434                                 x = insert_tag_to_parent_tag(start, grand_parent, parent);
00435                         } else {
00436                                 x = xmlnode_new_child(start, grand_parent);
00437                         }
00438                         g_free(grand_parent);
00439                 } else {
00440                         /*
00441                          * We found *something* to be the parent node.
00442                          * Note: may be the "root" node!
00443                          */
00444                         xmlnode *y;
00445                         if((y = xmlnode_get_child(x, new_tag)) != NULL) {
00446                                 return(y);
00447                         }
00448                 }
00449         }
00450 
00451         /*
00452          * insert the new tag into its parent node
00453          */
00454         return(xmlnode_new_child((x == NULL? start : x), new_tag));
00455 }

00485

00487                         g_ascii_strncasecmp(vc_node->name, "vCard", 5))) {
00488                 xmlnode_free(vc_node);
00489                 vc_node = NULL;
00490         }
00491 
00492         if ((img = purple_buddy_icons_find_account_icon(gc->account))) {
00493                 gconstpointer avatar_data;

00497 
00498                 if(!vc_node) {

00500                         for(tag_attr = vcard_tag_attr_list; tag_attr->attr != NULL; ++tag_attr)
00501                                 xmlnode_set_attrib(vc_node, tag_attr->attr, tag_attr->value);
00502                 }

00505                 avatar_len = purple_imgstore_get_size(img);
00506                 /* Get rid of an old PHOTO if one exists.
00507                  * TODO: This may want to be modified to remove all old PHOTO

00509                  * multiple PHOTO entries in their vCard. */
00510                 if((photo = xmlnode_get_child(vc_node, "PHOTO"))) {
00511                         xmlnode_free(photo);
00512                 }
00513                 photo = xmlnode_new_child(vc_node, "PHOTO");

00524                 purple_imgstore_unref(img);
00525         } else if (vc_node) {
00526                 xmlnode *photo;
00527                 /* TODO: Remove all PHOTO children? (see above note) */
00528                 if ((photo = xmlnode_get_child(vc_node, "PHOTO"))) {
00529                         xmlnode_free(photo);
00530                 }
00531         }
00532

00534                 iq = jabber_iq_new(js, JABBER_IQ_SET);

00900 static void jabber_vcard_save_mine(JabberStream *js, const char *from,
00901                                    JabberIqType type, const char *id,
00902                                    xmlnode *packet, gpointer data)
00903 {
00904         xmlnode *vcard, *photo, *binval;
00905         char *txt, *vcard_hash = NULL;
00906         PurpleAccount *account;
00907 
00908         if (type == JABBER_IQ_ERROR) {
00909                 xmlnode *error;
00910                 purple_debug_warning("jabber", "Server returned error while retrieving vCard\n");
00911 
00912                 error = xmlnode_get_child(packet, "error");
00913                 if (!error || !xmlnode_get_child(error, "item-not-found"))
00914                         return;
00915         }
00916 
00917         account = purple_connection_get_account(js->gc);
00918 
00919         if((vcard = xmlnode_get_child(packet, "vCard")) ||
00920                         (vcard = xmlnode_get_child_with_namespace(packet, "query", "vcard-temp")))
00921         {
00922                 txt = xmlnode_to_str(vcard, NULL);
00923                 purple_account_set_user_info(account, txt);
00924                 g_free(txt);
00925         } else {
00926                 /* if we have no vCard, then lets not overwrite what we might have locally */
00927         }
00928 
00929         js->vcard_fetched = TRUE;
00930 
00931         if (vcard && (photo = xmlnode_get_child(vcard, "PHOTO")) &&
00932                      (binval = xmlnode_get_child(photo, "BINVAL"))) {
00933                 gsize size;
00934                 char *bintext = xmlnode_get_data(binval);
00935                 if (bintext) {
00936                         guchar *data = purple_base64_decode(bintext, &size);
00937                         g_free(bintext);
00938 
00939                         if (data) {
00940                                 vcard_hash = jabber_calculate_data_hash(data, size, "sha1");
00941                                 g_free(data);
00942                         }
00943                 }
00944         }
00945 
00946         /* Republish our vcard if the photo is different than the server's */
00947         if (js->initial_avatar_hash && !purple_strequal(vcard_hash, js->initial_avatar_hash)) {
00948                 /*
00949                  * Google Talk has developed the behavior that it will not accept
00950                  * a vcard set in the first 10 seconds (or so) of the connection;
00951                  * it returns an error (namespaces trimmed):
00952                  * <error code="500" type="wait"><internal-server-error/></error>.
00953                  */
00954                 if (js->googletalk)
00955                         js->vcard_timer = purple_timeout_add_seconds(10, set_own_vcard_cb,
00956                                                                      js);
00957                 else
00958                         jabber_set_info(js->gc, purple_account_get_user_info(account));
00959         } else if (vcard_hash) {
00960                 /* A photo is in the vCard. Advertise its hash */
00961                 js->avatar_hash = vcard_hash;
00962                 vcard_hash = NULL;
00963 
00964                 /* Send presence to update vcard-temp:x:update */
00965                 jabber_presence_send(js, FALSE);
00966         }
00967 
00968         g_free(vcard_hash);
00969 }
