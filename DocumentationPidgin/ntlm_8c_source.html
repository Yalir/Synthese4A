00175 setup_des_key(const guint8 key_56[], guint8 *key)
00176 {
00177         key[0] = key_56[0];
00178         key[1] = ((key_56[0] << 7) & 0xFF) | (key_56[1] >> 1);
00179         key[2] = ((key_56[1] << 6) & 0xFF) | (key_56[2] >> 2);
00180         key[3] = ((key_56[2] << 5) & 0xFF) | (key_56[3] >> 3);
00181         key[4] = ((key_56[3] << 4) & 0xFF) | (key_56[4] >> 4);
