00270 void jabber_parser_process(JabberStream *js, const char *buf, int len)
00271 {
00272         int ret;
00273 
00274         if (js->context == NULL) {
00275                 /* libxml inconsistently starts parsing on creating the
00276                  * parser, so do a ParseChunk right afterwards to force it. */
00277                 js->context = xmlCreatePushParserCtxt(&jabber_parser_libxml, js, buf, len, NULL);
00278                 xmlParseChunk(js->context, "", 0, 0);
00279         } else if ((ret = xmlParseChunk(js->context, buf, len, 0)) != XML_ERR_OK) {
00280                 xmlError *err = xmlCtxtGetLastError(js->context);
00281                 /*
00282                  * libxml2 uses a global setting to determine whether or not to store
00283                  * warnings.  Other libraries may set this, which causes err to be
00284                  * NULL. See #8136 for details.
00285                  */
00286                 xmlErrorLevel level = XML_ERR_WARNING;
00287 
00288                 if (err)
00289                         level = err->level;
00290 
00291                 switch (level) {
00292                         case XML_ERR_NONE:
00293                                 purple_debug_info("jabber", "xmlParseChunk returned info %i\n", ret);
00294                                 break;
00295                         case XML_ERR_WARNING:
00296                                 purple_debug_warning("jabber", "xmlParseChunk returned warning %i\n", ret);
00297                                 break;
00298                         case XML_ERR_ERROR:
00299                                 purple_debug_error("jabber", "xmlParseChunk returned error %i\n", ret);
00300                                 break;
00301                         case XML_ERR_FATAL:
00302                                 purple_debug_error("jabber", "xmlParseChunk returned fatal %i\n", ret);
00303                                 purple_connection_error_reason (js->gc,
00304                                                                 PURPLE_CONNECTION_ERROR_NETWORK_ERROR,
00305                                                                 _("XML Parse error"));
00306                                 break;
00307                 }
00308         }
00309 
00310         if (js->protocol_version.major == 0 && js->protocol_version.minor == 9 &&
00311                         !js->gc->disconnect_timeout &&
00312                         (js->state == JABBER_STREAM_INITIALIZING ||
00313                          js->state == JABBER_STREAM_INITIALIZING_ENCRYPTION)) {
00314                 /*
00315                  * Legacy servers don't advertise features, so if we've just gotten
00316                  * the opening <stream:stream> and there was no version, we need to
00317                  * immediately start legacy IQ auth.
00318                  */
00319                 jabber_stream_set_state(js, JABBER_STREAM_AUTHENTICATING);
00320                 jabber_auth_start_old(js);
00321         }
00322 }
