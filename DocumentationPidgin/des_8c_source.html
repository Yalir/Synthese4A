00179 /*
00180  *  Macro to swap bits across two words
00181  **/
00182 #define DO_PERMUTATION(a, temp, b, offset, mask)    \
00183         temp = ((a>>offset) ^ b) & mask;            \
00184         b ^= temp;                      \
00185         a ^= temp<<offset;

00188 /*
00189  *  This performs the 'initial permutation' for the data to be encrypted or decrypted
00190  **/
00191 #define INITIAL_PERMUTATION(left, temp, right)      \
00192         DO_PERMUTATION(left, temp, right, 4, 0x0f0f0f0f)    \
00193         DO_PERMUTATION(left, temp, right, 16, 0x0000ffff)   \
00194         DO_PERMUTATION(right, temp, left, 2, 0x33333333)    \
00195         DO_PERMUTATION(right, temp, left, 8, 0x00ff00ff)    \
00196         DO_PERMUTATION(left, temp, right, 1, 0x55555555)

00210 /*
00211  * A full DES round including 'expansion function', 'sbox substitution'
00212  * and 'primitive function P' but without swapping the left and right word.
00213  **/
00214 #define DES_ROUND(from, to, work, subkey)       \
00215         work = ((from<<1) | (from>>31)) ^ *subkey++;    \
00216         to ^= sbox8[  work      & 0x3f ];           \
00217         to ^= sbox6[ (work>>8)  & 0x3f ];           \
00218         to ^= sbox4[ (work>>16) & 0x3f ];           \
00219         to ^= sbox2[ (work>>24) & 0x3f ];           \
00220         work = ((from>>3) | (from<<29)) ^ *subkey++;    \
00221         to ^= sbox7[  work      & 0x3f ];           \
00222         to ^= sbox5[ (work>>8)  & 0x3f ];           \
00223         to ^= sbox3[ (work>>16) & 0x3f ];           \

00331 /*
00332  *  Fill a DES context with subkeys calculated from a 64bit key.
00333  *  Does not check parity bits, but simply ignore them.
00334  *  Does not check for weak keys.

00336 static void
00337 des_set_key (PurpleCipherContext *context, const guchar * key)
00338 {
00339         struct _des_ctx *ctx = purple_cipher_context_get_data(context);
00340         int i;
00341 
00342         des_key_schedule (key, ctx->encrypt_subkeys);
00343 
00344         for(i=0; i<32; i+=2)
00345         {
00346                 ctx->decrypt_subkeys[i] = ctx->encrypt_subkeys[30-i];
00347                 ctx->decrypt_subkeys[i+1] = ctx->encrypt_subkeys[31-i];
00348         }

00352 /*
00353  *  Electronic Codebook Mode DES encryption/decryption of data according
00354  *  to 'mode'.
00355  **/
00356 static int
00357 des_ecb_crypt (struct _des_ctx *ctx, const guint8 * from, guint8 * to, int mode)
00358 {
00359         guint32 left, right, work;
00360         guint32 *keys;
00361 
00362         keys = mode ? ctx->decrypt_subkeys : ctx->encrypt_subkeys;
00363 
00364         READ_64BIT_DATA (from, left, right)
00365         INITIAL_PERMUTATION (left, work, right)
00366 
00367         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00368         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00369         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00370         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00371         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00372         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00373         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00374         DES_ROUND (right, left, work, keys) DES_ROUND (left, right, work, keys)
00375 
00376         FINAL_PERMUTATION (right, work, left)
00377         WRITE_64BIT_DATA (to, right, left)
00378 
00379         return 0;
00380 }
