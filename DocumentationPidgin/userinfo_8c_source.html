00183         PurplePresence *presence = NULL;
00184         PurpleStatus *status = NULL;
00185         gchar *message = NULL, *itmsurl = NULL, *tmp;
00186         gboolean escaping_needed = TRUE;
00187 
00188         od = purple_connection_get_protocol_data(gc);
00189 
00190         if (b == NULL && userinfo == NULL)
00191                 return;
00192 
00193         if (b == NULL)
00194                 b = purple_find_buddy(purple_connection_get_account(gc), userinfo->bn);
00195         else
00196                 userinfo = aim_locate_finduserinfo(od, purple_buddy_get_name(b));
00197 
00198         if (b) {
00199                 presence = purple_buddy_get_presence(b);
00200                 status = purple_presence_get_active_status(presence);
00201         }
00202 
00203         /* If we have both b and userinfo we favor userinfo, because if we're
00204            viewing someone's profile then we want the HTML away message, and
00205            the "message" attribute of the status contains only the plaintext
00206            message. */
00207         if (userinfo) {
00208                 if ((userinfo->flags & AIM_FLAG_AWAY) && use_html_status && userinfo->away_len > 0 && userinfo->away != NULL && userinfo->away_encoding != NULL) {
00209                         /* Away message */
00210                         message = oscar_encoding_to_utf8(userinfo->away_encoding, userinfo->away, userinfo->away_len);
00211                         escaping_needed = FALSE;
00212                 } else {
00213                         /*
00214                          * Available message or non-HTML away message (because that's
00215                          * all we have right now.
00216                          */
00217                         if ((userinfo->status != NULL) && userinfo->status[0] != '\0') {
00218                                 message = oscar_encoding_to_utf8(userinfo->status_encoding, userinfo->status, userinfo->status_len);
00219                         }
00220 #if defined (_WIN32) || defined (__APPLE__)
00221                         if (userinfo->itmsurl && (userinfo->itmsurl[0] != '\0')) {
00222                                 itmsurl = oscar_encoding_to_utf8(userinfo->itmsurl_encoding, userinfo->itmsurl, userinfo->itmsurl_len);
00223                         }
00224 #endif
00225                 }
00226         } else {
00227                 message = g_strdup(purple_status_get_attr_string(status, "message"));
00228                 itmsurl = g_strdup(purple_status_get_attr_string(status, "itmsurl"));
00229         }
00230 
00231         if (message) {
00232                 tmp = oscar_util_format_string(message, purple_account_get_username(account));
00233                 g_free(message);
00234                 message = tmp;
00235                 if (escaping_needed) {
00236                         tmp = purple_markup_escape_text(message, -1);
00237                         g_free(message);
00238                         message = tmp;
00239                 }
00240         }
00241 
00242         if (use_html_status && itmsurl) {
00243                 tmp = g_strdup_printf("<a href=\"%s\">%s</a>", itmsurl, message);
00244                 g_free(message);
00245                 message = tmp;
00246         }
00247 
00248         if (b) {
00249                 if (purple_presence_is_online(presence)) {
00250                         gboolean is_away = ((status && !purple_status_is_available(status)) || (userinfo && (userinfo->flags & AIM_FLAG_AWAY)));
00251                         if (oscar_util_valid_name_icq(purple_buddy_get_name(b)) || is_away || !message || !(*message)) {
00252                                 /* Append the status name for online ICQ statuses, away AIM statuses, and for all buddies with no message.
00253                                  * If the status name and the message are the same, only show one. */
00254                                 const char *status_name = purple_status_get_name(status);
00255                                 if (status_name && message && !strcmp(status_name, message))
00256                                         status_name = NULL;
00257 
00258                                 tmp = g_strdup_printf("%s%s%s",
00259                                                                            status_name ? status_name : "",
00260                                                                            ((status_name && message) && *message) ? ": " : "",
00261                                                                            (message && *message) ? message : "");
00262                                 g_free(message);
00263                                 message = tmp;
00264                         }
00265 
00266                 } else if (aim_ssi_waitingforauth(od->ssi.local,
00267                         aim_ssi_itemlist_findparentname(od->ssi.local, purple_buddy_get_name(b)),
00268                         purple_buddy_get_name(b)))
00269                 {
00270                         /* Note if an offline buddy is not authorized */
00271                         tmp = g_strdup_printf("%s%s%s",
00272                                         _("Not Authorized"),
00273                                         (message && *message) ? ": " : "",
00274                                         (message && *message) ? message : "");
00275                         g_free(message);
00276                         message = tmp;
00277                 } else {
00278                         g_free(message);
00279                         message = g_strdup(_("Offline"));
00280                 }
00281         }
00282 
00283         if (presence) {
00284                 const char *mood;
00285                 const char *comment;
00286                 char *description;
00287                 status = purple_presence_get_status(presence, "mood");
00288                 mood = icq_get_custom_icon_description(purple_status_get_attr_string(status, PURPLE_MOOD_NAME));
00289                 if (mood) {
00290                         comment = purple_status_get_attr_string(status, PURPLE_MOOD_COMMENT);
00291                         if (comment) {
00292                                 char *escaped_comment = purple_markup_escape_text(comment, -1);
00293                                 description = g_strdup_printf("%s (%s)", _(mood), escaped_comment);
00294                                 g_free(escaped_comment);
00295                         } else {
00296                                 description = g_strdup(_(mood));
00297                         }
00298                         purple_notify_user_info_add_pair(user_info, _("Mood"), description);
00299                         g_free(description);
00300                 }
00301         }
00302 
00303         purple_notify_user_info_add_pair(user_info, _("Status"), message);
00304         g_free(message);
00305 }
