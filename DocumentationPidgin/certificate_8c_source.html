00802 /* Since the libpurple CertificatePools get registered before plugins are
00803    loaded, an X.509 Scheme is generally not available when x509_ca_init is
00804    called, but x509_ca requires X.509 operations in order to properly load.
00805 
00806    To solve this, I present the lazy_init function. It attempts to finish
00807    initialization of the Pool, but it usually fails when it is called from
00808    x509_ca_init. However, this is OK; initialization is then simply deferred
00809    until someone tries to use functions from the pool. */
00810 static gboolean
00811 x509_ca_lazy_init(void)
00812 {
00813         PurpleCertificateScheme *x509;
00814         GDir *certdir;
00815         const gchar *entry;
00816         GPatternSpec *pempat, *crtpat;
00817         GList *iter = NULL;
00818         GSList *crts = NULL;
00819 
00820         if (x509_ca_initialized) return TRUE;
00821 
00822         /* Check that X.509 is registered */
00823         x509 = purple_certificate_find_scheme(x509_ca.scheme_name);
00824         if ( !x509 ) {
00825                 purple_debug_warning("certificate/x509/ca",
00826                                   "Lazy init failed because an X.509 Scheme "
00827                                   "is not yet registered. Maybe it will be "
00828                                   "better later.\n");
00829                 return FALSE;
00830         }
00831 
00832         /* Use a glob to only read .pem files */
00833         pempat = g_pattern_spec_new("*.pem");
00834         crtpat = g_pattern_spec_new("*.crt");
00835 
00836         /* Populate the certificates pool from the search path(s) */
00837         for (iter = x509_ca_paths; iter; iter = iter->next) {
00838                 certdir = g_dir_open(iter->data, 0, NULL);
00839                 if (!certdir) {
00840                         purple_debug_error("certificate/x509/ca", "Couldn't open location '%s'\n", (const char *)iter->data);
00841                         continue;
00842                 }
00843 
00844                 while ( (entry = g_dir_read_name(certdir)) ) {
00845                         gchar *fullpath;
00846                         PurpleCertificate *crt;
00847 
00848                         if (!g_pattern_match_string(pempat, entry) && !g_pattern_match_string(crtpat, entry)) {
00849                                 continue;
00850                         }
00851 
00852                         fullpath = g_build_filename(iter->data, entry, NULL);
00853 
00854                         /* TODO: Respond to a failure in the following? */
00855                         crts = purple_certificates_import(x509, fullpath);
00856 
00857                         while (crts && crts->data) {
00858                                 crt = crts->data;
00859                                 if (x509_ca_quiet_put_cert(crt)) {
00860                                         gchar *name;
00861                                         name = purple_certificate_get_subject_name(crt);
00862                                         purple_debug_info("certificate/x509/ca",
00863                                                           "Loaded %s from %s\n",
00864                                                           name ? name : "(unknown)", fullpath);
00865                                         g_free(name);
00866                                 } else {
00867                                         purple_debug_error("certificate/x509/ca",
00868                                                           "Failed to load certificate from %s\n",
00869                                                           fullpath);
00870                                 }
00871                                 purple_certificate_destroy(crt);
00872                                 crts = g_slist_delete_link(crts, crts);
00873                         }
00874 
00875                         g_free(fullpath);
00876                 }
00877                 g_dir_close(certdir);
00878         }
00879 
00880         g_pattern_spec_free(pempat);
00881         g_pattern_spec_free(crtpat);
00882 
00883         purple_debug_info("certificate/x509/ca",
00884                           "Lazy init completed.\n");
00885         x509_ca_initialized = TRUE;
00886         return TRUE;
00887 }

01417 static void
01418 x509_tls_cached_complete(PurpleCertificateVerificationRequest *vrq,
01419                          PurpleCertificateInvalidityFlags flags)
01420 {
01421         PurpleCertificatePool *tls_peers;
01422         PurpleCertificate *peer_crt = vrq->cert_chain->data;
01423 
01424         if (flags & PURPLE_CERTIFICATE_FATALS_MASK) {
01425                 /* TODO: Also print any other warnings? */
01426                 const gchar *error;
01427                 gchar *tmp, *secondary;
01428 
01429                 if (flags & PURPLE_CERTIFICATE_INVALID_CHAIN)
01430                         error = invalidity_reason_to_string(PURPLE_CERTIFICATE_INVALID_CHAIN);
01431                 else if (flags & PURPLE_CERTIFICATE_REVOKED)
01432                         error = invalidity_reason_to_string(PURPLE_CERTIFICATE_REVOKED);
01433                 else
01434                         error = invalidity_reason_to_string(PURPLE_CERTIFICATE_UNKNOWN_ERROR);
01435 
01436                 tmp = g_strdup_printf(_("The certificate for %s could not be validated."),
01437                                         vrq->subject_name);
01438                 secondary = g_strconcat(tmp, " ", error, NULL);
01439                 g_free(tmp);
01440 
01441                 purple_notify_error(NULL, /* TODO: Probably wrong. */
01442                                         _("SSL Certificate Error"),
01443                                         _("Unable to validate certificate"),
01444                                         secondary);
01445                 g_free(secondary);
01446 
01447                 purple_certificate_verify_complete(vrq, PURPLE_CERTIFICATE_INVALID);
01448                 return;
01449         } else if (flags & PURPLE_CERTIFICATE_NON_FATALS_MASK) {
01450                 /* Non-fatal error. Prompt the user. */
01451                 gchar *tmp;
01452                 GString *errors;
01453                 guint32 i = 1;
01454 
01455                 tmp = g_strdup_printf(_("The certificate for %s could not be validated."),
01456                                         vrq->subject_name);
01457                 errors = g_string_new(tmp);
01458                 g_free(tmp);
01459 
01460                 errors = g_string_append_c(errors, '\n');
01461 
01462                 /* Special case a name mismatch because we want to display the two names... */
01463                 if (flags & PURPLE_CERTIFICATE_NAME_MISMATCH) {
01464                         gchar *sn = purple_certificate_get_subject_name(peer_crt);
01465 
01466                         if (sn) {
01467                                 g_string_append_printf(errors, _("The certificate claims to be "
01468                                                         "from \"%s\" instead. This could mean that you are "
01469                                                         "not connecting to the service you believe you are."),
01470                                                         sn);
01471                                 g_free(sn);
01472 
01473                                 flags &= ~PURPLE_CERTIFICATE_NAME_MISMATCH;
01474                         }
01475                 }
01476 
01477                 while (i != PURPLE_CERTIFICATE_LAST) {
01478                         if (flags & i) {
01479                                 errors = g_string_append_c(errors, '\n');
01480                                 g_string_append(errors, invalidity_reason_to_string(i));
01481                         }
01482 
01483                         i <<= 1;
01484                 }
01485 
01486                 x509_tls_cached_user_auth(vrq, errors->str);
01487                 g_string_free(errors, TRUE);
01488                 return;
01489         }
01490 
01491         /* If we reach this point, the certificate is good. */
01492 
01493         /* Look up the local cache and store it there for future use */
01494         tls_peers = purple_certificate_find_pool(x509_tls_cached.scheme_name,
01495                                                  "tls_peers");
01496         if (tls_peers) {
01497                 if (!purple_certificate_pool_store(tls_peers,vrq->subject_name,
01498                                                    peer_crt)) {
01499                         purple_debug_error("certificate/x509/tls_cached",
01500                                            "FAILED to cache peer certificate\n");
01501                 }
01502         } else {
01503                 purple_debug_error("certificate/x509/tls_cached",
01504                                    "Unable to locate tls_peers certificate cache.\n");
01505         }
01506 
01507         purple_certificate_verify_complete(vrq, PURPLE_CERTIFICATE_VALID);
01508 }

01510 static void
01511 x509_tls_cached_cert_in_cache(PurpleCertificateVerificationRequest *vrq,
01512                               PurpleCertificateInvalidityFlags flags)
01513 {
01514         /* TODO: Looking this up by name over and over is expensive.
01515            Fix, please! */
01516         PurpleCertificatePool *tls_peers =
01517                 purple_certificate_find_pool(x509_tls_cached.scheme_name,
01518                                              "tls_peers");
01519 
01520         /* The peer's certificate should be the first in the list */
01521         PurpleCertificate *peer_crt =
01522                 (PurpleCertificate *) vrq->cert_chain->data;
01523 
01524         PurpleCertificate *cached_crt;
01525         GByteArray *peer_fpr, *cached_fpr;
01526 
01527         /* Load up the cached certificate */
01528         cached_crt = purple_certificate_pool_retrieve(
01529                 tls_peers, vrq->subject_name);
01530         if ( !cached_crt ) {
01531                 purple_debug_warning("certificate/x509/tls_cached",
01532                                    "Lookup failed on cached certificate!\n"
01533                                    "Falling back to full verification.\n");
01534                 /* vrq now becomes the problem of unknown_peer */
01535                 x509_tls_cached_unknown_peer(vrq, flags);
01536                 return;
01537         }
01538 
01539         /* Now get SHA1 sums for both and compare them */
01540         /* TODO: This is not an elegant way to compare certs */
01541         peer_fpr = purple_certificate_get_fingerprint_sha1(peer_crt);
01542         cached_fpr = purple_certificate_get_fingerprint_sha1(cached_crt);
01543         if (!memcmp(peer_fpr->data, cached_fpr->data, peer_fpr->len)) {
01544                 purple_debug_info("certificate/x509/tls_cached",
01545                                   "Peer cert matched cached\n");
01546                 x509_tls_cached_complete(vrq, flags);
01547         } else {
01548                 purple_debug_error("certificate/x509/tls_cached",
01549                                   "Peer cert did NOT match cached\n");
01550                 /* vrq now becomes the problem of the user */
01551                 x509_tls_cached_unknown_peer(vrq, flags);
01552         }
01553 
01554         purple_certificate_destroy(cached_crt);
01555         g_byte_array_free(peer_fpr, TRUE);
01556         g_byte_array_free(cached_fpr, TRUE);
01557 }

01589 /* For when we've never communicated with this party before */
01590 /* TODO: Need ways to specify possibly multiple problems with a cert, or at
01591    least  reprioritize them.
01592  */
01593 static void
01594 x509_tls_cached_unknown_peer(PurpleCertificateVerificationRequest *vrq,
01595                              PurpleCertificateInvalidityFlags flags)
01596 {
01597         PurpleCertificatePool *ca;
01598         PurpleCertificate *peer_crt;
01599         PurpleCertificate *ca_crt, *end_crt;
01600         PurpleCertificate *failing_crt;
01601         GList *chain = vrq->cert_chain;
01602         GSList *ca_crts, *cur;
01603         GByteArray *last_fpr, *ca_fpr;
01604         gboolean valid = FALSE;
01605         gchar *ca_id, *ca2_id;
01606 
01607         peer_crt = (PurpleCertificate *) chain->data;
01608 
01609         /* TODO: Figure out a way to check for a bad signature, as opposed to
01610            "not self-signed" */
01611         if ( purple_certificate_signed_by(peer_crt, peer_crt) ) {
01612                 flags |= PURPLE_CERTIFICATE_SELF_SIGNED;
01613 
01614                 purple_debug_info("certificate/x509/tls_cached",
01615                                   "Certificate for %s is self-signed.\n",
01616                                   vrq->subject_name);
01617 
01618                 x509_tls_cached_check_subject_name(vrq, flags);
01619                 return;
01620         } /* if (self signed) */
01621 
01622         ca = purple_certificate_find_pool(x509_tls_cached.scheme_name, "ca");
01623 
01624         /* Next, check that the certificate chain is valid */
01625         if (!purple_certificate_check_signature_chain_with_failing(chain,
01626                                 &failing_crt))
01627         {
01628                 gboolean chain_validated = FALSE;
01629                 /*
01630                  * Check if the failing certificate is in the CA store. If it is, then
01631                  * consider this fully validated. This works around issues with some
01632                  * prominent intermediate CAs whose signature is md5WithRSAEncryption.
01633                  * I'm looking at CACert Class 3 here. See #4458 for details.
01634                  */
01635                 if (ca) {
01636                         gchar *uid = purple_certificate_get_unique_id(failing_crt);
01637                         PurpleCertificate *ca_crt = purple_certificate_pool_retrieve(ca, uid);
01638                         if (ca_crt != NULL) {
01639                                 GByteArray *failing_fpr;
01640                                 GByteArray *ca_fpr;
01641                                 failing_fpr = purple_certificate_get_fingerprint_sha1(failing_crt);
01642                                 ca_fpr = purple_certificate_get_fingerprint_sha1(ca_crt);
01643                                 if (byte_arrays_equal(failing_fpr, ca_fpr)) {
01644                                         purple_debug_info("certificate/x509/tls_cached",
01645                                                         "Full chain verification failed (probably a bad "
01646                                                         "signature algorithm), but found the last "
01647                                                         "certificate %s in the CA pool.\n", uid);
01648                                         chain_validated = TRUE;
01649                                 }
01650 
01651                                 g_byte_array_free(failing_fpr, TRUE);
01652                                 g_byte_array_free(ca_fpr, TRUE);
01653                         }
01654 
01655                         purple_certificate_destroy(ca_crt);
01656                         g_free(uid);
01657                 }
01658 
01659                 /*
01660                  * If we get here, either the cert matched the stuff right above
01661                  * or it didn't, in which case we give up and complain to the user.
01662                  */
01663                 if (!chain_validated)
01664                         /* TODO: Tell the user where the chain broke? */
01665                         flags |= PURPLE_CERTIFICATE_INVALID_CHAIN;
01666 
01667                 x509_tls_cached_check_subject_name(vrq, flags);
01668                 return;
01669         } /* if (signature chain not good) */
01670 
01671         /* Next, attempt to verify the last certificate is signed by a trusted
01672          * CA, or is a trusted CA (based on fingerprint).
01673          */
01674         /* If, for whatever reason, there is no Certificate Authority pool
01675            loaded, we'll verify the subject name and then warn about thsi. */
01676         if ( !ca ) {
01677                 purple_debug_error("certificate/x509/tls_cached",
01678                                    "No X.509 Certificate Authority pool "
01679                                    "could be found!\n");
01680 
01681                 flags |= PURPLE_CERTIFICATE_NO_CA_POOL;
01682 
01683                 x509_tls_cached_check_subject_name(vrq, flags);
01684                 return;
01685         }
01686 
01687         end_crt = g_list_last(chain)->data;
01688 
01689         /* Attempt to look up the last certificate, and the last certificate's
01690          * issuer. 
01691          */
01692         ca_id  = purple_certificate_get_issuer_unique_id(end_crt);
01693         ca2_id = purple_certificate_get_unique_id(end_crt);
01694         purple_debug_info("certificate/x509/tls_cached",
01695                           "Checking for a CA with DN=%s\n",
01696                           ca_id);
01697         purple_debug_info("certificate/x509/tls_cached",
01698                           "Also checking for a CA with DN=%s\n",
01699                           ca2_id);
01700         ca_crts = g_slist_concat(x509_ca_get_certs(ca_id), x509_ca_get_certs(ca2_id));
01701         g_free(ca_id);
01702         g_free(ca2_id);
01703         if ( NULL == ca_crts ) {
01704                 flags |= PURPLE_CERTIFICATE_CA_UNKNOWN;
01705 
01706                 purple_debug_warning("certificate/x509/tls_cached",
01707                                   "No Certificate Authorities with either DN found "
01708                                   "found. I'll prompt the user, I guess.\n");
01709 
01710                 x509_tls_cached_check_subject_name(vrq, flags);
01711                 return;
01712         }
01713 
01714         /*
01715          * Check the fingerprints; if they match, then this certificate *is* one
01716          * of the designated "trusted roots", and we don't need to verify the
01717          * signature. This is good because some of the older roots are self-signed
01718          * with bad hash algorithms that we don't want to allow in any other
01719          * circumstances (one of Verisign's root CAs is self-signed with MD2).
01720          *
01721          * If the fingerprints don't match, we'll fall back to checking the
01722          * signature.
01723          */
01724         last_fpr = purple_certificate_get_fingerprint_sha1(end_crt);
01725         for (cur = ca_crts; cur; cur = cur->next) {
01726                 ca_crt = cur->data;
01727                 ca_fpr = purple_certificate_get_fingerprint_sha1(ca_crt);
01728 
01729                 if ( byte_arrays_equal(last_fpr, ca_fpr) ||
01730                                 purple_certificate_signed_by(end_crt, ca_crt) )
01731                 {
01732                         /* TODO: If signed_by ever returns a reason, maybe mention
01733                            that, too. */
01734                         /* TODO: Also mention the CA involved. While I could do this
01735                            now, a full DN is a little much with which to assault the
01736                            user's poor, leaky eyes. */
01737                         valid = TRUE;
01738                         g_byte_array_free(ca_fpr, TRUE);
01739                         break;
01740                 }
01741 
01742                 g_byte_array_free(ca_fpr, TRUE);
01743         }
01744 
01745         if (valid == FALSE)
01746                 flags |= PURPLE_CERTIFICATE_INVALID_CHAIN;
01747 
01748         g_slist_foreach(ca_crts, (GFunc)purple_certificate_destroy, NULL);
01749         g_slist_free(ca_crts);
01750         g_byte_array_free(last_fpr, TRUE);
01751 
01752         x509_tls_cached_check_subject_name(vrq, flags);
01753 }

01755 static void
01756 x509_tls_cached_start_verify(PurpleCertificateVerificationRequest *vrq)
01757 {
01758         const gchar *tls_peers_name = "tls_peers"; /* Name of local cache */
01759         PurpleCertificatePool *tls_peers;
01760         time_t now, activation, expiration;
01761         PurpleCertificateInvalidityFlags flags = PURPLE_CERTIFICATE_NO_PROBLEMS;
01762         gboolean ret;
01763 
01764         g_return_if_fail(vrq);
01765 
01766         purple_debug_info("certificate/x509/tls_cached",
01767                           "Starting verify for %s\n",
01768                           vrq->subject_name);
01769 
01770         /*
01771          * Verify the first certificate (the main one) has been activated and
01772          * isn't expired, i.e. activation < now < expiration.
01773          */
01774         now = time(NULL);
01775         ret = purple_certificate_get_times(vrq->cert_chain->data, &activation,
01776                                            &expiration);
01777         if (!ret) {
01778                 flags |= PURPLE_CERTIFICATE_EXPIRED | PURPLE_CERTIFICATE_NOT_ACTIVATED;
01779                 purple_debug_error("certificate/x509/tls_cached",
01780                                 "Failed to get validity times for certificate %s\n",
01781                                 vrq->subject_name);
01782         } else if (now > expiration) {
01783                 flags |= PURPLE_CERTIFICATE_EXPIRED;
01784                 purple_debug_error("certificate/x509/tls_cached",
01785                                 "Certificate %s expired at %s\n",
01786                                 vrq->subject_name, ctime(&expiration));
01787         } else if (now < activation) {
01788                 flags |= PURPLE_CERTIFICATE_NOT_ACTIVATED;
01789                 purple_debug_error("certificate/x509/tls_cached",
01790                                 "Certificate %s is not yet valid, will be at %s\n",
01791                                 vrq->subject_name, ctime(&activation));
01792         }
01793 
01794         tls_peers = purple_certificate_find_pool(x509_tls_cached.scheme_name,tls_peers_name);
01795 
01796         if (!tls_peers) {
01797                 purple_debug_error("certificate/x509/tls_cached",
01798                                    "Couldn't find local peers cache %s\n",
01799                                    tls_peers_name);
01800 
01801                 /* vrq now becomes the problem of unknown_peer */
01802                 x509_tls_cached_unknown_peer(vrq, flags);
01803                 return;
01804         }
01805 
01806         /* Check if the peer has a certificate cached already */
01807         purple_debug_info("certificate/x509/tls_cached",
01808                           "Checking for cached cert...\n");
01809         if (purple_certificate_pool_contains(tls_peers, vrq->subject_name)) {
01810                 purple_debug_info("certificate/x509/tls_cached",
01811                                   "...Found cached cert\n");
01812                 /* vrq is now the responsibility of cert_in_cache */
01813                 x509_tls_cached_cert_in_cache(vrq, flags);
01814         } else {
01815                 purple_debug_warning("certificate/x509/tls_cached",
01816                                   "...Not in cache\n");
01817                 /* vrq now becomes the problem of unknown_peer */
01818                 x509_tls_cached_unknown_peer(vrq, flags);
01819         }
01820 }
