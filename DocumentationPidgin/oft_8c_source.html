00021 /*
00022  * I feel like this is a good place to explain OFT, so I'm going to
00023  * do just that.  Each OFT packet has a header type.  I guess this
00024  * is pretty similar to the subtype of a SNAC packet.  The type
00025  * basically tells the other client the meaning of the OFT packet.
00026  * There are two distinct types of file transfer, which I usually
00027  * call "sendfile" and "getfile."  Sendfile is when you send a file
00028  * to another AIM user.  Getfile is when you share a group of files,
00029  * and other users request that you send them the files.
00030  *
00031  * A typical sendfile file transfer goes like this:
00032  *   1) Sender sends a channel 2 ICBM telling the other user that
00033  *      we want to send them a file.  At the same time, we open a
00034  *      listener socket (this should be done before sending the
00035  *      ICBM) on some port, and wait for them to connect to us.
00036  *      The ICBM we sent should contain our IP address and the port
00037  *      number that we're listening on.
00038  *   2) The receiver connects to the sender on the given IP address
00039  *      and port.  After the connection is established, the receiver
00040  *      sends an ICBM signifying that we are ready and waiting.
00041  *   3) The sender sends an OFT PROMPT message over the OFT
00042  *      connection.
00043  *   4) The receiver of the file sends back an exact copy of this
00044  *      OFT packet, except the cookie is filled in with the cookie
00045  *      from the ICBM.  I think this might be an attempt to verify
00046  *      that the user that is connected is actually the guy that
00047  *      we sent the ICBM to.  Oh, I've been calling this the ACK.
00048  *   5) The sender starts sending raw data across the connection
00049  *      until the entire file has been sent.
00050  *   6) The receiver knows the file is finished because the sender
00051  *      sent the file size in an earlier OFT packet.  So then the
00052  *      receiver sends the DONE thingy (after filling in the
00053  *      "received" checksum and size) and closes the connection.
00054  */

00108 static guint32
00109 peer_oft_checksum_chunk(const guint8 *buffer, int bufferlen, guint32 prevchecksum, int odd)
00110 {
00111         guint32 checksum, oldchecksum;
00112         int i = 0;
00113         unsigned short val;
00114 
00115         checksum = (prevchecksum >> 16) & 0xffff;
00116         if (odd)
00117         {
00118                 /*
00119                  * This is one hell of a hack, but it should always work.
00120                  * Essentially, I am reindexing the array so that index 1
00121                  * is the first element.  Since the odd and even bytes are
00122                  * detected by the index number.
00123                  */
00124                 i = 1;
00125                 bufferlen++;
00126                 buffer--;
00127         }
00128         for (; i < bufferlen; i++)
00129         {
00130                 oldchecksum = checksum;
00131                 if (i & 1)
00132                         val = buffer[i];
00133                 else
00134                         val = buffer[i] << 8;
00135                 checksum -= val;
00136                 /*
00137                  * The following appears to be necessary.... It happens
00138                  * every once in a while and the checksum doesn't fail.
00139                  */
00140                 if (checksum > oldchecksum)
00141                         checksum--;
00142         }
00143         checksum = ((checksum & 0x0000ffff) + (checksum >> 16));
00144         checksum = ((checksum & 0x0000ffff) + (checksum >> 16));
00145         return checksum << 16;
00146 }

00148 static gboolean
00149 peer_oft_checksum_file_piece(gpointer data)
00150 {
00151         ChecksumData *checksum_data;
00152         gboolean repeat;
00153 
00154         checksum_data = data;
00155         repeat = FALSE;
00156 
00157         if (checksum_data->total < checksum_data->size)
00158         {
00159                 size_t bytes = MIN(CHECKSUM_BUFFER_SIZE,
00160                                 checksum_data->size - checksum_data->total);
00161 
00162                 bytes = fread(checksum_data->buffer, 1, bytes, checksum_data->file);
00163                 if (bytes != 0)
00164                 {
00165                         checksum_data->checksum = peer_oft_checksum_chunk(checksum_data->buffer, bytes, checksum_data->checksum, checksum_data->total & 1);
00166                         checksum_data->total += bytes;
00167                         repeat = TRUE;
00168                 }
00169         }
00170 
00171         if (!repeat)
00172         {
00173                 purple_debug_info("oscar", "Checksum of %s calculated\n",
00174                                 purple_xfer_get_local_filename(checksum_data->xfer));
00175                 if (checksum_data->callback != NULL)
00176                         checksum_data->callback(checksum_data);
00177                 peer_oft_checksum_destroy(checksum_data);
00178         }
00179 
00180         return repeat;
00181 }

00459                 purple_xfer_set_bytes_sent(checksum_data->xfer, conn->xferdata.nrecvd);

00498 {

00514         peer_connection_schedule_destroy(conn, conn->disconnect_reason, NULL);

00772  *   connection.
00773  *  -We get drunk because file transfer kicks ass.

00778  *   oscar_xfer_init is called
