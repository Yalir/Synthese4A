00088 purple_pref *find_pref(const char *name)
00089 {
00090         g_return_val_if_fail(name != NULL && name[0] == '/', NULL);
00091 
00092         if (name[1] == '\0')
00093                 return &prefs;
00094         else
00095         {
00096                 /* When we're initializing, the debug system is
00097                  * initialized before the prefs system, but debug
00098                  * calls will end up calling prefs functions, so we
00099                  * need to deal cleanly here. */
00100                 if (prefs_hash)
00101                         return g_hash_table_lookup(prefs_hash, name);
00102                 else
00103                         return NULL;

00113  * This function recursively creates the xmlnode tree from the prefs
00114  * tree structure.  Yay recursion!

00116 static void
00117 pref_to_xmlnode(xmlnode *parent, struct purple_pref *pref)
00118 {
00119         xmlnode *node, *childnode;
00120         struct purple_pref *child;
00121         char buf[21];
00122         GList *cur;
00123 
00124         /* Create a new node */
00125         node = xmlnode_new_child(parent, "pref");
00126         xmlnode_set_attrib(node, "name", pref->name);
00127 
00128         /* Set the type of this node (if type == PURPLE_PREF_NONE then do nothing) */
00129         if (pref->type == PURPLE_PREF_INT) {
00130                 xmlnode_set_attrib(node, "type", "int");
00131                 g_snprintf(buf, sizeof(buf), "%d", pref->value.integer);
00132                 xmlnode_set_attrib(node, "value", buf);
00133         }
00134         else if (pref->type == PURPLE_PREF_STRING) {
00135                 xmlnode_set_attrib(node, "type", "string");
00136                 xmlnode_set_attrib(node, "value", pref->value.string ? pref->value.string : "");
00137         }
00138         else if (pref->type == PURPLE_PREF_STRING_LIST) {
00139                 xmlnode_set_attrib(node, "type", "stringlist");
00140                 for (cur = pref->value.stringlist; cur != NULL; cur = cur->next)
00141                 {
00142                         childnode = xmlnode_new_child(node, "item");
00143                         xmlnode_set_attrib(childnode, "value", cur->data ? cur->data : "");
00144                 }
00145         }
00146         else if (pref->type == PURPLE_PREF_PATH) {
00147                 char *encoded = g_filename_to_utf8(pref->value.string ? pref->value.string : "", -1, NULL, NULL, NULL);
00148                 xmlnode_set_attrib(node, "type", "path");
00149                 xmlnode_set_attrib(node, "value", encoded);
00150                 g_free(encoded);
00151         }
00152         else if (pref->type == PURPLE_PREF_PATH_LIST) {
00153                 xmlnode_set_attrib(node, "type", "pathlist");
00154                 for (cur = pref->value.stringlist; cur != NULL; cur = cur->next)
00155                 {
00156                         char *encoded = g_filename_to_utf8(cur->data ? cur->data : "", -1, NULL, NULL, NULL);
00157                         childnode = xmlnode_new_child(node, "item");
00158                         xmlnode_set_attrib(childnode, "value", encoded);
00159                         g_free(encoded);
00160                 }
00161         }
00162         else if (pref->type == PURPLE_PREF_BOOLEAN) {
00163                 xmlnode_set_attrib(node, "type", "bool");
00164                 g_snprintf(buf, sizeof(buf), "%d", pref->value.boolean);
00165                 xmlnode_set_attrib(node, "value", buf);
00166         }
00167 
00168         /* All My Children */
00169         for (child = pref->first_child; child != NULL; child = child->sibling)
00170                 pref_to_xmlnode(node, child);
00171 }

00281                         pref_value == NULL) {

00292                         pref_name_full = g_string_prepend(pref_name_full, tmp->data);
00293                         pref_name_full = g_string_prepend_c(pref_name_full, '/');
00294                 }

00300                                 pref->value.stringlist = g_list_append(pref->value.stringlist,

00308         } else {
00309                 char *decoded;
00310 
00311                 if(!pref_name || purple_strequal(pref_name, "/"))
00312                         return;
00313 
00314                 pref_name_full = g_string_new(pref_name);
00315 
00316                 for(tmp = prefs_stack; tmp; tmp = tmp->next) {
00317                         pref_name_full = g_string_prepend_c(pref_name_full, '/');

00338                                 break;

00340                                 if (pref_value) {
00341                                         decoded = g_filename_from_utf8(pref_value, -1, NULL, NULL, NULL);
00342                                         purple_prefs_set_path(pref_name_full->str, decoded);

00345                                         purple_prefs_set_path(pref_name_full->str, NULL);

00573 
00574         if(!parent)
00575                 return NULL;

00577         my_name = get_path_basename(name);
00578 
00579         for(sibling = parent->first_child; sibling; sibling = sibling->sibling) {
00580                 if(purple_strequal(sibling->name, my_name)) {
00581                         g_free(my_name);
00582                         return NULL;
00583                 }
00584         }
00585 
00586         me = g_new0(struct purple_pref, 1);
00587         me->type = type;
00588         me->name = my_name;
00589 
00590         me->parent = parent;
00591         if(parent->first_child) {
00592                 /* blatant abuse of a for loop */
00593                 for(sibling = parent->first_child; sibling->sibling;
00594                                 sibling = sibling->sibling);
00595                 sibling->sibling = me;
00596         } else {
00597                 parent->first_child = me;
00598         }
00599 
00600         g_hash_table_insert(prefs_hash, g_strdup(name), (gpointer)me);

01096 static void
01097 purple_prefs_rename_node(struct purple_pref *oldpref, struct purple_pref *newpref)
01098 {
01099         struct purple_pref *child, *next;
01100         char *oldname, *newname;
01101 
01102         /* if we're a parent, rename the kids first */
01103         for(child = oldpref->first_child; child != NULL; child = next)
01104         {
01105                 struct purple_pref *newchild;
01106                 next = child->sibling;
01107                 for(newchild = newpref->first_child; newchild != NULL; newchild = newchild->sibling)
01108                 {
01109                         if(purple_strequal(child->name, newchild->name))
01110                         {
01111                                 purple_prefs_rename_node(child, newchild);
01112                                 break;
01113                         }
01114                 }
01115                 if(newchild == NULL) {
01116                         /* no rename happened, we weren't able to find the new pref */
01117                         char *tmpname = pref_full_name(child);
01118                         purple_debug_error("prefs", "Unable to find rename pref for %s\n", tmpname);
01119                         g_free(tmpname);
01120                 }
01121         }
01122 
01123         oldname = pref_full_name(oldpref);
01124         newname = pref_full_name(newpref);
01125 
01126         if (oldpref->type != newpref->type)
01127         {
01128                 purple_debug_error("prefs", "Unable to rename %s to %s: differing types\n", oldname, newname);
01129                 g_free(oldname);
01130                 g_free(newname);
01131                 return;
01132         }
01133 
01134         purple_debug_info("prefs", "Renaming %s to %s\n", oldname, newname);
01135         g_free(oldname);
01136 
01137         switch(oldpref->type) {
01138                 case PURPLE_PREF_NONE:
01139                         break;
01140                 case PURPLE_PREF_BOOLEAN:
01141                         purple_prefs_set_bool(newname, oldpref->value.boolean);
01142                         break;
01143                 case PURPLE_PREF_INT:
01144                         purple_prefs_set_int(newname, oldpref->value.integer);
01145                         break;
01146                 case PURPLE_PREF_STRING:
01147                         purple_prefs_set_string(newname, oldpref->value.string);
01148                         break;
01149                 case PURPLE_PREF_STRING_LIST:
01150                         purple_prefs_set_string_list(newname, oldpref->value.stringlist);
01151                         break;
01152                 case PURPLE_PREF_PATH:
01153                         purple_prefs_set_path(newname, oldpref->value.string);
01154                         break;
01155                 case PURPLE_PREF_PATH_LIST:
01156                         purple_prefs_set_path_list(newname, oldpref->value.stringlist);
01157                         break;
01158         }
01159         g_free(newname);
01160 
01161         remove_pref(oldpref);
01162 }
