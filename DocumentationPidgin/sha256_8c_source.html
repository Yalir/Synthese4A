00058 static void
00059 sha256_hash_block(struct SHA256Context *sha256_ctx) {
00060         gint i;
00061         guint32 A, B, C, D, E, F, G, H, T1, T2;
00062 
00063         for(i = 16; i < 64; i++) {
00064                 sha256_ctx->W[i] =
00065                           (SHA256_ROTR(sha256_ctx->W[i-2], 17) ^ SHA256_ROTR(sha256_ctx->W[i-2],  19) ^ (sha256_ctx->W[i-2] >> 10))
00066                         + sha256_ctx->W[i-7]
00067                         + (SHA256_ROTR(sha256_ctx->W[i-15], 7) ^ SHA256_ROTR(sha256_ctx->W[i-15], 18) ^ (sha256_ctx->W[i-15] >> 3))
00068                         + sha256_ctx->W[i-16];
00069         }
00070 
00071         A = sha256_ctx->H[0];
00072         B = sha256_ctx->H[1];
00073         C = sha256_ctx->H[2];
00074         D = sha256_ctx->H[3];
00075         E = sha256_ctx->H[4];
00076         F = sha256_ctx->H[5];
00077         G = sha256_ctx->H[6];
00078         H = sha256_ctx->H[7];
00079 
00080         for(i = 0; i < 64; i++) {
00081                 T1 = H
00082                         + (SHA256_ROTR(E, 6) ^ SHA256_ROTR(E, 11) ^ SHA256_ROTR(E, 25))
00083                         + ((E & F) ^ ((~E) & G))
00084                         + sha256_K[i] + sha256_ctx->W[i];
00085                 T2 = (SHA256_ROTR(A, 2) ^ SHA256_ROTR(A, 13) ^ SHA256_ROTR(A, 22))
00086                         + ((A & B) ^ (A & C) ^ (B & C));
00087                 H = G;
00088                 G = F;
00089                 F = E;
00090                 E = D + T1;
00091                 D = C;
00092                 C = B;
00093                 B = A;
00094                 A = T1 + T2;
00095         }
00096 
00097         sha256_ctx->H[0] += A;
00098         sha256_ctx->H[1] += B;
00099         sha256_ctx->H[2] += C;
00100         sha256_ctx->H[3] += D;
00101         sha256_ctx->H[4] += E;
00102         sha256_ctx->H[5] += F;
00103         sha256_ctx->H[6] += G;
00104         sha256_ctx->H[7] += H;
00105 }

00214 sha256_digest(PurpleCipherContext *context, size_t in_len, guchar digest[32],
00215               size_t *out_len)
00216 {
00217         struct SHA256Context *sha256_ctx;
00218         guchar pad0x80 = 0x80, pad0x00 = 0x00;
00219         guchar padlen[8];

00221 
00222         g_return_val_if_fail(in_len >= 32, FALSE);
00223 
00224         sha256_ctx = purple_cipher_context_get_data(context);
00225 
00226         g_return_val_if_fail(sha256_ctx, FALSE);
00227 
00228         padlen[0] = (guchar)((sha256_ctx->sizeHi >> 24) & 255);
00229         padlen[1] = (guchar)((sha256_ctx->sizeHi >> 16) & 255);
00230         padlen[2] = (guchar)((sha256_ctx->sizeHi >> 8) & 255);
00231         padlen[3] = (guchar)((sha256_ctx->sizeHi >> 0) & 255);
00232         padlen[4] = (guchar)((sha256_ctx->sizeLo >> 24) & 255);
00233         padlen[5] = (guchar)((sha256_ctx->sizeLo >> 16) & 255);
00234         padlen[6] = (guchar)((sha256_ctx->sizeLo >> 8) & 255);
00235         padlen[7] = (guchar)((sha256_ctx->sizeLo >> 0) & 255);
00236 
00237         /* pad with a 1, then zeroes, then length */
00238         purple_cipher_context_append(context, &pad0x80, 1);
00239         while(sha256_ctx->lenW != 56)
00240                 purple_cipher_context_append(context, &pad0x00, 1);
00241         purple_cipher_context_append(context, padlen, 8);
00242 
00243         for(i = 0; i < 32; i++) {
00244                 digest[i] = (guchar)(sha256_ctx->H[i / 4] >> 24);
00245                 sha256_ctx->H[i / 4] <<= 8;
00246         }
00247 
00248         purple_cipher_context_reset(context, NULL);
00249 
00250         if(out_len)
00251                 *out_len = 32;
00252 
00253         return TRUE;
