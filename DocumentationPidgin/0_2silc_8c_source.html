00086                   SILC_UMODE_HYPER |
00087                   SILC_UMODE_BUSY |
00088                   SILC_UMODE_INDISPOSED |
00089                   SILC_UMODE_PAGE);
00090 
00091         if (!strcmp(state, "hyper"))
00092                 mode |= SILC_UMODE_HYPER;
00093         else if (!strcmp(state, "away"))
00094                 mode |= SILC_UMODE_GONE;
00095         else if (!strcmp(state, "busy"))
00096                 mode |= SILC_UMODE_BUSY;
00097         else if (!strcmp(state, "indisposed"))
00098                 mode |= SILC_UMODE_INDISPOSED;
00099         else if (!strcmp(state, "page"))
00100                 mode |= SILC_UMODE_PAGE;
00101 
00102         /* Send UMODE */
00103         idp = silc_id_payload_encode(sg->conn->local_id, SILC_ID_CLIENT);
00104         SILC_PUT32_MSB(mode, mb);
00105         silc_client_command_send(sg->client, sg->conn, SILC_COMMAND_UMODE,
00106                                  ++sg->conn->cmd_ident, 2,
00107                                  1, idp->data, idp->len,
00108                                  2, mb, sizeof(mb));

01131 silcpurple_send_im(PurpleConnection *gc, const char *who, const char *message,
01132                  PurpleMessageFlags flags)
01133 {
01134         SilcPurple sg = gc->proto_data;
01135         SilcClient client = sg->client;
01136         SilcClientConnection conn = sg->conn;
01137         SilcClientEntry *clients;
01138         SilcUInt32 clients_count, mflags;
01139         char *nickname, *msg, *tmp;
01140         int ret = 0;
01141         gboolean sign = purple_account_get_bool(sg->account, "sign-verify", FALSE);
01142 #ifdef HAVE_SILCMIME_H
01143         SilcDList list;
01144 #endif
01145 
01146         if (!who || !message)
01147                 return 0;
01148 
01149         mflags = SILC_MESSAGE_FLAG_UTF8;
01150 
01151         tmp = msg = purple_unescape_html(message);
01152 
01153         if (!g_ascii_strncasecmp(msg, "/me ", 4)) {
01154                 msg += 4;
01155                 if (!*msg) {
01156                         g_free(tmp);
01157                         return 0;
01158                 }
01159                 mflags |= SILC_MESSAGE_FLAG_ACTION;
01160         } else if (strlen(msg) > 1 && msg[0] == '/') {
01161                 if (!silc_client_command_call(client, conn, msg + 1))
01162                         purple_notify_error(gc, _("Call Command"), _("Cannot call command"),
01163                                         _("Unknown command"));
01164                 g_free(tmp);
01165                 return 0;
01166         }
01167 
01168 
01169         if (!silc_parse_userfqdn(who, &nickname, NULL)) {
01170                 g_free(tmp);
01171                 return 0;
01172         }
01173 
01174         if (sign)
01175                 mflags |= SILC_MESSAGE_FLAG_SIGNED;
01176 
01177         /* Find client entry */
01178         clients = silc_client_get_clients_local(client, conn, nickname, who,
01179                                                 &clients_count);
01180         if (!clients) {
01181                 /* Resolve unknown user */
01182                 SilcPurpleIM im = silc_calloc(1, sizeof(*im));
01183                 if (!im) {
01184                         g_free(tmp);
01185                         return 0;
01186                 }
01187                 im->nick = g_strdup(who);
01188                 im->message = g_strdup(message);
01189                 im->message_len = strlen(im->message);
01190                 im->flags = mflags;
01191                 im->gflags = flags;
01192                 silc_client_get_clients(client, conn, nickname, NULL,
01193                                         silcpurple_send_im_resolved, im);
01194                 silc_free(nickname);
01195                 g_free(tmp);
01196                 return 0;
01197         }
01198 
01199 #ifdef HAVE_SILCMIME_H
01200         /* Check for images */
01201         if (flags & PURPLE_MESSAGE_IMAGES) {
01202                 list = silcpurple_image_message(message, &mflags);
01203                 if (list) {
01204                         /* Send one or more MIME message.  If more than one, they
01205                            are MIME fragments due to over large message */
01206                         SilcBuffer buf;
01207 
01208                         silc_dlist_start(list);
01209                         while ((buf = silc_dlist_get(list)) != SILC_LIST_END)
01210                                 ret =
01211                                 silc_client_send_private_message(client, conn,
01212                                                                  clients[0], mflags,
01213                                                                  buf->data, buf->len,
01214                                                                  TRUE);
01215                         silc_mime_partial_free(list);
01216                         g_free(tmp);
01217                         silc_free(nickname);
01218                         silc_free(clients);
01219                         return ret;
01220                 }
01221         }
01222 #endif
01223 
01224         /* Send private message directly */
01225         ret = silc_client_send_private_message(client, conn, clients[0],
01226                                                mflags,
01227                                                (unsigned char *)msg,
01228                                                strlen(msg), TRUE);
01229 
01230         g_free(tmp);
01231         silc_free(nickname);
01232         silc_free(clients);
01233         return ret;
01234 }
