00047 static void
00048 sha1_hash_block(struct SHA1Context *sha1_ctx) {
00049         gint i;
00050         guint32 A, B, C, D, E, T;
00051 
00052         for(i = 16; i < 80; i++) {
00053                 sha1_ctx->W[i] = SHA1_ROTL(sha1_ctx->W[i -  3] ^
00054                                 sha1_ctx->W[i -  8] ^
00055                                 sha1_ctx->W[i - 14] ^
00056                                 sha1_ctx->W[i - 16], 1);
00057         }
00058 
00059         A = sha1_ctx->H[0];
00060         B = sha1_ctx->H[1];
00061         C = sha1_ctx->H[2];
00062         D = sha1_ctx->H[3];
00063         E = sha1_ctx->H[4];
00064 
00065         for(i = 0; i < 20; i++) {
00066                 T = (SHA1_ROTL(A, 5) + (((C ^ D) & B) ^ D) + E + sha1_ctx->W[i] + 0x5A827999) & 0xFFFFFFFF;
00067                 E = D;
00068                 D = C;
00069                 C = SHA1_ROTL(B, 30);
00070                 B = A;
00071                 A = T;
00072         }
00073 
00074         for(i = 20; i < 40; i++) {
00075                 T = (SHA1_ROTL(A, 5) + (B ^ C ^ D) + E + sha1_ctx->W[i] + 0x6ED9EBA1) & 0xFFFFFFFF;
00076                 E = D;
00077                 D = C;
00078                 C = SHA1_ROTL(B, 30);
00079                 B = A;
00080                 A = T;
00081         }
00082 
00083         for(i = 40; i < 60; i++) {
00084                 T = (SHA1_ROTL(A, 5) + ((B & C) | (D & (B | C))) + E + sha1_ctx->W[i] + 0x8F1BBCDC) & 0xFFFFFFFF;
00085                 E = D;
00086                 D = C;
00087                 C = SHA1_ROTL(B, 30);
00088                 B = A;
00089                 A = T;
00090         }
00091 
00092         for(i = 60; i < 80; i++) {
00093                 T = (SHA1_ROTL(A, 5) + (B ^ C ^ D) + E + sha1_ctx->W[i] + 0xCA62C1D6) & 0xFFFFFFFF;
00094                 E = D;
00095                 D = C;
00096                 C = SHA1_ROTL(B, 30);
00097                 B = A;
00098                 A = T;
00099         }
00100 
00101         sha1_ctx->H[0] += A;
00102         sha1_ctx->H[1] += B;
00103         sha1_ctx->H[2] += C;
00104         sha1_ctx->H[3] += D;
00105         sha1_ctx->H[4] += E;
00106 }

00189 sha1_append(PurpleCipherContext *context, const guchar *data, size_t len) {
00190         struct SHA1Context *sha1_ctx;
00191         gint i;
00192 
00193         sha1_ctx = purple_cipher_context_get_data(context);
00194 
00195         g_return_if_fail(sha1_ctx);
00196 
00197         for(i = 0; i < len; i++) {
00198                 sha1_ctx->W[sha1_ctx->lenW / 4] <<= 8;
00199                 sha1_ctx->W[sha1_ctx->lenW / 4] |= data[i];
00200 
00201                 if((++sha1_ctx->lenW) % 64 == 0) {
00202                         sha1_hash_block(sha1_ctx);
00203                         sha1_ctx->lenW = 0;
00204                 }
00205 
00206                 sha1_ctx->sizeLo += 8;
00207                 sha1_ctx->sizeHi += (sha1_ctx->sizeLo < 8);

00211 static gboolean
00212 sha1_digest(PurpleCipherContext *context, size_t in_len, guchar digest[20],
00213             size_t *out_len)
00214 {
00215         struct SHA1Context *sha1_ctx;
00216         guchar pad0x80 = 0x80, pad0x00 = 0x00;
00217         guchar padlen[8];
00218         gint i;
00219 
00220         g_return_val_if_fail(in_len >= 20, FALSE);
00221 
00222         sha1_ctx = purple_cipher_context_get_data(context);
00223 
00224         g_return_val_if_fail(sha1_ctx, FALSE);
00225 
00226         padlen[0] = (guchar)((sha1_ctx->sizeHi >> 24) & 255);
00227         padlen[1] = (guchar)((sha1_ctx->sizeHi >> 16) & 255);
00228         padlen[2] = (guchar)((sha1_ctx->sizeHi >> 8) & 255);
00229         padlen[3] = (guchar)((sha1_ctx->sizeHi >> 0) & 255);
00230         padlen[4] = (guchar)((sha1_ctx->sizeLo >> 24) & 255);
00231         padlen[5] = (guchar)((sha1_ctx->sizeLo >> 16) & 255);
00232         padlen[6] = (guchar)((sha1_ctx->sizeLo >> 8) & 255);
00233         padlen[7] = (guchar)((sha1_ctx->sizeLo >> 0) & 255);
00234 
00235         /* pad with a 1, then zeroes, then length */
00236         purple_cipher_context_append(context, &pad0x80, 1);
00237         while(sha1_ctx->lenW != 56)
00238                 purple_cipher_context_append(context, &pad0x00, 1);
00239         purple_cipher_context_append(context, padlen, 8);
00240 
00241         for(i = 0; i < 20; i++) {
00242                 digest[i] = (guchar)(sha1_ctx->H[i / 4] >> 24);
00243                 sha1_ctx->H[i / 4] <<= 8;
00244         }
00245 
00246         purple_cipher_context_reset(context, NULL);
00247 
00248         if(out_len)
00249                 *out_len = 20;
00250 
00251         return TRUE;
00252 }
