00044 /* Escape windows dir separators.  This is needed when paths are saved,
00045    and on being read back have their '\' chars used as an escape char.
00046    Returns an allocated string which needs to be freed.
00047 */
00048 char *wpurple_escape_dirsep(const char *filename) {
00049         int sepcount = 0;
00050         const char *tmp = filename;
00051         char *ret;
00052         int cnt = 0;
00053 
00054         g_return_val_if_fail(filename != NULL, NULL);
00055 
00056         while(*tmp) {
00057                 if(*tmp == '\\')
00058                         sepcount++;
00059                 tmp++;
00060         }
00061         ret = g_malloc0(strlen(filename) + sepcount + 1);
00062         while(*filename) {
00063                 ret[cnt] = *filename;
00064                 if(*filename == '\\')
00065                         ret[++cnt] = '\\';
00066                 filename++;
00067                 cnt++;
00068         }
00069         ret[cnt] = '\0';
00070         return ret;
00071 }

00075 FARPROC wpurple_find_and_loadproc(const char *dllname, const char *procedure) {
00076         HMODULE hmod;
00077         BOOL did_load = FALSE;
00078         FARPROC proc = 0;
00079 
00080         wchar_t *wc_dllname = g_utf8_to_utf16(dllname, -1, NULL, NULL, NULL);
00081 
00082         if(!(hmod = GetModuleHandleW(wc_dllname))) {
00083                 purple_debug_warning("wpurple", "%s not already loaded; loading it...\n", dllname);
00084                 if(!(hmod = LoadLibraryW(wc_dllname))) {
00085                         purple_debug_error("wpurple", "Could not load: %s (%s)\n", dllname,
00086                                 g_win32_error_message(GetLastError()));
00087                         g_free(wc_dllname);
00088                         return NULL;
00089                 }
00090                 else
00091                         did_load = TRUE;
00092         }
00093 
00094         g_free(wc_dllname);
00095         wc_dllname = NULL;
00096 
00097         if((proc = GetProcAddress(hmod, procedure))) {
00098                 purple_debug_info("wpurple", "This version of %s contains %s\n",
00099                         dllname, procedure);
00100                 return proc;
00101         }
00102         else {
00103                 purple_debug_warning("wpurple", "Function %s not found in dll %s\n",
00104                         procedure, dllname);
00105                 if(did_load) {
00106                         /* unload dll */
00107                         FreeLibrary(hmod);
00108                 }
00109                 return NULL;
00110         }
00111 }
