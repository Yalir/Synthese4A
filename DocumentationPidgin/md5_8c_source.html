00038         (n) = ((guint32)(b) [(i)    ]      )    \
00039         | ((guint32)(b) [(i) + 1] <<  8)    \

00220         struct MD5Context *md5_context = NULL;
00221         guint32 left = 0, fill = 0;
00222 
00223         g_return_if_fail(context != NULL);

00226         g_return_if_fail(md5_context != NULL);
00227 
00228         left = md5_context->total[0] & 0x3F;
00229         fill = 64 - left;
00230 
00231         md5_context->total[0] += len;
00232         md5_context->total[0] &= 0xFFFFFFFF;
00233 
00234         if(md5_context->total[0] < len)
00235                 md5_context->total[1]++;
00236 
00237         if(left && len >= fill) {
00238                 memcpy((md5_context->buffer + left), data, fill);
00239                 md5_process(md5_context, md5_context->buffer);
00240                 len -= fill;
00241                 data += fill;
00242                 left = 0;
00243         }
00244 
00245         while(len >= 64) {
00246                 md5_process(md5_context, data);
00247                 len -= 64;
