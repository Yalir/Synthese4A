00093 static GList *
00094 plugin_actions(PurplePlugin *plugin, gpointer context)
00095 {
00096         GList *actions = NULL;
00097 
00098         /* Here we take advantage of return values to avoid the need for a temp variable */
00099         actions = g_list_prepend(actions,
00100                 purple_plugin_action_new("Show Error Notification", notify_error_cb));
00101 
00102         actions = g_list_prepend(actions,
00103                 purple_plugin_action_new("Show Info Notification", notify_info_cb));
00104 
00105         actions = g_list_prepend(actions,
00106                 purple_plugin_action_new("Show Warning Notification", notify_warn_cb));
00107 
00108         actions = g_list_prepend(actions,
00109                 purple_plugin_action_new("Show Formatted Notification", notify_format_cb));
00110 
00111         actions = g_list_prepend(actions,
00112                 purple_plugin_action_new("Show URI Notification", notify_uri_cb));
00113 
00114         return g_list_reverse(actions);
00115 }
