00021 Code_t ZCheckAuthentication(notice, from)
00022     ZNotice_t *notice;
00023     struct sockaddr_in *from;
00024 {
00025 #ifdef ZEPHYR_USES_KERBEROS
00026     int result;
00027     ZChecksum_t our_checksum;
00028     CREDENTIALS cred;
00029 
00030     /* If the value is already known, return it. */
00031     if (notice->z_checked_auth != ZAUTH_UNSET)
00032         return (notice->z_checked_auth);
00033 
00034     if (!notice->z_auth)
00035         return (ZAUTH_NO);
00036 
00037     if ((result = krb_get_cred(SERVER_SERVICE, SERVER_INSTANCE,
00038                                __Zephyr_realm, &cred)) != 0)
00039         return (ZAUTH_NO);
00040 
00041 #ifdef NOENCRYPTION
00042     our_checksum = 0;
00043 #else
00044     our_checksum = des_quad_cksum(notice->z_packet, NULL,
00045                                   notice->z_default_format+
00046                                   strlen(notice->z_default_format)+1-
00047                                   notice->z_packet, 0, (C_Block *)cred.session);
00048 #endif
00049     /* if mismatched checksum, then the packet was corrupted */
00050     return ((our_checksum == notice->z_checksum) ? ZAUTH_YES : ZAUTH_FAILED);
00051

00053     return (notice->z_auth ? ZAUTH_YES : ZAUTH_NO);
