00636                 return FALSE;
00637         }
00638 
00639         if (g_str_equal(creator, "initiator"))
00640                 is_creator = jingle_session_is_initiator(session);
00641         else

00643         g_free(creator);
00644

00663 static GList *
00664 jingle_rtp_parse_codecs(xmlnode *description)
00665 {
00666         GList *codecs = NULL;
00667         xmlnode *codec_element = NULL;
00668         const char *encoding_name,*id, *clock_rate;
00669         PurpleMediaCodec *codec;
00670         const gchar *media = xmlnode_get_attrib(description, "media");
00671         PurpleMediaSessionType type;
00672 
00673         if (media == NULL) {
00674                 purple_debug_warning("jingle-rtp", "missing media type\n");
00675                 return NULL;
00676         }
00677 
00678         if (g_str_equal(media, "video")) {
00679                 type = PURPLE_MEDIA_VIDEO;
00680         } else if (g_str_equal(media, "audio")) {
00681                 type = PURPLE_MEDIA_AUDIO;
00682         } else {
00683                 purple_debug_warning("jingle-rtp", "unknown media type: %s\n",
00684                                 media);
00685                 return NULL;
00686         }
00687 
00688         for (codec_element = xmlnode_get_child(description, "payload-type") ;
00689                  codec_element ;
00690                  codec_element = xmlnode_get_next_twin(codec_element)) {
00691                 xmlnode *param;
00692                 gchar *codec_str;
00693                 encoding_name = xmlnode_get_attrib(codec_element, "name");
00694 
00695                 id = xmlnode_get_attrib(codec_element, "id");
00696                 clock_rate = xmlnode_get_attrib(codec_element, "clockrate");
00697 
00698                 codec = purple_media_codec_new(atoi(id), encoding_name,
00699                                      type,
00700                                      clock_rate ? atoi(clock_rate) : 0);
00701 
00702                 for (param = xmlnode_get_child(codec_element, "parameter");
00703                                 param; param = xmlnode_get_next_twin(param)) {
00704                         purple_media_codec_add_optional_parameter(codec,
00705                                         xmlnode_get_attrib(param, "name"),
00706                                         xmlnode_get_attrib(param, "value"));
00707                 }
00708 
00709                 codec_str = purple_media_codec_to_string(codec);
00710                 purple_debug_info("jingle-rtp", "received codec: %s\n", codec_str);
00711                 g_free(codec_str);
00712 
00713                 codecs = g_list_append(codecs, codec);
00714         }
00715         return codecs;

00801 static void
00802 jingle_rtp_handle_action_internal(JingleContent *content, xmlnode *xmlcontent, JingleActionType action)
00803 {
00804         switch (action) {
00805                 case JINGLE_SESSION_ACCEPT:
00806                 case JINGLE_SESSION_INITIATE: {
00807                         JingleSession *session;
00808                         JingleTransport *transport;
00809                         xmlnode *description;
00810                         GList *candidates;

00815 
00816                         session = jingle_content_get_session(content);
00817 
00818                         if (action == JINGLE_SESSION_INITIATE &&
00819                                         !jingle_rtp_init_media(content)) {
00820                                 /* XXX: send error */
00821                                 jabber_iq_send(jingle_session_terminate_packet(
00822                                                 session, "general-error"));
00823                                 g_object_unref(session);
00824                                 break;
00825                         }
00826 
00827                         transport = jingle_transport_parse(
00828                                         xmlnode_get_child(xmlcontent, "transport"));
00829                         description = xmlnode_get_child(xmlcontent, "description");
00830                         candidates = jingle_rtp_transport_to_candidates(transport);
00831                         codecs = jingle_rtp_parse_codecs(description);
00832                         name = jingle_content_get_name(content);
00833                         remote_jid = jingle_session_get_remote_jid(session);
00834 
00835                         media = jingle_rtp_get_media(session);
00836                         purple_media_set_remote_codecs(media,
00837                                         name, remote_jid, codecs);
00838                         purple_media_add_remote_candidates(media,
00839                                         name, remote_jid, candidates);
00840 
00841                         if (action == JINGLE_SESSION_ACCEPT)
00842                                 purple_media_stream_info(media,
00843                                                 PURPLE_MEDIA_INFO_ACCEPT,
00844                                                 name, remote_jid, FALSE);
00845 
00846                         g_free(remote_jid);
00847                         g_free(name);
00848                         g_object_unref(session);
00849                         break;
00850                 }
00851                 case JINGLE_SESSION_TERMINATE: {
00852                         JingleSession *session = jingle_content_get_session(content);
00853                         PurpleMedia *media = jingle_rtp_get_media(session);
00854 
00855                         if (media != NULL) {
00856                                 purple_media_end(media, NULL, NULL);
00857                         }
00858 
00859                         g_object_unref(session);
00860                         break;
00861                 }
00862                 case JINGLE_TRANSPORT_INFO: {
00863                         JingleSession *session = jingle_content_get_session(content);
00864                         JingleTransport *transport = jingle_transport_parse(
00865                                         xmlnode_get_child(xmlcontent, "transport"));
00866                         GList *candidates = jingle_rtp_transport_to_candidates(transport);
00867                         gchar *name = jingle_content_get_name(content);
00868                         gchar *remote_jid =
00869                                         jingle_session_get_remote_jid(session);
00870 
00871                         purple_media_add_remote_candidates(
00872                                         jingle_rtp_get_media(session),
00873                                         name, remote_jid, candidates);
00874 
00875                         g_free(remote_jid);
00876                         g_free(name);
00877                         g_object_unref(session);
00878                         break;
00879                 }
00880                 case JINGLE_DESCRIPTION_INFO: {
00881                         JingleSession *session =
00882                                         jingle_content_get_session(content);
00883                         xmlnode *description = xmlnode_get_child(
00884                                         xmlcontent, "description");
00885                         GList *codecs, *iter, *iter2, *remote_codecs =
00886                                         jingle_rtp_parse_codecs(description);
00887                         gchar *name = jingle_content_get_name(content);
00888                         gchar *remote_jid =
00889                                         jingle_session_get_remote_jid(session);
00890                         PurpleMedia *media;
00891 
00892                         media = jingle_rtp_get_media(session);
00893 
00894                         /*
00895                          * This may have problems if description-info is
00896                          * received without the optional parameters for a
00897                          * codec with configuration info (such as THEORA
00898                          * or H264). The local configuration info may be
00899                          * set for the remote codec.
00900                          *
00901                          * As of 2.6.3 there's no API to support getting

00907 
00908                         for (iter = codecs; iter; iter = g_list_next(iter)) {
00909                                 guint id;
00910 
00911                                 id = purple_media_codec_get_id(iter->data);
00912                                 iter2 = remote_codecs;
00913 
00914                                 for (; iter2; iter2 = g_list_next(iter2)) {
00915                                         if (purple_media_codec_get_id(
00916                                                         iter2->data) != id)
00917                                                 continue;
00918 
00919                                         g_object_unref(iter->data);
00920                                         iter->data = iter2->data;

00927                         codecs = g_list_concat(codecs, remote_codecs);
00928 
00929                         purple_media_set_remote_codecs(media,
00930                                         name, remote_jid, codecs);
00931 
00932                         purple_media_codec_list_free (codecs);
00933                         g_free(remote_jid);
00934                         g_free(name);

00938                 default:

00943 gboolean
00944 jingle_rtp_initiate_media(JabberStream *js, const gchar *who,
00945                       PurpleMediaSessionType type)
00946 {
00947         /* create content negotiation */
00948         JingleSession *session;
00949         JingleContent *content;
00950         JingleTransport *transport;
00951         JabberBuddy *jb;
00952         JabberBuddyResource *jbr;
00953         const gchar *transport_type;
00954 
00955         gchar *resource = NULL, *me = NULL, *sid = NULL;
00956 
00957         /* construct JID to send to */
00958         jb = jabber_buddy_find(js, who, FALSE);
00959         if (!jb) {
00960                 purple_debug_error("jingle-rtp", "Could not find Jabber buddy\n");
00961                 return FALSE;
00962         }
00963 
00964         resource = jabber_get_resource(who);
00965         jbr = jabber_buddy_find_resource(jb, resource);
00966         g_free(resource);
00967 
00968         if (!jbr) {
00969                 purple_debug_error("jingle-rtp", "Could not find buddy's resource\n");
00970         }
00971 
00972         if (jabber_resource_has_capability(jbr, JINGLE_TRANSPORT_ICEUDP)) {
00973                 transport_type = JINGLE_TRANSPORT_ICEUDP;
00974         } else if (jabber_resource_has_capability(jbr, JINGLE_TRANSPORT_RAWUDP)) {
00975                 transport_type = JINGLE_TRANSPORT_RAWUDP;
00976         } else {
00977                 purple_debug_error("jingle-rtp", "Resource doesn't support "
00978                                 "the same transport types\n");
00979                 return FALSE;
00980         }
00981 
00982         /* set ourselves as initiator */
00983         me = g_strdup_printf("%s@%s/%s", js->user->node, js->user->domain, js->user->resource);
00984 
00985         sid = jabber_get_next_id(js);
00986         session = jingle_session_create(js, sid, me, who, TRUE);
00987         g_free(sid);
00988 
00989 
00990         if (type & PURPLE_MEDIA_AUDIO) {
00991                 transport = jingle_transport_create(transport_type);
00992                 content = jingle_content_create(JINGLE_APP_RTP, "initiator",
00993                                 "session", "audio-session", "both", transport);
00994                 jingle_session_add_content(session, content);
00995                 JINGLE_RTP(content)->priv->media_type = g_strdup("audio");
00996                 jingle_rtp_init_media(content);
00997         }
00998         if (type & PURPLE_MEDIA_VIDEO) {
00999                 transport = jingle_transport_create(transport_type);
01000                 content = jingle_content_create(JINGLE_APP_RTP, "initiator",
01001                                 "session", "video-session", "both", transport);
01002                 jingle_session_add_content(session, content);
01003                 JINGLE_RTP(content)->priv->media_type = g_strdup("video");
01004                 jingle_rtp_init_media(content);
01005         }
01006 
01007         g_free(me);
01008 
01009         if (jingle_rtp_get_media(session) == NULL) {
01010                 return FALSE;
01011         }
01012 
01013         return TRUE;
01014 }
