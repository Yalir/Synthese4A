00024 /* probes the given plugin to determine if its a plugin */
00025 static gboolean probe_mono_plugin(PurplePlugin *plugin)
00026 {
00027         MonoAssembly *assm;
00028         MonoMethod *m = NULL;
00029         MonoObject *plugin_info;
00030         gboolean found_load = FALSE, found_unload = FALSE, found_destroy = FALSE;
00031         gpointer iter = NULL;
00032 
00033         PurplePluginInfo *info;
00034         PurpleMonoPlugin *mplug;
00035 
00036         char *file = plugin->path;
00037 
00038         assm = mono_domain_assembly_open(ml_get_domain(), file);
00039 
00040         if (!assm) {
00041                 return FALSE;
00042         }
00043 
00044         purple_debug(PURPLE_DEBUG_INFO, "mono", "Probing plugin\n");
00045 
00046         if (ml_is_api_dll(mono_assembly_get_image(assm))) {
00047                 purple_debug_info("mono", "Found our PurpleAPI.dll\n");
00048                 mono_assembly_close(assm);
00049                 return FALSE;
00050         }
00051 
00052         mplug = g_new0(PurpleMonoPlugin, 1);
00053 
00054         mplug->signal_data = NULL;
00055 
00056         mplug->assm = assm;
00057 
00058         mplug->klass = ml_find_plugin_class(mono_assembly_get_image(mplug->assm));
00059         if (!mplug->klass) {
00060                 purple_debug(PURPLE_DEBUG_ERROR, "mono", "no plugin class in \'%s\'\n", file);
00061                 mono_assembly_close(assm);
00062                 g_free(mplug);
00063                 return FALSE;
00064         }
00065 
00066         mplug->obj = mono_object_new(ml_get_domain(), mplug->klass);
00067         if (!mplug->obj) {
00068                 purple_debug(PURPLE_DEBUG_ERROR, "mono", "obj not valid\n");
00069                 mono_assembly_close(assm);
00070                 g_free(mplug);
00071                 return FALSE;
00072         }
00073 
00074         mono_runtime_object_init(mplug->obj);
00075 
00076         while ((m = mono_class_get_methods(mplug->klass, &iter))) {
00077                 purple_debug_info("mono", "plugin method: %s\n", mono_method_get_name(m));
00078                 if (strcmp(mono_method_get_name(m), "Load") == 0) {
00079                         mplug->load = m;
00080                         found_load = TRUE;
00081                 } else if (strcmp(mono_method_get_name(m), "Unload") == 0) {
00082                         mplug->unload = m;
00083                         found_unload = TRUE;
00084                 } else if (strcmp(mono_method_get_name(m), "Destroy") == 0) {
00085                         mplug->destroy = m;
00086                         found_destroy = TRUE;
00087                 }
00088         }
00089 
00090         if (!(found_load && found_unload && found_destroy)) {
00091                 purple_debug(PURPLE_DEBUG_ERROR, "mono", "did not find the required methods\n");
00092                 mono_assembly_close(assm);
00093                 g_free(mplug);
00094                 return FALSE;
00095         }
00096 
00097         plugin_info = ml_get_info_prop(mplug->obj);
00098 
00099         /* now that the methods are filled out we can populate
00100            the info struct with all the needed info */
00101 
00102         info = g_new0(PurplePluginInfo, 1);
00103         info->id = ml_get_prop_string(plugin_info, "Id");
00104         info->name = ml_get_prop_string(plugin_info, "Name");
00105         info->version = ml_get_prop_string(plugin_info, "Version");
00106         info->summary = ml_get_prop_string(plugin_info, "Summary");
00107         info->description = ml_get_prop_string(plugin_info, "Description");
00108         info->author = ml_get_prop_string(plugin_info, "Author");
00109         info->homepage = ml_get_prop_string(plugin_info, "Homepage");
00110 
00111         info->magic = PURPLE_PLUGIN_MAGIC;
00112         info->major_version = PURPLE_MAJOR_VERSION;
00113         info->minor_version = PURPLE_MINOR_VERSION;
00114         info->type = PURPLE_PLUGIN_STANDARD;
00115 
00116         /* this plugin depends on us; duh */
00117         info->dependencies = g_list_append(info->dependencies, MONO_PLUGIN_ID);
00118         mplug->plugin = plugin;
00119 
00120         plugin->info = info;
00121         info->extra_info = mplug;
00122 
00123         ml_add_plugin(mplug);
00124 
00125         return purple_plugin_register(plugin);
00126 }
