01503 static void process_input_message(struct simple_account_data *sip, struct sipmsg *msg) {
01504         gboolean found = FALSE;
01505         if(msg->response == 0) { /* request */
01506                 if(!strcmp(msg->method, "MESSAGE")) {
01507                         process_incoming_message(sip, msg);
01508                         found = TRUE;
01509                 } else if(!strcmp(msg->method, "NOTIFY")) {
01510                         process_incoming_notify(sip, msg);
01511                         found = TRUE;
01512                 } else if(!strcmp(msg->method, "SUBSCRIBE")) {
01513                         process_incoming_subscribe(sip, msg);
01514                         found = TRUE;
01515                 } else {
01516                         send_sip_response(sip->gc, msg, 501, "Not implemented", NULL);
01517                 }
01518         } else { /* response */
01519                 struct transaction *trans = transactions_find(sip, msg);
01520                 if(trans) {
01521                         if(msg->response == 407) {
01522                                 gchar *resend, *auth;
01523                                 const gchar *ptmp;
01524 
01525                                 if(sip->proxy.retries > 3) return;
01526                                 sip->proxy.retries++;

01528 
01529                                 ptmp = sipmsg_find_header(msg, "Proxy-Authenticate");
01530 
01531                                 fill_auth(sip, ptmp, &sip->proxy);
01532                                 auth = auth_header(sip, &sip->proxy, trans->msg->method, trans->msg->target);
01533                                 sipmsg_remove_header(trans->msg, "Proxy-Authorization");
01534                                 sipmsg_add_header(trans->msg, "Proxy-Authorization", auth);

01539                                 g_free(resend);

01541                                 if(msg->response == 100) {
01542                                         /* ignore provisional response */
01543                                         purple_debug_info("simple", "got trying response\n");
01544                                 } else {
01545                                         sip->proxy.retries = 0;
01546                                         if(!strcmp(trans->msg->method, "REGISTER")) {
01547 
01548                                                 /* This is encountered when a REGISTER request was ...
01549                                                  */
01550                                                 if(msg->response == 401) {
01551                                                         /* denied until further authentication was provided. */
01552                                                         sip->registrar.retries++;
01553                                                 }
01554                                                 else if (msg->response != 200) {
01555                                                         /* denied for some other reason! */
01556                                                         sip->registrar.retries++;
01557                                                 }
01558                                                 else {
01559                                                         /* accepted! */
01560                                                         sip->registrar.retries = 0;
01561                                                 }
01562                                         } else {
01563                                                 if(msg->response == 401) {

01568                                                         const gchar *ptmp;

01574

01581                                                         /* resend request */
01582                                                         sendout_pkt(sip->gc, resend);
01583                                                         g_free(resend);
01584                                                 } else {
01585                                                         /* Reset any count of retries that may have
01586                                                          * accumulated in the above branch.
01587                                                          */
01588                                                         sip->registrar.retries = 0;
01589                                                 }
01590                                         }
01591                                         if(trans->callback) {
01592                                                 /* call the callback to process response*/
01593                                                 (trans->callback)(sip, msg, trans);
01594                                         }
01595                                         transactions_remove(sip, trans);
01596                                 }
01597                         }
01598                         found = TRUE;
01599                 } else {
01600                         purple_debug(PURPLE_DEBUG_MISC, "simple", "received response to unknown transaction");
01601                 }
01602         }
01603         if(!found) {
01604                 purple_debug(PURPLE_DEBUG_MISC, "simple", "received a unknown sip message with method %s and response %d\n", msg->method, msg->response);
01605         }
01606 }

01608 static void process_input(struct simple_account_data *sip, struct sip_connection *conn)
01609 {
01610         char *cur;
01611         char *dummy;
01612         struct sipmsg *msg;
01613         int restlen;
01614         cur = conn->inbuf;
01615 
01616         /* according to the RFC remove CRLF at the beginning */
01617         while(*cur == '\r' || *cur == '\n') {
01618                 cur++;
01619         }
01620         if(cur != conn->inbuf) {
01621                 memmove(conn->inbuf, cur, conn->inbufused - (cur - conn->inbuf));
01622                 conn->inbufused = strlen(conn->inbuf);
01623         }
01624 
01625         /* Received a full Header? */
01626         if((cur = strstr(conn->inbuf, "\r\n\r\n")) != NULL) {
01627                 time_t currtime = time(NULL);
01628                 cur += 2;
01629                 cur[0] = '\0';
01630                 purple_debug_info("simple", "\n\nreceived - %s\n######\n%s\n#######\n\n", ctime(&currtime), conn->inbuf);
01631                 msg = sipmsg_parse_header(conn->inbuf);
01632 
01633                 if(!msg) {
01634                         /* Should we re-use this error message (from lower in the function)? */
01635                         purple_debug_misc("simple", "received a incomplete sip msg: %s\n", conn->inbuf);
01636                         return;
01637                 }
01638 
01639                 cur[0] = '\r';
01640                 cur += 2;
01641                 restlen = conn->inbufused - (cur - conn->inbuf);
01642                 if(restlen >= msg->bodylen) {
01643                         dummy = g_malloc(msg->bodylen + 1);
01644                         memcpy(dummy, cur, msg->bodylen);
01645                         dummy[msg->bodylen] = '\0';
01646                         msg->body = dummy;
01647                         cur += msg->bodylen;
01648                         memmove(conn->inbuf, cur, conn->inbuflen - (cur - conn->inbuf));
01649                         conn->inbufused = strlen(conn->inbuf);
01650                 } else {
01651                         sipmsg_free(msg);
01652                         return;
01653                 }
01654                 purple_debug(PURPLE_DEBUG_MISC, "simple", "in process response response: %d\n", msg->response);
01655                 process_input_message(sip, msg);
01656                 sipmsg_free(msg);
01657         } else {
01658                 purple_debug(PURPLE_DEBUG_MISC, "simple", "received a incomplete sip msg: %s\n", conn->inbuf);
01659         }
01660 }
