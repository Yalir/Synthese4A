00400 static PurpleNetworkListenData *
00401 purple_network_do_listen(unsigned short port, int socket_family, int socket_type, PurpleNetworkListenCallback cb, gpointer cb_data)
00402 {
00403         int listenfd = -1;
00404         int flags;
00405         const int on = 1;
00406         PurpleNetworkListenData *listen_data;
00407         unsigned short actual_port;
00408 #ifdef HAVE_GETADDRINFO
00409         int errnum;
00410         struct addrinfo hints, *res, *next;
00411         char serv[6];
00412 
00413         /*
00414          * Get a list of addresses on this machine.
00415          */
00416         g_snprintf(serv, sizeof(serv), "%hu", port);
00417         memset(&hints, 0, sizeof(struct addrinfo));
00418         hints.ai_flags = AI_PASSIVE;
00419         hints.ai_family = socket_family;
00420         hints.ai_socktype = socket_type;
00421         errnum = getaddrinfo(NULL /* any IP */, serv, &hints, &res);
00422         if (errnum != 0) {
00423 #ifndef _WIN32
00424                 purple_debug_warning("network", "getaddrinfo: %s\n", purple_gai_strerror(errnum));
00425                 if (errnum == EAI_SYSTEM)
00426                         purple_debug_warning("network", "getaddrinfo: system error: %s\n", g_strerror(errno));
00427 #else
00428                 purple_debug_warning("network", "getaddrinfo: Error Code = %d\n", errnum);
00429 #endif
00430                 return NULL;
00431         }
00432 
00433         /*
00434          * Go through the list of addresses and attempt to listen on
00435          * one of them.
00436          * XXX - Try IPv6 addresses first?
00437          */
00438         for (next = res; next != NULL; next = next->ai_next) {
00439                 listenfd = socket(next->ai_family, next->ai_socktype, next->ai_protocol);
00440                 if (listenfd < 0)
00441                         continue;
00442                 if (setsockopt(listenfd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) != 0)
00443                         purple_debug_warning("network", "setsockopt(SO_REUSEADDR): %s\n", g_strerror(errno));
00444                 if (bind(listenfd, next->ai_addr, next->ai_addrlen) == 0)
00445                         break; /* success */
00446                 /* XXX - It is unclear to me (datallah) whether we need to be
00447                    using a new socket each time */
00448                 close(listenfd);
00449         }
00450 
00451         freeaddrinfo(res);
00452 
00453         if (next == NULL)
00454                 return NULL;
00455 #else
00456         struct sockaddr_in sockin;
00457 
00458         if (socket_family != AF_INET && socket_family != AF_UNSPEC) {
00459                 purple_debug_warning("network", "Address family %d only "
00460                                      "supported when built with getaddrinfo() "
00461                                      "support\n", socket_family);
00462                 return NULL;
00463         }
00464 
00465         if ((listenfd = socket(AF_INET, socket_type, 0)) < 0) {
00466                 purple_debug_warning("network", "socket: %s\n", g_strerror(errno));
00467                 return NULL;
00468         }
00469 
00470         if (setsockopt(listenfd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) != 0)
00471                 purple_debug_warning("network", "setsockopt: %s\n", g_strerror(errno));
00472 
00473         memset(&sockin, 0, sizeof(struct sockaddr_in));
00474         sockin.sin_family = PF_INET;
00475         sockin.sin_port = htons(port);
00476 
00477         if (bind(listenfd, (struct sockaddr *)&sockin, sizeof(struct sockaddr_in)) != 0) {
00478                 purple_debug_warning("network", "bind: %s\n", g_strerror(errno));
00479                 close(listenfd);
00480                 return NULL;
00481         }
00482 #endif
00483 
00484         if (socket_type == SOCK_STREAM && listen(listenfd, 4) != 0) {
00485                 purple_debug_warning("network", "listen: %s\n", g_strerror(errno));
00486                 close(listenfd);
00487                 return NULL;
00488         }
00489         flags = fcntl(listenfd, F_GETFL);
00490         fcntl(listenfd, F_SETFL, flags | O_NONBLOCK);
00491 #ifndef _WIN32
00492         fcntl(listenfd, F_SETFD, FD_CLOEXEC);
00493 #endif
00494         actual_port = purple_network_get_port_from_fd(listenfd);
00495 
00496         purple_debug_info("network", "Listening on port: %hu\n", actual_port);
00497 
00498         listen_data = g_new0(PurpleNetworkListenData, 1);
00499         listen_data->listenfd = listenfd;
00500         listen_data->adding = TRUE;
00501         listen_data->retry = TRUE;
00502         listen_data->cb = cb;
00503         listen_data->cb_data = cb_data;
00504         listen_data->socket_type = socket_type;
00505 
00506         if (!purple_socket_speaks_ipv4(listenfd) || !listen_map_external ||
00507                         !purple_prefs_get_bool("/purple/network/map_ports"))
00508         {
00509                 purple_debug_info("network", "Skipping external port mapping.\n");
00510                 /* The pmp_map_cb does what we want to do */
00511                 listen_data->timer = purple_timeout_add(0, purple_network_finish_pmp_map_cb, listen_data);
00512         }
00513         /* Attempt a NAT-PMP Mapping, which will return immediately */
00514         else if (purple_pmp_create_map(((socket_type == SOCK_STREAM) ? PURPLE_PMP_TYPE_TCP : PURPLE_PMP_TYPE_UDP),
00515                                                           actual_port, actual_port, PURPLE_PMP_LIFETIME))
00516         {
00517                 purple_debug_info("network", "Created NAT-PMP mapping on port %i\n", actual_port);
00518                 /* We want to return listen_data now, and on the next run loop trigger the cb and destroy listen_data */
00519                 listen_data->timer = purple_timeout_add(0, purple_network_finish_pmp_map_cb, listen_data);
00520         }
00521         else
00522         {
00523                 /* Attempt a UPnP Mapping */
00524                 listen_data->mapping_data = purple_upnp_set_port_mapping(
00525                                                  actual_port,
00526                                                  (socket_type == SOCK_STREAM) ? "TCP" : "UDP",
00527                                                  purple_network_set_upnp_port_mapping_cb, listen_data);
00528         }
00529 
00530         return listen_data;
00531 }

00703 static gpointer wpurple_network_change_thread(gpointer data)
00704 {
00705         WSAQUERYSET qs;
00706         WSAEVENT *nla_event;
00707         time_t last_trigger = time(NULL) - 31;
00708         char buf[4096];
00709         WSAQUERYSET *res = (LPWSAQUERYSET) buf;
00710         DWORD size;
00711 
00712         if ((nla_event = WSACreateEvent()) == WSA_INVALID_EVENT) {
00713                 int errorid = WSAGetLastError();
00714                 gchar *msg = g_win32_error_message(errorid);
00715                 purple_timeout_add(0, _print_debug_msg,
00716                                                    g_strdup_printf("Couldn't create WSA event. "
00717                                                                                    "Message: %s (%d).\n", msg, errorid));
00718                 g_free(msg);
00719                 g_thread_exit(NULL);
00720                 return NULL;
00721         }
00722 
00723         while (TRUE) {
00724                 int retval;
00725                 DWORD retLen = 0;
00726                 WSACOMPLETION completion;
00727                 WSAOVERLAPPED overlapped;
00728 
00729                 g_static_mutex_lock(&mutex);
00730                 if (network_initialized == FALSE) {
00731                         /* purple_network_uninit has been called */
00732                         WSACloseEvent(nla_event);
00733                         g_static_mutex_unlock(&mutex);
00734                         g_thread_exit(NULL);
00735                         return NULL;
00736                 }
00737 
00738                 if (network_change_handle == NULL) {
00739                         memset(&qs, 0, sizeof(WSAQUERYSET));
00740                         qs.dwSize = sizeof(WSAQUERYSET);
00741                         qs.dwNameSpace = NS_NLA;
00742                         if (WSALookupServiceBegin(&qs, 0, &network_change_handle) == SOCKET_ERROR) {
00743                                 int errorid = WSAGetLastError();
00744                                 gchar *msg = g_win32_error_message(errorid);
00745                                 purple_timeout_add(0, _print_debug_msg,
00746                                                                    g_strdup_printf("Couldn't retrieve NLA SP lookup handle. "
00747                                                                                                    "NLA service is probably not running. Message: %s (%d).\n",
00748                                                                                                         msg, errorid));
00749                                 g_free(msg);
00750                                 WSACloseEvent(nla_event);
00751                                 g_static_mutex_unlock(&mutex);
00752                                 g_thread_exit(NULL);
00753                                 return NULL;
00754                         }
00755                 }
00756                 g_static_mutex_unlock(&mutex);
00757 
00758                 memset(&completion, 0, sizeof(WSACOMPLETION));
00759                 completion.Type = NSP_NOTIFY_EVENT;
00760                 overlapped.hEvent = nla_event;
00761                 completion.Parameters.Event.lpOverlapped = &overlapped;
00762 
00763                 if (MyWSANSPIoctl(network_change_handle, SIO_NSP_NOTIFY_CHANGE, NULL, 0, NULL, 0, &retLen, &completion) == SOCKET_ERROR) {
00764                         int errorid = WSAGetLastError();
00765                         if (errorid == WSA_INVALID_HANDLE) {
00766                                 purple_timeout_add(0, _print_debug_msg,
00767                                                                    g_strdup("Invalid NLA handle; resetting.\n"));
00768                                 g_static_mutex_lock(&mutex);
00769                                 retval = WSALookupServiceEnd(network_change_handle);
00770                                 network_change_handle = NULL;
00771                                 g_static_mutex_unlock(&mutex);
00772                                 continue;
00773                         /* WSA_IO_PENDING indicates successful async notification will happen */
00774                         } else if (errorid != WSA_IO_PENDING) {
00775                                 gchar *msg = g_win32_error_message(errorid);
00776                                 purple_timeout_add(0, _print_debug_msg,
00777                                                                    g_strdup_printf("Unable to wait for changes. Message: %s (%d).\n",
00778                                                                                                    msg, errorid));
00779                                 g_free(msg);
00780                         }
00781                 }
00782 
00783                 /* Make sure at least 30 seconds have elapsed since the last
00784                  * notification so we don't peg the cpu if this keeps changing. */
00785                 if ((time(NULL) - last_trigger) < 30)
00786                         Sleep(30000);
00787 
00788                 /* This will block until NLA notifies us */
00789                 retval = WaitForSingleObjectEx(nla_event, WSA_INFINITE, TRUE);
00790 
00791                 last_trigger = time(NULL);
00792 
00793                 g_static_mutex_lock(&mutex);
00794                 if (network_initialized == FALSE) {
00795                         /* Time to die */
00796                         WSACloseEvent(nla_event);
00797                         g_static_mutex_unlock(&mutex);
00798                         g_thread_exit(NULL);
00799                         return NULL;
00800                 }
00801 
00802                 size = sizeof(buf);
00803                 while ((retval = WSALookupServiceNext(network_change_handle, 0, &size, res)) == ERROR_SUCCESS) {
00804                         /*purple_timeout_add(0, _print_debug_msg,
00805                                                            g_strdup_printf("thread found network '%s'\n",
00806                                                                                            res->lpszServiceInstanceName ? res->lpszServiceInstanceName : "(NULL)"));*/
00807                         size = sizeof(buf);
00808                 }
00809 
00810                 WSAResetEvent(nla_event);
00811                 g_static_mutex_unlock(&mutex);
00812 
00813                 purple_timeout_add(0, wpurple_network_change_thread_cb, NULL);
00814         }
00815 
00816         g_thread_exit(NULL);
00817         return NULL;
00818 }

00963 static void
00964 purple_network_ip_lookup_cb(GSList *hosts, gpointer data,
00965         const char *error_message)
00966 {
00967         const gchar **ip = (const gchar **) data;
00968 
00969         if (error_message) {
00970                 purple_debug_error("network", "lookup of IP address failed: %s\n",
00971                         error_message);
00972                 g_slist_free(hosts);
00973                 return;
00974         }
00975 
00976         if (hosts && g_slist_next(hosts)) {
00977                 struct sockaddr *addr = g_slist_next(hosts)->data;
00978                 char dst[INET6_ADDRSTRLEN];
00979 
00980                 if (addr->sa_family == AF_INET6) {
00981                         inet_ntop(addr->sa_family, &((struct sockaddr_in6 *) addr)->sin6_addr,
00982                                 dst, sizeof(dst));
00983                 } else {
00984                         inet_ntop(addr->sa_family, &((struct sockaddr_in *) addr)->sin_addr,
00985                                 dst, sizeof(dst));
00986                 }
00987 
00988                 *ip = g_strdup(dst);
00989                 purple_debug_info("network", "set IP address: %s\n", *ip);
00990         }
00991 
00992         while (hosts != NULL) {
00993                 hosts = g_slist_delete_link(hosts, hosts);
00994                 /* Free the address */
00995                 g_free(hosts->data);
00996                 hosts = g_slist_delete_link(hosts, hosts);
00997         }
00998 }
