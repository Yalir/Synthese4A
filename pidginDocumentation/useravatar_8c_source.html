00216 do_got_own_avatar_cb(JabberStream *js, const char *from, xmlnode *items)
00217 {
00218         xmlnode *item = NULL, *metadata = NULL, *info = NULL;
00219         PurpleAccount *account = purple_connection_get_account(js->gc);
00220         const char *server_hash = NULL;
00221 
00222         if (items && (item = xmlnode_get_child(items, "item")) &&
00223                         (metadata = xmlnode_get_child(item, "metadata")) &&
00224                         (info = xmlnode_get_child(metadata, "info"))) {
00225                 server_hash = xmlnode_get_attrib(info, "id");
00226         }
00227 
00228         /*
00229          * If we have an avatar and the server returned an error/malformed data,
00230          * push our avatar. If the server avatar doesn't match the local one, push
00231          * our avatar.
00232          */
00233         if ((!items || !metadata) ||
00234                         !purple_strequal(server_hash, js->initial_avatar_hash)) {
00235                 PurpleStoredImage *img = purple_buddy_icons_find_account_icon(account);
00236                 jabber_avatar_set(js, img);
00237                 purple_imgstore_unref(img);
00238         }

00319 update_buddy_metadata(JabberStream *js, const char *from, xmlnode *items)
00320 {
00321         PurpleBuddy *buddy = purple_find_buddy(purple_connection_get_account(js->gc), from);
00322         const char *checksum;
00323         xmlnode *item, *metadata;
00324         if(!buddy)
00325                 return;
00326 
00327         if (!items)
00328                 return;
00329 
00330         item = xmlnode_get_child(items,"item");
00331         if (!item)
00332                 return;
00333 
00334         metadata = xmlnode_get_child(item, "metadata");
00335         if(!metadata)
00336                 return;
00337 
00338         checksum = purple_buddy_icons_get_checksum_for_user(buddy);
00339 
00340         /* <stop/> was the pre-v1.1 method of publishing an empty avatar */
00341         if(xmlnode_get_child(metadata, "stop")) {
00342                 purple_buddy_icons_set_for_user(purple_connection_get_account(js->gc), from, NULL, 0, NULL);
00343         } else {
00344                 xmlnode *info, *goodinfo = NULL;
00345                 gboolean has_children = FALSE;
00346 
00347                 /* iterate over all info nodes to get one we can use */
00348                 for(info = metadata->child; info; info = info->next) {
00349                         if(info->type == XMLNODE_TYPE_TAG)
00350                                 has_children = TRUE;
00351                         if(info->type == XMLNODE_TYPE_TAG && !strcmp(info->name,"info")) {
00352                                 const char *type = xmlnode_get_attrib(info,"type");
00353                                 const char *id = xmlnode_get_attrib(info,"id");
00354 
00355                                 if(checksum && id && !strcmp(id, checksum)) {
00356                                         /* we already have that avatar, so we don't have to do anything */
00357                                         goodinfo = NULL;
00358                                         break;
00359                                 }
00360                                 /* We'll only pick the png one for now. It's a very nice image format anyways. */
00361                                 if(type && id && !goodinfo && !strcmp(type, "image/png"))
00362                                         goodinfo = info;
00363                         }
00364                 }
00365                 if(has_children == FALSE) {
00366                         purple_buddy_icons_set_for_user(purple_connection_get_account(js->gc), from, NULL, 0, NULL);
00367                 } else if(goodinfo) {
00368                         const char *url = xmlnode_get_attrib(goodinfo, "url");
00369                         const char *id = xmlnode_get_attrib(goodinfo,"id");
00370 
00371                         /* the avatar might either be stored in a pep node, or on a HTTP(S) URL */
00372                         if(!url) {
00373                                 jabber_pep_request_item(js, from, NS_AVATAR_1_1_DATA, id,
00374                                                         do_buddy_avatar_update_data);
00375                         } else {
00376                                 PurpleUtilFetchUrlData *url_data;
00377                                 JabberBuddyAvatarUpdateURLInfo *info = g_new0(JabberBuddyAvatarUpdateURLInfo, 1);
00378                                 info->js = js;

00380                                 url_data = purple_util_fetch_url_len(url, TRUE, NULL, TRUE,
00381                                                                                   MAX_HTTP_BUDDYICON_BYTES,
00382                                                                                   do_buddy_avatar_update_fromurl, info);
