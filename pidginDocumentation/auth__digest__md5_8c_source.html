00047 GHashTable* jabber_auth_digest_md5_parse(const char *challenge)
00048 {
00049         const char *token_start, *val_start, *val_end, *cur;
00050         GHashTable *ret = g_hash_table_new_full(g_str_hash, g_str_equal,
00051                         g_free, g_free);
00052 
00053         cur = challenge;
00054         while(*cur != '\0') {
00055                 /* Find the end of the token */
00056                 gboolean in_quotes = FALSE;
00057                 char *name, *value = NULL;
00058                 token_start = cur;
00059                 while(*cur != '\0' && (in_quotes || (!in_quotes && *cur != ','))) {
00060                         if (*cur == '"')
00061                                 in_quotes = !in_quotes;
00062                         cur++;
00063                 }
00064 
00065                 /* Find start of value.  */
00066                 val_start = strchr(token_start, '=');
00067                 if (val_start == NULL || val_start > cur)
00068                         val_start = cur;
00069 
00070                 if (token_start != val_start) {
00071                         name = g_strndup(token_start, val_start - token_start);
00072 
00073                         if (val_start != cur) {
00074                                 val_start++;
00075                                 while (val_start != cur && (*val_start == ' ' || *val_start == '\t'
00076                                                 || *val_start == '\r' || *val_start == '\n'
00077                                                 || *val_start == '"'))
00078                                         val_start++;
00079 
00080                                 val_end = cur;
00081                                 while (val_end >= val_start && (*val_end == ' ' || *val_end == ',' || *val_end == '\t'
00082                                                 || *val_end == '\r' || *val_end == '\n'
00083                                                 || *val_end == '"'  || *val_end == '\0'))
00084                                         val_end--;
00085 
00086                                 if (val_end - val_start + 1 >= 0)
00087                                         value = g_strndup(val_start, val_end - val_start + 1);
00088                         }
00089 
00090                         g_hash_table_replace(ret, name, value);
00091                 }
00092 
00093                 /* Find the start of the next token, if there is one */
00094                 if (*cur != '\0') {
00095                         cur++;
00096                         while (*cur == ' ' || *cur == ',' || *cur == '\t'
00097                                         || *cur == '\r' || *cur == '\n')
00098                                 cur++;
00099                 }
00100         }
00101 
00102         return ret;
00103 }

00117                                         NULL, NULL, NULL)) == NULL) {

00121                                                 "utf-8", NULL, NULL, NULL)) == NULL)) {
00122                 convpasswd = g_strdup(passwd);

00125         cipher = purple_ciphers_find_cipher("md5");

00130         purple_cipher_context_digest(context, sizeof(result), result, NULL);
00131 
00132         a1 = g_strdup_printf("xxxxxxxxxxxxxxxx:%s:%s", nonce, cnonce);
00133         a1len = strlen(a1);

00139 
00140         ha1 = purple_base16_encode(result, 16);
00141 
00142         purple_cipher_context_reset(context, NULL);
00143         purple_cipher_context_append(context, (const guchar *)a2, strlen(a2));
00144         purple_cipher_context_digest(context, sizeof(result), result, NULL);
00145 
00146         ha2 = purple_base16_encode(result, 16);
00147

00152         purple_cipher_context_digest(context, sizeof(result), result, NULL);

00173         char *enc_in = xmlnode_get_data(packet);

00178 
00179         if (!enc_in) {
00180                 *msg = g_strdup(_("Invalid response from server"));
00181                 return JABBER_SASL_STATE_FAIL;
00182         }
00183 
00184         dec_in = (char *)purple_base64_decode(enc_in, NULL);
00185         purple_debug_misc("jabber", "decoded challenge (%"
00186                         G_GSIZE_FORMAT "): %s\n",
00187                         dec_in != NULL ? strlen(dec_in) : 0,
00188                         dec_in != NULL  ? dec_in : "(null)");
00189 
00190         parts = jabber_auth_digest_md5_parse(dec_in);
00191 
00192         if (g_hash_table_lookup(parts, "rspauth")) {
00193                 char *rspauth = g_hash_table_lookup(parts, "rspauth");
00194                 char *expected_rspauth = js->auth_mech_data;
00195 
00196                 if (rspauth && purple_strequal(rspauth, expected_rspauth)) {
00197                         reply = xmlnode_new("response");
00198                         xmlnode_set_namespace(reply, NS_XMPP_SASL);
00199                 } else {
00200                         *msg = g_strdup(_("Invalid challenge from server"));
00201                         state = JABBER_SASL_STATE_FAIL;
00202                 }
00203                 g_free(js->auth_mech_data);
00204                 js->auth_mech_data = NULL;
00205         } else {
00206                 /* assemble a response, and send it */
00207                 /* see RFC 2831 */
00208                 char *realm;
00209                 char *nonce;
00210 
00211                 /* Make sure the auth string contains everything that should be there.
00212                    This isn't everything in RFC2831, but it is what we need. */
00213 
00214                 nonce = g_hash_table_lookup(parts, "nonce");
00215 
00216                 /* we're actually supposed to prompt the user for a realm if
00217                  * the server doesn't send one, but that really complicates things,
00218                  * so i'm not gonna worry about it until is poses a problem to
00219                  * someone, or I get really bored */
00220                 realm = g_hash_table_lookup(parts, "realm");
00221                 if(!realm)
00222                         realm = js->user->domain;
00223 
00224                 if (nonce == NULL || realm == NULL) {
00225                         *msg = g_strdup(_("Invalid challenge from server"));
00226                         state = JABBER_SASL_STATE_FAIL;
00227                 } else {
00228                         GString *response = g_string_new("");
00229                         char *a2;
00230                         char *auth_resp;
00231                         char *cnonce;
00232 
00233                         cnonce = g_strdup_printf("%x%u%x", g_random_int(), (int)time(NULL),
00234                                         g_random_int());
00235 
00236                         a2 = g_strdup_printf("AUTHENTICATE:xmpp/%s", realm);
00237                         auth_resp = generate_response_value(js->user,
00238                                         purple_connection_get_password(js->gc), nonce, cnonce, a2, realm);
00239                         g_free(a2);
00240 
00241                         a2 = g_strdup_printf(":xmpp/%s", realm);
00242                         js->auth_mech_data = generate_response_value(js->user,
00243                                         purple_connection_get_password(js->gc), nonce, cnonce, a2, realm);
00244                         g_free(a2);
00245 
00246                         g_string_append_printf(response, "username=\"%s\"", js->user->node);
00247                         g_string_append_printf(response, ",realm=\"%s\"", realm);
00248                         g_string_append_printf(response, ",nonce=\"%s\"", nonce);
00249                         g_string_append_printf(response, ",cnonce=\"%s\"", cnonce);
00250                         g_string_append_printf(response, ",nc=00000001");
00251                         g_string_append_printf(response, ",qop=auth");
00252                         g_string_append_printf(response, ",digest-uri=\"xmpp/%s\"", realm);
00253                         g_string_append_printf(response, ",response=%s", auth_resp);
00254                         g_string_append_printf(response, ",charset=utf-8");
00255 
00256                         g_free(auth_resp);
00257                         g_free(cnonce);
00258 
00259                         enc_out = purple_base64_encode((guchar *)response->str, response->len);
00260 
00261                         purple_debug_misc("jabber", "decoded response (%"
00262                                         G_GSIZE_FORMAT "): %s\n",
00263                                         response->len, response->str);
00264 
00265                         reply = xmlnode_new("response");
00266                         xmlnode_set_namespace(reply, NS_XMPP_SASL);
00267                         xmlnode_insert_data(reply, enc_out, -1);
00268 
00269                         g_free(enc_out);
00270 
00271                         g_string_free(response, TRUE);
00272                 }
00273         }
00274 
00275         g_free(enc_in);
