00085 static void
00086 file_recv_request_cb(PurpleXfer *xfer, gpointer handle)
00087 {
00088         PurpleAccount *account;
00089         PurpleBlistNode *node;
00090         const char *pref;
00091         char *filename;
00092         char *dirname;
00093 
00094     int accept_setting;
00095 
00096         account = xfer->account;
00097         node = PURPLE_BLIST_NODE(purple_find_buddy(account, xfer->who));
00098 
00099         /* If person is on buddy list, use the buddy setting; otherwise, use the
00100            stranger setting. */
00101         if (node) {
00102                 node = purple_blist_node_get_parent(node);
00103                 g_return_if_fail(PURPLE_BLIST_NODE_IS_CONTACT(node));
00104                 accept_setting = purple_blist_node_get_int(node, "autoaccept");
00105         } else {
00106                 accept_setting = purple_prefs_get_int(PREF_STRANGER);
00107         }
00108 
00109         switch (accept_setting)
00110         {
00111                 case FT_ASK:
00112                         break;
00113                 case FT_ACCEPT:
00114             pref = purple_prefs_get_string(PREF_PATH);
00115                         if (ensure_path_exists(pref))
00116                         {
00117                                 int count = 1;
00118                                 const char *escape;
00119                                 gchar **name_and_ext;
00120                                 const gchar *name;
00121                                 gchar *ext;
00122 
00123                                 if (purple_prefs_get_bool(PREF_NEWDIR))
00124                                         dirname = g_build_filename(pref, purple_normalize(account, xfer->who), NULL);
00125                                 else
00126                                         dirname = g_build_filename(pref, NULL);
00127 
00128                                 if (!ensure_path_exists(dirname))
00129                                 {
00130                                         g_free(dirname);
00131                                         break;
00132                                 }
00133 
00134                                 /* Escape filename (if escaping is turned on) */
00135                                 if (purple_prefs_get_bool(PREF_ESCAPE)) {
00136                                         escape = purple_escape_filename(xfer->filename);
00137                                 } else {
00138                                         escape = xfer->filename;
00139                                 }
00140                                 filename = g_build_filename(dirname, escape, NULL);
00141 
00142                                 /* Split at the first dot, to avoid uniquifying "foo.tar.gz" to "foo.tar-2.gz" */
00143                                 name_and_ext = g_strsplit(escape, ".", 2);
00144                                 name = name_and_ext[0];
00145                                 g_return_if_fail(name != NULL);
00146                                 if (name_and_ext[1] != NULL) {
00147                                         /* g_strsplit does not include the separator in each chunk. */
00148                                         ext = g_strdup_printf(".%s", name_and_ext[1]);
00149                                 } else {
00150                                         ext = g_strdup("");
00151                                 }
00152 
00153                                 /* Make sure the file doesn't exist. Do we want some better checking than this? */
00154                                 /* FIXME: There is a race here: if the newly uniquified file name gets created between
00155                                  *        this g_file_test and the transfer starting, the file created in the meantime
00156                                  *        will be clobbered. But it's not at all straightforward to fix.
00157                                  */
00158                                 while (g_file_test(filename, G_FILE_TEST_EXISTS)) {
00159                                         char *file = g_strdup_printf("%s-%d%s", name, count++, ext);
00160                                         g_free(filename);
00161                                         filename = g_build_filename(dirname, file, NULL);
00162                                         g_free(file);
00163                                 }
00164 
00165                                 purple_xfer_request_accepted(xfer, filename);
00166 
00167                                 g_strfreev(name_and_ext);
00168                                 g_free(ext);
00169                                 g_free(dirname);
00170                                 g_free(filename);
00171                         }
00172 
00173                         purple_signal_connect(purple_xfers_get_handle(), "file-recv-complete", handle,
00174                                                                 PURPLE_CALLBACK(auto_accept_complete_cb), xfer);
00175                         break;
00176                 case FT_REJECT:
00177                         xfer->status = PURPLE_XFER_STATUS_CANCEL_LOCAL;
00178                         break;
00179         }
00180 }

00230 static gboolean
00231 plugin_load(PurplePlugin *plugin)
00232 {
00233         /* migrate the old pref (we should only care if the plugin is actually *used*) */
00234         /*
00235          * TODO: We should eventually call purple_prefs_remove(PREFS_STRANGER_OLD)
00236          *       to clean up after ourselves, but we don't want to do it yet
00237          *       so that we don't break users who share a .purple directory
00238          *       between old libpurple clients and new libpurple clients.
00239          *                                             --Mark Doliner, 2011-01-03
00240          */
00241         if (!purple_prefs_exists(PREF_STRANGER)) {
00242                 if (purple_prefs_get_bool(PREF_STRANGER_OLD))
00243                         purple_prefs_add_int(PREF_STRANGER, FT_REJECT);
00244                 else
00245                         purple_prefs_set_int(PREF_STRANGER, FT_ASK);
00246         }
00247 
00248         purple_signal_connect(purple_xfers_get_handle(), "file-recv-request", plugin,
00249                                                 PURPLE_CALLBACK(file_recv_request_cb), plugin);
00250         purple_signal_connect(purple_blist_get_handle(), "blist-node-extended-menu", plugin,
00251                                                 PURPLE_CALLBACK(context_menu), plugin);
00252         return TRUE;
00253 }
