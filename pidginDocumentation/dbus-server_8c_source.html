00233                 DBusError *error, int first_arg_type, va_list var_args)
00234 {
00235         int spec_type, msg_type, i;
00236 
00237         spec_type = first_arg_type;
00238 
00239         for (i = 0; spec_type != DBUS_TYPE_INVALID; i++)
00240         {
00241                 msg_type = dbus_message_iter_get_arg_type(iter);
00242 
00243                 if (msg_type != spec_type)
00244                 {
00245                         dbus_set_error(error, DBUS_ERROR_INVALID_ARGS,
00246                                         "Argument %d is specified to be of type \"%i\", but "
00247                                         "is actually of type \"%i\"\n", i,
00248                                         spec_type, msg_type);
00249                         return FALSE;
00250                 }
00251 
00252                 if (!TYPE_IS_CONTAINER(spec_type))
00253                 {
00254                         gpointer ptr;
00255                         ptr = va_arg (var_args, gpointer);
00256                         dbus_message_iter_get_basic(iter, ptr);
00257                 }
00258                 else
00259                 {
00260                         DBusMessageIter *sub;
00261                         sub = va_arg (var_args, DBusMessageIter*);
00262                         dbus_message_iter_recurse(iter, sub);
00263                         purple_debug_info("dbus", "subiter %p:%p\n", sub, * (gpointer*) sub);
00264                         break; /* for testing only! */
00265                 }
00266 
00267                 spec_type = va_arg(var_args, int);
00268                 if (!dbus_message_iter_next(iter) && spec_type != DBUS_TYPE_INVALID)
00269                 {
00270                         dbus_set_error (error, DBUS_ERROR_INVALID_ARGS,
00271                                         "Message has only %d arguments, but more were expected", i);
00272                         return FALSE;
00273                 }
00274         }

00374 GHashTable *
00375 purple_dbus_iter_hash_table(DBusMessageIter *iter, DBusError *error)
00376 {
00377         GHashTable *hash;
00378 
00379         /* we do not need to destroy strings because they are part of the message */
00380         hash = g_hash_table_new(g_str_hash, g_str_equal);
00381 
00382         do {
00383                 char *key, *value;
00384                 DBusMessageIter subiter;
00385 
00386                 if (dbus_message_iter_get_arg_type(iter) != DBUS_TYPE_DICT_ENTRY)
00387                         goto error;
00388                         /* With all due respect to Dijkstra,
00389                          * this goto is for exception
00390                          * handling, and it is ok because it
00391                          * avoids duplication of the code
00392                          * responsible for destroying the hash
00393                          * table.  Exceptional instructions
00394                          * for exceptional situations.
00395                          */
00396 
00397                 dbus_message_iter_recurse(iter, &subiter);
00398                 if (!purple_dbus_message_iter_get_args(&subiter, error,
00399                                 DBUS_TYPE_STRING, &key,
00400                                 DBUS_TYPE_STRING, &value,
00401                                 DBUS_TYPE_INVALID))
00402                         goto error; /* same here */
00403 
00404                 g_hash_table_insert(hash, key, value);
00405         } while (dbus_message_iter_next(iter));
00406 
00407         return hash;
00408 
00409 error:
00410         g_hash_table_destroy(hash);
00411         return NULL;
00412 }

00492 {

00494         GString *str;
00495         GList *bindings_list, *node;
00496         const char *signals;
00497         const char *type;
00498         const char *pointer_type;
00499 
00500         str = g_string_sized_new(0x1000); /* TODO: why this size? */
00501 
00502         g_string_append(str, "<!DOCTYPE node PUBLIC '-//freedesktop//DTD D-BUS Object Introspection 1.0//EN' 'http://www.freedesktop.org/standards/dbus/1.0/introspect.dtd'>\n");

00504         g_string_append(str, "  <interface name='org.freedesktop.DBus.Introspectable'>\n    <method name='Introspect'>\n      <arg name='data' direction='out' type='s'/>\n    </method>\n  </interface>\n\n");
00505 
00506         g_string_append_printf(str, "  <interface name='%s'>\n", DBUS_INTERFACE_PURPLE);
00507

00526                         {
00527                                 const char *name, *direction, *type;

00535                                                 name, type, direction);

00665 purple_dbus_convert_signal_name(const char *purple_name)
00666 {
00667         int purple_index, g_index;
00668         char *g_name = g_new(char, strlen(purple_name) + 1);
00669         gboolean capitalize_next = TRUE;
00670 
00671         for (purple_index = g_index = 0; purple_name[purple_index]; purple_index++)
00672                 if (purple_name[purple_index] != '-' && purple_name[purple_index] != '_')
00673                 {
00674                         if (capitalize_next)
00675                                 g_name[g_index++] = g_ascii_toupper(purple_name[purple_index]);
00676                         else
00677                                 g_name[g_index++] = purple_name[purple_index];
00678                         capitalize_next = FALSE;

00680                         capitalize_next = TRUE;

00689 static gboolean
00690 purple_dbus_message_append_purple_values(DBusMessageIter *iter,
00691                 int number, PurpleValue **purple_values, va_list data)
00692 {
00693         int i;
00694         gboolean error = FALSE;
00695 
00696         for (i = 0; i < number; i++)
00697         {
00698                 const char *str;
00699                 int id;
00700                 gint xint;
00701                 guint xuint;
00702                 gint64 xint64;
00703                 guint64 xuint64;
00704                 gboolean xboolean;
00705                 gpointer ptr = NULL;
00706                 gpointer val;
00707 
00708                 if (purple_value_is_outgoing(purple_values[i]))
00709                 {
00710                         ptr = my_arg(gpointer);
00711                         g_return_val_if_fail(ptr, TRUE);
00712                 }
00713 
00714                 switch (purple_values[i]->type)
00715                 {
00716                 case PURPLE_TYPE_INT:
00717                 case PURPLE_TYPE_ENUM:
00718                         xint = my_arg(gint);
00719                         dbus_message_iter_append_basic(iter, DBUS_TYPE_INT32, &xint);
00720                         break;
00721                 case PURPLE_TYPE_UINT:
00722                         xuint = my_arg(guint);
00723                         dbus_message_iter_append_basic(iter, DBUS_TYPE_UINT32, &xuint);
00724                         break;
00725                 case PURPLE_TYPE_INT64:
00726                         xint64 = my_arg(gint64);
00727                         dbus_message_iter_append_basic(iter, DBUS_TYPE_INT64, &xint64);
00728                         break;
00729                 case PURPLE_TYPE_UINT64:
00730                         xuint64 = my_arg(guint64);
00731                         dbus_message_iter_append_basic(iter, DBUS_TYPE_UINT64, &xuint64);
00732                         break;
00733                 case PURPLE_TYPE_BOOLEAN:
00734                         xboolean = my_arg(gboolean);
00735                         dbus_message_iter_append_basic(iter, DBUS_TYPE_BOOLEAN, &xboolean);
00736                         break;
00737                 case PURPLE_TYPE_STRING:
00738                         str = null_to_empty(my_arg(char*));
00739                         if (!g_utf8_validate(str, -1, NULL)) {
00740                                 gchar *tmp;
00741                                 purple_debug_error("dbus", "Invalid UTF-8 string passed to signal, emitting salvaged string!\n");
00742                                 tmp = purple_utf8_salvage(str);
00743                                 dbus_message_iter_append_basic(iter, DBUS_TYPE_STRING, &tmp);
00744                                 g_free(tmp);
00745                         } else {
00746                                 dbus_message_iter_append_basic(iter, DBUS_TYPE_STRING, &str);
00747                         }
00748                         break;
00749                 case PURPLE_TYPE_SUBTYPE: /* registered pointers only! */
00750                 case PURPLE_TYPE_POINTER:
00751                 case PURPLE_TYPE_OBJECT:
00752                 case PURPLE_TYPE_BOXED:
00753                         val = my_arg(gpointer);
00754                         id = purple_dbus_pointer_to_id(val);
00755                         if (id == 0 && val != NULL)
00756                                 error = TRUE;      /* Some error happened. */
00757                         dbus_message_iter_append_basic(iter,
00758                                         (sizeof(id) == sizeof(dbus_int32_t)) ? DBUS_TYPE_INT32 : DBUS_TYPE_INT64, &id);
00759                         break;
00760                 default: /* no conversion implemented */
00761                         g_return_val_if_reached(TRUE);
00762                 }
00763         }
00764         return error;
00765 }

00771                 PurpleValue **values, va_list vargs)

00774         DBusMessageIter iter;

00776

00780         if (purple_dbus_connection == NULL)

00785         /*
00786          * The test below is a hack that prevents our "dbus-method-called"
