00150 jabber_adhoc_parse(JabberStream *js, const char *from,
00151                    JabberIqType type, const char *id,
00152                    xmlnode *packet, gpointer data)
00153 {
00154         xmlnode *command = xmlnode_get_child_with_namespace(packet, "command", "http://jabber.org/protocol/commands");
00155         const char *status = xmlnode_get_attrib(command,"status");
00156         xmlnode *xdata = xmlnode_get_child_with_namespace(command,"x","jabber:x:data");
00157 
00158         if (type == JABBER_IQ_ERROR) {
00159                 char *msg = jabber_parse_error(js, packet, NULL);
00160                 if(!msg)
00161                         msg = g_strdup(_("Unknown Error"));
00162 
00163                 purple_notify_error(NULL, _("Ad-Hoc Command Failed"),
00164                                                         _("Ad-Hoc Command Failed"), msg);
00165                 g_free(msg);
00166                 return;
00167         }
00168 
00169         if(!status)
00170                 return;
00171 
00172         if(!strcmp(status,"completed")) {
00173                 /* display result */
00174                 xmlnode *note = xmlnode_get_child(command,"note");
00175 
00176                 if(note) {
00177                         char *data = xmlnode_get_data(note);
00178                         purple_notify_info(NULL, from, data, NULL);
00179                         g_free(data);
00180                 }
00181 
00182                 if(xdata)
00183                         jabber_x_data_request(js, xdata, (jabber_x_data_cb)do_adhoc_ignoreme, NULL);
00184                 return;
00185         }
00186         if(!strcmp(status,"executing")) {
00187                 /* this command needs more steps */
00188                 xmlnode *actions, *action;
00189                 int actionindex = 0;
00190                 GList *actionslist = NULL;
00191                 JabberAdHocActionInfo *actionInfo;
00192                 if(!xdata)
00193                         return; /* shouldn't happen */
00194 
00195                 actions = xmlnode_get_child(command,"actions");
00196                 if(!actions) {
00197                         JabberXDataAction *defaultaction = g_new0(JabberXDataAction, 1);
00198                         defaultaction->name = g_strdup(_("execute"));
00199                         defaultaction->handle = g_strdup("execute");
00200                         actionslist = g_list_append(actionslist, defaultaction);
00201                 } else {
00202                         const char *defaultactionhandle = xmlnode_get_attrib(actions, "execute");
00203                         int index = 0;
00204                         for(action = actions->child; action; action = action->next, ++index) {
00205                                 if(action->type == XMLNODE_TYPE_TAG) {
00206                                         JabberXDataAction *newaction = g_new0(JabberXDataAction, 1);
00207                                         newaction->name = g_strdup(_(action->name));
00208                                         newaction->handle = g_strdup(action->name);
00209                                         actionslist = g_list_append(actionslist, newaction);
00210                                         if(defaultactionhandle && !strcmp(defaultactionhandle, action->name))
00211                                                 actionindex = index;
00212                                 }
00213                         }
00214                 }
00215 
00216                 actionInfo = g_new0(JabberAdHocActionInfo, 1);
00217                 actionInfo->sessionid = g_strdup(xmlnode_get_attrib(command,"sessionid"));
00218                 actionInfo->who = g_strdup(from);
00219                 actionInfo->node = g_strdup(xmlnode_get_attrib(command,"node"));
00220                 actionInfo->actionslist = actionslist;
00221 
00222                 jabber_x_data_request_with_actions(js,xdata,actionslist,actionindex,do_adhoc_action_cb,actionInfo);
