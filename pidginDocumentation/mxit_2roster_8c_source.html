00356 /*------------------------------------------------------------------------
00357  * A contact update packet was received from the MXit server, so update the buddy's
00358  * information.
00359  *
00360  *  @param session              The MXit session object
00361  *  @param contact              The contact
00362  */
00363 void mxit_update_contact( struct MXitSession* session, struct contact* contact )
00364 {
00365         PurpleBuddy*            buddy   = NULL;
00366         PurpleGroup*            group   = NULL;
00367         const char*                     id              = NULL;
00368 
00369         purple_debug_info( MXIT_PLUGIN_ID, "mxit_update_contact: user='%s' alias='%s' group='%s'\n", contact->username, contact->alias, contact->groupname );
00370 
00371         /*
00372          * libPurple requires all contacts to be in a group.
00373          * So if this MXit contact isn't in a group, pretend it is.
00374          */
00375         if ( *contact->groupname == '\0' ) {
00376                 g_strlcpy( contact->groupname, MXIT_DEFAULT_GROUP, sizeof( contact->groupname ) );
00377         }
00378 
00379         /* find or create a group for this contact */
00380         group = purple_find_group( contact->groupname );
00381         if ( !group )
00382                 group = purple_group_new( contact->groupname );
00383 
00384         /* see if the buddy is not in the group already */
00385         buddy = purple_find_buddy_in_group( session->acc, contact->username, group );
00386         if ( !buddy ) {
00387                 /* buddy not found in the group */
00388 
00389                 /* lets try finding him in all groups */
00390                 buddy = purple_find_buddy( session->acc, contact->username );
00391                 if ( buddy ) {
00392                         /* ok, so we found him in another group. to switch him between groups we must delete him and add him again. */
00393                         purple_blist_remove_buddy( buddy );
00394                         buddy = NULL;
00395                 }
00396 
00397                 /* create new buddy */
00398                 buddy = purple_buddy_new( session->acc, contact->username, contact->alias );
00399                 purple_buddy_set_protocol_data(buddy, contact);
00400 
00401                 /* add new buddy to list */
00402                 purple_blist_add_buddy( buddy, NULL, group, NULL );
00403         }
00404         else {
00405                 /* buddy was found in the group */
00406 
00407                 gpointer data = NULL;
00408 
00409                 /* now update the buddy's alias */
00410                 purple_blist_alias_buddy( buddy, contact->alias );
00411 
00412                 /* replace the buddy's contact struct */
00413                 if ( ( data = purple_buddy_get_protocol_data( buddy ) ) )
00414                         free( data );
00415                 purple_buddy_set_protocol_data( buddy, contact );
00416         }
00417 
00418         /* load buddy's avatar id */
00419         id = purple_buddy_icons_get_checksum_for_user( buddy );
00420         if ( id )
00421                 contact->avatarId = g_strdup( id );
00422         else
00423                 contact->avatarId = NULL;
00424 
00425         /* update the buddy's status (reference: "libpurple/prpl.h") */
00426         purple_prpl_got_user_status( session->acc, contact->username, mxit_statuses[contact->presence].id, NULL );
00427 
00428         /* update the buddy's mood */
00429         if ( contact->mood == MXIT_MOOD_NONE )
00430                 purple_prpl_got_user_status_deactive( session->acc, contact->username, "mood" );
00431         else
00432                 purple_prpl_got_user_status( session->acc, contact->username, "mood", PURPLE_MOOD_NAME, mxit_moods[contact->mood-1].mood, NULL );

00721 /*------------------------------------------------------------------------
00722  * The user has added a buddy to the list, so send an invite request.
00723  *
00724  *  @param gc           The connection object
00725  *  @param buddy        The new buddy
00726  *  @param group        The group of the new buddy
00727  *  @param message      The invite message
00728  */
00729 void mxit_add_buddy( PurpleConnection* gc, PurpleBuddy* buddy, PurpleGroup* group, const char* message )
00730 {
00731         struct MXitSession*     session = (struct MXitSession*) gc->proto_data;
00732         GSList*                         list    = NULL;
00733         PurpleBuddy*            mxbuddy = NULL;
00734         unsigned int            i;
00735         const gchar *           buddy_name = purple_buddy_get_name( buddy );
00736         const gchar *           buddy_alias = purple_buddy_get_alias( buddy );
00737         const gchar *           group_name = purple_group_get_name( group );
00738 
00739         purple_debug_info( MXIT_PLUGIN_ID, "mxit_add_buddy '%s' (group='%s')\n", buddy_name, group_name );
00740 
00741         list = purple_find_buddies( session->acc, buddy_name );
00742         if ( g_slist_length( list ) == 1 ) {
00743                 purple_debug_info( MXIT_PLUGIN_ID, "mxit_add_buddy (scenario 1) (list:%i)\n", g_slist_length( list ) );
00744                 /*
00745                  * we only send an invite to MXit when the user is not already inside our
00746                  * blist.  this is done because purple does an add_buddy() call when
00747                  * you accept an invite.  so in that case the user is already
00748                  * in our blist and ready to be chatted to.
00749                  */
00750 
00751                 if ( buddy_name[0] == '#' ) {
00752                         gchar *tmp = (gchar*) purple_base64_decode( buddy_name + 1, NULL );
00753                         mxit_send_invite( session, tmp, FALSE, buddy_alias, group_name, message );
00754                         g_free( tmp );
00755                 }
00756                 else
00757                         mxit_send_invite( session, buddy_name, TRUE, buddy_alias, group_name, message );
00758         }
00759         else {
00760                 purple_debug_info( MXIT_PLUGIN_ID, "mxit_add_buddy (scenario 2) (list:%i)\n", g_slist_length( list ) );
00761                 /*
00762                  * we already have the buddy in our list, so we will only update
00763                  * his information here and not send another invite message
00764                  */
00765 
00766                 /* find the correct buddy */
00767                 for ( i = 0; i < g_slist_length( list ); i++ ) {
00768                         mxbuddy = g_slist_nth_data( list, i );
00769 
00770                         if ( purple_buddy_get_protocol_data( mxbuddy ) != NULL ) {
00771                                 /* this is our REAL MXit buddy! */
00772 
00773                                 /* now update the buddy's alias */
00774                                 purple_blist_alias_buddy( mxbuddy, buddy_alias );
00775 
00776                                 /* now update the buddy's group */
00777 //                              mxbuddy = mxit_update_buddy_group( session, mxbuddy, group );
00778 
00779                                 /* send the update to the MXit server */
00780                                 mxit_send_update_contact( session, purple_buddy_get_name( mxbuddy ), purple_buddy_get_alias( mxbuddy ), group_name );
00781                         }
00782                 }
00783         }
00784 
00785         /*
00786          * we remove the buddy here from the buddy list because the MXit server
00787          * will send us a proper contact update packet if this succeeds.  now
00788          * we do not have to worry about error handling in case of adding an
00789          * invalid contact.  so the user will still see the contact as offline
00790          * until he eventually accepts the invite.
00791          */
00792         purple_blist_remove_buddy( buddy );
00793 
00794         g_slist_free( list );
00795 }
