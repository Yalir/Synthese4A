00648         /* This is a hack to work around the fact that marv gets messages
00649          * with null bytes in them while using some weird irc server at work
00650          */
00651         while ((cur < (irc->inbuf + irc->inbufused)) && !*cur)
00652                 cur++;
00653 
00654         while (cur < irc->inbuf + irc->inbufused &&
00655                ((end = strstr(cur, "\r\n")) || (end = strstr(cur, "\n")))) {
00656                 int step = (*end == '\r' ? 2 : 1);
00657                 *end = '\0';
00658                 irc_parse_msg(irc, cur);
00659                 cur = end + step;
00660         }
00661         if (cur != irc->inbuf + irc->inbufused) { /* leftover */
00662                 irc->inbufused -= (cur - irc->inbuf);
00663                 memmove(irc->inbuf, cur, irc->inbufused);
00664         } else {

00666         }
