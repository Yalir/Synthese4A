00078 /* blk0() and blk() perform the initial expand. */
00079 /* I got the idea of expanding during the round function from SSLeay */
00080 #ifndef GG_CONFIG_BIGENDIAN
00081 #define blk0(i) (block->l[i] = (rol(block->l[i],24)&0xFF00FF00) \
00082     |(rol(block->l[i],8)&0x00FF00FF))
00083 #else
00084 #define blk0(i) block->l[i]
00085 #endif
00086 #define blk(i) (block->l[i&15] = rol(block->l[(i+13)&15]^block->l[(i+8)&15] \
00087     ^block->l[(i+2)&15]^block->l[i&15],1))
00088 
00089 /* (R0+R1), R2, R3, R4 are the different operations used in SHA1 */
00090 #define R0(v,w,x,y,z,i) z+=((w&(x^y))^y)+blk0(i)+0x5A827999+rol(v,5);w=rol(w,30);
00091 #define R1(v,w,x,y,z,i) z+=((w&(x^y))^y)+blk(i)+0x5A827999+rol(v,5);w=rol(w,30);
00092 #define R2(v,w,x,y,z,i) z+=(w^x^y)+blk(i)+0x6ED9EBA1+rol(v,5);w=rol(w,30);
00093 #define R3(v,w,x,y,z,i) z+=(((w|x)&y)|(w&x))+blk(i)+0x8F1BBCDC+rol(v,5);w=rol(w,30);
00094 #define R4(v,w,x,y,z,i) z+=(w^x^y)+blk(i)+0xCA62C1D6+rol(v,5);w=rol(w,30);

00099 static void SHA1_Transform(uint32_t state[5], const unsigned char buffer[64])
00100 {
00101 uint32_t a, b, c, d, e;
00102 typedef union {
00103     unsigned char c[64];
00104     uint32_t l[16];
00105 } CHAR64LONG16;
00106 CHAR64LONG16* block;
00107 static unsigned char workspace[64];
00108     block = (CHAR64LONG16*)workspace;
00109     memcpy(block, buffer, 64);
00110     /* Copy context->state[] to working vars */
00111     a = state[0];
00112     b = state[1];
00113     c = state[2];
00114     d = state[3];
00115     e = state[4];
00116     /* 4 rounds of 20 operations each. Loop unrolled. */
00117     R0(a,b,c,d,e, 0); R0(e,a,b,c,d, 1); R0(d,e,a,b,c, 2); R0(c,d,e,a,b, 3);
00118     R0(b,c,d,e,a, 4); R0(a,b,c,d,e, 5); R0(e,a,b,c,d, 6); R0(d,e,a,b,c, 7);
00119     R0(c,d,e,a,b, 8); R0(b,c,d,e,a, 9); R0(a,b,c,d,e,10); R0(e,a,b,c,d,11);
00120     R0(d,e,a,b,c,12); R0(c,d,e,a,b,13); R0(b,c,d,e,a,14); R0(a,b,c,d,e,15);
00121     R1(e,a,b,c,d,16); R1(d,e,a,b,c,17); R1(c,d,e,a,b,18); R1(b,c,d,e,a,19);
00122     R2(a,b,c,d,e,20); R2(e,a,b,c,d,21); R2(d,e,a,b,c,22); R2(c,d,e,a,b,23);
00123     R2(b,c,d,e,a,24); R2(a,b,c,d,e,25); R2(e,a,b,c,d,26); R2(d,e,a,b,c,27);
00124     R2(c,d,e,a,b,28); R2(b,c,d,e,a,29); R2(a,b,c,d,e,30); R2(e,a,b,c,d,31);
00125     R2(d,e,a,b,c,32); R2(c,d,e,a,b,33); R2(b,c,d,e,a,34); R2(a,b,c,d,e,35);
00126     R2(e,a,b,c,d,36); R2(d,e,a,b,c,37); R2(c,d,e,a,b,38); R2(b,c,d,e,a,39);
00127     R3(a,b,c,d,e,40); R3(e,a,b,c,d,41); R3(d,e,a,b,c,42); R3(c,d,e,a,b,43);
00128     R3(b,c,d,e,a,44); R3(a,b,c,d,e,45); R3(e,a,b,c,d,46); R3(d,e,a,b,c,47);
00129     R3(c,d,e,a,b,48); R3(b,c,d,e,a,49); R3(a,b,c,d,e,50); R3(e,a,b,c,d,51);
00130     R3(d,e,a,b,c,52); R3(c,d,e,a,b,53); R3(b,c,d,e,a,54); R3(a,b,c,d,e,55);
00131     R3(e,a,b,c,d,56); R3(d,e,a,b,c,57); R3(c,d,e,a,b,58); R3(b,c,d,e,a,59);
00132     R4(a,b,c,d,e,60); R4(e,a,b,c,d,61); R4(d,e,a,b,c,62); R4(c,d,e,a,b,63);
00133     R4(b,c,d,e,a,64); R4(a,b,c,d,e,65); R4(e,a,b,c,d,66); R4(d,e,a,b,c,67);
00134     R4(c,d,e,a,b,68); R4(b,c,d,e,a,69); R4(a,b,c,d,e,70); R4(e,a,b,c,d,71);
00135     R4(d,e,a,b,c,72); R4(c,d,e,a,b,73); R4(b,c,d,e,a,74); R4(a,b,c,d,e,75);
00136     R4(e,a,b,c,d,76); R4(d,e,a,b,c,77); R4(c,d,e,a,b,78); R4(b,c,d,e,a,79);
00137     /* Add the working vars back into context.state[] */
00138     state[0] += a;
00139     state[1] += b;
00140     state[2] += c;
00141     state[3] += d;
00142     state[4] += e;
00143     /* Wipe variables */
00144     a = b = c = d = e = 0;
00145 }

00164 static void SHA1_Update(SHA_CTX* context, const unsigned char* data, unsigned int len)
00165 {
00166 unsigned int i, j;
00167 
00168     j = (context->count[0] >> 3) & 63;
00169     if ((context->count[0] += len << 3) < (len << 3)) context->count[1]++;
00170     context->count[1] += (len >> 29);
00171     if ((j + len) > 63) {
00172         memcpy(&context->buffer[j], data, (i = 64-j));
00173         SHA1_Transform(context->state, context->buffer);
00174         for ( ; i + 63 < len; i += 64) {
00175             SHA1_Transform(context->state, &data[i]);
00176         }
00177         j = 0;
00178     }
00179     else i = 0;
00180     memcpy(&context->buffer[j], &data[i], len - i);
00181 }

00186 static void SHA1_Final(unsigned char digest[20], SHA_CTX* context)
00187 {
00188 uint32_t i, j;
00189 unsigned char finalcount[8];
00190 
00191     for (i = 0; i < 8; i++) {
00192         finalcount[i] = (unsigned char)((context->count[(i >= 4 ? 0 : 1)]
00193          >> ((3-(i & 3)) * 8) ) & 255);  /* Endian independent */
00194     }
00195     SHA1_Update(context, (unsigned char *)"\200", 1);
00196     while ((context->count[0] & 504) != 448) {
00197         SHA1_Update(context, (unsigned char *)"\0", 1);
00198     }
00199     SHA1_Update(context, finalcount, 8);  /* Should cause a SHA1_Transform() */
00200     for (i = 0; i < 20; i++) {
00201         digest[i] = (unsigned char)
00202          ((context->state[i>>2] >> ((3-(i & 3)) * 8) ) & 255);
00203     }
00204     /* Wipe variables */
00205     i = j = 0;
00206     memset(context->buffer, 0, 64);
00207     memset(context->state, 0, 20);
00208     memset(context->count, 0, 8);
00209     memset(&finalcount, 0, 8);
00210 #ifdef SHA1HANDSOFF  /* make SHA1_Transform overwrite it's own static vars */
00211     SHA1_Transform(context->state, context->buffer);
00212 #endif
00213 }
