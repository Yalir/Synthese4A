02564 {

02566         FILE *file;
02567         size_t real_size, byteswritten;
02568         struct stat st;
02569 #ifndef HAVE_FILENO
02570         int fd;
02571 #endif
02572 
02573         purple_debug_info("util", "Writing file %s\n",
02574                                         filename_full);
02575 
02576         g_return_val_if_fail((size >= -1), FALSE);
02577 
02578         filename_temp = g_strdup_printf("%s.save", filename_full);
02579 
02580         /* Remove an old temporary file, if one exists */
02581         if (g_file_test(filename_temp, G_FILE_TEST_EXISTS))
02582         {
02583                 if (g_unlink(filename_temp) == -1)
02584                 {
02585                         purple_debug_error("util", "Error removing old file "
02586                                            "%s: %s\n",
02587                                            filename_temp, g_strerror(errno));
02588                 }
02589         }
02590 
02591         /* Open file */
02592         file = g_fopen(filename_temp, "wb");
02593         if (file == NULL)
02594         {
02595                 purple_debug_error("util", "Error opening file %s for "
02596                                    "writing: %s\n",
02597                                    filename_temp, g_strerror(errno));
02598                 g_free(filename_temp);
02599                 return FALSE;
02600         }
02601 
02602         /* Write to file */
02603         real_size = (size == -1) ? strlen(data) : (size_t) size;
02604         byteswritten = fwrite(data, 1, real_size, file);
02605 
02606 #ifdef HAVE_FILENO

02608          * guarantee that file data is flushed before file metadata,
02609          * so this procedure is insufficient without some flushage. */
02610         if (fflush(file) < 0) {
02611                 purple_debug_error("util", "Error flushing %s: %s\n",
02612                                    filename_temp, g_strerror(errno));
02613                 g_free(filename_temp);
02614                 fclose(file);

02616         }
02617         if (fsync(fileno(file)) < 0) {
02618                 purple_debug_error("util", "Error syncing file contents for %s: %s\n",
02619                                    filename_temp, g_strerror(errno));
02620                 g_free(filename_temp);
02621                 fclose(file);

02623         }
02624 #endif
02625

02627         if (fclose(file) != 0)

02629                 purple_debug_error("util", "Error closing file %s: %s\n",
02630                                    filename_temp, g_strerror(errno));
02631                 g_free(filename_temp);

02633         }

02637          * above, but for systems without fileno(). */
02638         if ((fd = open(filename_temp, O_RDWR)) < 0) {
02639                 purple_debug_error("util", "Error opening file %s for flush: %s\n",
02640                                    filename_temp, g_strerror(errno));
02641                 g_free(filename_temp);

02643         }

02645                 purple_debug_error("util", "Error syncing %s: %s\n",
02646                                    filename_temp, g_strerror(errno));
02647                 g_free(filename_temp);
02648                 close(fd);
02649                 return FALSE;
02650         }
02651         if (close(fd) < 0) {
02652                 purple_debug_error("util", "Error closing %s after sync: %s\n",
02653                                    filename_temp, g_strerror(errno));
02654                 g_free(filename_temp);
02655                 return FALSE;
02656         }
02657 #endif
02658 
02659         /* Ensure the file is the correct size */
02660         if (byteswritten != real_size)
02661         {
02662                 purple_debug_error("util", "Error writing to file %s: Wrote %"
02663                                    G_GSIZE_FORMAT " bytes "
02664                                    "but should have written %" G_GSIZE_FORMAT
02665                                    "; is your disk full?\n",
02666                                    filename_temp, byteswritten, real_size);
02667                 g_free(filename_temp);
02668                 return FALSE;
02669         }
02670         /* Use stat to be absolutely sure. */
02671         if ((g_stat(filename_temp, &st) == -1) || (st.st_size != real_size))
02672         {
02673                 purple_debug_error("util", "Error writing data to file %s: "
02674                                    "Incomplete file written; is your disk "
02675                                    "full?\n",
02676                                    filename_temp);
02677                 g_free(filename_temp);
02678                 return FALSE;
02679         }
02680 
02681 #ifndef _WIN32
02682         /* Set file permissions */
02683         if (chmod(filename_temp, S_IRUSR | S_IWUSR) == -1)
02684         {
02685                 purple_debug_error("util", "Error setting permissions of file %s: %s\n",
02686                                                  filename_temp, g_strerror(errno));
02687         }
02688 #endif
02689 
02690         /* Rename to the REAL name */
02691         if (g_rename(filename_temp, filename_full) == -1)
02692         {
02693                 purple_debug_error("util", "Error renaming %s to %s: %s\n",
02694                                    filename_temp, filename_full,
02695                                    g_strerror(errno));
02696         }
02697 
02698         g_free(filename_temp);
02699 
02700         return TRUE;
02701 }

03379 {

03382         const char *tmp, *param_string;
03383         char *cmd;
03384         GHashTable *params = NULL;
03385         int len;
03386         if (!(tmp = strchr(uri, ':')) || tmp == uri) {
03387                 purple_debug_error("util", "Malformed protocol handler message - missing protocol.\n");
03388                 return;
03389         }
03390 
03391         len = MIN(sizeof(proto) - 1, (tmp - uri));
03392 
03393         strncpy(proto, uri, len);
03394         proto[len] = '\0';
03395 
03396         tmp++;
03397 
03398         if (g_str_equal(proto, "xmpp"))
03399                 delimiter = ';';
03400         else
03401                 delimiter = '&';
03402 
03403         purple_debug_info("util", "Processing message '%s' for protocol '%s' using delimiter '%c'.\n", tmp, proto, delimiter);
03404 
03405         if ((param_string = strchr(tmp, '?'))) {
03406                 const char *keyend = NULL, *pairstart;
03407                 char *key, *value = NULL;
03408 
03409                 cmd = g_strndup(tmp, (param_string - tmp));
03410                 param_string++;
03411 
03412                 params = g_hash_table_new_full(g_str_hash, g_str_equal, g_free, g_free);
03413                 pairstart = tmp = param_string;
03414 
03415                 while (*tmp || *pairstart) {
03416                         if (*tmp == delimiter || !(*tmp)) {
03417                                 /* If there is no explicit value */

03419                                         keyend = tmp;
03420 
03421                                 if (keyend && keyend != pairstart) {
03422                                         char *p;
03423                                         key = g_strndup(pairstart, (keyend - pairstart));
03424                                         /* If there is an explicit value */
03425                                         if (keyend != tmp && keyend != (tmp - 1))
03426                                                 value = g_strndup(keyend + 1, (tmp - keyend - 1));
03427                                         for (p = key; *p; ++p)
03428                                                 *p = g_ascii_tolower(*p);
03429                                         g_hash_table_insert(params, key, value);
03430                                 }
03431                                 keyend = value = NULL;
03432                                 pairstart = (*tmp) ? tmp + 1 : tmp;
03433                         } else if (*tmp == '=')
03434                                 keyend = tmp;
03435 
03436                         if (*tmp)
03437                                 tmp++;
03438                 }
03439         } else
03440                 cmd = g_strdup(tmp);
03441 
03442         purple_signal_emit_return_1(purple_get_core(), "uri-handler", proto, cmd, params);
03443 
03444         g_free(cmd);

03455  *       force https login, and if you access the web messenger login
03456  *       page via http then it redirects you to the https version, but
03457  *       purple_util_fetch_url() ignores the "https" and attempts to
03458  *       fetch the URL via http again, which gets redirected again.
03459  */
03460 gboolean
03461 purple_url_parse(const char *url, char **ret_host, int *ret_port,
03462                            char **ret_path, char **ret_user, char **ret_passwd)
03463 {
03464         gboolean is_https = FALSE;
03465         const char * scan_info;
03466         char port_str[6];
03467         int f;
03468         const char *at, *slash;
03469         const char *turl;
03470         char host[256], path[256], user[256], passwd[256];
03471         int port = 0;
03472         /* hyphen at end includes it in control set */
03473 
03474 #define ADDR_CTRL "A-Za-z0-9.-"
03475 #define PORT_CTRL "0-9"
03476 #define PAGE_CTRL "A-Za-z0-9.~_/:*!@&%%?=+^-"
03477 #define USER_CTRL "A-Za-z0-9.~_
03495         /* Only care about @ char BEFORE the first / */
03496         at = strchr(url, '@');
03497         slash = strchr(url, '/');
03498         f = 0;
03499         if (at && (!slash || at < slash)) {
03500                 scan_info = "%255[" USER_CTRL "]:%255[" PASSWD_CTRL "]^@";
03501                 f = sscanf(url, scan_info, user, passwd);
03502 
03503                 if (f == 1) {
03504                         /* No passwd, possibly just username supplied */
03505                         scan_info = "%255[" USER_CTRL "]^@";
03506                         f = sscanf(url, scan_info, user);
03507                 }
03508 
03509                 url = at+1; /* move pointer after the @ char */
03510         }
03511 
03512         if (f < 1) {
03513                 *user = '\0';
03514                 *passwd = '\0';
03515         } else if (f == 1)
03516                 *passwd = '\0';
03517 
03518         scan_info = "%255[" ADDR_CTRL "]:%5[" PORT_CTRL "]/%255[" PAGE_CTRL "]";
03519         f = sscanf(url, scan_info, host, port_str, path);
03520 
03521         if (f == 1)
03522         {
03523                 scan_info = "%255[" ADDR_CTRL "]/%255[" PAGE_CTRL "]";
03524                 f = sscanf(url, scan_info, host, path);
03525                 /* Use the default port */
03526                 if (is_https)
03527                         g_snprintf(port_str, sizeof(port_str), "443");
03528                 else
03529                         g_snprintf(port_str, sizeof(port_str), "80");
03530         }
03531 
03532         if (f == 0)
03533                 *host = '\0';
03534 
03535         if (f <= 1)
03536                 *path = '\0';
03537 
03538         sscanf(port_str, "%d", &port);
03539 
03540         if (ret_host != NULL) *ret_host = g_strdup(host);
03541         if (ret_port != NULL) *ret_port = port;
03542         if (ret_path != NULL) *ret_path = g_strdup(path);
03543         if (ret_user != NULL) *ret_user = g_strdup(user);
03544         if (ret_passwd != NULL) *ret_passwd = g_strdup(passwd);
03545 
03546         return ((*host != '\0') ? TRUE : FALSE);
03547

03549 #undef PORT_CTRL
03550 #undef PAGE_CTRL
03551 #undef USER_CTRL
03552 #undef PASSWD_CTRL
03553 }

03737 /* Process in-place */
03738 static void
03739 process_chunked_data(char *data, gsize *len)
03740 {
03741         gsize sz;
03742         gsize newlen = 0;
03743         char *p = data;
03744         char *s = data;
03745 
03746         while (*s) {
03747                 /* Read the size of this chunk */
03748                 if (sscanf(s, "%" G_GSIZE_MODIFIER "x", &sz) != 1)
03749                 {
03750                         purple_debug_error("util", "Error processing chunked data: "
03751                                         "Expected data length, found: %s\n", s);
03752                         break;
03753                 }
03754                 if (sz == 0) {
03755                         /* We've reached the last chunk */
03756                         /*
03757                          * TODO: The spec allows "footers" to follow the last chunk.
03758                          *       If there is more data after this line then we should
03759                          *       treat it like a header.
03760                          */
03761                         break;
03762                 }
03763 
03764                 /* Advance to the start of the data */
03765                 s = strstr(s, "\r\n");
03766                 if (s == NULL)
03767                         break;
03768                 s += 2;
03769 
03770                 if (s + sz > data + *len) {
03771                         purple_debug_error("util", "Error processing chunked data: "
03772                                         "Chunk size %" G_GSIZE_FORMAT " bytes was longer "
03773                                         "than the data remaining in the buffer (%"
03774                                         G_GSIZE_FORMAT " bytes)\n", sz, data + *len - s);
03775                 }
03776 
03777                 /* Move all data overtop of the chunk length that we read in earlier */
03778                 g_memmove(p, s, sz);
03779                 p += sz;
03780                 s += sz;
03781                 newlen += sz;
03782                 if (*s != '\r' && *(s + 1) != '\n') {
03783                         purple_debug_error("util", "Error processing chunked data: "
03784                                         "Expected \\r\\n, found: %s\n", s);
03785                         break;
03786                 }
03787                 s += 2;
03788         }
03789 
03790         /* NULL terminate the data */
03791         *p = 0;
03792 
03793         *len = newlen;
03794 }

03797 url_fetch_recv_cb(gpointer url_data, gint source, PurpleInputCondition cond)
03798 {

03803         gboolean got_eof = FALSE;
03804 
03805         /*
03806          * Read data in a loop until we can't read any more!  This is a
03807          * little confusing because we read using a different function
03808          * depending on whether the socket is ssl or cleartext.
03809          */
03810         while ((gfud->is_ssl && ((len = purple_ssl_read(gfud->ssl_connection, buf, sizeof(buf))) > 0)) ||
03811                         (!gfud->is_ssl && (len = read(source, buf, sizeof(buf))) > 0))

03815                                                     gfud->website.address, gfud->max_len);
03816                         return;
03817                 }

03820                 if((gfud->len + len) >= gfud->data_len) {
03821                         while((gfud->len + len) >= gfud->data_len)
03822                                 gfud->data_len += sizeof(buf);

03825                 }
03826

03828 
03829                 gfud->len += len;
03830 
03831                 memcpy(data_cursor, buf, len);
03832 
03833                 gfud->webdata[gfud->len] = '\0';
03834 
03835                 if(!gfud->got_headers) {
03836                         char *end_of_headers;
03837

03841                                 char *new_data;

03843                                 size_t content_len;
03844

03846                                         header_len, gfud->webdata);

03849                                 if(parse_redirect(gfud->webdata, header_len, gfud))
03850                                         return;

03853

03855                                 content_len = parse_content_len(gfud->webdata, header_len);
03856                                 gfud->chunked = content_is_chunked(gfud->webdata, header_len);
03857 
03858                                 if (content_len == 0) {

03860                                         content_len = 8192;

03862                                         gfud->has_explicit_data_len = TRUE;
03863                                 }
03864 
03865

03867                                 if (gfud->include_headers) {

04297 /* Originally lifted from
04298  * http://www.oreillynet.com/pub/a/network/excerpt/spcookbook_chap03/index3.html
04299  * ... and slightly modified to be a bit more rfc822 compliant
04300  * ... and modified a bit more to make domain checking rfc1035 compliant
04301  *     with the exception permitted in rfc1101 for domains to start with digit
04302  *     but not completely checking to avoid conflicts with IP addresses
04303  */
04304 gboolean
04305 purple_email_is_valid(const char *address)
04306 {
04307         const char *c, *domain;
04308         static char *rfc822_specials = "()<>@,;:\\\"[]";
04309 
04310         g_return_val_if_fail(address != NULL, FALSE);
04311 
04312         if (*address == '.') return FALSE;
04313 
04314         /* first we validate the name portion (name@domain) (rfc822)*/
04315         for (c = address;  *c;  c++) {
04316                 if (*c == '\"' && (c == address || *(c - 1) == '.' || *(c - 1) == '\"')) {
04317                         while (*++c) {
04318                                 if (*c == '\\') {
04319                                         if (*c++ && *c < 127 && *c != '\n' && *c != '\r') continue;
04320                                         else return FALSE;
04321                                 }
04322                                 if (*c == '\"') break;
04323                                 if (*c < ' ' || *c >= 127) return FALSE;
04324                         }
04325                         if (!*c++) return FALSE;
04326                         if (*c == '@') break;
04327                         if (*c != '.') return FALSE;
04328                         continue;
04329                 }
04330                 if (*c == '@') break;
04331                 if (*c <= ' ' || *c >= 127) return FALSE;
04332                 if (strchr(rfc822_specials, *c)) return FALSE;
04333         }
04334 
04335         /* It's obviously not an email address if we didn't find an '@' above */
04336         if (*c == '\0') return FALSE;
04337 
04338         /* strictly we should return false if (*(c - 1) == '.') too, but I think
04339          * we should permit user.@domain type addresses - they do work :) */
04340         if (c == address) return FALSE;
04341 
04342         /* next we validate the domain portion (name@domain) (rfc1035 & rfc1011) */
04343         if (!*(domain = ++c)) return FALSE;
04344         do {
04345                 if (*c == '.' && (c == domain || *(c - 1) == '.' || *(c - 1) == '-'))
04346                         return FALSE;
04347                 if (*c == '-' && (*(c - 1) == '.' || *(c - 1) == '@')) return FALSE;
04348                 if ((*c < '0' && *c != '-' && *c != '.') || (*c > '9' && *c < 'A') ||
04349                         (*c > 'Z' && *c < 'a') || (*c > 'z')) return FALSE;
04350         } while (*++c);
04351 
04352         if (*(c - 1) == '-') return FALSE;
04353 
04354         return ((c - domain) > 3 ? TRUE : FALSE);
04355 }
