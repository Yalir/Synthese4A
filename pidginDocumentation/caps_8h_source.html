00050 /*
00051  * This stores a set of exts "known" for a specific node (which indicates
00052  * a specific client -- for reference, Pidgin, Finch, Meebo, et al share one
00053  * node.) In XEP-0115 v1.3, exts are used for features that may or may not be
00054  * present at a given time (PEP things, buzz might be disabled, etc).
00055  *
00056  * This structure is shared among all JabberCapsClientInfo instances matching
00057  * a specific node (if the capstable key->hash == NULL, which indicates that
00058  * the ClientInfo is using v1.3 caps as opposed to v1.5 caps).
00059  *
00060  * It's only exposed so that jabber_resource_has_capability can use it.
00061  * Everyone else, STAY AWAY!
00062  */

00065         GHashTable *exts; /* char *ext_name -> GList *features */
