00387     GOfflineMessage *message = (GOfflineMessage *)offline_messages->data;
00388     purple_debug_info("nullprpl", "delivering offline message to %s: %s\n",
00389                       acct->username, message->message);

00391                 message->mtime);

00395     g_free(message->message);
00396     g_free(message);

00409 static int nullprpl_send_im(PurpleConnection *gc, const char *who,
00410                             const char *message, PurpleMessageFlags flags)
00411 {
00412   const char *from_username = gc->account->username;
00413   PurpleMessageFlags receive_flags = ((flags & ~PURPLE_MESSAGE_SEND)
00414                                       | PURPLE_MESSAGE_RECV);
00415   PurpleAccount *to_acct = purple_accounts_find(who, NULLPRPL_ID);
00416   PurpleConnection *to;
00417 
00418   purple_debug_info("nullprpl", "sending message from %s to %s: %s\n",
00419                     from_username, who, message);
00420 
00421   /* is the sender blocked by the recipient's privacy settings? */
00422   if (to_acct && !purple_privacy_check(to_acct, gc->account->username)) {
00423     char *msg = g_strdup_printf(
00424       _("Your message was blocked by %s's privacy settings."), who);
00425     purple_debug_info("nullprpl",
00426                       "discarding; %s is blocked by %s's privacy settings\n",
00427                       from_username, who);
00428     purple_conv_present_error(who, gc->account, msg);
00429     g_free(msg);
00430     return 0;
00431   }
00432 
00433   /* is the recipient online? */
00434   to = get_nullprpl_gc(who);
00435   if (to) {  /* yes, send */
00436     serv_got_im(to, from_username, message, receive_flags, time(NULL));
00437 
00438   } else {  /* nope, store as an offline message */
00439     GOfflineMessage *offline_message;
00440     GList *messages;
00441 
00442     purple_debug_info("nullprpl",
00443                       "%s is offline, sending as offline message\n", who);
00444     offline_message = g_new0(GOfflineMessage, 1);
00445     offline_message->from = g_strdup(from_username);
00446     offline_message->message = g_strdup(message);
00447     offline_message->mtime = time(NULL);
00448     offline_message->flags = receive_flags;
00449 
00450     messages = g_hash_table_lookup(goffline_messages, who);
00451     messages = g_list_append(messages, offline_message);
00452     g_hash_table_insert(goffline_messages, g_strdup(who), messages);
00453   }
00454 
00455    return 1;
00456 }

00765 static PurpleCmdRet send_whisper(PurpleConversation *conv, const gchar *cmd,
00766                                  gchar **args, gchar **error, void *userdata) {
00767   const char *to_username;
00768   const char *message;
00769   const char *from_username;
00770   PurpleConvChat *chat;
00771   PurpleConvChatBuddy *chat_buddy;
00772   PurpleConnection *to;
00773 
00774   /* parse args */
00775   to_username = args[0];
00776   message = args[1];
00777 
00778   if (!to_username || strlen(to_username) == 0) {
00779     *error = g_strdup(_("Whisper is missing recipient."));
00780     return PURPLE_CMD_RET_FAILED;
00781   } else if (!message || strlen(message) == 0) {
00782     *error = g_strdup(_("Whisper is missing message."));
00783     return PURPLE_CMD_RET_FAILED;
00784   }
00785 
00786   from_username = conv->account->username;
00787   purple_debug_info("nullprpl", "%s whispers to %s in chat room %s: %s\n",
00788                     from_username, to_username, conv->name, message);
00789 
00790   chat = purple_conversation_get_chat_data(conv);
00791   chat_buddy = purple_conv_chat_cb_find(chat, to_username);
00792   to = get_nullprpl_gc(to_username);
00793 
00794   if (!chat_buddy) {
00795     /* this will be freed by the caller */
00796     *error = g_strdup_printf(_("%s is not logged in."), to_username);
00797     return PURPLE_CMD_RET_FAILED;
00798   } else if (!to) {
00799     *error = g_strdup_printf(_("%s is not in this chat room."), to_username);
00800     return PURPLE_CMD_RET_FAILED;
00801   } else {
00802     /* write the whisper in the sender's chat window  */
00803     char *message_to = g_strdup_printf("%s (to %s)", message, to_username);
00804     purple_conv_chat_write(chat, from_username, message_to,
00805                            PURPLE_MESSAGE_SEND | PURPLE_MESSAGE_WHISPER,
00806                            time(NULL));
00807     g_free(message_to);
00808 
00809     /* send the whisper */
00810     serv_chat_whisper(to, chat->id, from_username, message);
00811 
00812     return PURPLE_CMD_RET_OK;
00813   }
00814 }

00966 static PurpleRoomlist *nullprpl_roomlist_get_list(PurpleConnection *gc) {
00967   const char *username = gc->account->username;
00968   PurpleRoomlist *roomlist = purple_roomlist_new(gc->account);
00969   GList *fields = NULL;
00970   PurpleRoomlistField *field;
00971   GList *chats;
00972   GList *seen_ids = NULL;
00973 
00974   purple_debug_info("nullprpl", "%s asks for room list; returning:\n", username);
00975 
00976   /* set up the room list */
00977   field = purple_roomlist_field_new(PURPLE_ROOMLIST_FIELD_STRING, "room",
00978                                     "room", TRUE /* hidden */);
00979   fields = g_list_append(fields, field);
00980 
00981   field = purple_roomlist_field_new(PURPLE_ROOMLIST_FIELD_INT, "Id", "Id", FALSE);
00982   fields = g_list_append(fields, field);
00983 
00984   purple_roomlist_set_fields(roomlist, fields);
00985 
00986   /* add each chat room. the chat ids are cached in seen_ids so that each room
00987    * is only returned once, even if multiple users are in it. */
00988   for (chats  = purple_get_chats(); chats; chats = g_list_next(chats)) {
00989     PurpleConversation *conv = (PurpleConversation *)chats->data;
00990     PurpleRoomlistRoom *room;
00991     const char *name = conv->name;
00992     int id = purple_conversation_get_chat_data(conv)->id;
00993 
00994     /* have we already added this room? */
00995     if (g_list_find_custom(seen_ids, name, (GCompareFunc)strcmp))
00996       continue;                                /* yes! try the next one. */
00997 
00998     /* This cast is OK because this list is only staying around for the life
00999      * of this function and none of the conversations are being deleted
01000      * in that timespan. */
01001     seen_ids = g_list_prepend(seen_ids, (char *)name); /* no, it's new. */
01002     purple_debug_info("nullprpl", "%s (%d), ", name, id);
01003 
01004     room = purple_roomlist_room_new(PURPLE_ROOMLIST_ROOMTYPE_ROOM, name, NULL);
01005     purple_roomlist_room_add_field(roomlist, room, name);
01006     purple_roomlist_room_add_field(roomlist, room, &id);
01007     purple_roomlist_room_add(roomlist, room);
01008   }
01009 
01010   g_list_free(seen_ids);
01011   purple_timeout_add(1 /* ms */, nullprpl_finish_get_roomlist, roomlist);
01012   return roomlist;
01013 }
