00156 static void
00157 perl_init(void)
00158 {
00159         /* changed the name of the variable from load_file to perl_definitions
00160          * since now it does much more than defining the load_file sub.
00161          * Moreover, deplaced the initialisation to the xs_init function.
00162          * (TheHobbit) */
00163         char *perl_args[] = { "", "-e", "0", "-w" };
00164         char perl_definitions[] =
00165         {
00166                 /* We use to function one to load a file the other to execute
00167                  * the string obtained from the first and holding the file
00168                  * contents. This allows to have a really local $/ without
00169                  * introducing temp variables to hold the old value. Just a
00170                  * question of style:) */
00171                 "package Purple::PerlLoader;"
00172                 "use Symbol;"
00173 
00174                 "sub load_file {"
00175                   "my $f_name=shift;"
00176                   "local $/=undef;"
00177                   "open FH,$f_name or return \"__FAILED__\";"
00178                   "$_=<FH>;"
00179                   "close FH;"
00180                   "return $_;"
00181                 "}"
00182 
00183                 "sub destroy_package {"
00184                   "eval { $_[0]->UNLOAD() if $_[0]->can('UNLOAD'); };"
00185                   "Symbol::delete_package($_[0]);"
00186                 "}"
00187 
00188                 "sub load_n_eval {"
00189                   "my ($f_name, $package) = @_;"
00190                   "destroy_package($package);"
00191                   "my $strin=load_file($f_name);"
00192                   "return 2 if($strin eq \"__FAILED__\");"
00193                   "my $eval = qq{package $package; $strin;};"
00194 
00195                   "{"
00196                   "  eval $eval;"
00197                   "}"
00198 
00199                   "if($@) {"
00200                     /*"  #something went wrong\n"*/
00201                     "die(\"Errors loading file $f_name: $@\");"
00202                   "}"
00203 
00204                   "return 0;"
00205                 "}"
00206         };
00207 
00208         my_perl = perl_alloc();
00209         PERL_SET_CONTEXT(my_perl);
00210         PL_perl_destruct_level = 1;
00211         perl_construct(my_perl);
00212 #ifdef DEBUG
00213         perl_parse(my_perl, xs_init, 4, perl_args, NULL);
00214 #else
00215         perl_parse(my_perl, xs_init, 3, perl_args, NULL);
00216 #endif
00217 #ifdef HAVE_PERL_EVAL_PV
00218         eval_pv(perl_definitions, TRUE);
00219 #else
00220         perl_eval_pv(perl_definitions, TRUE); /* deprecated */
00221 #endif
00222         perl_run(my_perl);
