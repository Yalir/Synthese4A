00657 void irc_parse_msg(struct irc_conn *irc, char *input)
00658 {
00659         struct _irc_msg *msgent;
00660         char *cur, *end, *tmp, *from, *msgname, *fmt, **args, *msg;
00661         guint i;

00663 
00664         irc->recv_time = time(NULL);
00665 
00666         /*
00667          * The data passed to irc-receiving-text is the raw protocol data.
00668          * TODO: It should be passed as an array of bytes and a length
00669          * instead of a null terminated string.
00670          */
00671         purple_signal_emit(_irc_plugin, "irc-receiving-text", gc, &input);
00672 
00673         if (!strncmp(input, "PING ", 5)) {
00674                 msg = irc_format(irc, "vv", "PONG", input + 5);
00675                 irc_send(irc, msg);
00676                 g_free(msg);
00677                 return;
00678         } else if (!strncmp(input, "ERROR ", 6)) {
00679                 if (g_utf8_validate(input, -1, NULL)) {
00680                         char *tmp = g_strdup_printf("%s\n%s", _("Disconnected."), input);
00681                         purple_connection_error_reason (gc,
00682                                 PURPLE_CONNECTION_ERROR_NETWORK_ERROR, tmp);
00683                         g_free(tmp);
00684                 } else
00685                         purple_connection_error_reason (gc,
00686                                 PURPLE_CONNECTION_ERROR_NETWORK_ERROR,
00687                                 _("Disconnected."));
00688                 return;
00689 #ifdef HAVE_CYRUS_SASL
00690         } else if (!strncmp(input, "AUTHENTICATE ", 13)) {
00691                 irc_msg_auth(irc, input + 13);
00692                 return;
00693 #endif
00694         }
00695 
00696         if (input[0] != ':' || (cur = strchr(input, ' ')) == NULL) {
00697                 irc_parse_error_cb(irc, input);
00698                 return;
00699         }
00700 
00701         from = g_strndup(&input[1], cur - &input[1]);
00702         cur++;
00703         end = strchr(cur, ' ');
00704         if (!end)
00705                 end = cur + strlen(cur);
00706 
00707         tmp = g_strndup(cur, end - cur);
00708         msgname = g_ascii_strdown(tmp, -1);
00709         g_free(tmp);
00710 
00711         if ((msgent = g_hash_table_lookup(irc->msgs, msgname)) == NULL) {
00712                 irc_msg_default(irc, "", from, &input);
00713                 g_free(msgname);
00714                 g_free(from);
00715                 return;
00716         }
00717         g_free(msgname);
00718 
00719         args = g_new0(char *, strlen(msgent->format));
00720         for (cur = end, fmt = msgent->format, i = 0; fmt[i] && *cur++; i++) {
00721                 switch (fmt[i]) {
00722                 case 'v':
00723                         if (!(end = strchr(cur, ' '))) end = cur + strlen(cur);
00724                         /* This is a string of unknown encoding which we do not
00725                          * want to transcode, but it may or may not be valid
00726                          * UTF-8, so we'll salvage it.  If a nick/channel/target
00727                          * field has inadvertently been marked verbatim, this
00728                          * could cause weirdness. */
00729                         tmp = g_strndup(cur, end - cur);
00730                         args[i] = purple_utf8_salvage(tmp);
00731                         g_free(tmp);
00732                         cur += end - cur;
00733                         break;
00734                 case 't':
00735                 case 'n':
00736                 case 'c':
00737                         if (!(end = strchr(cur, ' '))) end = cur + strlen(cur);
00738                         tmp = g_strndup(cur, end - cur);
00739                         args[i] = irc_recv_convert(irc, tmp);
00740                         g_free(tmp);
00741                         cur += end - cur;
00742                         break;
00743                 case ':':
00744                         if (*cur == ':') cur++;
00745                         args[i] = irc_recv_convert(irc, cur);
00746                         cur = cur + strlen(cur);
00747                         break;
00748                 case '*':
00749                         /* Ditto 'v' above; we're going to salvage this in case
00750                          * it leaks past the IRC prpl */
00751                         args[i] = purple_utf8_salvage(cur);
00752                         cur = cur + strlen(cur);
00753                         break;
00754                 default:
00755                         purple_debug(PURPLE_DEBUG_ERROR, "irc", "invalid message format character '%c'\n", fmt[i]);
00756                         break;
00757                 }
00758         }
00759         tmp = irc_recv_convert(irc, from);
00760         (msgent->cb)(irc, msgent->name, tmp, args);
00761         g_free(tmp);
00762         for (i = 0; i < strlen(msgent->format); i++) {
00763                 g_free(args[i]);
00764         }
00765         g_free(args);
00766         g_free(from);
00767 }
