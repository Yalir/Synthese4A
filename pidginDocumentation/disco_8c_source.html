00445                                 dst, sizeof(dst));

00447                 } else {
00448                         inet_ntop(addr->sa_family, &((struct sockaddr_in *) addr)->sin_addr,
00449                                 dst, sizeof(dst));
00450                         port = ntohs(((struct sockaddr_in *) addr)->sin_port);
00451                 }
00452 
00453                 if (js->stun_ip)
00454                         g_free(js->stun_ip);
00455                 js->stun_ip = g_strdup(dst);
00456                 js->stun_port = port;
00457

00459                                   "%s:%d\n", dst, port);
00460 
00461                 /* unmark ongoing query */
00462                 js->stun_query = NULL;
00463         }
00464 
00465         while (hosts != NULL) {
00466                 hosts = g_slist_delete_link(hosts, hosts);
00467                 /* Free the address */
