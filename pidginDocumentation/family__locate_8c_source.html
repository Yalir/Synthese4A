00763 /*
00764  * AIM is fairly regular about providing user info.  This is a generic
00765  * routine to extract it in its standard form.
00766  */
00767 int
00768 aim_info_extract(OscarData *od, ByteStream *bs, aim_userinfo_t *outinfo)
00769 {
00770         int curtlv, tlvcnt;
00771         guint8 bnlen;
00772 
00773         if (!bs || !outinfo)
00774                 return -EINVAL;
00775 
00776         /* Clear out old data first */
00777         memset(outinfo, 0x00, sizeof(aim_userinfo_t));
00778 
00779         /*
00780          * Username.  Stored as an unterminated string prepended with a
00781          * byte containing its length.
00782          */
00783         bnlen = byte_stream_get8(bs);
00784         outinfo->bn = byte_stream_getstr(bs, bnlen);
00785 
00786         /*
00787          * Warning Level.  Stored as an unsigned short.
00788          */
00789         outinfo->warnlevel = byte_stream_get16(bs);
00790 
00791         /*
00792          * TLV Count. Unsigned short representing the number of
00793          * Type-Length-Value triples that follow.
00794          */
00795         tlvcnt = byte_stream_get16(bs);
00796 
00797         /*
00798          * Parse out the Type-Length-Value triples as they're found.
00799          */
00800         for (curtlv = 0; curtlv < tlvcnt; curtlv++) {
00801                 guint16 type, length;
00802                 int endpos;
00803                 int curpos;
00804 
00805                 type = byte_stream_get16(bs);
00806                 length = byte_stream_get16(bs);
00807                 curpos = byte_stream_curpos(bs);
00808                 endpos = curpos + MIN(length, byte_stream_bytes_left(bs));
00809 
00810                 if (type == 0x0001) {
00811                         /*
00812                          * User flags
00813                          *
00814                          * Specified as any of the following ORed together:
00815                          *      0x0001  Unconfirmed account
00816                          *      0x0002  Unknown bit 2
00817                          *      0x0004  AOL Main Service user
00818                          *      0x0008  Unknown bit 4
00819                          *      0x0010  Free (AIM) user
00820                          *      0x0020  Away
00821                          *      0x0040  ICQ user (AIM bit also set)
00822                          *      0x0080  Mobile device
00823                          *      0x0400  Bot (like ActiveBuddy)
00824                          */
00825                         outinfo->flags = byte_stream_get16(bs);
00826                         outinfo->present |= AIM_USERINFO_PRESENT_FLAGS;
00827 
00828                 } else if (type == 0x0002) {
00829                         /*
00830                          * Account creation time
00831                          *
00832                          * The time/date that the user originally registered for
00833                          * the service, stored in time_t format.
00834                          *
00835                          * I'm not sure how this differs from type 5 ("member
00836                          * since").
00837                          *
00838                          * Note: This is the field formerly known as "member
00839                          * since".  All these years and I finally found out
00840                          * that I got the name wrong.
00841                          */
00842                         outinfo->createtime = byte_stream_get32(bs);
00843                         outinfo->present |= AIM_USERINFO_PRESENT_CREATETIME;
00844 
00845                 } else if (type == 0x0003) {
00846                         /*
00847                          * On-Since date
00848                          *
00849                          * The time/date that the user started their current
00850                          * session, stored in time_t format.
00851                          */
00852                         outinfo->onlinesince = byte_stream_get32(bs);
00853                         outinfo->present |= AIM_USERINFO_PRESENT_ONLINESINCE;
00854 
00855                 } else if (type == 0x0004) {
00856                         /*
00857                          * Idle time
00858                          *
00859                          * Number of minutes since the user actively used the
00860                          * service.
00861                          *
00862                          * Note that the client tells the server when to start
00863                          * counting idle times, so this may or may not be
00864                          * related to reality.
00865                          */
00866                         outinfo->idletime = byte_stream_get16(bs);
00867                         outinfo->present |= AIM_USERINFO_PRESENT_IDLE;
00868 
00869                 } else if (type == 0x0005) {
00870                         /*
00871                          * Member since date
00872                          *
00873                          * The time/date that the user originally registered for
00874                          * the service, stored in time_t format.
00875                          *

00877                          * creation time"), particularly in the self-info.

00885                          * ICQ Online Status

00888                          * of values done by Scott <darkagl@pcnet.com>
00889                          */

00898

00901                          * ICQ User IP Address

00960                          */
