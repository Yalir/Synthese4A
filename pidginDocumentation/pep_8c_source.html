00088         JabberIq *iq = jabber_iq_new(js, JABBER_IQ_GET);
00089         xmlnode *pubsub, *items;

00091         if (to)

00096 
00097         items = xmlnode_new_child(pubsub, "items");
00098         xmlnode_set_attrib(items,"node",node);
00099 
00100         if (id) {
00101                 xmlnode *item = xmlnode_new_child(items, "item");
00102                 xmlnode_set_attrib(item, "id", id);

00104                 /* Most recent item */
00105                 xmlnode_set_attrib(items, "max_items", "1");

00116 void jabber_handle_event(JabberMessage *jm) {
00117         /* this may be called even when the own server doesn't support pep! */
00118         JabberPEPHandler *jph;
00119         GList *itemslist;
00120         char *jid;
00121 
00122         if (jm->type != JABBER_MESSAGE_EVENT)
00123                 return;
00124 
00125         jid = jabber_get_bare_jid(jm->from);
00126 
00127         for(itemslist = jm->eventitems; itemslist; itemslist = itemslist->next) {
00128                 xmlnode *items = (xmlnode*)itemslist->data;
00129                 const char *nodename = xmlnode_get_attrib(items,"node");
00130 
00131                 if(nodename && (jph = g_hash_table_lookup(pep_handlers, nodename)))
00132                         jph(jm->js, jid, items);
00133         }
00134 
00135         /* discard items we don't have a handler for */
00136         g_free(jid);
00137 }
