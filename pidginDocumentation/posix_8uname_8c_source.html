00042 int
00043 jabber_win32_uname( struct utsname *uts )
00044 {
00045   DWORD sLength;
00046   OSVERSIONINFO OS_version;
00047   SYSTEM_INFO System_Info;
00048 
00049 /* XXX Should these be in the global runtime */
00050   enum WinOS {Win95, Win98, WinNT, unknown};
00051   int MingwOS;
00052 
00053   memset( uts, 0, sizeof ( *uts ) );
00054   OS_version.dwOSVersionInfoSize = sizeof( OSVERSIONINFO );
00055 
00056   GetVersionEx ( &OS_version );
00057   GetSystemInfo ( &System_Info );
00058 
00059   g_strlcpy( uts->sysname, "WIN32_" , sizeof(uts->sysname));
00060   switch( OS_version.dwPlatformId )
00061   {
00062     case VER_PLATFORM_WIN32_NT:
00063       g_strlcat( uts->sysname, "WinNT", sizeof(uts->sysname) );
00064       MingwOS = WinNT;
00065       break;
00066     case VER_PLATFORM_WIN32_WINDOWS:
00067       switch ( OS_version.dwMinorVersion )
00068       {
00069         case 0:
00070           g_strlcat( uts->sysname, "Win95", sizeof(uts->sysname) );
00071           MingwOS = Win95;
00072           break;
00073         case 10:
00074           g_strlcat( uts->sysname, "Win98", sizeof(uts->sysname) );
00075           MingwOS = Win98;
00076           break;
00077         default:
00078           g_strlcat( uts->sysname, "Win??", sizeof(uts->sysname) );
00079           MingwOS = unknown;
00080           break;
00081       }
00082       break;
00083     default:
00084       g_strlcat( uts->sysname, "Win??", sizeof(uts->sysname) );
00085       MingwOS = unknown;
00086       break;
00087   }
00088 
00089 #ifdef __MINGW32__
00090   sprintf( uts->version, "%i", __MINGW32_MAJOR_VERSION );
00091   sprintf( uts->release, "%i", __MINGW32_MINOR_VERSION );
00092 #endif
00093 
00094   switch( System_Info.wProcessorArchitecture )
00095   {
00096     case PROCESSOR_ARCHITECTURE_PPC:
00097       g_strlcpy( uts->machine, "ppc" , sizeof( uts->machine ) );
00098       break;
00099     case PROCESSOR_ARCHITECTURE_ALPHA:
00100       g_strlcpy( uts->machine, "alpha" , sizeof( uts->machine ) );
00101       break;
00102     case PROCESSOR_ARCHITECTURE_MIPS:
00103       g_strlcpy( uts->machine, "mips" , sizeof( uts->machine ) );
00104       break;
00105     case PROCESSOR_ARCHITECTURE_INTEL:
00106       /* dwProcessorType is only valid in Win95 and Win98
00107          wProcessorLevel is only valid in WinNT */
00108       switch( MingwOS )
00109       {
00110         case Win95:
00111         case Win98:
00112           switch( System_Info.dwProcessorType )
00113           {
00114             case PROCESSOR_INTEL_386:
00115             case PROCESSOR_INTEL_486:
00116             case PROCESSOR_INTEL_PENTIUM:
00117               sprintf( uts->machine, "i%ld", System_Info.dwProcessorType );
00118               break;
00119             default:
00120               g_strlcpy( uts->machine, "i386" , sizeof( uts->machine ) );
00121               break;
00122           }
00123           break;
00124         case WinNT:
00125           sprintf( uts->machine, "i%d86", System_Info.wProcessorLevel );
00126           break;
00127         default:
00128           g_strlcpy( uts->machine, "unknown" , sizeof( uts->machine ) );
00129           break;
00130       }
00131       break;
00132     default:
00133       g_strlcpy( uts->machine, "unknown" , sizeof( uts->machine ) );
00134       break;
00135   }
00136 
00137   sLength = sizeof ( uts->nodename ) - 1;
00138   GetComputerNameA( uts->nodename, &sLength );
00139   return 1;
00140 }
