00036 OscarData *
00037 oscar_data_new(void)
00038 {
00039         OscarData *od;
00040         aim_module_t *cur;
00041         GString *msg;
00042 
00043         od = g_new0(OscarData, 1);
00044 
00045         aim_initsnachash(od);
00046         od->snacid_next = 0x00000001;
00047         od->buddyinfo = g_hash_table_new_full(g_str_hash, g_str_equal, g_free, g_free);
00048         od->handlerlist = g_hash_table_new_full(g_direct_hash, g_direct_equal, NULL, g_free);
00049 
00050         /*
00051          * Register all the modules for this session...
00052          */
00053         aim__registermodule(od, misc_modfirst); /* load the catch-all first */
00054         aim__registermodule(od, service_modfirst);
00055         aim__registermodule(od, locate_modfirst);
00056         aim__registermodule(od, buddylist_modfirst);
00057         aim__registermodule(od, msg_modfirst);
00058         aim__registermodule(od, admin_modfirst);
00059         aim__registermodule(od, popups_modfirst);
00060         aim__registermodule(od, bos_modfirst);
00061         aim__registermodule(od, search_modfirst);
00062         aim__registermodule(od, stats_modfirst);
00063         aim__registermodule(od, chatnav_modfirst);
00064         aim__registermodule(od, chat_modfirst);
00065         aim__registermodule(od, bart_modfirst);
00066         /* missing 0x11 - 0x12 */
00067         aim__registermodule(od, ssi_modfirst);
00068         /* missing 0x14 */
00069         aim__registermodule(od, icq_modfirst);
00070         /* missing 0x16 */
00071         /* auth_modfirst is only needed if we're connecting with the old-style BUCP login */
00072         aim__registermodule(od, auth_modfirst);
00073         aim__registermodule(od, email_modfirst);
00074 
00075         msg = g_string_new("Registered modules: ");
00076         for (cur = od->modlistv; cur; cur = cur->next) {
00077                 g_string_append_printf(
00078                         msg,
00079                         "%s (family=0x%04x, version=0x%04x, toolid=0x%04x, toolversion=0x%04x), ",
00080                         cur->name,
00081                         cur->family,
00082                         cur->version,
00083                         cur->toolid,
00084                         cur->toolversion);
00085         }
00086         purple_debug_misc("oscar", "%s\n", msg->str);
00087         g_string_free(msg, TRUE);
00088 
00089         return od;
00090 }

00097 void
00098 oscar_data_destroy(OscarData *od)
00099 {
00100         aim_cleansnacs(od, -1);
00101 
00102         /* Only used when connecting with clientLogin */
00103         if (od->url_data != NULL)
00104                 purple_util_fetch_url_cancel(od->url_data);
00105 
00106         while (od->requesticon)
00107         {
00108                 g_free(od->requesticon->data);
00109                 od->requesticon = g_slist_delete_link(od->requesticon, od->requesticon);
00110         }
00111         g_free(od->email);
00112         g_free(od->newp);
00113         g_free(od->oldp);
00114         if (od->getblisttimer > 0)
00115                 purple_timeout_remove(od->getblisttimer);
00116         while (od->oscar_connections != NULL)
00117                 flap_connection_destroy(od->oscar_connections->data,
00118                                 OSCAR_DISCONNECT_DONE, NULL);
00119 
00120         while (od->peer_connections != NULL)
00121                 peer_connection_destroy(od->peer_connections->data,
00122                                 OSCAR_DISCONNECT_LOCAL_CLOSED, NULL);
00123 
00124         aim__shutdownmodules(od);
00125 
00126         g_hash_table_destroy(od->buddyinfo);
00127         g_hash_table_destroy(od->handlerlist);
00128 
00129         g_free(od);
