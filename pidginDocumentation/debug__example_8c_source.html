00063 /* As we've covered before, libpurple calls this function, if present, when it
00064  * loads the plugin.  Here we're using it to show off the capabilities of the
00065  * debug API and just blindly returning TRUE to tell libpurple it's safe to
00066  * continue loading. */
00067 static gboolean
00068 plugin_load(PurplePlugin *plugin)
00069 {
00070         /* Define these for convenience--we're just using them to show the
00071          * similarities of the debug functions to the standard printf(). */
00072         gint i = 256;
00073         gfloat f = 512.1024;
00074         const gchar *s = "example string";
00075 
00076         /* Introductory message */
00077         purple_debug_info(PLUGIN_ID,
00078                 "Called plugin_load.  Beginning debug demonstration\n");
00079 
00080         /* Show off the debug API a bit */
00081         purple_debug_misc(PLUGIN_ID,
00082                 "MISC level debug message.  i = %d, f = %f, s = %s\n", i, f, s);
00083 
00084         purple_debug_info(PLUGIN_ID,
00085                 "INFO level debug message.  i = %d, f = %f, s = %s\n", i, f, s);
00086 
00087         purple_debug_warning(PLUGIN_ID,
00088                 "WARNING level debug message.  i = %d, f = %f, s = %s\n", i, f, s);
00089 
00090         purple_debug_error(PLUGIN_ID,
00091                 "ERROR level debug message.  i = %d, f = %f, s = %s\n", i, f, s);
00092 
00093         purple_debug_fatal(PLUGIN_ID,
00094                 "FATAL level debug message. i = %d, f = %f, s = %s\n", i, f, s);
00095 
00096         /* Now just return TRUE to tell libpurple to finish loading. */
00097         return TRUE;
00098 }
