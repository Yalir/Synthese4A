00079 /* Indexes of this array + 1 map HTML font size to scale of normal font size. *
00080  * Based on _point_sizes from libpurple/gtkimhtml.c
00081  *                                 1    2  3    4     5      6       7 */
00082 static gdouble _font_scale[] = { .85, .95, 1, 1.2, 1.44, 1.728, 2.0736 };
00083 
00084 /* Purple maximum font size.  Equivalent to sizeof(_font_scale) / sizeof(_font_scale[0]) */
00085 #define MAX_FONT_SIZE                   7
00086 
00087 #define POINTS_PER_INCH                 72      /* How many pt's in an inch */
00088 
00089 /* Text formatting bits for <f s=#> */
00090 #define MSIM_TEXT_BOLD                  1
00091 #define MSIM_TEXT_ITALIC                2
00092 #define MSIM_TEXT_UNDERLINE             4

00094 /* Default baseline size of purple's fonts, in points. What is size 3 in points.
00095  * _font_scale specifies scaling factor relative to this point size. Note this
00096  * is only the default; it is configurable in account options. */
00097 #define MSIM_BASE_FONT_POINT_SIZE       8
00098 
00099 /* Default display's DPI. 96 is common but it can differ. Also configurable
00100  * in account options. */

00103 /* round is part of C99, but sometimes is unavailable before then.
00104  * Based on http://forums.belution.com/en/cpp/000/050/13.shtml
00105  */
00106 static double msim_round(double value)
00107 {
00108         if (value < 0) {
00109                 return -(floor(-value + 0.5));
00110         } else {
00111                 return   floor( value + 0.5);
00112         }
00113 }

00119 static guint
00120 msim_point_to_purple_size(MsimSession *session, guint point)
00121 {
00122         guint size, this_point, base;
00123 
00124         base = purple_account_get_int(session->account, "base_font_size", MSIM_BASE_FONT_POINT_SIZE);
00125 
00126         for (size = 0; size < MAX_FONT_SIZE; ++size) {
00127                 this_point = (guint)msim_round(base * _font_scale[size]);
00128 
00129                 if (this_point >= point) {
00130                         purple_debug_info("msim", "msim_point_to_purple_size: %d pt -> size=%d\n",
00131                                         point, size);
00132                         return size;
00133                 }
00134         }
00135 
00136         /* No HTML font size was this big; return largest possible. */
00137         return this_point;
00138 }

00194 static void
00195 msim_markup_f_to_html(MsimSession *session, xmlnode *root, gchar **begin, gchar **end)
00196 {
00197         const gchar *face, *height_str, *decor_str;
00198         GString *gs_end, *gs_begin;
00199         guint decor, height;
00200 
00201         face = xmlnode_get_attrib(root, "f");
00202         height_str = xmlnode_get_attrib(root, "h");
00203         decor_str = xmlnode_get_attrib(root, "s");
00204 
00205         /* Validate the font face, to avoid constructing invalid HTML later */
00206         if (face != NULL && strchr(face, '\'') != NULL)
00207                 face = NULL;
00208 
00209         height = height_str != NULL ? atol(height_str) : 12;
00210         decor = decor_str != NULL ? atol(decor_str) : 0;
00211 
00212         /*
00213          * The HTML we're constructing here is a bit redudant.  Ideally we
00214          * would use only the font-family and font-size CSS span, but Pidgin
00215          * doesn't support it (it's included for other UIs).  For Pidgin we
00216          * wrap the whole thing in an ugly font tag, and Pidgin will happily
00217          * ignore the <span>.
00218          */
00219         gs_begin = g_string_new("");
00220         if (height && !face) {
00221                 guint point_size = msim_height_to_point(session, height);
00222                 g_string_printf(gs_begin,
00223                                 "<font size='%d'><span style='font-size: %dpt'>",
00224                                 msim_point_to_purple_size(session, point_size),
00225                                 point_size);
00226         } else if (height && face) {
00227                 guint point_size = msim_height_to_point(session, height);
00228                 g_string_printf(gs_begin,
00229                                 "<font face='%s' size='%d'><span style='font-family: %s; font-size: %dpt'>",
00230                                 face, msim_point_to_purple_size(session, point_size),
00231                                 face, point_size);
00232         } else {
00233                 g_string_printf(gs_begin, "<font><span>");
00234         }
00235 
00236         gs_end = g_string_new("</span></font>");
00237 
00238         if (decor & MSIM_TEXT_BOLD) {
00239                 g_string_append(gs_begin, "<b>");
00240                 g_string_prepend(gs_end, "</b>");
00241         }
00242 
00243         if (decor & MSIM_TEXT_ITALIC) {
00244                 g_string_append(gs_begin, "<i>");
00245                 g_string_append(gs_end, "</i>");
00246         }
00247 
00248         if (decor & MSIM_TEXT_UNDERLINE) {
00249                 g_string_append(gs_begin, "<u>");
00250                 g_string_append(gs_end, "</u>");
00251         }
00252 
00253         *begin = g_string_free(gs_begin, FALSE);
00254         *end = g_string_free(gs_end, FALSE);
00255 }

00265 msim_color_to_purple(const char *msim)
00266 {
00267         guint red, green, blue;
00268

00270                 return g_strdup("black");

00410 static int
00411 msim_markup_tag_to_html(MsimSession *session, xmlnode *root, gchar **begin,
00412                 gchar **end)

00414         g_return_val_if_fail(root != NULL, 0);

00427                 msim_markup_i_to_html(session, root, begin, end);
00428         } else {
00429                 purple_debug_info("msim", "msim_markup_tag_to_html: "
00430                                 "unknown tag name=%s, ignoring\n",
00431                                 root->name ? root->name : "(NULL)");
00432                 *begin = g_strdup("");
00433                 *end = g_strdup("");
00434         }
00435         return 0;

00441 static int
00442 html_tag_to_msim_markup(MsimSession *session, xmlnode *root, gchar **begin,
00443                 gchar **end)
00444 {
00445         int ret = 0;
00446 
00447         if (!purple_utf8_strcasecmp(root->name, "root") ||
00448             !purple_utf8_strcasecmp(root->name, "html")) {
00449                 *begin = g_strdup("");
00450                 *end = g_strdup("");
00451         /* TODO: Coalesce nested tags into one <f> tag!
00452          * Currently, the 's' value will be overwritten when b/i/u is nested
00453          * within another one, and only the inner-most formatting will be
00454          * applied to the text. */
00455         } else if (!purple_utf8_strcasecmp(root->name, "b")) {
00456                 if (root->child->type == XMLNODE_TYPE_DATA) {
00457                         *begin = g_strdup_printf("<f s='%d'>", MSIM_TEXT_BOLD);
00458                         *end = g_strdup("</f>");
00459                 } else {
00460                         if (!purple_utf8_strcasecmp(root->child->name,"i")) {
00461                                 ret++;
00462                                 if (root->child->child->type == XMLNODE_TYPE_DATA) {
00463                                         *begin = g_strdup_printf("<f s='%d'>", (MSIM_TEXT_BOLD + MSIM_TEXT_ITALIC));
00464                                         *end = g_strdup("</f>");
00465                                 } else {
00466                                         if (!purple_utf8_strcasecmp(root->child->child->name,"u")) {
00467                                                 ret++;
00468                                                 *begin = g_strdup_printf("<f s='%d'>", (MSIM_TEXT_BOLD + MSIM_TEXT_ITALIC + MSIM_TEXT_UNDERLINE));
00469                                                 *end = g_strdup("</f>");
00470                                         }
00471                                 }
00472                         } else if (!purple_utf8_strcasecmp(root->child->name,"u")) {
00473                                 ret++;
00474                                 *begin = g_strdup_printf("<f s='%d'>", (MSIM_TEXT_BOLD + MSIM_TEXT_UNDERLINE));
00475                                 *end = g_strdup("</f>");
00476                         }
00477                 }
00478         } else if (!purple_utf8_strcasecmp(root->name, "i")) {
00479                 if (root->child->type == XMLNODE_TYPE_DATA) {
00480                         *begin = g_strdup_printf("<f s='%d'>", MSIM_TEXT_ITALIC);
00481                         *end = g_strdup("</f>");
00482                 } else {
00483                         if (!purple_utf8_strcasecmp(root->child->name,"u")) {
00484                                 ret++;
00485                                 *begin = g_strdup_printf("<f s='%d'>", (MSIM_TEXT_ITALIC + MSIM_TEXT_UNDERLINE));
00486                                 *end = g_strdup("</f>");
00487                         }
00488                 }
00489         } else if (!purple_utf8_strcasecmp(root->name, "u")) {
00490                 *begin = g_strdup_printf("<f s='%d'>", MSIM_TEXT_UNDERLINE);
00491                 *end = g_strdup("</f>");
00492         } else if (!purple_utf8_strcasecmp(root->name, "a")) {
00493                 const gchar *href;
00494                 gchar *link_text;
00495 
00496                 href = xmlnode_get_attrib(root, "href");
00497 
00498                 if (!href) {
00499                         href = xmlnode_get_attrib(root, "HREF");
00500                 }
00501 
00502                 link_text = xmlnode_get_data(root);
00503 
00504                 if (href) {
00505                         if (g_str_equal(link_text, href)) {
00506                                 /* Purple gives us: <a href="URL">URL</a>
00507                                  * Translate to <a h='URL' />
00508                                  * Displayed as text of URL with link to URL
00509                                  */
00510                                 *begin = g_strdup_printf("<a h='%s' />", href);
00511                         } else {
00512                                 /* But if we get: <a href="URL">text</a>
00513                                  * Translate to: text: <a h='URL' />
00514                                  *
00515                                  * Because official client only supports self-closed <a>
00516                                  * tags; you can't change the link text.
00517                                  */
00518                                 *begin = g_strdup_printf("%s: <a h='%s' />", link_text, href);
00519                         }
00520                 } else {
00521                         *begin = g_strdup("<a />");
00522                 }
00523 
00524                 /* Sorry, kid. MySpace doesn't support you within <a> tags. */
00525                 xmlnode_free(root->child);
00526                 g_free(link_text);
00527                 root->child = NULL;
00528 
00529                 *end = g_strdup("");
00530         } else if (!purple_utf8_strcasecmp(root->name, "font")) {
00531                 GString *tmpbegin, *tmpend;
00532                 const gchar *size;
00533                 const gchar *face;
00534                 const gchar *color;
00535 
00536                 size = xmlnode_get_attrib(root, "size");
00537                 face = xmlnode_get_attrib(root, "face");
00538                 color = xmlnode_get_attrib(root, "color");
00539 
00540                 tmpbegin = g_string_new("<f");
00541                 tmpend = g_string_new("</f>");
00542 
00543                 if (face != NULL)
00544                         g_string_append_printf(tmpbegin, " f='%s'", face);
00545 
00546                 if (size != NULL)
00547                         g_string_append_printf(tmpbegin, " h='%d'",
00548                                          msim_point_to_height(session,
00549                                                  msim_purple_size_to_point(session, atoi(size))));
00550 
00551                 /* Close the <f> tag */
00552                 g_string_append(tmpbegin, ">");
00553 
00554                 if (color != NULL) {
00555                         g_string_append_printf(tmpbegin, "<c v='%s'>", color);
00556                         g_string_prepend(tmpend, "</c>");
00557                 }
00558 
00559                 *begin = g_string_free(tmpbegin, FALSE);
00560                 *end = g_string_free(tmpend, FALSE);
00561 
00562         } else if (!purple_utf8_strcasecmp(root->name, "body")) {
00563                 const gchar *bgcolor;
00564 
00565                 bgcolor = xmlnode_get_attrib(root, "bgcolor");
00566 
00567                 if (bgcolor != NULL) {
00568                         *begin = g_strdup_printf("<b v='%s'>", bgcolor);
00569                         *end = g_strdup("</b>");
00570                 }
00571 
00572         } else {
00573                 gchar *err;
00574 
00575 #ifdef MSIM_MARKUP_SHOW_UNKNOWN_TAGS
00576                 *begin = g_strdup_printf("[%s]", root->name);
00577                 *end = g_strdup_printf("[/%s]", root->name);
00578 #else
00579                 *begin = g_strdup("");
00580                 *end = g_strdup("");
00581 #endif
00582 
00583                 err = g_strdup_printf("html_tag_to_msim_markup: unrecognized "
00584                         "HTML tag %s was sent by the IM client; ignoring",
00585                         root->name ? root->name : "(NULL)");
00586                 msim_unrecognized(NULL, NULL, err);
00587                 g_free(err);
00588         }
00589         return ret;
00590 }

00599 static void
00600 msim_convert_xmlnode(MsimSession *session, GString *out, xmlnode *root, MSIM_XMLNODE_CONVERT f, int nodes_processed)
00601 {
00602         xmlnode *node;
00603         gchar *begin, *end, *tmp;
00604         int descended = nodes_processed;
00605 
00606         if (!root || !root->name)
00607                 return;
00608 
00609         purple_debug_info("msim", "msim_convert_xmlnode: got root=%s\n",
00610                         root->name);
00611 
00612         begin = end = NULL;
00613 
00614         if (descended == 0) /* We've not formatted this yet.. :) */
00615                 descended = f(session, root, &begin, &end); /* Get the value that our format function has already descended for us */
00616 
00617         g_string_append(out, begin);
00618         g_free(begin);
