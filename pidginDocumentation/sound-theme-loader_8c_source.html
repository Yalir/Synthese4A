00034 static PurpleTheme *
00035 purple_sound_loader_build(const gchar *dir)
00036 {
00037         xmlnode *root_node = NULL, *sub_node;
00038         gchar *filename_full, *data = NULL;
00039         PurpleSoundTheme *theme = NULL;
00040         const gchar *name;
00041 
00042         /* Find the theme file */
00043         g_return_val_if_fail(dir != NULL, NULL);
00044         filename_full = g_build_filename(dir, "theme.xml", NULL);
00045 
00046         if (g_file_test(filename_full, G_FILE_TEST_IS_REGULAR))
00047                 root_node = xmlnode_from_file(dir, "theme.xml", "sound themes", "sound-theme-loader");
00048 
00049         g_free(filename_full);
00050         if (root_node == NULL)
00051                 return NULL;
00052 
00053         name = xmlnode_get_attrib(root_node, "name");
00054 
00055         if (name && purple_strequal(xmlnode_get_attrib(root_node, "type"), "sound")) {
00056                 /* Parse the tree */
00057                 sub_node = xmlnode_get_child(root_node, "description");
00058                 data = xmlnode_get_data(sub_node);
00059 
00060                 if (xmlnode_get_attrib(root_node, "name") != NULL) {
00061                         theme = g_object_new(PURPLE_TYPE_SOUND_THEME,
00062                                         "type", "sound",
00063                                         "name", name,
00064                                         "author", xmlnode_get_attrib(root_node, "author"),
00065                                         "image", xmlnode_get_attrib(root_node, "image"),
00066                                         "directory", dir,
00067                                         "description", data, NULL);
00068 
00069                         sub_node = xmlnode_get_child(root_node, "event");
00070 
00071                         while (sub_node) {
00072                                 purple_sound_theme_set_file(theme,
00073                                                 xmlnode_get_attrib(sub_node, "name"),
00074                                                 xmlnode_get_attrib(sub_node, "file"));
00075                                 sub_node = xmlnode_get_next_twin(sub_node);
00076                         }
00077                 }
00078         } else purple_debug_warning("sound-theme-loader", "Missing attribute or problem with the root element\n");
00079 
00080         xmlnode_free(root_node);
00081         g_free(data);
00082         return PURPLE_THEME(theme);
00083 }
