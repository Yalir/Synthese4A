00152 silcpurple_buddy_keyagr_do(PurpleConnection *gc, const char *name,
00153                          gboolean force_local)
00154 {
00155         SilcPurple sg = gc->proto_data;
00156         SilcClientEntry *clients;
00157         SilcUInt32 clients_count;
00158         char *local_ip = NULL, *remote_ip = NULL;
00159         gboolean local = TRUE;
00160         char *nickname;
00161         SilcPurpleKeyAgr a;
00162 
00163         if (!sg->conn || !name)
00164                 return;
00165 
00166         if (!silc_parse_userfqdn(name, &nickname, NULL))
00167                 return;
00168 
00169         /* Find client entry */
00170         clients = silc_client_get_clients_local(sg->client, sg->conn, nickname, name,
00171                                                 &clients_count);
00172         if (!clients) {
00173                 /* Resolve unknown user */
00174                 SilcPurpleResolve r = silc_calloc(1, sizeof(*r));
00175                 if (!r)
00176                         return;
00177                 r->nick = g_strdup(name);
00178                 r->gc = gc;
00179                 silc_client_get_clients(sg->client, sg->conn, nickname, NULL,
00180                                         silcpurple_buddy_keyagr_resolved, r);
00181                 silc_free(nickname);
00182                 return;
00183         }
00184 
00185         /* Resolve the local IP from the outgoing socket connection.  We resolve
00186            it to check whether we have a private range IP address or public IP
00187            address.  If we have public then we will assume that we are not behind
00188            NAT and will provide automatically the point of connection to the
00189            agreement.  If we have private range address we assume that we are
00190            behind NAT and we let the responder provide the point of connection.
00191 
00192            The algorithm also checks the remote IP address of server connection.
00193            If it is private range address and we have private range address we
00194            assume that we are chatting in LAN and will provide the point of
00195            connection.
00196 
00197            Naturally this algorithm does not always get things right. */
00198 
00199         if (silc_net_check_local_by_sock(sg->conn->sock->sock, NULL, &local_ip)) {
00200                 /* Check if the IP is private */
00201                 if (!force_local && silcpurple_ip_is_private(local_ip)) {
00202                         local = FALSE;
00203 
00204                         /* Local IP is private, resolve the remote server IP to see whether
00205                            we are talking to Internet or just on LAN. */
00206                         if (silc_net_check_host_by_sock(sg->conn->sock->sock, NULL,
00207                                                         &remote_ip))
00208                                 if (silcpurple_ip_is_private(remote_ip))
00209                                         /* We assume we are in LAN.  Let's provide
00210                                            the connection point. */
00211                                         local = TRUE;
00212                 }
00213         }
00214 
00215         if (force_local)
00216                 local = TRUE;
00217 
00218         if (local && !local_ip)
00219                 local_ip = silc_net_localip();
00220 
00221         a = silc_calloc(1, sizeof(*a));
00222         if (!a)
00223                 return;
00224         a->responder = local;
00225 
00226         /* Send the key agreement request */
00227         silc_client_send_key_agreement(sg->client, sg->conn, clients[0],
00228                                        local ? local_ip : NULL, NULL, 0, 60,
00229                                        silcpurple_buddy_keyagr_cb, a);
00230 
00231         silc_free(local_ip);
00232         silc_free(remote_ip);
00233         silc_free(clients);
00234 }

00744 static void
00745 silcpurple_add_buddy_save(bool success, void *context)
00746 {
00747         SilcPurpleBuddyRes r = context;
00748         PurpleBuddy *b = r->b;
00749         SilcClient client = r->client;
00750         SilcClientEntry client_entry;
00751         SilcAttributePayload attr;
00752         SilcAttribute attribute;
00753         SilcVCardStruct vcard;
00754         SilcAttributeObjMime message, extension;
00755 #ifdef SILC_ATTRIBUTE_USER_ICON
00756         SilcAttributeObjMime usericon;
00757 #endif
00758         SilcAttributeObjPk serverpk, usersign, serversign;
00759         gboolean usign_success = TRUE, ssign_success = TRUE;
00760         char filename[512], filename2[512], *fingerprint = NULL, *tmp;
00761         SilcUInt32 len;
00762         int i;
00763 
00764         if (!success) {
00765                 /* The user did not trust the public key. */
00766                 silcpurple_add_buddy_pk_no(r);
00767                 silc_free(r);
00768                 return;
00769         }
00770 
00771         if (r->offline) {
00772                 /* User is offline.  Associate the imported public key with
00773                    this user. */
00774                 fingerprint = silc_hash_fingerprint(NULL, r->offline_pk,
00775                                                     r->offline_pk_len);
00776                 for (i = 0; i < strlen(fingerprint); i++)
00777                         if (fingerprint[i] == ' ')
00778                                 fingerprint[i] = '_';
00779                 g_snprintf(filename, sizeof(filename) - 1,
00780                            "%s" G_DIR_SEPARATOR_S "clientkeys" G_DIR_SEPARATOR_S "clientkey_%s.pub",
00781                            silcpurple_silcdir(), fingerprint);
00782                 purple_blist_node_set_string((PurpleBlistNode *)b, "public-key", filename);
00783                 purple_prpl_got_user_status(purple_buddy_get_account(r->b), purple_buddy_get_name(r->b), SILCPURPLE_STATUS_ID_OFFLINE, NULL);
00784                 silc_free(fingerprint);
00785                 silc_free(r->offline_pk);
00786                 silc_free(r);
00787                 return;
00788         }
00789 
00790         /* Get the client entry. */
00791         client_entry = silc_client_get_client_by_id(r->client, r->conn,
00792                                                     &r->client_id);
00793         if (!client_entry) {
00794                 silc_free(r);
00795                 return;
00796         }
00797 
00798         memset(&vcard, 0, sizeof(vcard));
00799         memset(&message, 0, sizeof(message));
00800         memset(&extension, 0, sizeof(extension));
00801 #ifdef SILC_ATTRIBUTE_USER_ICON
00802         memset(&usericon, 0, sizeof(usericon));
00803 #endif
00804         memset(&serverpk, 0, sizeof(serverpk));
00805         memset(&usersign, 0, sizeof(usersign));
00806         memset(&serversign, 0, sizeof(serversign));
00807 
00808         /* Now that we have the public key and we trust it now we
00809            save the attributes of the buddy and update its status. */
00810 
00811         if (client_entry->attrs) {
00812                 silc_dlist_start(client_entry->attrs);
00813                 while ((attr = silc_dlist_get(client_entry->attrs))
00814                        != SILC_LIST_END) {
00815                         attribute = silc_attribute_get_attribute(attr);
00816 
00817                         switch (attribute) {
00818                         case SILC_ATTRIBUTE_USER_INFO:
00819                                 if (!silc_attribute_get_object(attr, (void *)&vcard,
00820                                                                sizeof(vcard)))
00821                                         continue;
00822                                 break;
00823 
00824                         case SILC_ATTRIBUTE_STATUS_MESSAGE:
00825                                 if (!silc_attribute_get_object(attr, (void *)&message,
00826                                                                sizeof(message)))
00827                                         continue;
00828                                 break;
00829 
00830                         case SILC_ATTRIBUTE_EXTENSION:
00831                                 if (!silc_attribute_get_object(attr, (void *)&extension,
00832                                                                sizeof(extension)))
00833                                         continue;
00834                                 break;
00835 
00836 #ifdef SILC_ATTRIBUTE_USER_ICON
00837                         case SILC_ATTRIBUTE_USER_ICON:
00838                                 if (!silc_attribute_get_object(attr, (void *)&usericon,
00839                                                                sizeof(usericon)))
00840                                         continue;
00841                                 break;
00842 #endif
00843 
00844                         case SILC_ATTRIBUTE_SERVER_PUBLIC_KEY:
00845                                 if (serverpk.type)
00846                                         continue;
00847                                 if (!silc_attribute_get_object(attr, (void *)&serverpk,
00848                                                                sizeof(serverpk)))
00849                                         continue;
00850                                 break;
00851 
00852                         case SILC_ATTRIBUTE_USER_DIGITAL_SIGNATURE:
00853                                 if (usersign.data)
00854                                         continue;
00855                                 if (!silc_attribute_get_object(attr, (void *)&usersign,
00856                                                                sizeof(usersign)))
00857                                         continue;
00858                                 break;
00859 
00860                         case SILC_ATTRIBUTE_SERVER_DIGITAL_SIGNATURE:
00861                                 if (serversign.data)
00862                                         continue;
00863                                 if (!silc_attribute_get_object(attr, (void *)&serversign,
00864                                                                sizeof(serversign)))
00865                                         continue;
00866                                 break;
00867 
00868                         default:
00869                                 break;
00870                         }
00871                 }
00872         }
00873 
00874         /* Verify the attribute signatures */
00875 
00876         if (usersign.data) {
00877                 SilcPKCS pkcs;
00878                 unsigned char *verifyd;
00879                 SilcUInt32 verify_len;
00880 
00881                 silc_pkcs_alloc((unsigned char*)"rsa", &pkcs);
00882                 verifyd = silc_attribute_get_verify_data(client_entry->attrs,
00883                                                          FALSE, &verify_len);
00884                 if (verifyd && silc_pkcs_public_key_set(pkcs, client_entry->public_key)){
00885                         if (!silc_pkcs_verify_with_hash(pkcs, client->sha1hash,
00886                                                         usersign.data,
00887                                                         usersign.data_len,
00888                                                         verifyd, verify_len))
00889                                 usign_success = FALSE;
00890                 }
00891                 silc_free(verifyd);
00892         }
00893 
00894         if (serversign.data && !strcmp(serverpk.type, "silc-rsa")) {
00895                 SilcPublicKey public_key;
00896                 SilcPKCS pkcs;
00897                 unsigned char *verifyd;
00898                 SilcUInt32 verify_len;
00899 
00900                 if (silc_pkcs_public_key_decode(serverpk.data, serverpk.data_len,
00901                                                 &public_key)) {
00902                         silc_pkcs_alloc((unsigned char *)"rsa", &pkcs);
00903                         verifyd = silc_attribute_get_verify_data(client_entry->attrs,
00904                                                                  TRUE, &verify_len);
00905                         if (verifyd && silc_pkcs_public_key_set(pkcs, public_key)) {
00906                                 if (!silc_pkcs_verify_with_hash(pkcs, client->sha1hash,
00907                                                                serversign.data,
00908                                                                serversign.data_len,
00909                                                                verifyd, verify_len))
00910                                         ssign_success = FALSE;
00911                         }
00912                         silc_pkcs_public_key_free(public_key);
00913                         silc_free(verifyd);
00914                 }
00915         }
00916 
00917         fingerprint = silc_fingerprint(client_entry->fingerprint,
00918                                        client_entry->fingerprint_len);
00919         for (i = 0; i < strlen(fingerprint); i++)
00920                 if (fingerprint[i] == ' ')
00921                         fingerprint[i] = '_';
00922 
00923         if (usign_success || ssign_success) {
00924                 struct passwd *pw;
00925                 struct stat st;
00926 
00927                 memset(filename2, 0, sizeof(filename2));
00928 
00929                 /* Filename for dir */
00930                 tmp = fingerprint + strlen(fingerprint) - 9;
00931                 g_snprintf(filename, sizeof(filename) - 1,
00932                            "%s" G_DIR_SEPARATOR_S "friends" G_DIR_SEPARATOR_S "%s",
00933                            silcpurple_silcdir(), tmp);
00934 
00935                 pw = getpwuid(getuid());
00936                 if (!pw)
00937                         return;
00938 
00939                 /* Create dir if it doesn't exist */
00940                 if ((g_stat(filename, &st)) == -1) {
00941                         if (errno == ENOENT) {
00942                                 if (pw->pw_uid == geteuid()) {
00943                                         int ret = g_mkdir(filename, 0755);
00944                                         if (ret < 0)
00945                                                 return;
00946                                 }
00947                         }
00948                 }
00949 
00950                 /* Save VCard */
00951                 g_snprintf(filename2, sizeof(filename2) - 1,
00952                            "%s" G_DIR_SEPARATOR_S "vcard", filename);
00953                 if (vcard.full_name) {
00954                         tmp = (char *)silc_vcard_encode(&vcard, &len);
00955                         silc_file_writefile(filename2, tmp, len);
00956                         silc_free(tmp);
00957                 }
00958 
00959                 /* Save status message */
00960                 if (message.mime) {
00961                         memset(filename2, 0, sizeof(filename2));
00962                         g_snprintf(filename2, sizeof(filename2) - 1,
00963                                    "%s" G_DIR_SEPARATOR_S "status_message.mime",
00964                                    filename);
00965                         silc_file_writefile(filename2, (char *)message.mime,
00966                                             message.mime_len);
00967                 }
00968 
00969                 /* Save extension data */
00970                 if (extension.mime) {
00971                         memset(filename2, 0, sizeof(filename2));
00972                         g_snprintf(filename2, sizeof(filename2) - 1,
00973                                    "%s" G_DIR_SEPARATOR_S "extension.mime",
00974                                    filename);
00975                         silc_file_writefile(filename2, (char *)extension.mime,
00976                                             extension.mime_len);
00977                 }
00978 
00979 #ifdef SILC_ATTRIBUTE_USER_ICON
00980                 /* Save user icon */
00981                 if (usericon.mime) {
00982                         SilcMime m = silc_mime_decode(usericon.mime,
00983                                                       usericon.mime_len);
00984                         if (m) {
00985                                 const char *type = silc_mime_get_field(m, "Content-Type");
00986                                 if (!strcmp(type, "image/jpeg") ||
00987                                     !strcmp(type, "image/gif") ||
00988                                     !strcmp(type, "image/bmp") ||
00989                                     !strcmp(type, "image/png")) {
00990                                         const unsigned char *data;
00991                                         SilcUInt32 data_len;
00992                                         data = silc_mime_get_data(m, &data_len);
00993                                         if (data) {
00994                                                 /* TODO: Check if SILC gives us something to use as the checksum instead */
00995                                                 purple_buddy_icons_set_for_user(purple_buddy_get_account(r->b), purple_buddy_get_name(r->b), g_memdup(data, data_len), data_len, NULL);
00996                                         }
00997                                 }
00998                                 silc_mime_free(m);
00999                         }
01000                 }
01001 #endif
01002         }
01003 
01004         /* Save the public key path to buddy properties, as it is used
01005            to identify the buddy in the network (and not the nickname). */
01006         memset(filename, 0, sizeof(filename));
01007         g_snprintf(filename, sizeof(filename) - 1,
01008                    "%s" G_DIR_SEPARATOR_S "clientkeys" G_DIR_SEPARATOR_S "clientkey_%s.pub",
01009                    silcpurple_silcdir(), fingerprint);
01010         purple_blist_node_set_string((PurpleBlistNode *)b, "public-key", filename);
01011 
01012         /* Update online status */
01013         purple_prpl_got_user_status(purple_buddy_get_account(r->b), purple_buddy_get_name(r->b), SILCPURPLE_STATUS_ID_AVAILABLE, NULL);
01014 
01015         /* Finally, start watching this user so we receive its status
01016            changes from the server */
01017         g_snprintf(filename2, sizeof(filename2) - 1, "+%s", filename);
01018         silc_client_command_call(r->client, r->conn, NULL, "WATCH", "-pubkey",
01019                                  filename2, NULL);
01020 
01021         silc_free(fingerprint);
01022         silc_free(r);
01023 }

01198                             SilcClientConnection conn,

01200                             SilcUInt32 clients_count,
01201                             void *context)
01202 {

01204         PurpleBuddy *b = r->b;
01205         SilcAttributePayload pub;
01206         SilcAttributeObjPk userpk;
01207         unsigned char *pk;
01208         SilcUInt32 pk_len;
01209         const char *filename;
01210 
01211         filename = purple_blist_node_get_string((PurpleBlistNode *)b, "public-key");
01212 
01213         /* If the buddy is offline/nonexistent, we will require user
01214            to associate a public key with the buddy or the buddy
01215            cannot be added. */
01216         if (!clients_count) {
01217                 if (r->init) {
01218                         silc_free(r);
01219                         return;
01220                 }
01221 
01222                 r->offline = TRUE;
01223                 /* If the user has already associated a public key, try loading it
01224                  * before prompting the user to load it again */
01225                 if (filename != NULL)
01226                         silcpurple_add_buddy_ask_import(r, filename);
01227                 else
01228                         silcpurple_add_buddy_ask_pk(r);
01229                 return;
01230         }
01231 
01232         /* If more than one client was found with nickname, we need to verify
01233            from user which one is the correct. */
01234         if (clients_count > 1 && !r->pubkey_search) {
01235                 if (r->init) {
01236                         silc_free(r);
01237                         return;
01238                 }
01239 
01240                 silcpurple_add_buddy_select(r, clients, clients_count);
01241                 return;
01242         }
01243 
01244         /* If we searched using public keys and more than one entry was found
01245            the same person is logged on multiple times. */
01246         if (clients_count > 1 && r->pubkey_search && b->name) {
01247                 if (r->init) {
01248                         /* Find the entry that closest matches to the
01249                            buddy nickname. */
01250                         int i;
01251                         for (i = 0; i < clients_count; i++) {
01252                                 if (!g_ascii_strncasecmp(b->name, clients[i]->nickname,
01253                                                  strlen(b->name))) {
01254                                         clients[0] = clients[i];
01255                                         break;
01256                                 }
01257                         }
01258                 } else {
01259                         /* Verify from user which one is correct */
01260                         silcpurple_add_buddy_select(r, clients, clients_count);
01261                         return;
01262                 }
01263         }
01264

01266            that before adding the buddy. */

01269         r->client_id = *clients[0]->id;
01270 
01271         /* Get the public key from attributes, if not present then
01272            resolve it with GETKEY unless we have it cached already. */

01275                                         SILC_ATTRIBUTE_USER_PUBLIC_KEY);

01277                                                        sizeof(userpk))) {
01278                         /* Get public key with GETKEY */

01280                                                  "GETKEY", clients[0]->nickname, NULL);
01281                         silc_client_command_pending(conn, SILC_COMMAND_GETKEY,
01282                                                     conn->cmd_ident,
01283                                                     (SilcCommandCb)silcpurple_add_buddy_getkey_cb,
01284                                                     r);
01285                         return;
01286                 }
01287                 if (!silc_pkcs_public_key_decode(userpk.data, userpk.data_len,
01288                                                  &clients[0]->public_key))
01289                         return;
01290                 silc_free(userpk.data);
01291         } else if (filename && !clients[0]->public_key) {
01292                 if (!silc_pkcs_load_public_key(filename, &clients[0]->public_key,
01293                                                SILC_PKCS_FILE_PEM) &&
01294                     !silc_pkcs_load_public_key(filename, &clients[0]->public_key,
01295                                                SILC_PKCS_FILE_BIN)) {
01296                         /* Get public key with GETKEY */
01297                         silc_client_command_call(client, conn, NULL,
01298                                                  "GETKEY", clients[0]->nickname, NULL);
01299                         silc_client_command_pending(conn, SILC_COMMAND_GETKEY,
01300                                                     conn->cmd_ident,
01301                                                     (SilcCommandCb)silcpurple_add_buddy_getkey_cb,
01302                                                     r);
01303                         return;
01304                 }
01305         } else if (!clients[0]->public_key) {
01306                 /* Get public key with GETKEY */
01307                 silc_client_command_call(client, conn, NULL,
01308                                          "GETKEY", clients[0]->nickname, NULL);
01309                 silc_client_command_pending(conn, SILC_COMMAND_GETKEY,
01310                                             conn->cmd_ident,
01311                                             (SilcCommandCb)silcpurple_add_buddy_getkey_cb,
01312                                             r);
01313                 return;
01314         }
01315

01317         pk = silc_pkcs_public_key_encode(clients[0]->public_key, &pk_len);

01319                                    SILC_SOCKET_TYPE_CLIENT,

01321                                    silcpurple_add_buddy_save, r);
