01433 silcpurple_send_im(PurpleConnection *gc, const char *who, const char *message,
01434                    PurpleMessageFlags flags)
01435 {

01437         SilcClient client = sg->client;
01438         SilcClientConnection conn = sg->conn;
01439         SilcDList clients;
01440         SilcClientEntry client_entry;

01444         gboolean sign = purple_account_get_bool(sg->account, "sign-verify", FALSE);
01445         SilcDList list;
01446 
01447         if (!who || !message)
01448                 return 0;

01472

01488                                         silcpurple_send_im_resolved, im);

01515                 }

01517

01520                                                mflags, sg->sha1hash,
