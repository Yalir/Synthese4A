00185         char *tmp, *x;
00186         int i, j;
00187 
00188         if (!name)
00189                 return NULL;
00190 
00191         x = strchr(name, '-');
00192         if (!x)
00193                 return NULL;
00194 
00195         x = strchr(x + 1, '-');
00196         if (!x)
00197                 return NULL;
00198 
00199         tmp = g_strdup(++x);
00200 
00201         for (i = 0, j = 0; x[i]; i++) {
00202                 char hex[3];
00203                 if (x[i] != '%') {
00204                         tmp[j++] = x[i];
00205                         continue;
00206                 }
00207                 strncpy(hex, x + ++i, 2);
00208                 hex[2] = 0;
00209                 i++;
00210                 tmp[j++] = strtol(hex, NULL, 16);
00211         }
00212

04520 int oscar_send_chat(PurpleConnection *gc, int id, const char *message, PurpleMessageFlags flags)
04521 {
04522         OscarData *od = purple_connection_get_protocol_data(gc);
04523         PurpleConversation *conv = NULL;
04524         struct chat_connection *c = NULL;
04525         char *buf, *buf2, *buf3;
04526         guint16 charset;
04527         char *charsetstr;
04528         gsize len;
04529 
04530         if (!(conv = purple_find_chat(gc, id)))
04531                 return -EINVAL;
04532 
04533         if (!(c = find_oscar_chat_by_conv(gc, conv)))
04534                 return -EINVAL;
04535 
04536         buf = purple_strdup_withhtml(message);
04537 
04538         if (strstr(buf, "<IMG "))
04539                 purple_conversation_write(conv, "",
04540                         _("Your IM Image was not sent. "
04541                           "You cannot send IM Images in AIM chats."),
04542                         PURPLE_MESSAGE_ERROR, time(NULL));
04543 
04544         buf2 = oscar_encode_im(buf, &len, &charset, &charsetstr);
04545         /*
04546          * Evan S. suggested that maxvis really does mean "number of
04547          * visible characters" and not "number of bytes"
04548          */
04549         if ((len > c->maxlen) || (len > c->maxvis)) {
04550                 /* If the length was too long, try stripping the HTML and then running it back through
04551                  * purple_strdup_withhtml() and the encoding process. The result may be shorter. */
04552                 g_free(buf2);
04553 
04554                 buf3 = purple_markup_strip_html(buf);
04555                 g_free(buf);
04556 
04557                 buf = purple_strdup_withhtml(buf3);
04558                 g_free(buf3);
04559 
04560                 buf2 = oscar_encode_im(buf, &len, &charset, &charsetstr);
04561 
04562                 if ((len > c->maxlen) || (len > c->maxvis)) {
04563                         purple_debug_warning("oscar",
04564                                         "Could not send %s because (%" G_GSIZE_FORMAT " > maxlen %i) or (%" G_GSIZE_FORMAT " > maxvis %i)\n",
04565                                         buf2, len, c->maxlen, len, c->maxvis);
04566                         g_free(buf);
04567                         g_free(buf2);
04568                         return -E2BIG;
04569                 }
04570 
04571                 purple_debug_info("oscar", "Sending %s as %s because the original was too long.\n",
04572                                 message, buf2);
04573         }
04574 
04575         aim_chat_send_im(od, c->conn, 0, buf2, len, charsetstr, "en");
04576         g_free(buf2);
04577         g_free(buf);
04578 
04579         return 0;
04580 }

05643         }
