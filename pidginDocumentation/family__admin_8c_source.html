00061 static void
00062 infochange(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00063 {
00064         aim_rxcallback_t userfunc;
00065         char *url=NULL, *sn=NULL, *email=NULL;
00066         guint16 perms, tlvcount, err=0;
00067 
00068         perms = byte_stream_get16(bs);
00069         tlvcount = byte_stream_get16(bs);
00070 
00071         while (tlvcount && byte_stream_bytes_left(bs)) {
00072                 guint16 type, length;
00073 
00074                 type = byte_stream_get16(bs);
00075                 length = byte_stream_get16(bs);
00076 
00077                 switch (type) {
00078                         case 0x0001: {
00079                                 g_free(sn);
00080                                 sn = byte_stream_getstr(bs, length);
00081                         } break;
00082 
00083                         case 0x0004: {
00084                                 g_free(url);
00085                                 url = byte_stream_getstr(bs, length);
00086                         } break;
00087 
00088                         case 0x0008: {
00089                                 err = byte_stream_get16(bs);
00090                         } break;
00091 
00092                         case 0x0011: {
00093                                 g_free(email);
00094                                 if (length == 0)
00095                                         email = g_strdup("*suppressed");
00096                                 else
00097                                         email = byte_stream_getstr(bs, length);
00098                         } break;
00099                 }
00100 
00101                 tlvcount--;
00102         }
00103 
00104         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00105                 userfunc(od, conn, frame, (snac->subtype == 0x0005) ? 1 : 0, perms, err, url, sn, email);
00106 
00107         g_free(sn);
00108         g_free(url);
00109         g_free(email);
00110 }

00223 static int
00224 snachandler(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00225 {
00226         if ((snac->subtype == 0x0003) || (snac->subtype == 0x0005)) {
00227                 infochange(od, conn, mod, frame, snac, bs);
00228                 return 1;
00229         } else if (snac->subtype == SNAC_FAMILY_ADMIN)
00230                 return accountconfirm(od, conn, mod, frame, snac, bs);
00231 
00232         return 0;
00233 }
