00317 /*------------------------------------------------------------------------
00318  * The user has selected to change their PIN.
00319  *
00320  *  @param gc           The connection object
00321  *  @param fields       The fields from the request pop-up
00322  */
00323 static void mxit_change_pin_cb( PurpleConnection* gc, PurpleRequestFields* fields )
00324 {
00325         struct MXitSession*             session = (struct MXitSession*) gc->proto_data;
00326         const char*                             pin             = NULL;
00327         const char*                             pin2    = NULL;
00328         const char*                             err             = NULL;
00329         int                                             len;
00330         int                                             i;
00331 
00332         if ( !PURPLE_CONNECTION_IS_VALID( gc ) ) {
00333                 purple_debug_error( MXIT_PLUGIN_ID, "Unable to update PIN; account offline.\n" );
00334                 return;
00335         }
00336 
00337         /* validate pin */
00338         pin = purple_request_fields_get_string( fields, "pin" );
00339         if ( !pin ) {
00340                 err = _( "The PIN you entered is invalid." );
00341                 goto out;
00342         }
00343         len = strlen( pin );
00344         if ( ( len < 4 ) || ( len > 10 ) ) {
00345                 err = _( "The PIN you entered has an invalid length [4-10]." );
00346                 goto out;
00347         }
00348         for ( i = 0; i < len; i++ ) {
00349                 if ( !g_ascii_isdigit( pin[i] ) ) {
00350                         err = _( "The PIN is invalid. It should only consist of digits [0-9]." );
00351                         goto out;
00352                 }
00353         }
00354         pin2 = purple_request_fields_get_string( fields, "pin2" );
00355         if ( ( !pin2 ) || ( strcmp( pin, pin2 ) != 0 ) ) {
00356                 err = _( "The two PINs you entered do not match." );
00357                 goto out;
00358         }
00359 
00360 out:
00361         if ( !err ) {
00362                 /* update PIN in account */
00363                 purple_account_set_password( session->acc, pin );
00364 
00365                 /* update session object */
00366                 g_free( session->encpwd );
00367                 session->encpwd = mxit_encrypt_password( session );
00368 
00369                 /* send the update request to MXit */
00370                 mxit_send_extprofile_update( session, session->encpwd, 0, NULL );
00371         }
00372         else {
00373                 /* show error to user */
00374                 mxit_popup( PURPLE_NOTIFY_MSG_ERROR, _( "PIN Update Error" ), err );
00375         }
00376 }
