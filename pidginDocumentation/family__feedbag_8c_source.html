00021 /*
00022  * Family 0x0013 - Server-Side/Stored Information.
00023  *
00024  * Relatively new facility that allows certain types of information, such as
00025  * a user's buddy list, permit/deny list, and permit/deny preferences, to be
00026  * stored on the server, so that they can be accessed from any client.
00027  *
00028  * We keep 2 copies of SSI data:
00029  * 1) An exact copy of what is stored on the AIM servers.
00030  * 2) A local copy that we make changes to, and then send diffs
00031  *    between this and the exact copy to keep them in sync.
00032  *
00033  * All the "aim_ssi_itemlist_bleh" functions near the top just modify the list
00034  * that is given to them (i.e. they don't send SNACs).
00035  *
00036  * The SNAC sending and receiving functions are lower down in the file, and
00037  * they're simpler.  They are in the order of the subtypes they deal with,
00038  * starting with the request rights function (subtype 0x0002), then parse
00039  * rights (subtype 0x0003), then--well, you get the idea.
00040  *
00041  * This is entirely too complicated.
00042  * You don't know the half of it.
00043  *
00044  */

00176

00180         /* Set the group ID# and buddy ID# */

00182         new->bid = bid;

00184                 if ((new->gid == 0xFFFF) && name) {

00207                                                 exists = TRUE;

00220                                                 break;
00221                                         }
00222                         } while (exists);
00223                 }

00225

00227         new->type = type;
00228

00230         new->data = aim_tlvlist_copy(data);
00231

00233         if (*list) {

00235                         new->next = *list;

00241                         prev->next = new;
00242                 }
00243         } else {
00244                 new->next = *list;
00245                 *list = new;
00246         }
00247

00533 static int aim_ssi_sync(OscarData *od)
00534 {
00535         struct aim_ssi_item *cur1, *cur2;
00536         struct aim_ssi_tmp *cur, *new;
00537         int n = 0;
00538         GString *debugstr = g_string_new("");
00539 
00540         /*
00541          * The variable "n" is used to limit the number of addmoddel's that
00542          * are performed in a single SNAC.  It will hopefully keep the size
00543          * of the SNAC below the maximum SNAC size.
00544          */
00545 
00546         if (!od)
00547                 return -EINVAL;
00548 
00549         /* If we're waiting for an ack, we shouldn't do anything else */
00550         if (od->ssi.waiting_for_ack)
00551                 return 0;
00552 
00553         /*
00554          * Compare the 2 lists and create an aim_ssi_tmp for each difference.
00555          * We should only send either additions, modifications, or deletions
00556          * before waiting for an acknowledgement.  So first do deletions, then
00557          * additions, then modifications.  Also, both the official and the local
00558          * list should be in ascending numerical order for the group ID#s and the
00559          * buddy ID#s, which makes things more efficient.  I think.
00560          */
00561 
00562         /* Deletions */
00563         if (!od->ssi.pending) {
00564                 for (cur1=od->ssi.official; cur1 && (n < 15); cur1=cur1->next) {
00565                         if (!aim_ssi_itemlist_find(od->ssi.local, cur1->gid, cur1->bid)) {
00566                                 n++;
00567                                 new = g_new(struct aim_ssi_tmp, 1);
00568                                 new->action = SNAC_SUBTYPE_FEEDBAG_DEL;
00569                                 new->ack = 0xffff;
00570                                 new->name = NULL;
00571                                 new->item = cur1;
00572                                 new->next = NULL;
00573                                 if (od->ssi.pending) {
00574                                         for (cur=od->ssi.pending; cur->next; cur=cur->next);
00575                                         cur->next = new;
00576                                 } else
00577                                         od->ssi.pending = new;
00578                                 aim_ssi_item_debug_append(debugstr, "Deleting item ", cur1);
00579                         }
00580                 }
00581         }
00582 
00583         /* Additions */
00584         if (!od->ssi.pending) {
00585                 for (cur1=od->ssi.local; cur1 && (n < 15); cur1=cur1->next) {
00586                         if (!aim_ssi_itemlist_find(od->ssi.official, cur1->gid, cur1->bid)) {
00587                                 n++;
00588                                 new = g_new(struct aim_ssi_tmp, 1);
00589                                 new->action = SNAC_SUBTYPE_FEEDBAG_ADD;
00590                                 new->ack = 0xffff;
00591                                 new->name = NULL;
00592                                 new->item = cur1;
00593                                 new->next = NULL;
00594                                 if (od->ssi.pending) {
00595                                         for (cur=od->ssi.pending; cur->next; cur=cur->next);
00596                                         cur->next = new;
00597                                 } else
00598                                         od->ssi.pending = new;
00599                                 aim_ssi_item_debug_append(debugstr, "Adding item ", cur1);
00600                         }
00601                 }
00602         }
00603 
00604         /* Modifications */
00605         if (!od->ssi.pending) {
00606                 for (cur1=od->ssi.local; cur1 && (n < 15); cur1=cur1->next) {
00607                         cur2 = aim_ssi_itemlist_find(od->ssi.official, cur1->gid, cur1->bid);
00608                         if (cur2 && (aim_ssi_itemlist_cmp(cur1, cur2))) {
00609                                 n++;
00610                                 new = g_new(struct aim_ssi_tmp, 1);
00611                                 new->action = SNAC_SUBTYPE_FEEDBAG_MOD;
00612                                 new->ack = 0xffff;
00613                                 new->name = NULL;
00614                                 new->item = cur1;
00615                                 new->next = NULL;
00616                                 if (od->ssi.pending) {
00617                                         for (cur=od->ssi.pending; cur->next; cur=cur->next);
00618                                         cur->next = new;
00619                                 } else
00620                                         od->ssi.pending = new;
00621                                 aim_ssi_item_debug_append(debugstr, "Modifying item ", cur1);
00622                         }
00623                 }
00624         }
00625         if (debugstr->len > 0) {
00626                 purple_debug_info("oscar", "%s", debugstr->str);
00627                 if (purple_debug_is_verbose()) {
00628                         g_string_truncate(debugstr, 0);
00629                         for (cur1 = od->ssi.local; cur1; cur1 = cur1->next) 
00630                                 aim_ssi_item_debug_append(debugstr, "\t", cur1);
00631                         purple_debug_misc("oscar", "Dumping item list of account %s:\n%s",
00632                                 purple_connection_get_account(od->gc)->username, debugstr->str);
00633                 }
00634         }
00635         g_string_free(debugstr, TRUE);
00636 
00637         /* We're out of stuff to do, so tell the AIM servers we're done and exit */
00638         if (!od->ssi.pending) {
00639                 if (od->ssi.in_transaction) {
00640                         aim_ssi_modend(od);
00641                         od->ssi.in_transaction = FALSE;
00642                 }
00643                 return 0;
00644         }
00645 
00646         /* If this is the first in a series of add/mod/del
00647          * requests then send the "begin transaction" message. */
00648         if (!od->ssi.in_transaction)
00649         {
00650                 aim_ssi_modbegin(od);
00651                 od->ssi.in_transaction = TRUE;
00652         }
00653 
00654         /* Make sure we don't send anything else between now
00655          * and when we receive the ack for the following operation */
00656         od->ssi.waiting_for_ack = TRUE;
00657 
00658         /* Now go mail off our data and wait 4 to 6 weeks */
00659         return aim_ssi_addmoddel(od);;
00660 }

00726         /* DESTROY any buddies that are directly in the master group. */
00727         /* Do the same for buddies that are in a non-existant group. */

01477  * Subtype 0x000e - SSI Add/Mod/Del Ack.
01478  *
01479  * Response to add, modify, or delete SNAC (sent with aim_ssi_addmoddel).
01480  *
01481  */
01482 static int parseack(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
01483 {
01484         int ret = 0;
01485         aim_rxcallback_t userfunc;
01486         struct aim_ssi_tmp *cur, *del;
01487 
01488         /* Read in the success/failure flags from the ack SNAC */
01489         cur = od->ssi.pending;
01490         while (cur && (byte_stream_bytes_left(bs)>0)) {
01491                 cur->ack = byte_stream_get16(bs);
01492                 cur = cur->next;
01493         }
01494 
01495         /*
01496          * If outcome is 0, then add the item to the item list, or replace the other item,
01497          * or remove the old item.  If outcome is non-zero, then remove the item from the
01498          * local list, or unmodify it, or add it.
01499          */
01500         for (cur=od->ssi.pending; (cur && (cur->ack != 0xffff)); cur=cur->next) {
01501         if (cur->item) {
01502                 if (cur->ack) {
01503                         /* Our action was unsuccessful, so change the local list back to how it was */
01504                         if (cur->action == SNAC_SUBTYPE_FEEDBAG_ADD) {
01505                                 /* Remove the item from the local list */
01506                                 /* Make sure cur->item is still valid memory */
01507                                 if (aim_ssi_itemlist_valid(od->ssi.local, cur->item)) {
01508                                         cur->name = g_strdup(cur->item->name);
01509                                         aim_ssi_itemlist_del(&od->ssi.local, cur->item);
01510                                 }
01511                                 cur->item = NULL;
01512 
01513                         } else if (cur->action == SNAC_SUBTYPE_FEEDBAG_MOD) {
01514                                 /* Replace the local item with the item from the official list */
01515                                 if (aim_ssi_itemlist_valid(od->ssi.local, cur->item)) {
01516                                         struct aim_ssi_item *cur1;
01517                                         if ((cur1 = aim_ssi_itemlist_find(od->ssi.official, cur->item->gid, cur->item->bid))) {
01518                                                 g_free(cur->item->name);
01519                                                 cur->item->name = g_strdup(cur1->name);
01520                                                 aim_tlvlist_free(cur->item->data);
01521                                                 cur->item->data = aim_tlvlist_copy(cur1->data);
01522                                         }
01523                                 } else
01524                                         cur->item = NULL;
01525 
01526                         } else if (cur->action == SNAC_SUBTYPE_FEEDBAG_DEL) {
01527                                 /* Add the item back into the local list */
01528                                 if (aim_ssi_itemlist_valid(od->ssi.official, cur->item)) {
01529                                         aim_ssi_itemlist_add(&od->ssi.local, cur->item->name, cur->item->gid, cur->item->bid, cur->item->type, cur->item->data);
01530                                 } else
01531                                         cur->item = NULL;
01532                         }
01533 
01534                 } else {
01535                         /* Do the exact opposite */
01536                         if (cur->action == SNAC_SUBTYPE_FEEDBAG_ADD) {
01537                         /* Add the local item to the official list */
01538                                 if (aim_ssi_itemlist_valid(od->ssi.local, cur->item)) {
01539                                         aim_ssi_itemlist_add(&od->ssi.official, cur->item->name, cur->item->gid, cur->item->bid, cur->item->type, cur->item->data);
01540                                 } else
01541                                         cur->item = NULL;
01542 
01543                         } else if (cur->action == SNAC_SUBTYPE_FEEDBAG_MOD) {
01544                                 /* Replace the official item with the item from the local list */
01545                                 if (aim_ssi_itemlist_valid(od->ssi.local, cur->item)) {
01546                                         struct aim_ssi_item *cur1;
01547                                         if ((cur1 = aim_ssi_itemlist_find(od->ssi.official, cur->item->gid, cur->item->bid))) {
01548                                                 g_free(cur1->name);
01549                                                 cur1->name = g_strdup(cur->item->name);
01550                                                 aim_tlvlist_free(cur1->data);
01551                                                 cur1->data = aim_tlvlist_copy(cur->item->data);
01552                                         }
01553                                 } else
01554                                         cur->item = NULL;
01555 
01556                         } else if (cur->action == SNAC_SUBTYPE_FEEDBAG_DEL) {
01557                                 /* Remove the item from the official list */
01558                                 if (aim_ssi_itemlist_valid(od->ssi.official, cur->item))
01559                                         aim_ssi_itemlist_del(&od->ssi.official, cur->item);
01560                                 cur->item = NULL;
01561                         }
01562 
01563                 }
01564         } /* End if (cur->item) */
01565         } /* End for loop */
01566 
01567         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
01568                 ret = userfunc(od, conn, frame, od->ssi.pending);
01569 
01570         /* Free all aim_ssi_tmp's with an outcome */
01571         cur = od->ssi.pending;
01572         while (cur && (cur->ack != 0xffff)) {
01573                 del = cur;
01574                 cur = cur->next;
01575                 g_free(del->name);
01576                 g_free(del);
01577         }
01578         od->ssi.pending = cur;
01579 
01580         /* If we're not waiting for any more acks, then send more SNACs */
01581         if (!od->ssi.pending) {
01582                 od->ssi.waiting_for_ack = FALSE;
01583                 aim_ssi_sync(od);
01584         }
01585 
01586         return ret;
01587 }
