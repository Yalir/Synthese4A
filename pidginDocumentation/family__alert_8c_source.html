00096 static int
00097 parseinfo(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00098 {
00099         int ret = 0;
00100         aim_rxcallback_t userfunc;
00101         struct aim_emailinfo *new;
00102         GSList *tlvlist;
00103         guint8 *cookie8, *cookie16;
00104         int tmp, havenewmail = 0; /* Used to tell the client we have _new_ mail */
00105 
00106         char *alertitle = NULL, *alerturl = NULL;
00107 
00108         cookie8 = byte_stream_getraw(bs, 8); /* Possibly the code used to log you in to mail? */
00109         cookie16 = byte_stream_getraw(bs, 16); /* Mail cookie sent above */
00110 
00111         /* See if we already have some info associated with this cookie */
00112         for (new = od->emailinfo; (new && memcmp(cookie16, new->cookie16, 16)); new = new->next);
00113         if (new) {
00114                 /* Free some of the old info, if it exists */
00115                 g_free(new->cookie8);
00116                 g_free(new->cookie16);
00117                 g_free(new->url);
00118                 g_free(new->domain);
00119         } else {
00120                 /* We don't already have info, so create a new struct for it */
00121                 new = g_new0(struct aim_emailinfo, 1);
00122                 new->next = od->emailinfo;
00123                 od->emailinfo = new;
00124         }
00125 
00126         new->cookie8 = cookie8;
00127         new->cookie16 = cookie16;
00128 
00129         tlvlist = aim_tlvlist_readnum(bs, byte_stream_get16(bs));
00130 
00131         tmp = aim_tlv_get16(tlvlist, 0x0080, 1);
00132         if (tmp) {
00133                 if (new->nummsgs < tmp)
00134                         havenewmail = 1;
00135                 new->nummsgs = tmp;
00136         } else {
00137                 /* If they don't send a 0x0080 TLV, it means we definitely have new mail */
00138                 /* (ie. this is not just another status update) */
00139                 havenewmail = 1;
00140                 new->nummsgs++; /* We know we have at least 1 new email */
00141         }
00142         new->url = aim_tlv_getstr(tlvlist, 0x0007, 1);
00143         if (!(new->unread = aim_tlv_get8(tlvlist, 0x0081, 1))) {
00144                 havenewmail = 0;
00145                 new->nummsgs = 0;
00146         }
00147         new->domain = aim_tlv_getstr(tlvlist, 0x0082, 1);
00148         new->flag = aim_tlv_get16(tlvlist, 0x0084, 1);
00149 
00150         alertitle = aim_tlv_getstr(tlvlist, 0x0005, 1);
00151         alerturl  = aim_tlv_getstr(tlvlist, 0x000d, 1);
00152 
00153         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00154                 ret = userfunc(od, conn, frame, new, havenewmail, alertitle, (alerturl ? alerturl + 2 : NULL));
00155 
00156         aim_tlvlist_free(tlvlist);
00157 
00158         g_free(alertitle);
00159         g_free(alerturl);
00160 
00161         return ret;
00162 }
