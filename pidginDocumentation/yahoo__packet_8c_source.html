00113 /*
00114  * 'len' is the value given to us by the server that is supposed to
00115  * be the length of 'data'.  But apparently there's a time when this
00116  * length is incorrect.  Christopher Layne thinks it might be a bug
00117  * in their server code.
00118  *
00119  * The following information is from Christopher:
00120  *
00121  * It sometimes happens when Yahoo! sends a packet continuation within
00122  * chat.  Sometimes when joining a large chatroom the initial
00123  * SERVICE_CHATJOIN packet will be so large that it will need to be
00124  * split into multiple packets.  That's fine, except that the length
00125  * of the second packet is wrong.  The packet has the same length as
00126  * the first packet, and the length given in the header is the same,
00127  * however the actual data in the packet is shorter than this length.
00128  * So half of the packet contains good, valid data, and then the rest
00129  * of the packet is junk.  Luckily there is a null terminator after
00130  * the valid data and before the invalid data.
00131  *
00132  * What does all this mean?  It means that we parse through the data
00133  * pulling out key/value pairs until we've parsed 'len' bytes, or until
00134  * we run into a null terminator, whichever comes first.
00135  */
00136 void yahoo_packet_read(struct yahoo_packet *pkt, const guchar *data, int len)
00137 {
00138         int pos = 0;
00139         char key[64];
00140         const guchar *delimiter;
00141         gboolean accept;
00142         int x;
00143         struct yahoo_pair *pair;
00144 
00145         while (pos + 1 < len)
00146         {
00147                 if (data[pos] == '\0')
00148                         break;
00149 
00150                 pair = g_new0(struct yahoo_pair, 1);
00151 
00152                 x = 0;
00153                 while (pos + 1 < len) {
00154                         if (data[pos] == 0xc0 && data[pos + 1] == 0x80)
00155                                 break;
00156                         if (x >= sizeof(key)-1) {
00157                                 x++;
00158                                 pos++;
00159                                 continue;
00160                         }
00161                         key[x++] = data[pos++];
00162                 }
00163                 if (x >= sizeof(key)-1) {
00164                         x = 0;
00165                 }
00166                 key[x] = 0;
00167                 pos += 2;
00168                 pair->key = strtol(key, NULL, 10);
00169                 accept = x; /* if x is 0 there was no key, so don't accept it */
00170 
00171                 if (pos + 1 > len) {
00172                         /* Malformed packet! (Truncated--garbage or something) */
00173                         accept = FALSE;
00174                 }
00175 
00176                 if (accept) {
00177                         delimiter = (const guchar *)g_strstr_len((const char *)&data[pos], len - pos, "\xc0\x80");
00178                         if (delimiter == NULL)
00179                         {
00180                                 /* Malformed packet! (It doesn't end in 0xc0 0x80) */
00181                                 g_free(pair);
00182                                 pos = len;
00183                                 continue;
00184                         }
00185                         x = delimiter - data;
00186                         pair->value = g_strndup((const gchar *)&data[pos], x - pos);
00187                         pos = x;
00188                         pkt->hash = g_slist_prepend(pkt->hash, pair);
00189 
00190                         if (purple_debug_is_verbose() || g_getenv("PURPLE_YAHOO_DEBUG")) {
00191                                 char *esc;
00192                                 esc = g_strescape(pair->value, NULL);
00193                                 purple_debug_misc("yahoo", "Key: %d  \tValue: %s\n", pair->key, esc);
00194                                 g_free(esc);
00195                         }
00196                 } else {
00197                         g_free(pair);
00198                 }
00199                 pos += 2;
00200 
00201                 if (pos + 1 > len) break;
00202 
00203                 /* Skip over garbage we've noticed in the mail notifications */
00204                 if (data[0] == '9' && data[pos] == 0x01)
00205                         pos++;
00206         }
00207 
00208         /*
00209          * Originally this function used g_slist_append().  I changed
00210          * it to use g_slist_prepend() for improved performance.
00211          * Ideally the Yahoo! PRPL code would be indifferent to the
00212          * order of the key/value pairs, but I don't know if this is
00213          * the case for all incoming messages.  To be on the safe side
00214          * we reverse the list.
00215          */
00216         pkt->hash = g_slist_reverse(pkt->hash);
00217 }
