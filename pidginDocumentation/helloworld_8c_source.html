00055 /* This function is the callback for the plugin action we added. All we're
00056  * doing here is displaying a message. When the user selects the plugin

00061         purple_notify_message (helloworld_plugin, PURPLE_NOTIFY_MSG_INFO,

00066 /* we tell libpurple in the PurplePluginInfo struct to call this function to
00067  * get a list of plugin actions to use for the plugin.  This function gives
00068  * libpurple that list of actions. */
00069 static GList *
00070 plugin_actions (PurplePlugin * plugin, gpointer context)
00071 {
00072         /* some C89 (a.k.a. ANSI C) compilers will warn if any variable declaration
00073          * includes an initilization that calls a function.  To avoid that, we
00074          * generally initialize our variables first with constant values like NULL
00075          * or 0 and assign to them with function calls later */
00076         GList *list = NULL;
00077         PurplePluginAction *action = NULL;
00078 
00079         /* The action gets created by specifying a name to show in the UI and a
00080          * callback function to call. */
00081         action = purple_plugin_action_new ("Plugin Action Test", plugin_action_test_cb);
00082 
00083         /* libpurple requires a GList of plugin actions, even if there is only one
00084          * action in the list.  We append the action to a GList here. */
00085         list = g_list_append (list, action);
00086 
00087         /* Once the list is complete, we send it to libpurple. */
00088         return list;
00089 }
