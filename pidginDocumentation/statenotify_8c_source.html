00029                                                                                                  buddy_name, account);
00030 
00031         if (conv == NULL)
00032                 return;
00033         g_return_if_fail(conv->type == PURPLE_CONV_TYPE_IM);
00034

00036         if (buddy != purple_find_buddy(account, buddy_name))
00037                 return;

00042         g_snprintf(buf, sizeof(buf), message, escaped);
00043         g_free(escaped);
00044 
00045         purple_conv_im_write(conv->u.im, NULL, buf, PURPLE_MESSAGE_SYSTEM | PURPLE_MESSAGE_ACTIVE_ONLY | PURPLE_MESSAGE_NO_LINKIFY, time(NULL));

00049 buddy_status_changed_cb(PurpleBuddy *buddy, PurpleStatus *old_status,
00050                         PurpleStatus *status, void *data)
00051 {
00052         gboolean available, old_available;
00053 
00054         if (!purple_status_is_exclusive(status) ||
00055                         !purple_status_is_exclusive(old_status))
00056                 return;
00057 
00058         available = purple_status_is_available(status);
00059         old_available = purple_status_is_available(old_status);
00060 
00061         if (purple_prefs_get_bool("/plugins/core/statenotify/notify_away")) {
00062                 if (available && !old_available)
00063                         write_status(buddy, _("%s is no longer away."));
00064                 else if (!available && old_available)
00065                         write_status(buddy, _("%s has gone away."));
00066         }
