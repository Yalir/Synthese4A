00123 purple_cipher_digest_region(const gchar *name, const guchar *data,
00124                                                   size_t data_len, size_t in_len,
00125                                                   guchar digest[], size_t *out_len)
00126 {
00127         PurpleCipher *cipher;
00128         PurpleCipherContext *context;
00129         gboolean ret = FALSE;
00130 
00131         g_return_val_if_fail(name, FALSE);
00132         g_return_val_if_fail(data, FALSE);
00133 
00134         cipher = purple_ciphers_find_cipher(name);
00135 
00136         g_return_val_if_fail(cipher, FALSE);
00137 
00138         if(!cipher->ops->append || !cipher->ops->digest) {
00139                 purple_debug_warning("cipher", "purple_cipher_region failed: "
00140                                                 "the %s cipher does not support appending and or "
00141                                                 "digesting.", cipher->name);
00142                 return FALSE;
00143         }
00144 
00145         context = purple_cipher_context_new(cipher, NULL);
00146         purple_cipher_context_append(context, data, data_len);
00147         ret = purple_cipher_context_digest(context, in_len, digest, out_len);
00148         purple_cipher_context_destroy(context);
00149 
00150         return ret;
00151 }

00443                                                                    gchar digest_s[], size_t *out_len)
00444 {
00445         /* 8k is a bit excessive, will tweak later. */
00446         guchar digest[BUF_LEN * 4];
00447         gint n = 0;
00448         size_t dlen = 0;
00449 
00450         g_return_val_if_fail(context, FALSE);
00451         g_return_val_if_fail(digest_s, FALSE);
00452 
00453         if(!purple_cipher_context_digest(context, sizeof(digest), digest, &dlen))
00454                 return FALSE;
00455 
00456         /* in_len must be greater than dlen * 2 so we have room for the NUL. */
00457         if(in_len <= dlen * 2)
00458                 return FALSE;
00459 
00460         for(n = 0; n < dlen; n++)
00461                 sprintf(digest_s + (n * 2), "%02x", digest[n]);
00462 
00463         digest_s[n * 2] = '\0';
00464 
00465         if(out_len)
00466                 *out_len = dlen * 2;
00467 
00468         return TRUE;
00469 }
