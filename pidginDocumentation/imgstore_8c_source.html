00088 int
00089 purple_imgstore_add_with_id(gpointer data, size_t size, const char *filename)
00090 {
00091         PurpleStoredImage *img = purple_imgstore_add(data, size, filename);
00092         if (!img) {
00093                 return 0;
00094         }
00095 
00096         /*
00097          * Use the next unused id number.  We do it in a loop on the
00098          * off chance that nextid wraps back around to 0 and the hash
00099          * table still contains entries from the first time around.
00100          */
00101         do {
00102                 img->id = ++nextid;
00103         } while (img->id == 0 || g_hash_table_lookup(imgstore, &(img->id)) != NULL);
00104 
00105         g_hash_table_insert(imgstore, &(img->id), img);
00106 
00107         return img->id;
00108 }

00176 PurpleStoredImage *
00177 purple_imgstore_unref(PurpleStoredImage *img)
00178 {
00179         if (img == NULL)
00180                 return NULL;
00181 
00182         g_return_val_if_fail(img->refcount > 0, NULL);
00183 
00184         img->refcount--;
00185 
00186         if (img->refcount == 0)
00187         {
00188                 purple_signal_emit(purple_imgstore_get_handle(),
00189                                    "image-deleting", img);
00190                 if (img->id)
00191                         g_hash_table_remove(imgstore, &img->id);

00193                 g_free(img->data);
00194                 g_free(img->filename);
00195                 PURPLE_DBUS_UNREGISTER_POINTER(img);
