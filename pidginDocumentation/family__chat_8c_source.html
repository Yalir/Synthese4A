00064 /*
00065  * Subtype 0x0002 - General room information.  Lots of stuff.
00066  *
00067  * Values I know are in here but I haven't attached
00068  * them to any of the 'Unknown's:
00069  *      - Language (English)
00070  *
00071  */
00072 static int
00073 infoupdate(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00074 {
00075         aim_rxcallback_t userfunc;
00076         int ret = 0;
00077         guint8 detaillevel = 0;
00078         struct aim_chat_roominfo roominfo;
00079         GSList *tlvlist;
00080         guint16 maxmsglen, maxvisiblemsglen;
00081 
00082         aim_chat_readroominfo(bs, &roominfo);
00083 
00084         detaillevel = byte_stream_get8(bs);
00085 
00086         if (detaillevel != 0x02) {
00087                 purple_debug_misc("oscar", "faim: chat_roomupdateinfo: detail level %d not supported\n", detaillevel);
00088                 return 1;
00089         }
00090 
00091         byte_stream_get16(bs); /* skip the TLV count */
00092 
00093         /*
00094          * Everything else are TLVs.
00095          */
00096         tlvlist = aim_tlvlist_read(bs);
00097 
00098         /*
00099          * Type 0x00d1: Maximum Message Length
00100          */
00101         maxmsglen = aim_tlv_get16(tlvlist, 0x00d1, 1);
00102 
00103         /*
00104          * Type 0x00da: Maximum visible message length
00105          */
00106         maxvisiblemsglen = aim_tlv_get16(tlvlist, 0x00da, 1);
00107 
00108         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype))) {
00109                 ret = userfunc(od, conn, frame, maxmsglen, maxvisiblemsglen);
00110         }
00111 
00112         g_free(roominfo.name);
00113 
00114         aim_tlvlist_free(tlvlist);
00115 
00116         return ret;

00142 /*
00143  * Subtype 0x0005 - Send a Chat Message.
00144  *
00145  * Possible flags:
00146  *   AIM_CHATFLAGS_NOREFLECT   --  Unset the flag that requests messages
00147  *                                 should be sent to their sender.
00148  *   AIM_CHATFLAGS_AWAY        --  Mark the message as an autoresponse
00149  *                                 (Note that WinAIM does not honor this,
00150  *                                 and displays the message as normal.)
00151  *
00152  * XXX convert this to use tlvchains
00153  */
00154 int
00155 aim_chat_send_im(OscarData *od, FlapConnection *conn, guint16 flags, const gchar *msg, int msglen, const char *encoding, const char *language)
00156 {
00157         int i;
00158         ByteStream bs;
00159         IcbmCookie *cookie;
00160         aim_snacid_t snacid;
00161         guint8 ckstr[8];
00162         GSList *tlvlist = NULL, *inner_tlvlist = NULL;
00163 
00164         if (!od || !conn || !msg || (msglen <= 0))
00165                 return 0;
00166 
00167         byte_stream_new(&bs, 1142);
00168 
00169         snacid = aim_cachesnac(od, SNAC_FAMILY_CHAT, 0x0005, 0x0000, NULL, 0);
00170 
00171         /*
00172          * Cookie
00173          *
00174          * XXX mkcookie should generate the cookie and cache it in one
00175          * operation to preserve uniqueness.
00176          */
00177         for (i = 0; i < 8; i++)
00178                 ckstr[i] = (guint8)rand();
00179 
00180         cookie = aim_mkcookie(ckstr, AIM_COOKIETYPE_CHAT, NULL);
00181         cookie->data = NULL; /* XXX store something useful here */
00182 
00183         aim_cachecookie(od, cookie);
00184 
00185         /* ICBM Header */
00186         byte_stream_putraw(&bs, ckstr, 8); /* Cookie */
00187         byte_stream_put16(&bs, 0x0003); /* Channel */
00188 
00189         /*
00190          * Type 1: Flag meaning this message is destined to the room.
00191          */
00192         aim_tlvlist_add_noval(&tlvlist, 0x0001);
00193 
00194         /*
00195          * Type 6: Reflect
00196          */
00197         if (!(flags & AIM_CHATFLAGS_NOREFLECT))
00198                 aim_tlvlist_add_noval(&tlvlist, 0x0006);
00199 
00200         /*
00201          * Type 7: Autoresponse
00202          */
00203         if (flags & AIM_CHATFLAGS_AWAY)
00204                 aim_tlvlist_add_noval(&tlvlist, 0x0007);
00205 
00206         /*
00207          * SubTLV: Type 1: Message
00208          */
00209         aim_tlvlist_add_raw(&inner_tlvlist, 0x0001, msglen, (guchar *)msg);
00210 
00211         /*
00212          * SubTLV: Type 2: Encoding
00213          */
00214         if (encoding != NULL)
00215                 aim_tlvlist_add_str(&inner_tlvlist, 0x0002, encoding);
00216 
00217         /*
00218          * SubTLV: Type 3: Language
00219          */
00220         if (language != NULL)
00221                 aim_tlvlist_add_str(&inner_tlvlist, 0x0003, language);
00222 
00223         /*
00224          * Type 5: Message block.  Contains more TLVs.
00225          *
00226          * This could include other information... We just
00227          * put in a message TLV however.
00228          *
00229          */
00230         aim_tlvlist_add_frozentlvlist(&tlvlist, 0x0005, &inner_tlvlist);
00231 
00232         aim_tlvlist_write(&bs, &tlvlist);
00233 
00234         aim_tlvlist_free(inner_tlvlist);
00235         aim_tlvlist_free(tlvlist);
00236 
00237         flap_connection_send_snac(od, conn, SNAC_FAMILY_CHAT, 0x0005, snacid, &bs);
00238 
00239         byte_stream_destroy(&bs);
00240 
00241         return 0;
00242 }

00244 /*
00245  * Subtype 0x0006
00246  *
00247  * We could probably include this in the normal ICBM parsing
00248  * code as channel 0x0003, however, since only the start
00249  * would be the same, we might as well do it here.
00250  *
00251  * General outline of this SNAC:
00252  *   snac
00253  *   cookie
00254  *   channel id
00255  *   tlvlist
00256  *     unknown
00257  *     source user info
00258  *       name
00259  *       evility
00260  *       userinfo tlvs
00261  *         online time
00262  *         etc
00263  *     message metatlv
00264  *       message tlv
00265  *         message string
00266  *       possibly others
00267  *
00268  */
00269 static int
00270 incomingim_ch3(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00271 {
00272         int ret = 0, i;
00273         aim_rxcallback_t userfunc;
00274         aim_userinfo_t userinfo;
00275         guint8 cookie[8];
00276         guint16 channel;
00277         GSList *tlvlist;
00278         char *msg = NULL;
00279         int len = 0;
00280         char *encoding = NULL, *language = NULL;
00281         IcbmCookie *ck;
00282         aim_tlv_t *tlv;
00283         ByteStream tbs;
00284 
00285         memset(&userinfo, 0, sizeof(aim_userinfo_t));
00286 
00287         /*
00288          * Read ICBM Cookie.
00289          */
00290         for (i = 0; i < 8; i++)
00291                 cookie[i] = byte_stream_get8(bs);
00292 
00293         if ((ck = aim_uncachecookie(od, cookie, AIM_COOKIETYPE_CHAT))) {
00294                 g_free(ck->data);
00295                 g_free(ck);
00296         }
00297 
00298         /*
00299          * Channel ID
00300          *
00301          * Channel 0x0003 is used for chat messages.
00302          *
00303          */
00304         channel = byte_stream_get16(bs);
00305 
00306         if (channel != 0x0003) {
00307                 purple_debug_misc("oscar", "faim: chat_incoming: unknown channel! (0x%04x)\n", channel);
00308                 return 0;
00309         }
00310 
00311         /*
00312          * Start parsing TLVs right away.
00313          */
00314         tlvlist = aim_tlvlist_read(bs);
00315 
00316         /*
00317          * Type 0x0003: Source User Information
00318          */
00319         tlv = aim_tlv_gettlv(tlvlist, 0x0003, 1);
00320         if (tlv != NULL)
00321         {
00322                 byte_stream_init(&tbs, tlv->value, tlv->length);
00323                 aim_info_extract(od, &tbs, &userinfo);
00324         }
00325 
00326         /*
00327          * Type 0x0005: Message Block.  Conains more TLVs.
00328          */
00329         tlv = aim_tlv_gettlv(tlvlist, 0x0005, 1);
00330         if (tlv != NULL)
00331         {
00332                 GSList *inner_tlvlist;
00333                 aim_tlv_t *inner_tlv;
00334 
00335                 byte_stream_init(&tbs, tlv->value, tlv->length);
00336                 inner_tlvlist = aim_tlvlist_read(&tbs);
00337 
00338                 /*
00339                  * Type 0x0001: Message.
00340                  */
00341                 inner_tlv = aim_tlv_gettlv(inner_tlvlist, 0x0001, 1);
00342                 if (inner_tlv != NULL)
00343                 {
00344                         len = inner_tlv->length;
00345                         msg = aim_tlv_getvalue_as_string(inner_tlv);
00346                 }
00347 
00348                 /*
00349                  * Type 0x0002: Encoding.
00350                  */
00351                 encoding = aim_tlv_getstr(inner_tlvlist, 0x0002, 1);
00352 
00353                 /*
00354                  * Type 0x0003: Language.
00355                  */
00356                 language = aim_tlv_getstr(inner_tlvlist, 0x0003, 1);
00357 
00358                 aim_tlvlist_free(inner_tlvlist);
00359         }
00360 
00361         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00362                 ret = userfunc(od, conn, frame, &userinfo, len, msg, encoding, language);
00363 
00364         aim_info_free(&userinfo);
00365         g_free(msg);
00366         g_free(encoding);
00367         g_free(language);
00368         aim_tlvlist_free(tlvlist);
00369 
00370         return ret;
00371 }

00373 static int
00374 snachandler(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00375 {
00376         if (snac->subtype == 0x0002)
00377                 return infoupdate(od, conn, mod, frame, snac, bs);
00378         else if ((snac->subtype == 0x0003) || (snac->subtype == 0x0004))
00379                 return userlistchange(od, conn, mod, frame, snac, bs);
00380         else if (snac->subtype == 0x0006)
00381                 return incomingim_ch3(od, conn, mod, frame, snac, bs);
00382 
00383         return 0;
00384 }
