00173  *
00174  * This is the initial login request packet.
00175  *
00176  * NOTE!! If you want/need to make use of the aim_sendmemblock() function,
00177  * then the client information you send here must exactly match the
00178  * executable that you're pulling the data from.

00181  *   clientstring = "AOL Instant Messenger (TM) version 1.1.19 for Java built 03/24/98, freeMem 215871 totalMem 1048567, i686, Linus, #2 SMP Sun Feb 11 03:41:17 UTC 2001 2.4.1-ac9, IBM Corporation, 1.1.8, 45.3, Tue Mar 27 12:09:17 PST 2001"

00193  *   minor  = 0x0001

00285  * This is sent back as a general response to the login command.
00286  * It can be either an error or a success, depending on the
00287  * presence of certain TLVs.
00288  *
00289  * The client should check the value passed as errorcode. If
00290  * its nonzero, there was an error.

00292 static int
00293 parse(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00294 {
00295         GSList *tlvlist;
00296         aim_rxcallback_t userfunc;
00297         struct aim_authresp_info *info;
00298         int ret = 0;
00299 
00300         info = g_new0(struct aim_authresp_info, 1);
00301 
00302         /*
00303          * Read block of TLVs.  All further data is derived
00304          * from what is parsed here.
00305          */
00306         tlvlist = aim_tlvlist_read(bs);
00307 
00308         /*
00309          * No matter what, we should have a username.
00310          */
00311         if (aim_tlv_gettlv(tlvlist, 0x0001, 1)) {
00312                 info->bn = aim_tlv_getstr(tlvlist, 0x0001, 1);
00313                 purple_connection_set_display_name(od->gc, info->bn);
00314         }
00315 
00316         /*
00317          * Check for an error code.  If so, we should also
00318          * have an error url.
00319          */
00320         if (aim_tlv_gettlv(tlvlist, 0x0008, 1))
00321                 info->errorcode = aim_tlv_get16(tlvlist, 0x0008, 1);
00322         if (aim_tlv_gettlv(tlvlist, 0x0004, 1))
00323                 info->errorurl = aim_tlv_getstr(tlvlist, 0x0004, 1);
00324 
00325         /*
00326          * BOS server address.
00327          */
00328         if (aim_tlv_gettlv(tlvlist, 0x0005, 1))
00329                 info->bosip = aim_tlv_getstr(tlvlist, 0x0005, 1);
00330 
00331         /*
00332          * Authorization cookie.
00333          */
00334         if (aim_tlv_gettlv(tlvlist, 0x0006, 1)) {
00335                 aim_tlv_t *tmptlv;
00336 
00337                 tmptlv = aim_tlv_gettlv(tlvlist, 0x0006, 1);
00338                 if (tmptlv != NULL)
00339                 {
00340                         info->cookielen = tmptlv->length;
00341                         info->cookie = tmptlv->value;
00342                 }
00343         }
00344 
00345         /*
00346          * The email address attached to this account
00347          *   Not available for ICQ or @mac.com logins.
00348          *   If you receive this TLV, then you are allowed to use
00349          *   family 0x0018 to check the status of your email.
00350          * XXX - Not really true!
00351          */
00352         if (aim_tlv_gettlv(tlvlist, 0x0011, 1))
00353                 info->email = aim_tlv_getstr(tlvlist, 0x0011, 1);
00354 
00355         /*
00356          * The registration status.  (Not real sure what it means.)
00357          *   Not available for ICQ or @mac.com logins.
00358          *
00359          *   1 = No disclosure
00360          *   2 = Limited disclosure
00361          *   3 = Full disclosure
00362          *
00363          * This has to do with whether your email address is available
00364          * to other users or not.  AFAIK, this feature is no longer used.
00365          *
00366          * Means you can use the admin family? (0x0007)
00367          *
00368          */
00369         if (aim_tlv_gettlv(tlvlist, 0x0013, 1))
00370                 info->regstatus = aim_tlv_get16(tlvlist, 0x0013, 1);
00371 
00372         if (aim_tlv_gettlv(tlvlist, 0x0040, 1))
00373                 info->latestbeta.build = aim_tlv_get32(tlvlist, 0x0040, 1);
00374         if (aim_tlv_gettlv(tlvlist, 0x0041, 1))
00375                 info->latestbeta.url = aim_tlv_getstr(tlvlist, 0x0041, 1);
00376         if (aim_tlv_gettlv(tlvlist, 0x0042, 1))
00377                 info->latestbeta.info = aim_tlv_getstr(tlvlist, 0x0042, 1);
00378         if (aim_tlv_gettlv(tlvlist, 0x0043, 1))
00379                 info->latestbeta.name = aim_tlv_getstr(tlvlist, 0x0043, 1);
00380 
00381         if (aim_tlv_gettlv(tlvlist, 0x0044, 1))
00382                 info->latestrelease.build = aim_tlv_get32(tlvlist, 0x0044, 1);
00383         if (aim_tlv_gettlv(tlvlist, 0x0045, 1))
00384                 info->latestrelease.url = aim_tlv_getstr(tlvlist, 0x0045, 1);
00385         if (aim_tlv_gettlv(tlvlist, 0x0046, 1))
00386                 info->latestrelease.info = aim_tlv_getstr(tlvlist, 0x0046, 1);
00387         if (aim_tlv_gettlv(tlvlist, 0x0047, 1))
00388                 info->latestrelease.name = aim_tlv_getstr(tlvlist, 0x0047, 1);
00389 
00390         /*
00391          * URL to change password.
00392          */
00393         if (aim_tlv_gettlv(tlvlist, 0x0054, 1))
00394                 info->chpassurl = aim_tlv_getstr(tlvlist, 0x0054, 1);
00395 
00396         od->authinfo = info;
00397 
00398         if ((userfunc = aim_callhandler(od, snac ? snac->family : SNAC_FAMILY_AUTH, snac ? snac->subtype : 0x0003)))
00399                 ret = userfunc(od, conn, frame, info);
00400

00402

00407 /*
00408  * Subtype 0x0007 (kind of) - Send a fake type 0x0007 SNAC to the client
00409  *
00410  * This is a bit confusing.
00411  *
00412  * Normal SNAC login goes like this:
00413  *   - connect
00414  *   - server sends flap version
00415  *   - client sends flap version
00416  *   - client sends username (17/6)
00417  *   - server sends hash key (17/7)
00418  *   - client sends auth request (17/2 -- aim_send_login)
00419  *   - server yells
00420  *
00421  * XOR login (for ICQ) goes like this:
00422  *   - connect
00423  *   - server sends flap version
00424  *   - client sends auth request which contains flap version (aim_send_login)
00425  *   - server yells
00426  *
00427  * For the client API, we make them implement the most complicated version,
00428  * and for the simpler version, we fake it and make it look like the more
00429  * complicated process.
00430  *
00431  * This is done by giving the client a faked key, just so we can convince
00432  * them to call aim_send_login right away, which will detect the session
00433  * flag that says this is XOR login and ignore the key, sending an ICQ
00434  * login request instead of the normal SNAC one.
00435  *
00436  * As soon as AOL makes ICQ log in the same way as AIM, this is /gone/.
00437  */
00438 static int
00439 goddamnicq(OscarData *od, FlapConnection *conn, const char *sn)
00440 {
00441         FlapFrame frame;
00442         aim_rxcallback_t userfunc;
00443 
00444         if ((userfunc = aim_callhandler(od, SNAC_FAMILY_AUTH, 0x0007)))
00445                 userfunc(od, conn, &frame, "");
00446 
00447         return 0;
00448 }

00456  * invalid, a 0017/0003 is spit back, with the standard error contents.  If
00457  * valid, a 0017/0007 comes back, which is the signal to send it the main

00459  *
00460  */

00462 aim_request_login(OscarData *od, FlapConnection *conn, const char *sn)
00463 {
00464         FlapFrame *frame;
00465         aim_snacid_t snacid;

00467 
00468         if (!od || !conn || !sn)
00469                 return -EINVAL;
00470 
00471 #ifdef USE_XOR_FOR_ICQ
00472         if (aim_snvalid_icq(sn))
00473                 return goddamnicq(od, conn, sn);
00474 #endif
00475 
00476         frame = flap_frame_new(od, 0x02, 10+2+2+strlen(sn)+8);
00477 
00478         snacid = aim_cachesnac(od, SNAC_FAMILY_AUTH, 0x0006, 0x0000, NULL, 0);
00479         aim_putsnac(&frame->data, SNAC_FAMILY_AUTH, 0x0006, snacid);
00480 
00481         aim_tlvlist_add_str(&tlvlist, 0x0001, sn);
00482 
00483         /* Tell the server we support SecurID logins. */
00484         aim_tlvlist_add_noval(&tlvlist, 0x004b);

00497 /*
00498  * Subtype 0x0007
00499  *
00500  * Middle handler for 0017/0007 SNACs.  Contains the auth key prefixed
00501  * by only its length in a two byte word.
00502  *
00503  * Calls the client, which should then use the value to call aim_send_login.
00504  *
00505  */
00506 static int
00507 keyparse(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00508 {
00509         int keylen, ret = 1;
00510         aim_rxcallback_t userfunc;
00511         char *keystr;
00512         GSList *tlvlist;
00513         gboolean truncate_pass;
00514 
00515         keylen = byte_stream_get16(bs);
00516         keystr = byte_stream_getstr(bs, keylen);
00517         tlvlist = aim_tlvlist_read(bs);
00518 
00519         /*
00520          * If the truncate_pass TLV exists then we should truncate the
00521          * user's password to 8 characters.  This flag is sent to us
00522          * when logging in with an AOL user's username.
00523          */
00524         truncate_pass = aim_tlv_gettlv(tlvlist, 0x0026, 1) != NULL;
00525 
00526         /* XXX - When GiantGrayPanda signed on AIM I got a thing asking me to register
00527          * for the netscape network.  This SNAC had a type 0x0058 TLV with length 10.
00528          * Data is 0x0007 0004 3e19 ae1e 0006 0004 0000 0005 */
00529 
00530         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00531                 ret = userfunc(od, conn, frame, keystr, (int)truncate_pass);
00532 
00533         g_free(keystr);
00534         aim_tlvlist_free(tlvlist);
00535 
00536         return ret;
00537 }
