00038 /* Subtype 0x0003 - BOS Rights. */
00039 static int rights(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00040 {
00041         aim_rxcallback_t userfunc;
00042         GSList *tlvlist;
00043         guint16 maxpermits = 0, maxdenies = 0;
00044         int ret = 0;
00045 
00046         /*
00047          * TLVs follow
00048          */
00049         tlvlist = aim_tlvlist_read(bs);
00050 
00051         /*
00052          * TLV type 0x0001: Maximum number of buddies on permit list.
00053          */
00054         if (aim_tlv_gettlv(tlvlist, 0x0001, 1))
00055                 maxpermits = aim_tlv_get16(tlvlist, 0x0001, 1);
00056 
00057         /*
00058          * TLV type 0x0002: Maximum number of buddies on deny list.
00059          */
00060         if (aim_tlv_gettlv(tlvlist, 0x0002, 1))
00061                 maxdenies = aim_tlv_get16(tlvlist, 0x0002, 1);
00062 
00063         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00064                 ret = userfunc(od, conn, frame, maxpermits, maxdenies);
00065 
00066         aim_tlvlist_free(tlvlist);
00067 
00068         return ret;
00069 }
