01679 /* This is the y64 alphabet... it's like base64, but has a . and a _ */
01680 static const char base64digits[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789._";

01682 /* This is taken from Sylpheed by Hiroyuki Yamamoto. We have our own tobase64 function
01683  * in util.c, but it is different from the one yahoo uses */
01684 static void to_y64(char *out, const unsigned char *in, gsize inlen)
01685      /* raw bytes in quasi-big-endian order to base 64 string (NUL-terminated) */
01686 {
01687         for (; inlen >= 3; inlen -= 3)
01688                 {
01689                         *out++ = base64digits[in[0] >> 2];
01690                         *out++ = base64digits[((in[0] << 4) & 0x30) | (in[1] >> 4)];
01691                         *out++ = base64digits[((in[1] << 2) & 0x3c) | (in[2] >> 6)];
01692                         *out++ = base64digits[in[2] & 0x3f];
01693                         in += 3;
01694                 }
01695         if (inlen > 0)
01696                 {
01697                         unsigned char fragment;
01698 
01699                         *out++ = base64digits[in[0] >> 2];
01700                         fragment = (in[0] << 4) & 0x30;
01701                         if (inlen > 1)
01702                                 fragment |= in[1] >> 4;
01703                         *out++ = base64digits[fragment];
01704                         *out++ = (inlen < 2) ? '-' : base64digits[(in[1] << 2) & 0x3c];
01705                         *out++ = '-';
01706                 }
01707         *out = '\0';
01708 }

02534 /* callback function associated with receiving of data, not considering receipt of multiple YMSG packets in a single TCP packet */
02535 static void yahoo_p2p_read_pkt_cb(gpointer data, gint source, PurpleInputCondition cond)
02536 {
02537         guchar buf[1024];       /* is it safe to assume a fixed array length of 1024 ?? */
02538         int len;
02539         int pos = 0;
02540         int pktlen;
02541         struct yahoo_packet *pkt;
02542         guchar *start = NULL;
02543         struct yahoo_p2p_data *p2p_data;
02544         YahooData *yd;
02545 
02546         if(!(p2p_data = data))
02547                 return ;
02548         yd = p2p_data->gc->proto_data;
02549 
02550         len = read(source, buf, sizeof(buf));
02551         if ((len < 0) && ((errno == EAGAIN) || (errno == EWOULDBLOCK)))
02552                 return ; /* No Worries*/
02553         else if (len <= 0)
02554         {
02555                 purple_debug_warning("yahoo","p2p: Error in connection, or host disconnected\n");
02556                 /* remove from p2p connection lists, also calls yahoo_p2p_disconnect_destroy_data */
02557                 if( g_hash_table_lookup(yd->peers, p2p_data->host_username) )
02558                         g_hash_table_remove(yd->peers,p2p_data->host_username);
02559                 else
02560                         yahoo_p2p_disconnect_destroy_data(data);
02561                 return;
02562         }
02563 
02564         if(len < YAHOO_PACKET_HDRLEN)
02565                 return;
02566 
02567         if(strncmp((char *)buf, "YMSG", MIN(4, len)) != 0) {
02568                 /* Not a YMSG packet */
02569                 purple_debug_warning("yahoo","p2p: Got something other than YMSG packet\n");
02570 
02571                 start = memchr(buf + 1, 'Y', len - 1);
02572                 if (start == NULL)
02573                         return;
02574 
02575                 g_memmove(buf, start, len - (start - buf));
02576                 len -= start - buf;
02577         }
02578 
02579         pos += 4;       /* YMSG */
02580         pos += 2;
02581         pos += 2;
02582 
02583         pktlen = yahoo_get16(buf + pos); pos += 2;
02584         purple_debug_misc("yahoo", "p2p: %d bytes to read\n", len);
02585 
02586         pkt = yahoo_packet_new(0, 0, 0);
02587         pkt->service = yahoo_get16(buf + pos); pos += 2;
02588         pkt->status = yahoo_get32(buf + pos); pos += 4;
02589         pkt->id = yahoo_get32(buf + pos); pos += 4;
02590 
02591         purple_debug_misc("yahoo", "p2p: Yahoo Service: 0x%02x Status: %d\n",pkt->service, pkt->status);
02592         yahoo_packet_read(pkt, buf + pos, pktlen);
02593 
02594         /* packet processing */
02595         switch(pkt->service) {
02596                 case YAHOO_SERVICE_P2PFILEXFER:
02597                         yahoo_p2p_process_p2pfilexfer(data, source, pkt);
02598                         break;
02599                 case YAHOO_SERVICE_MESSAGE:
02600                         yahoo_process_message(p2p_data->gc, pkt, YAHOO_PKT_TYPE_P2P);
02601                         break;
02602                 case YAHOO_SERVICE_NOTIFY:
02603                         yahoo_process_notify(p2p_data->gc, pkt, YAHOO_PKT_TYPE_P2P);
02604                         break;
02605                 default:
02606                         purple_debug_warning("yahoo","p2p: p2p service %d Unhandled\n",pkt->service);
02607         }
02608 
02609         yahoo_packet_free(pkt);
02610 }

03436 static GHashTable *yahoo_login_page_hash(const char *buf, size_t len)
03437 {
03438         GHashTable *hash = g_hash_table_new_full(g_str_hash, g_str_equal, g_free, g_free);
03439         const char *c = buf;
03440         char *d;
03441         char name[64], value[64];
03442         int count;
03443         int input_len = strlen("<input ");
03444         int name_len = strlen("name=\"");
03445         int value_len = strlen("value=\"");
03446         while ((len > ((c - buf) + input_len))
03447                         && (c = strstr(c, "<input "))) {
03448                 if (!(c = g_strstr_len(c, len - (c - buf), "name=\"")))
03449                         continue;
03450                 c += name_len;
03451                 count = sizeof(name)-1;
03452                 for (d = name; (len > ((c - buf) + 1)) && *c!='"'
03453                                 && count; c++, d++, count--)
03454                         *d = *c;
03455                 *d = '\0';
03456                 count = sizeof(value)-1;
03457                 if (!(d = g_strstr_len(c, len - (c - buf), "value=\"")))
03458                         continue;
03459                 d += value_len;
03460                 if (strchr(c, '>') < d)
03461                         break;
03462                 for (c = d, d = value; (len > ((c - buf) + 1))
03463                                 && *c!='"' && count; c++, d++, count--)
03464                         *d = *c;
03465                 *d = '\0';
03466                 g_hash_table_insert(hash, g_strdup(name), g_strdup(value));
03467         }
03468         return hash;
03469 }
