00272  *  @param contact              The invited MultiMX room's contact information
00273  *  @param creator              The nickname of the room's creator / invitor
00274  */
00275 void multimx_invite(struct MXitSession* session, struct contact* contact, const char* creator)
00276 {
00277         GHashTable *components;
00278         struct multimx* multimx = NULL;
00279 
00280         purple_debug_info(MXIT_PLUGIN_ID, "Groupchat invite to '%s' (roomid='%s') by '%s'\n", contact->alias, contact->username, creator);

00291 
00292         /* Call libpurple - will trigger either 'mxit_chat_join' or 'mxit_chat_reject' */
00293         serv_got_chat_invite(session->con, contact->alias, creator, NULL, components);

00342 /*------------------------------------------------------------------------
00343  * Received a message from a MultiMX room.
00344  *
00345  */
00346 void multimx_message_received(struct RXMsgData* mx, char* msg, int msglen, short msgtype, int msgflags)
00347 {
00348         struct multimx* multimx = NULL;
00349 
00350         purple_debug_info(MXIT_PLUGIN_ID, "Groupchat message received: %s\n", msg);
00351 
00352         /* Find matching multimx group */
00353         multimx = find_room_by_username(mx->session, mx->from);
00354         if (multimx == NULL) {
00355                 purple_debug_error(MXIT_PLUGIN_ID, "Groupchat '%s' not found\n", mx->from);
00356                 return;
00357         }
00358 
00359         /* Determine if system message or a message from a contact */
00360         if (msg[0] == '<') {
00361                 /* Message contains embedded nickname - must be from contact */
00362                 unsigned int i;
00363 
00364                 for (i = 1; i < strlen(msg); i++) {             /* search for end of nickname */
00365                         if (msg[i] == '>') {
00366                                 msg[i] = '\0';
00367                                 g_free(mx->from);
00368                                 mx->from = g_strdup(&msg[1]);
00369                                 msg = &msg[i+2];                /* skip '>' and newline */
00370                                 break;
00371                         }
00372                 }
00373 
00374                 /* now do markup processing on the message */
00375                 mx->chatid = multimx->chatid;
00376                 mxit_parse_markup(mx, msg, strlen(msg), msgtype, msgflags);
00377         }
00378         else {
00379                 /* Must be a service message */
00380                 char* ofs;
00381 
00382                 PurpleConversation* convo = purple_find_conversation_with_account(PURPLE_CONV_TYPE_CHAT, multimx->roomname, mx->session->acc);
00383                 if (convo == NULL) {
00384                         purple_debug_error(MXIT_PLUGIN_ID, "Conversation '%s' not found\n", multimx->roomname);
00385                         return;
00386                 }
00387 
00388                 /* Determine if somebody has joined or left - update member-list */
00389                 if ((ofs = strstr(msg, " has joined")) != NULL) {
00390                         /* Somebody has joined */
00391                         *ofs = '\0';
00392                         member_added(convo, msg);
00393                         mx->processed = TRUE;
00394                 }
00395                 else if ((ofs = strstr(msg, " has left")) != NULL) {
00396                         /* Somebody has left */
00397                         *ofs = '\0';
00398                         member_removed(convo, msg);
00399                         mx->processed = TRUE;
00400                 }
00401                 else if ((ofs = strstr(msg, " has been kicked")) != NULL) {
00402                         /* Somebody has been kicked */
00403                         *ofs = '\0';
00404                         member_kicked(convo, msg);
00405                         mx->processed = TRUE;
00406                 }
00407                 else if (strcmp(msg, "You have been kicked.") == 0) {
00408                         /* You have been kicked */
00409                         you_kicked(convo, mx->session, multimx);
00410                         mx->processed = TRUE;
00411                 }
00412                 else if (g_str_has_prefix(msg, "The following users are in this MultiMx:") == TRUE) {
00413                         member_update(convo, msg + strlen("The following users are in this MultiMx:") + 1);
00414                         mx->processed = TRUE;
00415                 }
00416                 else {
00417                         /* Display server message in chat window */
00418                         serv_got_chat_in(mx->session->con, multimx->chatid, "MXit", PURPLE_MESSAGE_SYSTEM, msg, mx->timestamp);
00419                         mx->processed = TRUE;
00420                 }
00421         }
00422 }
