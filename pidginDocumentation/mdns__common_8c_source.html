00051 /* Make sure that the value isn't longer than it is supposed to be */
00052 static const char*
00053 get_max_txt_record_value(const char *key, const char *value)
00054 {
00055         /* "each constituent string of a DNS TXT record is limited to 255 bytes"
00056          * This includes the key and the '='
00057          */
00058         static char buffer[MAX_TXT_CONSTITUENT_LEN + 1];
00059         gchar *end_valid = NULL;
00060         int len = MIN(strlen(value), MAX_TXT_CONSTITUENT_LEN - (strlen(key) + 2));
00061 
00062         strncpy(buffer, value, len);
00063 
00064         buffer[len] = '\0';
00065 
00066         /* If we've cut part of a utf-8 character, kill it */
00067         if (!g_utf8_validate(buffer, -1, (const gchar **)&end_valid))
00068                 *end_valid = '\0';
00069 
00070         return buffer;
00071 }

00073 static GSList *generate_presence_txt_records(BonjourDnsSd *data) {
00074         GSList *ret = NULL;
00075         PurpleKeyValuePair *kvp;
00076         char portstring[6];
00077         const char *jid, *aim, *email;
00078 
00079         /* Convert the port to a string */
00080         snprintf(portstring, sizeof(portstring), "%d", data->port_p2pj);
00081 
00082         jid = purple_account_get_string(data->account, "jid", NULL);
00083         aim = purple_account_get_string(data->account, "AIM", NULL);
00084         email = purple_account_get_string(data->account, "email", NULL);
00085 
00086 #define _M_ADD_R(k, v) \
00087         kvp = g_new0(PurpleKeyValuePair, 1); \
00088         kvp->key = g_strdup(k); \
00089         kvp->value = g_strdup(get_max_txt_record_value(k, v)); \
00090         ret = g_slist_prepend(ret, kvp); \
00091 
00092         /* We should try to follow XEP-0174, but some clients have "issues", so we humor them.
00093          * See http://telepathy.freedesktop.org/wiki/SalutInteroperability
00094          */
00095 
00096         /* Large TXT records are problematic.
00097          * While it is technically possible for this to exceed a standard 512-byte
00098          * DNS message, it shouldn't happen unless we get wacky data entered for
00099          * some of the freeform fields.  It is even less likely to exceed the
00100          * recommended maximum of 1300 bytes.
00101          */
00102 
00103         /* Needed by iChat */
00104         _M_ADD_R("txtvers", "1")
00105         /* Needed by Gaim/Pidgin <= 2.0.1 (remove at some point) */
00106         _M_ADD_R("1st", data->first)
00107         /* Needed by Gaim/Pidgin <= 2.0.1 (remove at some point) */
00108         _M_ADD_R("last", data->last)
00109         /* Needed by Adium */
00110         _M_ADD_R("port.p2pj", portstring)
00111         /* Needed by iChat, Gaim/Pidgin <= 2.0.1 */
00112         _M_ADD_R("status", data->status)
00113         _M_ADD_R("node", "libpurple")
00114         _M_ADD_R("ver", VERSION)
00115         /* Currently always set to "!" since we don't support AV and wont ever be in a conference */
00116         _M_ADD_R("vc", data->vc)
00117         if (email != NULL && *email != '\0') {
00118                 _M_ADD_R("email", email)
00119         }
00120         if (jid != NULL && *jid != '\0') {
00121                 _M_ADD_R("jid", jid)
00122         }
00123         /* Nonstandard, but used by iChat */
00124         if (aim != NULL && *aim != '\0') {
00125                 _M_ADD_R("AIM", aim)
00126         }
00127         if (data->msg != NULL && *data->msg != '\0') {
00128                 _M_ADD_R("msg", data->msg)
00129         }
00130         if (data->phsh != NULL && *data->phsh != '\0') {
00131                 _M_ADD_R("phsh", data->phsh)
00132         }
00133 
00134         /* TODO: ext, nick */
00135         return ret;
00136 }

00257 bonjour_dns_sd_set_jid(PurpleAccount *account, const char *hostname)
00258 {
00259         PurpleConnection *conn = purple_account_get_connection(account);
00260         BonjourData *bd = conn->proto_data;
00261         const char *tmp, *account_name = purple_account_get_username(account);
00262 
00263         /* Previously we allowed the hostname part of the jid to be set
00264          * explicitly when it should always be the current hostname.
00265          * That is what this is intended to deal with.
00266          */
00267         if ((tmp = strchr(account_name, '@'))
00268             && strstr(tmp, hostname) == (tmp + 1)
00269             && *((tmp + 1) + strlen(hostname)) == '\0')
00270                 bd->jid = g_strdup(account_name);
00271         else {
00272                 const char *tmp2;
00273                 GString *str = g_string_new("");
00274                 /* Escape an '@' in the account name */
00275                 tmp = account_name;
00276                 while ((tmp2 = strchr(tmp, '@')) != NULL) {
00277                         g_string_append_len(str, tmp, tmp2 - tmp);
00278                         g_string_append(str, "\\40");
00279                         tmp = tmp2 + 1;
00280                 }
00281                 g_string_append(str, tmp);
00282                 g_string_append_c(str, '@');
00283                 g_string_append(str, hostname);
00284 
00285                 bd->jid = g_string_free(str, FALSE);
00286         }
