00027 /* This implements the secure file transfer protocol (SFTP) using the SILC
00028    SFTP library implementation.  The API we use from the SILC Toolkit is the
00029    SILC Client file transfer API, as it provides a simple file transfer we
00030    need in this case.  We could use the SILC SFTP API directly, but it would
00031    be an overkill since we'd effectively re-implement the file transfer what
00032    the SILC Client's file transfer API already provides.
00033 
00034    From Purple we do NOT use the FT API to do the transfer as it is very limiting.
00035    In fact it does not suite to file transfers like SFTP at all.  For example,
00036    it assumes that read operations are synchronous what they are not in SFTP.
00037    It also assumes that the file transfer socket is to be handled by the Purple
00038    eventloop, and this naturally is something we don't want to do in case of
00039    SILC Toolkit.  The FT API suites well to purely stream based file transfers
00040    like HTTP GET and similar.
00041 
00042    For this reason, we directly access the Purple GKT FT API and hack the FT
00043    API to merely provide the user interface experience and all the magic
00044    is done in the SILC Toolkit.  Ie. we update the statistics information in
00045    the FT API for user interface, and that's it.  A bit dirty but until the
00046    FT API gets better this is the way to go.  Good thing that FT API allowed
00047    us to do this. */

00063                      SilcClientConnection conn,
00064                      SilcClientMonitorStatus status,
00065                      SilcClientFileError error,
00066                      SilcUInt64 offset,
00067                      SilcUInt64 filesize,
00068                      SilcClientEntry client_entry,
00069                      SilcUInt32 session_id,
00070                      const char *filepath,
00071                      void *context)
00072 {
00073         SilcPurpleXfer xfer = context;
00074         PurpleConnection *gc = xfer->sg->gc;
00075         char tmp[256];
00076 
00077         if (status == SILC_CLIENT_FILE_MONITOR_CLOSED) {
00078                 /* All started sessions terminate here */
00079                 xfer->xfer->data = NULL;
00080                 purple_xfer_unref(xfer->xfer);
00081                 silc_free(xfer);
00082                 return;
00083         }
00084 
00085         if (status == SILC_CLIENT_FILE_MONITOR_DISCONNECT) {
00086                 purple_notify_error(gc, _("Secure File Transfer"),
00087                                     _("Error during file transfer"),
00088                                     _("Remote disconnected"));
00089                 xfer->xfer->status = PURPLE_XFER_STATUS_CANCEL_REMOTE;
00090                 purple_xfer_update_progress(xfer->xfer);
00091                 silc_client_file_close(client, conn, session_id);
00092                 return;
00093         }
00094 
00095         if (status == SILC_CLIENT_FILE_MONITOR_KEY_AGREEMENT)
00096                 return;
00097 
00098         if (status == SILC_CLIENT_FILE_MONITOR_ERROR) {
00099                 if (error == SILC_CLIENT_FILE_NO_SUCH_FILE) {
00100                         g_snprintf(tmp, sizeof(tmp), "No such file %s",
00101                                    filepath ? filepath : "[N/A]");
00102                         purple_notify_error(gc, _("Secure File Transfer"),
00103                                           _("Error during file transfer"), tmp);
00104                 } else if (error == SILC_CLIENT_FILE_PERMISSION_DENIED) {
00105                         purple_notify_error(gc, _("Secure File Transfer"),
00106                                           _("Error during file transfer"),
00107                                           _("Permission denied"));
00108                 } else if (error == SILC_CLIENT_FILE_KEY_AGREEMENT_FAILED) {
00109                         purple_notify_error(gc, _("Secure File Transfer"),
00110                                           _("Error during file transfer"),
00111                                           _("Key agreement failed"));
00112                 } else if (error == SILC_CLIENT_FILE_TIMEOUT) {
00113                         purple_notify_error(gc, _("Secure File Transfer"),
00114                                           _("Error during file transfer"),
00115                                           _("Connection timed out"));
00116                 } else if (error == SILC_CLIENT_FILE_CONNECT_FAILED) {
00117                         purple_notify_error(gc, _("Secure File Transfer"),
00118                                           _("Error during file transfer"),
00119                                           _("Creating connection failed"));
00120                 } else if (error == SILC_CLIENT_FILE_UNKNOWN_SESSION) {
00121                         purple_notify_error(gc, _("Secure File Transfer"),
00122                                           _("Error during file transfer"),
00123                                           _("File transfer session does not exist"));
00124                 }
00125                 xfer->xfer->status = PURPLE_XFER_STATUS_CANCEL_REMOTE;
00126                 purple_xfer_update_progress(xfer->xfer);
00127                 silc_client_file_close(client, conn, session_id);
00128                 return;
00129         }
00130 
00131         /* Update file transfer UI */
00132         if (!offset && filesize)
00133                 purple_xfer_set_size(xfer->xfer, filesize);
00134         if (offset && filesize) {
00135                 xfer->xfer->bytes_sent = offset;
00136                 xfer->xfer->bytes_remaining = filesize - offset;
00137         }
00138         purple_xfer_update_progress(xfer->xfer);
00139 
00140         if (status == SILC_CLIENT_FILE_MONITOR_SEND ||
00141             status == SILC_CLIENT_FILE_MONITOR_RECEIVE) {
00142                 if (offset == filesize) {
00143                         /* Download finished */
00144                         purple_xfer_set_completed(xfer->xfer, TRUE);
00145                         silc_client_file_close(client, conn, session_id);
00146                 }
00147         }
00148 }

00212 static void
00213 silcpurple_ftp_request_result(PurpleXfer *x)
00214 {
00215         SilcPurpleXfer xfer = x->data;
00216         SilcClientFileError status;
00217         PurpleConnection *gc = xfer->sg->gc;
00218         SilcClientConnectionParams params;
00219         gboolean local = xfer->hostname ? FALSE : TRUE;
00220         char *local_ip = NULL, *remote_ip = NULL;
00221         SilcSocket sock;
00222 
00223         if (purple_xfer_get_status(x) != PURPLE_XFER_STATUS_ACCEPTED)
00224                 return;
00225         if (!xfer)
00226                 return;
00227 
00228         silc_socket_stream_get_info(silc_packet_stream_get_stream(xfer->sg->conn->stream),
00229                                     &sock, NULL, NULL, NULL);
00230 
00231         if (local) {
00232                 /* Do the same magic what we do with key agreement (see silcpurple_buddy.c)
00233                    to see if we are behind NAT. */
00234                 if (silc_net_check_local_by_sock(sock, NULL, &local_ip)) {
00235                         /* Check if the IP is private */
00236                         if (silcpurple_ip_is_private(local_ip)) {
00237                                 local = TRUE;
00238                                 /* Local IP is private, resolve the remote server IP to see whether
00239                                    we are talking to Internet or just on LAN. */
00240                                 if (silc_net_check_host_by_sock(sock, NULL,
00241                                                                 &remote_ip))
00242                                   if (silcpurple_ip_is_private(remote_ip))
00243                                     /* We assume we are in LAN.  Let's provide the connection point. */
00244                                     local = TRUE;
00245                         }
00246                 }
00247 
00248                 if (local && !local_ip)
00249                   local_ip = silc_net_localip();
00250         }
00251 
00252         memset(&params, 0, sizeof(params));
00253         params.timeout_secs = 60;
00254         if (local)
00255           /* Provide connection point */
00256           params.local_ip = local_ip;
00257 
00258         /* Start the file transfer */
00259         status = silc_client_file_receive(xfer->sg->client, xfer->sg->conn,
00260                                           &params, xfer->sg->public_key,
00261                                           xfer->sg->private_key,
00262                                           silcpurple_ftp_monitor, xfer,
00263                                           NULL, xfer->session_id,
00264                                           silcpurple_ftp_ask_name, xfer);
00265         switch (status) {
00266         case SILC_CLIENT_FILE_OK:
00267                 silc_free(local_ip);
00268                 silc_free(remote_ip);
00269                 return;
00270                 break;
00271 
00272         case SILC_CLIENT_FILE_UNKNOWN_SESSION:
00273                 purple_notify_error(gc, _("Secure File Transfer"),
00274                                   _("No file transfer session active"), NULL);
00275                 break;
00276 
00277         case SILC_CLIENT_FILE_ALREADY_STARTED:
00278                 purple_notify_error(gc, _("Secure File Transfer"),
00279                                   _("File transfer already started"), NULL);
00280                 break;
00281 
00282         case SILC_CLIENT_FILE_KEY_AGREEMENT_FAILED:
00283                 purple_notify_error(gc, _("Secure File Transfer"),
00284                                   _("Could not perform key agreement for file transfer"),
00285                                   NULL);
00286                 break;
00287 
00288         default:
00289                 purple_notify_error(gc, _("Secure File Transfer"),
00290                                   _("Could not start the file transfer"), NULL);
00291                 break;
00292         }
00293 
00294         /* Error */
00295         purple_xfer_unref(xfer->xfer);
00296         g_free(xfer->hostname);
00297         silc_free(xfer);
00298         silc_free(local_ip);
00299         silc_free(remote_ip);
00300 }

00309                             SilcClientEntry client_entry, SilcUInt32 session_id,
00310                             const char *hostname, SilcUInt16 port)
00311 {
00312         PurpleConnection *gc = client->application;

00314         SilcPurpleXfer xfer;
00315 
00316         xfer = silc_calloc(1, sizeof(*xfer));
00317         if (!xfer) {
00318                 silc_client_file_close(sg->client, sg->conn, session_id);
00319                 return;
00320         }
00321 
00322         xfer->sg = sg;
00323         xfer->client_entry = client_entry;
00324         xfer->session_id = session_id;
00325         xfer->hostname = g_strdup(hostname);
00326         xfer->port = port;
00327         xfer->xfer = purple_xfer_new(xfer->sg->account, PURPLE_XFER_RECEIVE,
00328                                      xfer->client_entry->nickname);
00329         if (!xfer->xfer) {
00330                 silc_client_file_close(xfer->sg->client, xfer->sg->conn, xfer->session_id);
00331                 g_free(xfer->hostname);
00332                 silc_free(xfer);
00333                 return;
00334         }
00335         purple_xfer_set_init_fnc(xfer->xfer, silcpurple_ftp_request_result);
00336         purple_xfer_set_request_denied_fnc(xfer->xfer, silcpurple_ftp_request_denied);
00337         purple_xfer_set_cancel_recv_fnc(xfer->xfer, silcpurple_ftp_cancel);
00338         xfer->xfer->remote_ip = g_strdup(hostname);
00339         xfer->xfer->remote_port = port;
00340         xfer->xfer->data = xfer;
00341 
00342         /* File transfer request */
00343         purple_xfer_request(xfer->xfer);
