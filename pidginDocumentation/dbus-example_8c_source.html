00109 static gboolean
00110 plugin_load(PurplePlugin *plugin)
00111 {
00112         PURPLE_DBUS_RETURN_FALSE_IF_DISABLED(plugin);
00113 
00114         /* First, we have to register our four exported functions with the
00115            main purple dbus loop.  Without this statement, the purple dbus
00116            code wouldn't know about our functions. */
00117         PURPLE_DBUS_REGISTER_BINDINGS(plugin);
00118 
00119         /* Then, we register the hello object of type PurpleText.  Note that
00120            pointer registrations / unregistrations are completely dynamic;
00121            they don't have to be made when the plugin is loaded /
00122            unloaded.  Without this statement the dbus purple code wouldn't
00123            know about the hello object.  */
00124         PURPLE_DBUS_REGISTER_POINTER(&hello, PurpleText);
00125 
00126         hello.text = g_strdup("Hello.");
00127 
00128         return TRUE;
00129 }

00140         return TRUE;
