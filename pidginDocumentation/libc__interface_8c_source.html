00205 /* sys/ioctl.h */
00206 int wpurple_ioctl(int fd, int command, void* val) {
00207         switch( command ) {
00208         case FIONBIO:
00209         {
00210                 if (ioctlsocket(fd, FIONBIO, (unsigned long *)val) == SOCKET_ERROR) {
00211                         errno = WSAGetLastError();
00212                         return -1;
00213                 }
00214                 return 0;
00215         }
00216         case SIOCGIFCONF:
00217         {
00218                 INTERFACE_INFO InterfaceList[20];
00219                 unsigned long nBytesReturned;
00220                 if (WSAIoctl(fd, SIO_GET_INTERFACE_LIST,
00221                                 0, 0, &InterfaceList,
00222                                 sizeof(InterfaceList), &nBytesReturned,
00223                                 0, 0) == SOCKET_ERROR) {
00224                         errno = WSAGetLastError();
00225                         return -1;
00226                 } else {
00227                         int i;
00228                         struct ifconf *ifc = val;
00229                         char *tmp = ifc->ifc_buf;
00230                         int nNumInterfaces =
00231                                 nBytesReturned / sizeof(INTERFACE_INFO);
00232                         for (i = 0; i < nNumInterfaces; i++) {
00233                                 INTERFACE_INFO ii = InterfaceList[i];
00234                                 struct ifreq *ifr = (struct ifreq *) tmp;
00235                                 struct sockaddr_in *sa = (struct sockaddr_in *) &ifr->ifr_addr;
00236 
00237                                 sa->sin_family = ii.iiAddress.AddressIn.sin_family;
00238                                 sa->sin_port = ii.iiAddress.AddressIn.sin_port;
00239                                 sa->sin_addr.s_addr = ii.iiAddress.AddressIn.sin_addr.s_addr;
00240                                 tmp += sizeof(struct ifreq);
00241 
00242                                 /* Make sure that we can fit in the original buffer */
00243                                 if (tmp >= (ifc->ifc_buf + ifc->ifc_len + sizeof(struct ifreq))) {
00244                                         break;
00245                                 }
00246                         }
00247                         /* Replace the length with the actually used length */
00248                         ifc->ifc_len = ifc->ifc_len - (ifc->ifc_buf - tmp);
00249                         return 0;
00250                 }
00251         }
00252         default:
00253                 errno = EINVAL;
00254                 return -1;
00255         }/*end switch*/
00256 }

00268 wpurple_inet_ntop (int af, const void *src, char *dst, socklen_t cnt)
00269 {
00270   /* struct sockaddr can't accomodate struct sockaddr_in6. */
00271   union {
00272     struct sockaddr_in6 sin6;
00273     struct sockaddr_in sin;
00274   } sa;
00275   DWORD dstlen = cnt;
00276   size_t srcsize;
00277 
00278   ZeroMemory(&sa, sizeof(sa));
00279   switch (af)
00280     {
00281     case AF_INET:
00282       sa.sin.sin_family = AF_INET;
00283       sa.sin.sin_addr = *(struct in_addr *) src;
00284       srcsize = sizeof (sa.sin);
00285       break;
00286     case AF_INET6:
00287       sa.sin6.sin6_family = AF_INET6;
00288       sa.sin6.sin6_addr = *(struct in6_addr *) src;
00289       srcsize = sizeof (sa.sin6);
00290       break;
00291     default:
00292       abort ();
00293     }
00294 
00295   if (WSAAddressToString ((struct sockaddr *) &sa, srcsize, NULL, dst, &dstlen) != 0)
00296     {
00297       errno = WSAGetLastError();
00298       return NULL;
00299     }
00300   return (const char *) dst;
