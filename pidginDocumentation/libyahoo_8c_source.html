00060 static PurpleAccount *find_acct(const char *prpl, const char *acct_id)
00061 {
00062         PurpleAccount *acct = NULL;
00063 
00064         /* If we have a specific acct, use it */
00065         if (acct_id) {
00066                 acct = purple_accounts_find(acct_id, prpl);
00067                 if (acct && !purple_account_is_connected(acct))
00068                         acct = NULL;
00069         } else { /* Otherwise find an active account for the protocol */
00070                 GList *l = purple_accounts_get_all();
00071                 while (l) {
00072                         if (!strcmp(prpl, purple_account_get_protocol_id(l->data))
00073                                         && purple_account_is_connected(l->data)) {
00074                                 acct = l->data;
00075                                 break;
00076                         }
00077                         l = l->next;
00078                 }
00079         }
00080 
00081         return acct;
00082 }

00095 static gboolean yahoo_uri_handler(const char *proto, const char *cmd, GHashTable *params)
00096 {
00097         char *acct_id = g_hash_table_lookup(params, "account");
00098         PurpleAccount *acct;
00099 
00100         if (g_ascii_strcasecmp(proto, "ymsgr"))
00101                 return FALSE;
00102 
00103         acct = find_acct(purple_plugin_get_id(my_protocol), acct_id);
00104 
00105         if (!acct)
00106                 return FALSE;
00107 
00108         /* ymsgr:SendIM?screename&m=The+Message */
00109         if (!g_ascii_strcasecmp(cmd, "SendIM")) {
00110                 char *sname = NULL;
00111                 g_hash_table_foreach(params, yahoo_find_uri_novalue_param, &sname);
00112                 if (sname) {
00113                         char *message = g_hash_table_lookup(params, "m");
00114 
00115                         PurpleConversation *conv = purple_find_conversation_with_account(
00116                                 PURPLE_CONV_TYPE_IM, sname, acct);
00117                         if (conv == NULL)
00118                                 conv = purple_conversation_new(PURPLE_CONV_TYPE_IM, acct, sname);
00119                         purple_conversation_present(conv);
00120 
00121                         if (message) {
00122                                 /* Spaces are encoded as '+' */
00123                                 g_strdelimit(message, "+", ' ');
00124                                 purple_conv_send_confirm(conv, message);
00125                         }
00126                 }
00127                 /* else
00128                         **If pidgindialogs_im() was in the core, we could use it here.
00129                          * It is all purple_request_* based, but I'm not sure it really belongs in the core
00130                         pidgindialogs_im(); */
00131 
00132                 return TRUE;
00133         }
00134         /* ymsgr:Chat?roomname */
00135         else if (!g_ascii_strcasecmp(cmd, "Chat")) {
00136                 char *rname = NULL;
00137                 g_hash_table_foreach(params, yahoo_find_uri_novalue_param, &rname);
00138                 if (rname) {
00139                         /* This is somewhat hacky, but the params aren't useful after this command */
00140                         g_hash_table_insert(params, g_strdup("room"), g_strdup(rname));
00141                         g_hash_table_insert(params, g_strdup("type"), g_strdup("Chat"));
00142                         serv_join_chat(purple_account_get_connection(acct), params);
00143                 }
00144                 /* else
00145                         ** Same as above (except that this would have to be re-written using purple_request_*)
00146                         pidgin_blist_joinchat_show(); */
00147 
00148                 return TRUE;
00149         }
00150         /* ymsgr:AddFriend?name */
00151         else if (!g_ascii_strcasecmp(cmd, "AddFriend")) {
00152                 char *name = NULL;
00153                 g_hash_table_foreach(params, yahoo_find_uri_novalue_param, &name);
00154                 purple_blist_request_add_buddy(acct, name, NULL, NULL);
00155                 return TRUE;
00156         }
00157 
00158         return FALSE;
