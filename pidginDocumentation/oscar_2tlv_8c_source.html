00043 static GSList *
00044 aim_tlv_read(GSList *list, ByteStream *bs)
00045 {
00046         guint16 type, length;
00047         aim_tlv_t *tlv;
00048 
00049         type = byte_stream_get16(bs);
00050         length = byte_stream_get16(bs);
00051 
00052         if (length > byte_stream_bytes_left(bs)) {
00053                 aim_tlvlist_free(list);
00054                 return NULL;
00055         }
00056 
00057         tlv = createtlv(type, length, NULL);
00058         if (tlv->length > 0) {
00059                 tlv->value = byte_stream_getraw(bs, length);
00060                 if (!tlv->value) {
00061                         freetlv(tlv);
00062                         aim_tlvlist_free(list);
00063                         return NULL;
00064                 }
00065         }
00066 
00067         return g_slist_prepend(list, tlv);
00068 }

00192 /*
00193  * Compare two TLV lists for equality.  This probably is not the most
00194  * efficient way to do this.
00195  *
00196  * @param one One of the TLV chains to compare.
00197  * @param two The other TLV chain to compare.
00198  * @return Return 0 if the lists are the same, return 1 if they are different.
00199  */
00200 int aim_tlvlist_cmp(GSList *one, GSList *two)
00201 {
00202         ByteStream bs1, bs2;
00203 
00204         if (aim_tlvlist_size(one) != aim_tlvlist_size(two))
00205                 return 1;
00206 
00207         byte_stream_new(&bs1, aim_tlvlist_size(one));
00208         byte_stream_new(&bs2, aim_tlvlist_size(two));
00209 
00210         aim_tlvlist_write(&bs1, &one);
00211         aim_tlvlist_write(&bs2, &two);
00212 
00213         if (memcmp(bs1.data, bs2.data, bs1.len)) {
00214                 byte_stream_destroy(&bs1);
00215                 byte_stream_destroy(&bs2);
00216                 return 1;
00217         }
00218 
00219         byte_stream_destroy(&bs1);
00220         byte_stream_destroy(&bs2);
00221 
00222         return 0;
00223 }

00517 {
