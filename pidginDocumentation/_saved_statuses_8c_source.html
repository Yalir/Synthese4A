00170 static gint
00171 saved_statuses_sort_func(gconstpointer a, gconstpointer b)
00172 {
00173         const PurpleSavedStatus *saved_status_a = a;
00174         const PurpleSavedStatus *saved_status_b = b;
00175         time_t time_a = saved_status_a->lastused +
00176                                                 (MIN(saved_status_a->usage_count, 10) * 86400);
00177         time_t time_b = saved_status_b->lastused +
00178                                                 (MIN(saved_status_b->usage_count, 10) * 86400);
00179         if (time_a > time_b)
00180                 return -1;
00181         if (time_a < time_b)
00182                 return 1;
00183         return 0;

00452 static PurpleSavedStatus *
00453 parse_status(xmlnode *status)
00454 {
00455         PurpleSavedStatus *ret;
00456         xmlnode *node;
00457         const char *attrib;
00458         char *data;
00459         int i;
00460 
00461         ret = g_new0(PurpleSavedStatus, 1);
00462 
00463         attrib = xmlnode_get_attrib(status, "transient");
00464         if (!purple_strequal(attrib, "true"))
00465         {
00466                 /* Read the title */
00467                 attrib = xmlnode_get_attrib(status, "name");
00468                 ret->title = g_strdup(attrib);
00469         }
00470 
00471         if (ret->title != NULL)
00472         {
00473                 /* Ensure the title is unique */
00474                 i = 2;
00475                 while (purple_savedstatus_find(ret->title) != NULL)
00476                 {
00477                         g_free(ret->title);
00478                         ret->title = g_strdup_printf("%s %d", attrib, i);
00479                         i++;
00480                 }
00481         }
00482 
00483         /* Read the creation time */
00484         attrib = xmlnode_get_attrib(status, "created");
00485         set_creation_time(ret, (attrib != NULL ? atol(attrib) : 0));
00486 
00487         /* Read the last used time */
00488         attrib = xmlnode_get_attrib(status, "lastused");
00489         ret->lastused = (attrib != NULL ? atol(attrib) : 0);
00490 
00491         /* Read the usage count */
00492         attrib = xmlnode_get_attrib(status, "usage_count");
00493         ret->usage_count = (attrib != NULL ? atol(attrib) : 0);
00494 
00495         /* Read the primitive status type */
00496         node = xmlnode_get_child(status, "state");
00497         if ((node != NULL) && ((data = xmlnode_get_data(node)) != NULL))
00498         {
00499                 ret->type = purple_primitive_get_type_from_id(data);
00500                 g_free(data);
00501         }
00502 
00503         /* Read the message */
00504         node = xmlnode_get_child(status, "message");
00505         if ((node != NULL) && ((data = xmlnode_get_data(node)) != NULL))
00506         {
00507                 ret->message = data;
00508         }
00509 
00510         /* Read substatuses */
00511         for (node = xmlnode_get_child(status, "substatus"); node != NULL;
00512                         node = xmlnode_get_next_twin(node))
00513         {
00514                 PurpleSavedStatusSub *new;
00515                 new = parse_substatus(node);
00516                 if (new != NULL)
00517                         ret->substatuses = g_list_prepend(ret->substatuses, new);
00518         }
00519 
00520         PURPLE_DBUS_REGISTER_POINTER(ret, PurpleSavedStatus);
00521         return ret;
00522 }
