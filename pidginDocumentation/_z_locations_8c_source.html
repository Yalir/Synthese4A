00044     char *class;
00045     char *opcode;
00046     Z_AuthProc auth;
00047     char *format;
00048 {
00049     int retval;
00050     time_t ourtime;
00051     ZNotice_t notice, retnotice;
00052     char *bptr[3];
00053 #ifndef X_DISPLAY_MISSING
00054     char *display;
00055 #endif
00056 #ifndef WIN32
00057     char *ttyp;
00058     char *p;
00059 #endif
00060     struct hostent *hent;
00061     short wg_port = ZGetWGPort();
00062 
00063     (void) memset((char *)&notice, 0, sizeof(notice));
00064     notice.z_kind = ACKED;
00065     notice.z_port = (unsigned short) ((wg_port == -1) ? 0 : wg_port);
00066     notice.z_class = class;
00067     notice.z_class_inst = ZGetSender();
00068     notice.z_opcode = opcode;
00069     notice.z_sender = 0;
00070     notice.z_recipient = "";

00074     /*
00075       keep track of what we said before so that we can be consistent
00076       when changing location information.
00077       This is done mainly for the sake of the WindowGram client.
00078      */
00079 
00080     if (!reenter) {
00081             if (gethostname(host, MAXHOSTNAMELEN) < 0)
00082                     return (errno);
00083 
00084             hent = gethostbyname(host);
00085             if (hent) {
00086               (void) strncpy(host, hent->h_name, sizeof(host));
00087               host[sizeof(host) - 1] = '\0';
00088             }
00089 #ifndef X_DISPLAY_MISSING
00090             if ((display = getenv("DISPLAY")) && *display) {
00091                     mytty = g_strdup(display);
00092             } else {
00093 #endif
00094 #ifdef WIN32
00095                     mytty = g_strdup("WinPurple");
00096 #else
00097                     ttyp = ttyname(0);
00098                     if (ttyp && *ttyp) {
00099                         p = strchr(ttyp + 1, '/');
00100                         mytty = g_strdup((p) ? p + 1 : ttyp);
00101                     } else {
00102                         mytty = g_strdup("unknown");
00103                     }
00104 #endif
00105 #ifndef X_DISPLAY_MISSING
00106             }
00107 #endif
00108             reenter = 1;
00109     }
00110 
00111     ourtime = time((time_t *)0);
00112     bptr[0] = host;
00113     bptr[1] = ctime(&ourtime);
00114     bptr[1][strlen(bptr[1])-1] = '\0';
00115     bptr[2] = mytty;
00116 
00117     if ((retval = ZSendList(&notice, bptr, 3, auth)) != ZERR_NONE)
00118         return (retval);
00119 
00120     retval = Z_WaitForNotice (&retnotice, ZCompareUIDPred, &notice.z_uid,
00121                               SRV_TIMEOUT);
00122     if (retval != ZERR_NONE)
00123       return retval;
00124 
00125     if (retnotice.z_kind == SERVNAK) {
00126         if (!retnotice.z_message_len) {
00127             ZFreeNotice(&retnotice);
00128             return (ZERR_SERVNAK);
00129         }
00130         if (!strcmp(retnotice.z_message, ZSRVACK_NOTSENT)) {
00131             ZFreeNotice(&retnotice);
00132             return (ZERR_AUTHFAIL);
00133         }
00134         if (!strcmp(retnotice.z_message, ZSRVACK_FAIL)) {
00135             ZFreeNotice(&retnotice);
00136             return (ZERR_LOGINFAIL);
00137         }
00138         ZFreeNotice(&retnotice);
00139         return (ZERR_SERVNAK);
00140     }
00141 
00142     if (retnotice.z_kind != SERVACK) {
00143         ZFreeNotice(&retnotice);
00144         return (ZERR_INTERNAL);
00145     }
00146 
00147     if (!retnotice.z_message_len) {
00148         ZFreeNotice(&retnotice);
00149         return (ZERR_INTERNAL);
00150     }
00151 
00152     if (strcmp(retnotice.z_message, ZSRVACK_SENT) &&
00153         strcmp(retnotice.z_message, ZSRVACK_NOTSENT)) {
00154         ZFreeNotice(&retnotice);
00155         return (ZERR_INTERNAL);
00156     }
