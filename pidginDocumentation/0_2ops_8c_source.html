01916    if there is some protocol active in the client.  The `protocol' is the

01923 static void
01924 silc_failure(SilcClient client, SilcClientConnection conn,
01925              SilcProtocol protocol, void *failure)
01926 {
01927         PurpleConnection *gc = client->application;

01929 
01930         memset(buf, 0, sizeof(buf));
