00053 /*------------------------------------------------------------------------
00054  * Handle an URI clicked on the UI
00055  *
00056  * @param link  the link name which has been clicked
00057  */
00058 static void* mxit_link_click( const char* link64 )
00059 {
00060         PurpleAccount*          account;
00061         PurpleConnection*       gc;
00062         gchar**                         parts           = NULL;
00063         gchar*                          link            = NULL;
00064         gsize                           len;
00065         gboolean                        is_command      = FALSE;
00066 
00067         purple_debug_info( MXIT_PLUGIN_ID, "mxit_link_click (%s)\n", link64 );
00068 
00069         if ( g_ascii_strncasecmp( link64, MXIT_LINK_PREFIX, strlen( MXIT_LINK_PREFIX ) ) != 0 ) {
00070                 /* this is not for us */
00071                 goto skip;
00072         }
00073 
00074         /* decode the base64 payload */
00075         link = (gchar*) purple_base64_decode( link64 + strlen( MXIT_LINK_PREFIX ), &len );
00076         purple_debug_info( MXIT_PLUGIN_ID, "Clicked Link: '%s'\n", link );
00077 
00078         parts = g_strsplit( link, "|", 6 );
00079 
00080         /* check if this is a valid mxit link */
00081         if ( ( !parts ) || ( !parts[0] ) || ( !parts[1] ) || ( !parts[2] ) || ( !parts[3] ) || ( !parts[4] ) || ( !parts[5] ) ) {
00082                 /* this is not for us */
00083                 goto skip;
00084         }
00085         else if ( g_ascii_strcasecmp( parts[0], MXIT_LINK_KEY ) != 0 ) {
00086                 /* this is not for us */
00087                 goto skip;
00088         }
00089 
00090         /* find the account */
00091         account = purple_accounts_find( parts[1], parts[2] );
00092         if ( !account )
00093                 goto skip;
00094         gc = purple_account_get_connection( account );
00095         if ( !gc )
00096                 goto skip;
00097 
00098         /* determine if it's a command-response to send */
00099         is_command = ( atoi( parts[4] ) == 1 );
00100 
00101         /* send click message back to MXit */
00102         mxit_send_message( purple_connection_get_protocol_data( gc ), parts[3], parts[5], FALSE, is_command );
00103 
00104         g_free( link );
00105         link = NULL;
00106         g_strfreev( parts );
00107         parts = NULL;
00108 
00109         return (void*) link64;
00110 
00111 skip:
00112         /* this is not an internal mxit link */
00113 
00114         if ( link )
00115                 g_free( link );
00116         link = NULL;
00117 
00118         if ( parts )
00119                 g_strfreev( parts );
00120         parts = NULL;
00121 
00122         if ( mxit_pidgin_uri_cb )
00123                 return mxit_pidgin_uri_cb( link64 );
00124         else
00125                 return (void*) link64;
00126 }

00505 /*------------------------------------------------------------------------
00506  * Periodic task called every KEEPALIVE_INTERVAL (30 sec) to to maintain
00507  * idle connections, timeouts and the transmission queue to the MXit server.
00508  *
00509  *  @param gc           The connection object
00510  */
00511 static void mxit_keepalive( PurpleConnection *gc )
00512 {
00513         struct MXitSession*     session = (struct MXitSession*) gc->proto_data;
00514 
00515         /* if not logged in, there is nothing to do */
00516         if ( !( session->flags & MXIT_FLAG_LOGGEDIN ) )
00517                 return;
00518 
00519         /* pinging is only for socket connections (HTTP does polling) */
00520         if ( session->http )
00521                 return;
00522 
00523         if ( session->last_tx <= ( mxit_now_milli() - ( MXIT_PING_INTERVAL * 1000 ) ) ) {
00524                 /*
00525                  * this connection has been idle for too long, better ping
00526                  * the server before it kills our connection.
00527                  */
00528                 mxit_send_ping( session );
00529         }
00530 }

00675  */
00676 static unsigned int mxit_send_typing( PurpleConnection *gc, const char *name, PurpleTypingState state )
00677 {
00678         PurpleAccount*          account         = purple_connection_get_account( gc );
00679         struct MXitSession*     session         = purple_connection_get_protocol_data( gc );
00680         PurpleBuddy*            buddy;
00681         struct contact*         contact;
00682         gchar*                          messageId       = NULL;
00683 
00684         /* find the buddy information for this contact (reference: "libpurple/blist.h") */
00685         buddy = purple_find_buddy( account, name );
00686         if ( !buddy ) {
00687                 purple_debug_warning( MXIT_PLUGIN_ID, "mxit_send_typing: unable to find the buddy '%s'\n", name );
00688                 return 0;
00689         }
00690 
00691         contact = purple_buddy_get_protocol_data( buddy );
00692         if ( !contact )
00693                 return 0;
00694 
00695         /* does this contact support and want typing notification? */
00696         if ( ! ( contact->capabilities & MXIT_PFLAG_TYPING ) )
00697                 return 0;
00698 
00699         messageId = purple_uuid_random();               /* generate a unique message id */
00700 
00701         switch ( state ) {
00702                 case PURPLE_TYPING :            /* currently typing */
00703                         mxit_send_msgevent( session, name, messageId, CP_MSGEVENT_TYPING );
00704                         break;
00705 
00706                 case PURPLE_TYPED :                     /* stopped typing */
00707                 case PURPLE_NOT_TYPING :        /* not typing / erased all text */
00708                         mxit_send_msgevent( session, name, messageId, CP_MSGEVENT_STOPPED );
00709                         break;
00710 
00711                 default:
00712                         break;
00713         }
00714 
00715         g_free( messageId );

00717         return 0;
