00032 static int
00033 error(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00034 {
00035         int ret = 0;
00036         aim_snac_t *snac2;
00037         guint16 error, chatnav_error;
00038         GSList *tlvlist;
00039 
00040         snac2 = aim_remsnac(od, snac->id);
00041         if (!snac2) {
00042                 purple_debug_warning("oscar", "chatnav error: received response to unknown request (%08x)\n", snac->id);
00043                 return 0;
00044         }
00045 
00046         if (snac2->family != SNAC_FAMILY_CHATNAV) {
00047                 purple_debug_warning("oscar", "chatnav error: received response that maps to corrupt request (fam=%04x)\n", snac2->family);
00048                 g_free(snac2->data);
00049                 g_free(snac2);
00050                 return 0;
00051         }
00052 
00053         /*
00054          * We now know what the original SNAC subtype was.
00055          */
00056         if (snac2->type == 0x0008) /* create room */
00057         {
00058                 error = byte_stream_get16(bs);
00059                 tlvlist = aim_tlvlist_read(bs);
00060                 chatnav_error = aim_tlv_get16(tlvlist, 0x0008, 1);
00061 
00062                 purple_debug_warning("oscar",
00063                                 "Could not join room, error=0x%04hx, chatnav_error=0x%04hx\n",
00064                                 error, chatnav_error);
00065                 purple_notify_error(od->gc, NULL, _("Could not join chat room"),
00066                                 chatnav_error == 0x0033 ? _("Invalid chat room name") : _("Unknown error"));
00067 
00068                 ret = 1;
00069         }
00070 
00071         g_free(snac2->data);
00072         g_free(snac2);
00073 
00074         return ret;
00075 }

00088 /*
00089  * Subtype 0x0008
00090  */
00091 int aim_chatnav_createroom(OscarData *od, FlapConnection *conn, const char *name, guint16 exchange)
00092 {
00093         static const char ck[] = {"create"};
00094         static const char lang[] = {"en"};
00095         static const char charset[] = {"us-ascii"};
00096         ByteStream bs;
00097         aim_snacid_t snacid;
00098         GSList *tlvlist = NULL;
00099 
00100         byte_stream_new(&bs, 1142);
00101 
00102         snacid = aim_cachesnac(od, SNAC_FAMILY_CHATNAV, 0x0008, 0x0000, NULL, 0);
00103 
00104         /* exchange */
00105         byte_stream_put16(&bs, exchange);
00106 
00107         /*
00108          * This looks to be a big hack.  You'll note that this entire
00109          * SNAC is just a room info structure, but the hard room name,
00110          * here, is set to "create".
00111          *
00112          * Either this goes on the "list of questions concerning
00113          * why-the-hell-did-you-do-that", or this value is completely
00114          * ignored.  Without experimental evidence, but a good knowledge of
00115          * AOL style, I'm going to guess that it is the latter, and that
00116          * the value of the room name in create requests is ignored.
00117          */
00118         byte_stream_put8(&bs, strlen(ck));
00119         byte_stream_putstr(&bs, ck);
00120 
00121         /*
00122          * instance
00123          *
00124          * Setting this to 0xffff apparently assigns the last instance.
00125          *
00126          */
00127         byte_stream_put16(&bs, 0xffff);
00128 
00129         /* detail level */
00130         byte_stream_put8(&bs, 0x01);
00131 
00132         aim_tlvlist_add_str(&tlvlist, 0x00d3, name);
00133         aim_tlvlist_add_str(&tlvlist, 0x00d6, charset);
00134         aim_tlvlist_add_str(&tlvlist, 0x00d7, lang);
00135 
00136         /* tlvcount */
00137         byte_stream_put16(&bs, aim_tlvlist_count(tlvlist));
00138         aim_tlvlist_write(&bs, &tlvlist);
00139 
00140         aim_tlvlist_free(tlvlist);
00141 
00142         flap_connection_send_snac(od, conn, SNAC_FAMILY_CHATNAV, 0x0008, snacid, &bs);
00143 
00144         byte_stream_destroy(&bs);
00145 
00146         return 0;
00147 }

00362 /*
00363  * Subtype 0x0009
00364  *
00365  * Since multiple things can trigger this callback, we must lookup the
00366  * snacid to determine the original snac subtype that was called.
00367  *
00368  * XXX This isn't really how this works.  But this is:  Every d/9 response
00369  * has a 16bit value at the beginning. That matches to:
00370  *    Short Desc = 1
00371  *    Full Desc = 2
00372  *    Instance Info = 4
00373  *    Nav Short Desc = 8
00374  *    Nav Instance Info = 16
00375  * And then everything is really asynchronous.  There is no specific
00376  * attachment of a response to a create room request, for example.  Creating
00377  * the room yields no different a response than requesting the room's info.
00378  *
00379  */
00380 static int
00381 parseinfo(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00382 {
00383         aim_snac_t *snac2;
00384         int ret = 0;
00385 
00386         if (!(snac2 = aim_remsnac(od, snac->id))) {
00387                 purple_debug_misc("oscar", "faim: chatnav_parse_info: received response to unknown request! (%08x)\n", snac->id);
00388                 return 0;
00389         }
00390 
00391         if (snac2->family != SNAC_FAMILY_CHATNAV) {
00392                 purple_debug_misc("oscar", "faim: chatnav_parse_info: received response that maps to corrupt request! (fam=%04x)\n", snac2->family);
00393                 g_free(snac2->data);
00394                 g_free(snac2);
00395                 return 0;
00396         }
00397 
00398         /*
00399          * We now know what the original SNAC subtype was.
00400          */
00401         if (snac2->type == 0x0002) /* request chat rights */
00402                 ret = parseinfo_perms(od, conn, mod, frame, snac, bs, snac2);
00403         else if (snac2->type == 0x0003) /* request exchange info */
00404                 purple_debug_misc("oscar", "chatnav_parse_info: response to exchange info\n");
00405         else if (snac2->type == 0x0004) /* request room info */
00406                 purple_debug_misc("oscar", "chatnav_parse_info: response to room info\n");
00407         else if (snac2->type == 0x0005) /* request more room info */
00408                 purple_debug_misc("oscar", "chatnav_parse_info: response to more room info\n");
00409         else if (snac2->type == 0x0006) /* request occupant list */
00410                 purple_debug_misc("oscar", "chatnav_parse_info: response to occupant info\n");
00411         else if (snac2->type == 0x0007) /* search for a room */
00412                 purple_debug_misc("oscar", "chatnav_parse_info: search results\n");
00413         else if (snac2->type == 0x0008) /* create room */
00414                 ret = parseinfo_create(od, conn, mod, frame, snac, bs, snac2);
00415         else
00416                 purple_debug_misc("oscar", "chatnav_parse_info: unknown request subtype (%04x)\n", snac2->type);
00417 
00418         if (snac2)
00419                 g_free(snac2->data);
00420         g_free(snac2);
00421 
00422         return ret;
00423 }
