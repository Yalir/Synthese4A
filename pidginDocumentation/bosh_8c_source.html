00201         PurpleBOSHConnection *conn;
00202         char *host, *path, *user, *passwd;
00203         int port;
00204

00221

00224                                             "in BOSH URL.\n");
00225         }
00226 
00227         g_free(user);
00228         g_free(passwd);
00229 
00230         conn->js = js;
00231 
00232         /*
00233          * Random 64-bit integer masked off by 2^52 - 1.

00240 
00241         conn->pending = purple_circ_buffer_new(0 /* default grow size */);
00242 
00243         conn->state = BOSH_CONN_OFFLINE;
00244         if (purple_strcasestr(url, "https://") != NULL)
00245                 conn->ssl = TRUE;
00246         else
00247                 conn->ssl = FALSE;

00250

00280 static PurpleHTTPConnection *
00281 find_available_http_connection(PurpleBOSHConnection *conn)
00282 {
00283         int i;
00284 
00285         if (purple_debug_is_verbose())
00286                 debug_dump_http_connections(conn);
00287 
00288         /* Easy solution: Does everyone involved support pipelining? Hooray! Just use
00289          * one TCP connection! */
00290         if (conn->pipelining)
00291                 return conn->connections[0]->state == HTTP_CONN_CONNECTED ?
00292                                 conn->connections[0] : NULL;
00293 
00294         /* First loop, look for a connection that's ready */
00295         for (i = 0; i < NUM_HTTP_CONNECTIONS; ++i) {
00296                 if (conn->connections[i] &&
00297                                 conn->connections[i]->state == HTTP_CONN_CONNECTED &&
00298                                 conn->connections[i]->requests == 0)
00299                         return conn->connections[i];
00300         }
00301 
00302         /* Second loop, is something currently connecting? If so, just queue up. */
00303         for (i = 0; i < NUM_HTTP_CONNECTIONS; ++i) {
00304                 if (conn->connections[i] &&
00305                                 conn->connections[i]->state == HTTP_CONN_CONNECTING)
00306                         return NULL;
00307         }
00308 
00309         /* Third loop, is something offline that we can connect? */
00310         for (i = 0; i < NUM_HTTP_CONNECTIONS; ++i) {
00311                 if (conn->connections[i] &&
00312                                 conn->connections[i]->state == HTTP_CONN_OFFLINE) {
00313                         purple_debug_info("jabber", "bosh: Reconnecting httpconn "
00314                                                     "(%i, %p)\n", i, conn->connections[i]);
00315                         http_connection_connect(conn->connections[i]);
00316                         return NULL;
00317                 }
00318         }
00319 
00320         /* Fourth loop, look for one that's NULL and create a new connection */
00321         for (i = 0; i < NUM_HTTP_CONNECTIONS; ++i) {
00322                 if (!conn->connections[i]) {
00323                         conn->connections[i] = jabber_bosh_http_connection_init(conn);
00324                         purple_debug_info("jabber", "bosh: Creating and connecting new httpconn "
00325                                                     "(%i, %p)\n", i, conn->connections[i]);
00326 
00327                         http_connection_connect(conn->connections[i]);
00328                         return NULL;
00329                 }
00330         }
00331 
00332         purple_debug_warning("jabber", "Could not find a HTTP connection!\n");
00333 
00334         /* None available. */
00335         return NULL;
00336 }

00338 static void
00339 jabber_bosh_connection_send(PurpleBOSHConnection *conn,
00340                             const PurpleBOSHPacketType type, const char *data)
00341 {
00342         PurpleHTTPConnection *chosen;
00343         GString *packet = NULL;
00344 
00345         if (type != PACKET_FLUSH && type != PACKET_TERMINATE) {
00346                 /*
00347                  * Unless this is a flush (or session terminate, which needs to be
00348                  * sent immediately), queue up the data and start a timer to flush
00349                  * the buffer.
00350                  */
00351                 if (data) {
00352                         int len = data ? strlen(data) : 0;
00353                         purple_circ_buffer_append(conn->pending, data, len);
00354                 }
00355 
00356                 if (purple_debug_is_verbose())
00357                         purple_debug_misc("jabber", "bosh: %p has %" G_GSIZE_FORMAT " bytes in "
00358                                           "the buffer.\n", conn, conn->pending->bufused);
00359                 if (conn->send_timer == 0)
00360                         conn->send_timer = purple_timeout_add_seconds(BUFFER_SEND_IN_SECS,
00361                                         send_timer_cb, conn);
00362                 return;
00363         }
00364 
00365         chosen = find_available_http_connection(conn);
00366 
00367         if (!chosen) {
00368                 if (type == PACKET_FLUSH)
00369                         return;
00370                 /*
00371                  * For non-ordinary traffic, we can't 'buffer' it, so use the
00372                  * first connection.
00373                  */
00374                 chosen = conn->connections[0];
00375 
00376                 if (chosen->state != HTTP_CONN_CONNECTED) {
00377                         purple_debug_warning("jabber", "Unable to find a ready BOSH "
00378                                         "connection. Ignoring send of type 0x%02x.\n", type);
00379                         return;
00380                 }
00381         }
00382 
00383         /* We're flushing the send buffer, so remove the send timer */
00384         if (conn->send_timer != 0) {
00385                 purple_timeout_remove(conn->send_timer);
00386                 conn->send_timer = 0;
00387         }
00388 
00389         packet = g_string_new(NULL);
00390 
00391         g_string_printf(packet, "<body "
00392                         "rid='%" G_GUINT64_FORMAT "' "
00393                         "sid='%s' "
00394                         "to='%s' "
00395                         "xml:lang='en' "
00396                         "xmlns='" NS_BOSH "' "
00397                         "xmlns:xmpp='" NS_XMPP_BOSH "'",
00398                         ++conn->rid,
00399                         conn->sid,
00400                         conn->js->user->domain);
00401 
00402         if (conn->js->reinit) {
00403                 packet = g_string_append(packet, " xmpp:restart='true'/>");
00404                 /* TODO: Do we need to wait for a response? */
00405                 conn->js->reinit = FALSE;
00406         } else {
00407                 gsize read_amt;
00408                 if (type == PACKET_TERMINATE)
00409                         packet = g_string_append(packet, " type='terminate'");
00410 
00411                 packet = g_string_append_c(packet, '>');
00412 
00413                 while ((read_amt = purple_circ_buffer_get_max_read(conn->pending)) > 0) {
00414                         packet = g_string_append_len(packet, conn->pending->outptr, read_amt);
00415                         purple_circ_buffer_mark_read(conn->pending, read_amt);
00416                 }
00417 
00418                 if (data)
00419                         packet = g_string_append(packet, data);
00420                 packet = g_string_append(packet, "</body>");
00421         }
00422 
00423         http_connection_send_request(chosen, packet);
00424 }

00491 static void jabber_bosh_connection_received(PurpleBOSHConnection *conn, xmlnode *node) {
00492         xmlnode *child;
00493         JabberStream *js = conn->js;
00494 
00495         g_return_if_fail(node != NULL);
00496         if (jabber_bosh_connection_error_check(conn, node))
00497                 return;
00498 
00499         child = node->child;
00500         while (child != NULL) {
00501                 /* jabber_process_packet might free child */
00502                 xmlnode *next = child->next;
00503                 if (child->type == XMLNODE_TYPE_TAG) {
00504                         const char *xmlns = xmlnode_get_namespace(child);
00505                         /*
00506                          * Workaround for non-compliant servers that don't stamp
00507                          * the right xmlns on these packets.  See #11315.
00508                          */
00509                         if ((xmlns == NULL /* shouldn't happen, but is equally wrong */ ||
00510                                         g_str_equal(xmlns, NS_BOSH)) &&
00511                                 (g_str_equal(child->name, "iq") ||
00512                                  g_str_equal(child->name, "message") ||
00513                                  g_str_equal(child->name, "presence"))) {
00514                                 xmlnode_set_namespace(child, NS_XMPP_CLIENT);
00515                         }
00516                         jabber_process_packet(js, &child);
00517                 }
00518

00525         const char *sid, *version;
00526         const char *inactivity, *requests;
00527         xmlnode *packet;
00528 
00529         g_return_if_fail(node != NULL);
00530         if (jabber_bosh_connection_error_check(conn, node))
00531                 return;
00532 
00533         sid = xmlnode_get_attrib(node, "sid");
00534         version = xmlnode_get_attrib(node, "ver");
00535 
00536         inactivity = xmlnode_get_attrib(node, "inactivity");
00537         requests = xmlnode_get_attrib(node, "requests");
00538 
00539         if (sid) {
00540                 conn->sid = g_strdup(sid);
00541         } else {
00542                 purple_connection_error_reason(js->gc,
00543                         PURPLE_CONNECTION_ERROR_NETWORK_ERROR,
00544                         _("No session ID given"));
00545                 return;
00546         }
00547 
00548         if (version) {
00549                 const char *dot = strchr(version, '.');
00550                 int major, minor = 0;
00551 
00552                 purple_debug_info("jabber", "BOSH connection manager version %s\n", version);
00553 
00554                 major = atoi(version);
00555                 if (dot)
00556                         minor = atoi(dot + 1);
00557 
00558                 if (major != 1 || minor < 6) {
00559                         purple_connection_error_reason(js->gc,
00560                                 PURPLE_CONNECTION_ERROR_NETWORK_ERROR,
00561                                 _("Unsupported version of BOSH protocol"));
00562                         return;
00563                 }
00564         } else {
00565                 purple_debug_info("jabber", "Missing version in BOSH initiation\n");
00566         }
00567 
00568         if (inactivity) {
00569                 js->max_inactivity = atoi(inactivity);
00570                 if (js->max_inactivity <= 5) {
00571                         purple_debug_warning("jabber", "Ignoring bogusly small inactivity: %s\n",
00572                                              inactivity);
00573                         /* Leave it at the default */
00574                 } else {
00575                         /* TODO: Can this check fail? It shouldn't */
00576                         js->max_inactivity -= 5; /* rounding */
00577 
00578                         if (js->inactivity_timer == 0) {
00579                                 purple_debug_misc("jabber", "Starting BOSH inactivity timer "
00580                                                 "for %d secs (compensating for rounding)\n",
00581                                                 js->max_inactivity);
00582                                 jabber_stream_restart_inactivity_timer(js);
00583                         }
00584                 }
00585         }
00586 
00587         if (requests)
00588                 conn->max_requests = atoi(requests);
00589 
00590         jabber_stream_set_state(js, JABBER_STREAM_AUTHENTICATING);
00591 
00592         /* FIXME: Depending on receiving features might break with some hosts */
00593         packet = xmlnode_get_child(node, "features");
00594         conn->state = BOSH_CONN_ONLINE;
00595         conn->receive_cb = jabber_bosh_connection_received;

00754 static void
00755 jabber_bosh_http_connection_process(PurpleHTTPConnection *conn)
00756 {
00757         const char *cursor;
00758 
00759         cursor = conn->read_buf->str + conn->handled_len;
00760 
00761         if (purple_debug_is_verbose())
00762                 purple_debug_misc("jabber", "BOSH server sent: %s\n", cursor);
00763 
00764         /* TODO: Chunked encoding and check response version :/ */
00765         if (!conn->headers_done) {
00766                 const char *content_length = purple_strcasestr(cursor, "\r\nContent-Length:");
00767                 const char *connection = purple_strcasestr(cursor, "\r\nConnection:");
00768                 const char *end_of_headers = strstr(cursor, "\r\n\r\n");
00769 
00770                 /* Make sure Content-Length is in headers, not body */
00771                 if (content_length && (!end_of_headers || content_length < end_of_headers)) {
00772                         int len;
00773 
00774                         if (strstr(content_length, "\r\n") == NULL)
00775                                 /*
00776                                  * The packet ends in the middle of the Content-Length line.
00777                                  * We'll try again later when we have more.
00778                                  */
00779                                 return;
00780 
00781                         len = atoi(content_length + strlen("\r\nContent-Length:"));
00782                         if (len == 0)
00783                                 purple_debug_warning("jabber", "Found mangled Content-Length header, or server returned 0-length response.\n");
00784 
00785                         conn->body_len = len;
00786                 }
00787 
00788                 if (connection && (!end_of_headers || content_length < end_of_headers)) {
00789                         const char *tmp;
00790                         if (strstr(connection, "\r\n") == NULL)
00791                                 return;
00792 
00793 
00794                         tmp = connection + strlen("\r\nConnection:");
00795                         while (*tmp && (*tmp == ' ' || *tmp == '\t'))
00796                                 ++tmp;
00797 
00798                         if (!g_ascii_strncasecmp(tmp, "close", strlen("close"))) {
00799                                 conn->close = TRUE;
00800                                 jabber_bosh_disable_pipelining(conn->bosh);
00801                         }
00802                 }
00803 
00804                 if (end_of_headers) {
00805                         conn->headers_done = TRUE;
00806                         conn->handled_len = end_of_headers - conn->read_buf->str + 4;
00807                 } else {
00808                         conn->handled_len = conn->read_buf->len;
00809                         return;
00810                 }
00811         }
00812 
00813         /* Have we handled everything in the buffer? */
00814         if (conn->handled_len >= conn->read_buf->len)
00815                 return;
00816 
00817         /* Have we read all that the Content-Length promised us? */
00818         if (conn->read_buf->len - conn->handled_len < conn->body_len)
00819                 return;
00820 
00821         --conn->requests;
00822         --conn->bosh->requests;
00823 
00824         http_received_cb(conn->read_buf->str + conn->handled_len, conn->body_len,
00825                          conn->bosh);
00826 
00827         /* Connection: Close? */
00828         if (conn->close && conn->state == HTTP_CONN_CONNECTED) {
00829                 if (purple_debug_is_verbose())
00830                         purple_debug_misc("jabber", "bosh (%p), server sent Connection: "
00831                                                     "close\n", conn);
00832                 http_connection_disconnected(conn);
00833         }
00834 
00835         if (conn->bosh->state == BOSH_CONN_ONLINE &&
00836                         (conn->bosh->requests == 0 || conn->bosh->pending->bufused > 0)) {
00837                 purple_debug_misc("jabber", "BOSH: Sending an empty request\n");
00838                 jabber_bosh_connection_send(conn->bosh, PACKET_NORMAL, NULL);
00839         }
00840 
00841         g_string_free(conn->read_buf, TRUE);
00842         conn->read_buf = NULL;
00843         conn->headers_done = FALSE;
00844         conn->handled_len = conn->body_len = 0;
00845 }
