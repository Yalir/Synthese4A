00079 static guint jabber_caps_hash(gconstpointer data) {
00080         const JabberCapsTuple *key = data;
00081         guint nodehash = g_str_hash(key->node);
00082         guint verhash  = g_str_hash(key->ver);
00083         /*
00084          * 'hash' was optional in XEP-0115 v1.4 and g_str_hash crashes on NULL >:O.
00085          * Okay, maybe I've played too much Zelda, but that looks like
00086          * a Deku Shrub...
00087          */
00088         guint hashhash = (key->hash ? g_str_hash(key->hash) : 0);
00089         return nodehash ^ verhash ^ hashhash;
00090 }

00228 jabber_caps_load(void)
00229 {
00230         xmlnode *capsdata = purple_util_read_xml_from_file(JABBER_CAPS_FILENAME, "XMPP capabilities cache");
00231         xmlnode *client;
00232 
00233         if(!capsdata)
00234                 return;
00235 
00236         if (!g_str_equal(capsdata->name, "capabilities")) {
00237                 xmlnode_free(capsdata);
00238                 return;
00239         }
00240 
00241         for (client = capsdata->child; client; client = client->next) {
00242                 if (client->type != XMLNODE_TYPE_TAG)
00243                         continue;
00244                 if (g_str_equal(client->name, "client")) {
00245                         JabberCapsClientInfo *value = g_new0(JabberCapsClientInfo, 1);
00246                         JabberCapsTuple *key = (JabberCapsTuple*)&value->tuple;
00247                         xmlnode *child;
00248                         JabberCapsNodeExts *exts = NULL;
00249                         key->node = g_strdup(xmlnode_get_attrib(client,"node"));
00250                         key->ver  = g_strdup(xmlnode_get_attrib(client,"ver"));
00251                         key->hash = g_strdup(xmlnode_get_attrib(client,"hash"));
00252 
00253                         /* v1.3 capabilities */
00254                         if (key->hash == NULL)
00255                                 exts = jabber_caps_find_exts_by_node(key->node);
00256 
00257                         for (child = client->child; child; child = child->next) {
00258                                 if (child->type != XMLNODE_TYPE_TAG)
00259                                         continue;
00260                                 if (g_str_equal(child->name, "feature")) {
00261                                         const char *var = xmlnode_get_attrib(child, "var");
00262                                         if(!var)
00263                                                 continue;
00264                                         value->features = g_list_append(value->features,g_strdup(var));
00265                                 } else if (g_str_equal(child->name, "identity")) {
00266                                         const char *category = xmlnode_get_attrib(child, "category");
00267                                         const char *type = xmlnode_get_attrib(child, "type");
00268                                         const char *name = xmlnode_get_attrib(child, "name");
00269                                         const char *lang = xmlnode_get_attrib(child, "lang");
00270                                         JabberIdentity *id;
00271 
00272                                         if (!category || !type)
00273                                                 continue;
00274 
00275                                         id = g_new0(JabberIdentity, 1);
00276                                         id->category = g_strdup(category);
00277                                         id->type = g_strdup(type);
00278                                         id->name = g_strdup(name);
00279                                         id->lang = g_strdup(lang);
00280 
00281                                         value->identities = g_list_append(value->identities,id);
00282                                 } else if (g_str_equal(child->name, "x")) {
00283                                         /* TODO: See #7814 -- this might cause problems if anyone
00284                                          * ever actually specifies forms. In fact, for this to
00285                                          * work properly, that bug needs to be fixed in
00286                                          * xmlnode_from_str, not the output version... */
00287                                         value->forms = g_list_append(value->forms, xmlnode_copy(child));
00288                                 } else if (g_str_equal(child->name, "ext")) {
00289                                         if (key->hash != NULL)
00290                                                 purple_debug_warning("jabber", "Ignoring exts when reading new-style caps\n");
00291                                         else {
00292                                                 /* TODO: Do we care about reading in the identities listed here? */
00293                                                 const char *identifier = xmlnode_get_attrib(child, "identifier");
00294                                                 xmlnode *node;
00295                                                 GList *features = NULL;
00296 
00297                                                 if (!identifier)
00298                                                         continue;
00299 
00300                                                 for (node = child->child; node; node = node->next) {
00301                                                         if (node->type != XMLNODE_TYPE_TAG)
00302                                                                 continue;
00303                                                         if (g_str_equal(node->name, "feature")) {
00304                                                                 const char *var = xmlnode_get_attrib(node, "var");
00305                                                                 if (!var)
00306                                                                         continue;
00307                                                                 features = g_list_prepend(features, g_strdup(var));
00308                                                         }
00309                                                 }
00310 
00311                                                 if (features) {
00312                                                         g_hash_table_insert(exts->exts, g_strdup(identifier),
00313                                                                             features);
00314                                                 } else
00315                                                         purple_debug_warning("jabber", "Caps ext %s had no features.\n",
00316                                                                              identifier);
00317                                         }
00318                                 }
00319                         }
00320 
00321                         value->exts = exts;
00322                         g_hash_table_replace(capstable, key, value);
00323 
00324                 }
00325         }

00349                                 char **exts)
00350 {
00351         int i;
00352         g_return_val_if_fail(info != NULL, FALSE);
00353 
00354         if (!exts)
00355                 return TRUE;

00357         for (i = 0; exts[i]; ++i) {
00358                 /* Hack since we advertise the ext along with v1.5 caps but don't

00363                                 !g_hash_table_lookup(info->exts->exts, exts[i]))
00364                         return FALSE;
00365         }
00366 
00367         return TRUE;

00435 static void
00436 jabber_caps_client_iqcb(JabberStream *js, const char *from, JabberIqType type,
00437                         const char *id, xmlnode *packet, gpointer data)
00438 {
00439         xmlnode *query = xmlnode_get_child_with_namespace(packet, "query",
00440                 NS_DISCO_INFO);
00441         jabber_caps_cbplususerdata *userdata = data;
00442         JabberCapsClientInfo *info = NULL, *value;
00443         JabberCapsTuple key;
00444 
00445         if (!query || type == JABBER_IQ_ERROR) {
00446                 /* Any outstanding exts will be dealt with via ref-counting */
00447                 userdata->cb(NULL, NULL, userdata->cb_data);
00448                 cbplususerdata_unref(userdata);
00449                 return;
00450         }
00451 
00452         /* check hash */
00453         info = jabber_caps_parse_client_info(query);
00454 
00455         /* Only validate if these are v1.5 capabilities */
00456         if (userdata->hash) {
00457                 gchar *hash = NULL;
00458                 /*
00459                  * TODO: If you add *any* hash here, make sure the checksum buffer
00460                  * size in jabber_caps_calculate_hash is large enough. The cipher API
00461                  * doesn't seem to offer a "Get the hash size" function(?).
00462                  */
00463                 if (g_str_equal(userdata->hash, "sha-1")) {
00464                         hash = jabber_caps_calculate_hash(info, "sha1");
00465                 } else if (g_str_equal(userdata->hash, "md5")) {
00466                         hash = jabber_caps_calculate_hash(info, "md5");
00467                 }
00468 
00469                 if (!hash || !g_str_equal(hash, userdata->ver)) {
00470                         purple_debug_warning("jabber", "Could not validate caps info from "
00471                                              "%s. Expected %s, got %s\n",
00472                                              xmlnode_get_attrib(packet, "from"),
00473                                              userdata->ver, hash ? hash : "(null)");
00474 
00475                         userdata->cb(NULL, NULL, userdata->cb_data);
00476                         jabber_caps_client_info_destroy(info);
00477                         cbplususerdata_unref(userdata);
00478                         g_free(hash);
00479                         return;
00480                 }
00481 
00482                 g_free(hash);
00483         }
00484 
00485         if (!userdata->hash && userdata->node_exts) {
00486                 /* If the ClientInfo doesn't have information about the exts, give them
00487                  * ours (along with our ref) */
00488                 info->exts = userdata->node_exts;
00489                 userdata->node_exts = NULL;
00490         }
00491 
00492         key.node = userdata->node;
00493         key.ver  = userdata->ver;
00494         key.hash = userdata->hash;
00495 
00496         /* Use the copy of this data already in the table if it exists or insert
00497          * a new one if we need to */
00498         if ((value = g_hash_table_lookup(capstable, &key))) {
00499                 jabber_caps_client_info_destroy(info);
00500                 info = value;
00501         } else {
00502                 JabberCapsTuple *n_key = (JabberCapsTuple *)&info->tuple;
00503                 n_key->node = userdata->node;
00504                 n_key->ver  = userdata->ver;
00505                 n_key->hash = userdata->hash;
00506                 userdata->node = userdata->ver = userdata->hash = NULL;
00507 
00508                 /* The capstable gets a reference */
00509                 g_hash_table_insert(capstable, n_key, info);
00510                 schedule_caps_save();
00511         }
00512 
00513         userdata->info = info;
00514 
00515         if (userdata->extOutstanding == 0)
00516                 jabber_caps_get_info_complete(userdata);
00517 
00518         cbplususerdata_unref(userdata);
00519 }

00555                 if (node_exts) {

00820 gchar *jabber_caps_calculate_hash(JabberCapsClientInfo *info, const char *hash)
00821 {
00822         GList *node;
00823         PurpleCipherContext *context;
00824         guint8 checksum[20];
00825         gsize checksum_size = 20;
00826         gboolean success;
00827 
00828         if (!info || !(context = purple_cipher_context_new_by_name(hash, NULL)))
00829                 return NULL;
00830 
00831         /* sort identities, features and x-data forms */
00832         info->identities = g_list_sort(info->identities, jabber_identity_compare);
00833         info->features = g_list_sort(info->features, (GCompareFunc)strcmp);
00834         info->forms = g_list_sort(info->forms, jabber_xdata_compare);
00835 
00836         /* Add identities to the hash data */
00837         for (node = info->identities; node; node = node->next) {
00838                 JabberIdentity *id = (JabberIdentity*)node->data;
00839                 char *category = g_markup_escape_text(id->category, -1);
00840                 char *type = g_markup_escape_text(id->type, -1);
00841                 char *lang = NULL;
00842                 char *name = NULL;
00843                 char *tmp;
00844 
00845                 if (id->lang)
00846                         lang = g_markup_escape_text(id->lang, -1);
00847                 if (id->name)
00848                         name = g_markup_escape_text(id->name, -1);
00849 
00850                 tmp = g_strconcat(category, "/", type, "/", lang ? lang : "",
00851                                   "/", name ? name : "", "<", NULL);
00852 
00853                 purple_cipher_context_append(context, (const guchar *)tmp, strlen(tmp));
00854 
00855                 g_free(tmp);
00856                 g_free(category);
00857                 g_free(type);
00858                 g_free(lang);
00859                 g_free(name);
00860         }
00861 
00862         /* concat features to the verification string */
00863         for (node = info->features; node; node = node->next) {
00864                 append_escaped_string(context, node->data);
00865         }
00866 
00867         /* concat x-data forms to the verification string */
00868         for(node = info->forms; node; node = node->next) {
00869                 xmlnode *data = (xmlnode *)node->data;
00870                 gchar *formtype = jabber_x_data_get_formtype(data);
00871                 GList *fields = jabber_caps_xdata_get_fields(data);
00872 
00873                 /* append FORM_TYPE's field value to the verification string */
00874                 append_escaped_string(context, formtype);
00875                 g_free(formtype);
00876 
00877                 while (fields) {
00878                         GList *value;
00879                         JabberDataFormField *field = (JabberDataFormField*)fields->data;
00880 
00881                         if (!g_str_equal(field->var, "FORM_TYPE")) {
00882                                 /* Append the "var" attribute */
00883                                 append_escaped_string(context, field->var);
00884                                 /* Append <value/> elements' cdata */
00885                                 for (value = field->values; value; value = value->next) {
00886                                         append_escaped_string(context, value->data);
00887                                         g_free(value->data);
00888                                 }
00889                         }
00890 
00891                         g_free(field->var);
00892                         g_list_free(field->values);
00893 
00894                         fields = g_list_delete_link(fields, fields);
00895                 }
00896         }
00897 
00898         /* generate hash */
00899         success = purple_cipher_context_digest(context, checksum_size,
00900                                                checksum, &checksum_size);
00901 
00902         purple_cipher_context_destroy(context);
00903 
00904         return (success ? purple_base64_encode(checksum, checksum_size) : NULL);
00905 }
