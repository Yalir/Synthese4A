02305 static char *amsn_logger_read(PurpleLog *log, PurpleLogReadFlags *flags)
02306 {
02307         struct amsn_logger_data *data;
02308         FILE *file;
02309         char *contents;
02310         char *escaped;
02311         GString *formatted;
02312         char *start;

02314 
02315         if (flags != NULL)
02316                 *flags = PURPLE_LOG_READ_NO_NEWLINE;
02317 
02318         g_return_val_if_fail(log != NULL, g_strdup(""));
02319 
02320         data = log->logger_data;
02321 
02322         g_return_val_if_fail(data->path != NULL, g_strdup(""));
02323         g_return_val_if_fail(data->length > 0, g_strdup(""));
02324 
02325         contents = g_malloc(data->length + 2);
02326 
02327         file = g_fopen(data->path, "rb");
02328         g_return_val_if_fail(file != NULL, g_strdup(""));
02329 
02330         fseek(file, data->offset, SEEK_SET);
02331         data->length = fread(contents, 1, data->length, file);
02332         fclose(file);
02333 
02334         contents[data->length] = '\n';
02335         contents[data->length + 1] = '\0';
02336 
02337         escaped = g_markup_escape_text(contents, -1);
02338         g_free(contents);
02339         contents = escaped;
02340 
02341         formatted = g_string_sized_new(data->length + 2);
02342 
02343         start = contents;
02344         while (start && *start) {
02345                 char *end;
02346                 char *old_tag;
02347                 char *tag;
02348                 end = strchr(start, '\n');
02349                 if (!end)
02350                         break;
02351                 *end = '\0';
02352                 if (purple_str_has_prefix(start, AMSN_LOG_FORMAT_TAG) && in_span) {
02353                         /* New format for this line */
02354                         g_string_append(formatted, "</span><br>");
02355                         in_span = FALSE;
02356                 } else if (start != contents) {
02357                         /* Continue format from previous line */
02358                         g_string_append(formatted, "<br>");
02359                 }
02360                 old_tag = start;
02361                 tag = strstr(start, AMSN_LOG_FORMAT_TAG);
02362                 while (tag) {
02363                         g_string_append_len(formatted, old_tag, tag - old_tag);
02364                         tag += strlen(AMSN_LOG_FORMAT_TAG);
02365                         if (in_span) {
02366                                 g_string_append(formatted, "</span>");
02367                                 in_span = FALSE;
02368                         }
02369                         if (*tag == 'C') {
02370                                 /* |"LCxxxxxx is a hex colour */
02371                                 char colour[7];
02372                                 strncpy(colour, tag + 1, 6);
02373                                 colour[6] = '\0';
02374                                 g_string_append_printf(formatted, "<span style=\"color: #%s;\">", colour);
02375                                 /* This doesn't appear to work? */
02376                                 /* g_string_append_printf(formatted, "<span style=\"color: #%6s;\">", tag + 1); */
02377                                 in_span = TRUE;
02378                                 old_tag = tag + 7; /* C + xxxxxx */
02379                         } else {
02380                                 /* |"Lxxx is a 3-digit colour code */
02381                                 if (purple_str_has_prefix(tag, "RED")) {
02382                                         g_string_append(formatted, "<span style=\"color: red;\">");
02383                                         in_span = TRUE;
02384                                 } else if (purple_str_has_prefix(tag, "GRA")) {
02385                                         g_string_append(formatted, "<span style=\"color: gray;\">");
02386                                         in_span = TRUE;
02387                                 } else if (purple_str_has_prefix(tag, "NOR")) {
02388                                         g_string_append(formatted, "<span style=\"color: black;\">");
02389                                         in_span = TRUE;
02390                                 } else if (purple_str_has_prefix(tag, "ITA")) {
02391                                         g_string_append(formatted, "<span style=\"color: blue;\">");
02392                                         in_span = TRUE;
02393                                 } else if (purple_str_has_prefix(tag, "GRE")) {
02394                                         g_string_append(formatted, "<span style=\"color: darkgreen;\">");
02395                                         in_span = TRUE;
02396                                 } else {
02397                                         purple_debug_info("aMSN logger", "Unknown colour format: %3s\n", tag);
02398                                 }
02399                                 old_tag = tag + 3;
02400                         }
02401                         tag = strstr(tag, AMSN_LOG_FORMAT_TAG);
02402                 }
02403                 g_string_append(formatted, old_tag);
02404                 start = end + 1;
02405         }
02406         if (in_span)
02407                 g_string_append(formatted, "</span>");
02408
