00182 static char *jabber_prep_resource(char *input) {
00183         char hostname[256], /* current hostname */
00184                  *dot = NULL;
00185 
00186         /* Empty resource == don't send any */
00187         if (input == NULL || *input == '\0')
00188                 return NULL;
00189 
00190         if (strstr(input, "__HOSTNAME__") == NULL)
00191                 return g_strdup(input);
00192 
00193         /* Replace __HOSTNAME__ with hostname */
00194         if (gethostname(hostname, sizeof(hostname) - 1)) {
00195                 purple_debug_warning("jabber", "gethostname: %s\n", g_strerror(errno));
00196                 /* according to glibc doc, the only time an error is returned
00197                    is if the hostname is longer than the buffer, in which case
00198                    glibc 2.2+ would still fill the buffer with partial
00199                    hostname, so maybe we want to detect that and use it
00200                    instead
00201                 */
00202                 g_strlcpy(hostname, "localhost", sizeof(hostname));
00203         }
00204         hostname[sizeof(hostname) - 1] = '\0';
00205 
00206         /* We want only the short hostname, not the FQDN - this will prevent the
00207          * resource string from being unreasonably long on systems which stuff the
00208          * whole FQDN in the hostname */
00209         if((dot = strchr(hostname, '.')))
00210                 *dot = '\0';
00211 
00212         return purple_strreplace(input, "__HOSTNAME__", hostname);
00213 }

00334

00349                 jabber_message_parse(js, *packet);
00350         } else if (purple_strequal(xmlns, NS_XMPP_STREAMS)) {
00351                 if (g_str_equal(name, "features"))
00352                         jabber_stream_features_parse(js, *packet);
00353                 else if (g_str_equal(name, "error"))
00354                         jabber_stream_handle_error(js, *packet);

00356                 if (js->state != JABBER_STREAM_AUTHENTICATING)
00357                         purple_debug_warning("jabber", "Ignoring spurious SASL stanza %s\n", name);
00358                 else {
00359                         if (g_str_equal(name, "challenge"))
00360                                 jabber_auth_handle_challenge(js, *packet);
00361                         else if (g_str_equal(name, "success"))
00362                                 jabber_auth_handle_success(js, *packet);
00363                         else if (g_str_equal(name, "failure"))
00364                                 jabber_auth_handle_failure(js, *packet);
00365                 }
00366         } else if (purple_strequal(xmlns, NS_XMPP_TLS)) {
00367                 if (js->state != JABBER_STREAM_INITIALIZING_ENCRYPTION || js->gsc)
00368                         purple_debug_warning("jabber", "Ignoring spurious %s\n", name);

00370                         if (g_str_equal(name, "proceed"))
00371                                 tls_init(js);

03432 PurpleMediaCaps jabber_get_media_caps(PurpleAccount *account, const char *who)
03433 {
03434 #ifdef USE_VV
03435         JabberStream *js = (JabberStream *)
03436                         purple_account_get_connection(account)->proto_data;
03437         JabberBuddy *jb;
03438         JabberBuddyResource *jbr;
03439         PurpleMediaCaps total = PURPLE_MEDIA_CAPS_NONE;
03440         gchar *resource;
03441         GList *specific = NULL, *l;
03442 
03443         if (!js) {
03444                 purple_debug_info("jabber",
03445                                 "jabber_can_do_media: NULL stream\n");
03446                 return FALSE;
03447         }
03448 
03449         jb = jabber_buddy_find(js, who, FALSE);
03450 
03451         if (!jb || !jb->resources) {
03452                 /* no resources online, we're trying to get caps for someone
03453                  * whose presence we're not subscribed to, or
03454                  * someone who is offline. */
03455                 return total;
03456 
03457         } else if ((resource = jabber_get_resource(who)) != NULL) {
03458                 /* they've specified a resource, no need to ask or
03459                  * default or anything, just do it */
03460                 jbr = jabber_buddy_find_resource(jb, resource);
03461                 g_free(resource);
03462 
03463                 if (!jbr) {
03464                         purple_debug_error("jabber", "jabber_get_media_caps:"
03465                                         " Can't find resource %s\n", who);
03466                         return total;
03467                 }
03468 
03469                 l = specific = g_list_prepend(specific, jbr);
03470 
03471         } else {
03472                 /* we've got multiple resources, combine their caps */
03473                 l = jb->resources;
03474         }
03475 
03476         for (; l; l = l->next) {
03477                 PurpleMediaCaps caps = PURPLE_MEDIA_CAPS_NONE;
03478                 jbr = l->data;
03479 
03480                 if (jabber_resource_has_capability(jbr,
03481                                 JINGLE_APP_RTP_SUPPORT_AUDIO))
03482                         caps |= PURPLE_MEDIA_CAPS_AUDIO_SINGLE_DIRECTION |
03483                                         PURPLE_MEDIA_CAPS_AUDIO;
03484                 if (jabber_resource_has_capability(jbr,
03485                                 JINGLE_APP_RTP_SUPPORT_VIDEO))
03486                         caps |= PURPLE_MEDIA_CAPS_VIDEO_SINGLE_DIRECTION |
03487                                         PURPLE_MEDIA_CAPS_VIDEO;
03488                 if (caps & PURPLE_MEDIA_CAPS_AUDIO && caps &
03489                                 PURPLE_MEDIA_CAPS_VIDEO)
03490                         caps |= PURPLE_MEDIA_CAPS_AUDIO_VIDEO;
03491                 if (caps != PURPLE_MEDIA_CAPS_NONE) {
03492                         if (!jabber_resource_has_capability(jbr,
03493                                         JINGLE_TRANSPORT_ICEUDP) &&
03494                                         !jabber_resource_has_capability(jbr,
03495                                         JINGLE_TRANSPORT_RAWUDP)) {
03496                                 purple_debug_info("jingle-rtp", "Buddy doesn't "
03497                                                 "support the same transport types\n");
03498                                 caps = PURPLE_MEDIA_CAPS_NONE;
03499                         } else
03500                                 caps |= PURPLE_MEDIA_CAPS_MODIFY_SESSION |
03501                                                 PURPLE_MEDIA_CAPS_CHANGE_DIRECTION;
03502                 }
03503                 if (jabber_resource_has_capability(jbr, NS_GOOGLE_VOICE)) {
03504                         caps |= PURPLE_MEDIA_CAPS_AUDIO;
03505                         if (jabber_resource_has_capability(jbr, NS_GOOGLE_VIDEO))
03506                                 caps |= PURPLE_MEDIA_CAPS_AUDIO_VIDEO;
03507                 }
03508 
03509                 total |= caps;
03510         }
03511 
03512         if (specific) {
03513                 g_list_free(specific);
03514         }
03515 
03516         return total;
03517 #else
03518         return PURPLE_MEDIA_CAPS_NONE;
03519 #endif
03520 }

03522 gboolean jabber_can_receive_file(PurpleConnection *gc, const char *who)
03523 {
03524         JabberStream *js = gc->proto_data;
03525 
03526         if (js) {
03527                 JabberBuddy *jb = jabber_buddy_find(js, who, FALSE);
03528                 GList *iter;
03529                 gboolean has_resources_without_caps = FALSE;
03530 
03531                 /* if we didn't find a JabberBuddy, we don't have presence for this
03532                  buddy, let's assume they can receive files, disco should tell us
03533                  when actually trying */
03534                 if (jb == NULL)
03535                         return TRUE;
03536 
03537                 /* find out if there is any resources without caps */
03538                 for (iter = jb->resources; iter ; iter = g_list_next(iter)) {
03539                         JabberBuddyResource *jbr = (JabberBuddyResource *) iter->data;
03540 
03541                         if (!jabber_resource_know_capabilities(jbr)) {
03542                                 has_resources_without_caps = TRUE;
03543                         }
03544                 }
03545 
03546                 if (has_resources_without_caps) {
03547                         /* there is at least one resource which we don't have caps for,
03548                          let's assume they can receive files... */
03549                         return TRUE;
03550                 } else {
03551                         /* we have caps for all the resources, see if at least one has
03552                          right caps */
03553                         for (iter = jb->resources; iter ; iter = g_list_next(iter)) {
03554                                 JabberBuddyResource *jbr = (JabberBuddyResource *) iter->data;
03555 
03556                                 if (jabber_resource_has_capability(jbr, NS_SI_FILE_TRANSFER)
03557                                 && (jabber_resource_has_capability(jbr,
03558                                                 NS_BYTESTREAMS)
03559                                         || jabber_resource_has_capability(jbr, NS_IBB))) {
03560                                         return TRUE;
03561                                 }
03562                         }
03563                         return FALSE;
03564                 }
03565         } else {
03566                 return TRUE;
03567         }
03568 }
