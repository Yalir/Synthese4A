00978 static parse_tree *parse_buffer(gchar* source, gboolean do_parse) {

00986                         unsigned int end;
00987                         gchar *newstr;

00991                                 p++;
00992                                 continue;
00993                         }
00994 
00995                         /* Skip comments */
00996                         if(source[p] == ';') {
00997                                 while(source[p] != '\n' && p < strlen(source)) {
00998                                         p++;
00999                                 }
01000                                 continue;
01001                         }

01005                                 gboolean in_quote = FALSE;

01012                                                         escape_next = TRUE;
01013                                                 }

01015                                                         if(source[end] == '(') {
01016                                                                 nesting++;
01017                                                         }
01018                                                         if(source[end] == ')') {
01019                                                                 nesting--;
01020                                                         }
01021                                                 }
01022                                                 if(source[end] == '"') {
01023                                                         in_quote = !in_quote;
01024                                                 }
01025                                         } else {
01026                                                 escape_next = FALSE;
01027                                         }

01031

01039                                 }

01041
