00236 
00237                         /* Remove from the pending list */
00238                         pending_buddies = g_slist_remove(pending_buddies, args->bb);
00239                 }
00240         } else {
00241                 /* Remove from the pending list */
00242                 pending_buddies = g_slist_remove(pending_buddies, args->bb);

00253 {

00255         Win32BuddyImplData *idata = args->bb->mdns_impl_data;
00256 
00257         /* remove the input fd and destroy the service ref */
00258         purple_input_remove(args->resolver_query->input_handler);
00259         DNSServiceRefDeallocate(args->resolver_query->sdRef);
00260

00269                         purple_debug_error("bonjour", "service resolver - host resolution failed.\n");
00270                 else {
00271                         args->resolver_query->sdRef = getaddrinfo_sr;
00272                         args->resolver_query->input_handler = purple_input_add(DNSServiceRefSockFD(getaddrinfo_sr),
00273                                 PURPLE_INPUT_READ, _mdns_handle_event, args->resolver_query);
00274                         args->full_service_name = g_strdup(fullname);
00275 
00276                         /* TODO: Should this be per resolver? */
00277                         args->bb->port_p2pj = ntohs(port);
00278 
00279                         /* We don't want to hit the cleanup code */
00280                         return;
00281                 }
00282         }
00283 
00284         /* If we get this far, clean up */
00285 
00286         g_free(args->resolver_query);

00290         _cleanup_resolver_data(args->res_data);

00295                 /* See if this is now attached to a PurpleBuddy */

00297                         bonjour_buddy_signed_off(pb);
00298                 else {
00299                         /* Remove from the pending list */
00300                         pending_buddies = g_slist_remove(pending_buddies, args->bb);
