00099                 searchaddr = (const char *)snac2->data;

00101         tlvlist = aim_tlvlist_read(bs);

00103 
00104         /* XXX uhm.

00107          */
00108         while ((cur = aim_tlv_getstr(tlvlist, 0x0001, j+1)) && j < m)
00109         {
00110                 buf = g_realloc(buf, (j+1) * (MAXSNLEN+1));
00111 
00112                 strncpy(&buf[j * (MAXSNLEN+1)], cur, MAXSNLEN);
00113                 g_free(cur);
00114 
00115                 j++;
00116         }
00117         g_free(cur);
00118 
00119         aim_tlvlist_free(tlvlist);
00120 
00121         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
