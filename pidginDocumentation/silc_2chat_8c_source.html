00513 /* Private groups are "virtual" channels.  They are groups inside a channel.
00514    This is implemented by using channel private keys.  By knowing a channel

00516    group.  Other users, on the same channel, won't be able to see the
00517    messages of that group.  It is possible to have multiple groups inside
00518    a channel - and thus having multiple private keys on the channel. */

00539         if (!name) {

00543         f = purple_request_fields_get_field(fields, "passphrase");

01026         const char *channel, *passphrase, *parentch;
01027 
01028         if (!conn)
01029                 return;
01030 
01031         channel = g_hash_table_lookup(data, "channel");
01032         passphrase = g_hash_table_lookup(data, "passphrase");
01033 
01034         /* Check if we are joining a private group.  Handle it
01035            purely locally as it's not a real channel */
01036         if (strstr(channel, "[Private Group]")) {
01037                 SilcChannelEntry channel_entry;
01038                 SilcChannelPrivateKey key;
01039                 PurpleChat *c;
01040                 SilcPurplePrvgrp grp;
01041 
01042                 c = purple_blist_find_chat(sg->account, channel);
01043                 parentch = purple_blist_node_get_string((PurpleBlistNode *)c, "parentch");
01044                 if (!parentch)
01045                         return;
01046 
01047                 channel_entry = silc_client_get_channel(sg->client, sg->conn,
01048                                                         (char *)parentch);
01049                 if (!channel_entry ||
01050                     !silc_client_on_channel(channel_entry, sg->conn->local_entry)) {
01051                         char tmp[512];
01052                         g_snprintf(tmp, sizeof(tmp),
01053                                    _("You have to join the %s channel before you are "
01054                                      "able to join the private group"), parentch);
01055                         purple_notify_error(gc, _("Join Private Group"),
01056                                           _("Cannot join private group"), tmp);
01057                         return;
01058                 }
01059 
01060                 /* Add channel private key */
01061                 if (!silc_client_add_channel_private_key(client, conn,
01062                                                          channel_entry, channel,
01063                                                          NULL, NULL,
01064                                                          (unsigned char *)passphrase,
01065                                                          strlen(passphrase), &key))
01066                         return;
01067 
01068                 /* Join the group */
01069                 grp = silc_calloc(1, sizeof(*grp));
01070                 if (!grp)
01071                         return;
01072                 grp->id = ++sg->channel_ids + SILCPURPLE_PRVGRP;
01073                 grp->chid = SILC_PTR_TO_32(channel_entry->context);
01074                 grp->parentch = parentch;
01075                 grp->channel = channel;
01076                 grp->key = key;
01077                 sg->grps = g_list_append(sg->grps, grp);
01078                 serv_got_joined_chat(gc, grp->id, channel);
01079                 return;
01080         }
01081 
01082         /* XXX We should have other properties here as well:
01083            1. whether to try to authenticate to the channel
01084              1a. with default key,
01085              1b. with specific key.
01086            2. whether to try to authenticate to become founder.
01087              2a. with default key,
01088              2b. with specific key.
01089 
01090            Since now such variety is not possible in the join dialog
01091            we always use -founder and -auth options, which try to
01092            do both 1 and 2 with default keys. */
01093 
01094         /* Call JOIN */
01095         if ((passphrase != NULL) && (*passphrase != '\0'))
01096                 silc_client_command_call(client, conn, NULL, "JOIN",
01097                                          channel, passphrase, "-auth", "-founder", NULL);
01098         else
01099                 silc_client_command_call(client, conn, NULL, "JOIN",
01100                                          channel, "-auth", "-founder", NULL);
01101 }

01103 void silcpurple_chat_invite(PurpleConnection *gc, int id, const char *msg,
01104                             const char *name)
01105 {
01106         SilcPurple sg = gc->proto_data;
01107         SilcClient client = sg->client;
01108         SilcClientConnection conn = sg->conn;
01109         SilcHashTableList htl;
01110         SilcChannelUser chu;
01111         gboolean found = FALSE;
01112 
01113         if (!conn)
01114                 return;
01115 
01116         /* See if we are inviting on a private group.  Invite
01117            to the actual channel */
01118         if (id > SILCPURPLE_PRVGRP) {
01119                 GList *l;
01120                 SilcPurplePrvgrp prv;
01121 
01122                 for (l = sg->grps; l; l = l->next)
01123                         if (((SilcPurplePrvgrp)l->data)->id == id)
01124                                 break;
01125                 if (!l)
01126                         return;
01127                 prv = l->data;
01128                 id = prv->chid;
01129         }
01130 
01131         /* Find channel by id */
01132         silc_hash_table_list(conn->local_entry->channels, &htl);
01133         while (silc_hash_table_get(&htl, NULL, (void *)&chu)) {
01134                 if (SILC_PTR_TO_32(chu->channel->context) == id ) {
01135                         found = TRUE;
01136                         break;
01137                 }
01138         }
01139         silc_hash_table_list_reset(&htl);
01140         if (!found)
01141                 return;
01142 
01143         /* Call INVITE */
01144         silc_client_command_call(client, conn, NULL, "INVITE",
01145                                  chu->channel->channel_name,
01146                                  name, NULL);

01219                          PurpleMessageFlags msgflags)
01220 {

01230         char *msg2, *tmp;
01231         gboolean found = FALSE;
01232         gboolean sign = purple_account_get_bool(sg->account, "sign-verify", FALSE);
01233         SilcDList list;
01234 
01235         if (!msg || !conn)
01236                 return 0;
01237

01239 
01240         tmp = msg2 = purple_unescape_html(msg);
01241 
01242         if (!g_ascii_strncasecmp(msg2, "/me ", 4))
01243         {
01244                 msg2 += 4;

01246                         g_free(tmp);
01247                         return 0;
01248                 }

01250         } else if (strlen(msg) > 1 && msg[0] == '/') {
01251                 if (!silc_client_command_call(client, conn, msg + 1))
01252                         purple_notify_error(gc, _("Call Command"), _("Cannot call command"),
01253                                             _("Unknown command"));

01255                 return 0;
01256         }
01257 
01258 
01259         if (sign)
01260                 flags |= SILC_MESSAGE_FLAG_SIGNED;
01261

01263            private group */

01265                 GList *l;
01266                 SilcPurplePrvgrp prv;
01267 
01268                 for (l = sg->grps; l; l = l->next)
01269                         if (((SilcPurplePrvgrp)l->data)->id == id)
01270                                 break;

01272                         g_free(tmp);
01273                         return 0;
01274                 }

01277                                                   (char *)prv->parentch);
01278                 if (!channel) {
01279                         g_free(tmp);
01280                         return 0;
01281                 }
01282                 key = prv->key;
01283         }
01284 
01285         if (!channel) {
01286                 /* Find channel by id */
01287                 silc_hash_table_list(conn->local_entry->channels, &htl);
01288                 while (silc_hash_table_get(&htl, NULL, (void *)&chu)) {

01290                                 found = TRUE;
01291                                 break;
01292                         }
01293                 }
01294                 silc_hash_table_list_reset(&htl);
01295                 if (!found) {
01296                         g_free(tmp);
01297                         return 0;
01298                 }
01299                 channel = chu->channel;
01300         }
01301 
01302         /* Check for images */
01303         if (msgflags & PURPLE_MESSAGE_IMAGES) {
01304                 list = silcpurple_image_message(msg, &flags);

01309

01312                                 ret =
01313                                 silc_client_send_channel_message(client, conn,
01314                                                                  channel, key,
01315                                                                  flags, sg->sha1hash,
01316                                                                  buf->data,
01317                                                                  silc_buffer_len(buf));

01329                                                flags, sg->sha1hash,
01330                                                (unsigned char *)msg2,
01331                                                strlen(msg2));

01333                 serv_got_chat_in(gc, id, purple_connection_get_display_name(gc), msgflags, msg,
01334                                  time(NULL));
01335         }

01337
