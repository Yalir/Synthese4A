00033 static void jabber_nick_cb(JabberStream *js, const char *from, xmlnode *items) {
00034         /* it doesn't make sense to have more than one item here, so let's just pick the first one */
00035         xmlnode *item = xmlnode_get_child(items, "item");
00036         JabberBuddy *buddy = jabber_buddy_find(js, from, FALSE);
00037         xmlnode *nick;
00038         char *nickname = NULL;
00039 
00040         /* ignore the nick of people not on our buddy list */
00041         if (!buddy || !item)
00042                 return;
00043 
00044         nick = xmlnode_get_child_with_namespace(item, "nick", "http://jabber.org/protocol/nick");
00045         if (!nick)
00046                 return;
00047         nickname = xmlnode_get_data(nick);
00048         serv_got_alias(js->gc, from, nickname);
00049         g_free(nickname);
00050 }

00052 static void do_nick_set(JabberStream *js, const char *nick) {
00053         xmlnode *publish, *nicknode;
00054 
00055         publish = xmlnode_new("publish");
00056         xmlnode_set_attrib(publish,"node","http://jabber.org/protocol/nick");
00057         nicknode = xmlnode_new_child(xmlnode_new_child(publish, "item"), "nick");
00058         xmlnode_set_namespace(nicknode, "http://jabber.org/protocol/nick");
00059 
00060         if(nick && nick[0] != '\0')
00061                 xmlnode_insert_data(nicknode, nick, -1);
00062 
00063         jabber_pep_publish(js, publish);
00064         /* publish is freed by jabber_pep_publish -> jabber_iq_send -> jabber_iq_free
00065                 (yay for well-defined memory management rules) */
00066 }

00068 static void do_nick_got_own_nick_cb(JabberStream *js, const char *from, xmlnode *items) {
00069         char *oldnickname = NULL;
00070         xmlnode *item = NULL;
00071 
00072         if (items)
00073                 item = xmlnode_get_child(items,"item");
00074 
00075         if(item) {
00076                 xmlnode *nick = xmlnode_get_child_with_namespace(item,"nick","http://jabber.org/protocol/nick");
00077                 if(nick)
00078                         oldnickname = xmlnode_get_data(nick);
00079         }
00080 
00081         purple_request_input(js->gc, _("Set User Nickname"), _("Please specify a new nickname for you."),
00082                 _("This information is visible to all contacts on your contact list, so choose something appropriate."),
00083                 oldnickname, FALSE, FALSE, NULL, _("Set"), PURPLE_CALLBACK(do_nick_set), _("Cancel"), NULL,
00084                 purple_connection_get_account(js->gc), NULL, NULL, js);
00085         g_free(oldnickname);
00086 }

00088 static void do_nick_set_nick(PurplePluginAction *action) {
00089         PurpleConnection *gc = action->context;
00090         JabberStream *js = purple_connection_get_protocol_data(gc);
00091 
00092         /* since the nickname might have been changed by another resource of this account, we always have to request the old one
00093                 from the server to present as the default for the new one */
00094         jabber_pep_request_item(js, NULL, "http://jabber.org/protocol/nick", NULL, do_nick_got_own_nick_cb);
00095 }
