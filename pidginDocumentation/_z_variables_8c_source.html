00189 /* If the variable in the line bfr[] is the same as var, return index to
00190    the variable value, else return 0. */
00191 static int varline(bfr, var)
00192     char *bfr;
00193     char *var;
00194 {
00195     register char *cp;
00196 
00197 
00198     if (!bfr[0] || bfr[0] == '#')       /* comment or null line */
00199         return (0);
00200 
00201     cp = bfr;
00202     while (*cp && !isspace(*cp) && (*cp != '='))
00203         cp++;
00204 
00205 #ifndef WIN32
00206 #define max(a,b) ((a > b) ? (a) : (b))
00207 #endif
00208 
00209     if (g_ascii_strncasecmp(bfr, var, max(strlen(var), cp - bfr)))
00210         return(0);                      /* var is not the var in
00211                                            bfr ==> no match */
00212 
00213     cp = strchr(bfr, '=');
00214     if (!cp)
00215         return(0);
00216     cp++;
00217     while (*cp && isspace(*cp))         /* space up to variable value */
00218         cp++;
00219 
00220     return (cp - bfr);                  /* return index */
00221 }
