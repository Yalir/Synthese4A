00049 static inline guint32 lshift(guint32 x, unsigned int s)
00050 {
00051         x &= 0xFFFFFFFF;
00052         return ((x << s) & 0xFFFFFFFF) | (x >> (32 - s));
00053 }

00055 static inline guint32 F(guint32 x, guint32 y, guint32 z)
00056 {
00057         return (x & y) | ((~x) & z);
00058 }
00059 
00060 static inline guint32 G(guint32 x, guint32 y, guint32 z)
00061 {
00062         return (x & y) | (x & z) | (y & z);
00063 }

00065 static inline guint32 H(guint32 x, guint32 y, guint32 z)
00066 {
00067         return x ^ y ^ z;
00068 }

00072 #define ROUND3(a,b,c,d,k,s) (a = lshift(a + H(b,c,d) + k + (guint32)0x6ED9EBA1,s))

00189         static void
00190 md4_append(PurpleCipherContext *context, const guchar *data, size_t len)
00191 {
00192         struct MD4_Context *mctx = purple_cipher_context_get_data(context);
00193         const guint32 avail = sizeof(mctx->block) - (mctx->byte_count & 0x3f);
00194 
00195         mctx->byte_count += len;
00196 
00197         if (avail > len) {
00198                 memcpy((char *)mctx->block + (sizeof(mctx->block) - avail),
00199                                 data, len);
00200                 return;
00201         }
00202 
00203         memcpy((char *)mctx->block + (sizeof(mctx->block) - avail),
00204                         data, avail);
00205 
00206         md4_transform_helper(mctx);
00207         data += avail;
00208         len -= avail;
00209 
00210         while (len >= sizeof(mctx->block)) {
00211                 memcpy(mctx->block, data, sizeof(mctx->block));
00212                 md4_transform_helper(mctx);
00213                 data += sizeof(mctx->block);
00214                 len -= sizeof(mctx->block);
00215         }
00216 
00217         memcpy(mctx->block, data, len);
00218 }

00220         static gboolean
00221 md4_digest(PurpleCipherContext *context, size_t in_len, guchar *out,
00222                 size_t *out_len)
00223 {
00224         struct MD4_Context *mctx = purple_cipher_context_get_data(context);
00225         const unsigned int offset = mctx->byte_count & 0x3f;
00226         char *p = (char *)mctx->block + offset;
00227         int padding = 56 - (offset + 1);
00228 
00229 
00230         if(in_len<16) return FALSE;
00231         if(out_len) *out_len = 16;
00232         *p++ = 0x80;
00233         if (padding < 0) {
00234                 memset(p, 0x00, padding + sizeof (guint64));
00235                 md4_transform_helper(mctx);
00236                 p = (char *)mctx->block;
00237                 padding = 56;
00238         }
00239 
00240         memset(p, 0, padding);
00241         mctx->block[14] = mctx->byte_count << 3;
00242         mctx->block[15] = mctx->byte_count >> 29;
00243         le32_to_cpu_array(mctx->block, (sizeof(mctx->block) -
00244                                 sizeof(guint64)) / sizeof(guint32));
00245         md4_transform(mctx->hash, mctx->block);
00246         cpu_to_le32_array(mctx->hash, sizeof(mctx->hash) / sizeof(guint32));
00247         memcpy(out, mctx->hash, sizeof(mctx->hash));
00248         memset(mctx, 0, sizeof(*mctx));
00249         return TRUE;
00250 }
