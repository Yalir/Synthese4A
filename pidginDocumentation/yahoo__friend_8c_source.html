00147 void yahoo_process_presence(PurpleConnection *gc, struct yahoo_packet *pkt)
00148 {
00149         GSList *l = pkt->hash;
00150         YahooFriend *f;
00151         char *temp = NULL;
00152         char *who = NULL;
00153         int value = 0;
00154         YahooFederation fed = YAHOO_FEDERATION_NONE;
00155 
00156         while (l) {
00157                 struct yahoo_pair *pair = l->data;
00158 
00159                 switch (pair->key) {
00160                         case 7:
00161                                 temp = pair->value;
00162                                 break;
00163                         case 31:
00164                                 value = strtol(pair->value, NULL, 10);
00165                                 break;
00166                         case 241:
00167                                 fed = strtol(pair->value, NULL, 10);
00168                                 break;
00169                 }
00170 
00171                 l = l->next;
00172         }
00173 
00174         if (value != 1 && value != 2) {
00175                 purple_debug_error("yahoo", "Received unknown value for presence key: %d\n", value);
00176                 return;
00177         }
00178 
00179         switch (fed) {
00180                 case YAHOO_FEDERATION_MSN:
00181                         who = g_strconcat("msn/", temp, NULL);
00182                         break;
00183                 case YAHOO_FEDERATION_OCS:
00184                         who = g_strconcat("ocs/", temp, NULL);
00185                         break;
00186                 case YAHOO_FEDERATION_IBM:
00187                         who = g_strconcat("ibm/", temp, NULL);
00188                         break;
00189                 case YAHOO_FEDERATION_PBX:
00190                         who = g_strconcat("pbx/", temp, NULL);
00191                         break;
00192                 case YAHOO_FEDERATION_NONE:
00193                         who = g_strdup(temp);
00194                         break;
00195         }
00196         g_return_if_fail(who != NULL);
00197 
00198         f = yahoo_friend_find(gc, who);
00199         if (!f) {
00200                 g_free(who);
00201                 return;
00202         }
00203 
00204         if (pkt->service == YAHOO_SERVICE_PRESENCE_PERM) {
00205                 purple_debug_info("yahoo", "Setting permanent presence for %s to %d.\n", who, (value == 1));
00206                 /* If setting from perm offline to online when in invisible status,
00207                  * this has already been taken care of (when the temp status changed) */
00208                 if (value == 2 && f->presence == YAHOO_PRESENCE_ONLINE) {
00209                 } else {
00210                         if (value == 1) /* Setting Perm offline */
00211                                 f->presence = YAHOO_PRESENCE_PERM_OFFLINE;
00212                         else
00213                                 f->presence = YAHOO_PRESENCE_DEFAULT;
00214                 }
00215         } else {
00216                 purple_debug_info("yahoo", "Setting session presence for %s to %d.\n", who, (value == 1));
00217                 if (value == 1)
00218                         f->presence = YAHOO_PRESENCE_ONLINE;
00219                 else
00220                         f->presence = YAHOO_PRESENCE_DEFAULT;
00221         }
00222         g_free(who);
00223 }
