00070 void
00071 nm_release_conference(NMConference * conference)
00072 {
00073         GSList *node;
00074 
00075         g_return_if_fail(conference != NULL);
00076 
00077         purple_debug(PURPLE_DEBUG_INFO, "novell",
00078                            "In release conference %p, refs=%d\n",
00079                            conference, conference->ref_count);
00080         if (--conference->ref_count == 0) {
00081 
00082                 purple_debug(PURPLE_DEBUG_INFO, "novell",
00083                                    "Releasing conference %p, total=%d\n",
00084                                    conference, --conf_count);
00085 
00086                 if (conference->guid)
00087                         g_free(conference->guid);
00088 
00089                 if (conference->participants) {
00090                         for (node = conference->participants; node; node = node->next) {
00091                                 if (node->data) {
00092                                         NMUserRecord *user_record = node->data;
00093 
00094                                         nm_release_user_record(user_record);
00095                                         node->data = NULL;
00096                                 }
00097                         }
00098 
00099                         g_slist_free(conference->participants);
00100                 }
00101 
00102                 g_free(conference);
00103         }
00104 }

00146 nm_conference_remove_participant(NMConference * conference, const char *dn)
00147 {
00148         GSList *node, *element = NULL;
00149 
00150         if (conference == NULL || dn == NULL) {
00151                 return;

00153 
00154         for (node = conference->participants; node; node = node->next) {
00155                 NMUserRecord *user_record = node->data;
00156 
00157                 if (user_record) {
00158                         if (nm_utf8_str_equal(dn, nm_user_record_get_dn(user_record))) {
00159                                 element = node;

00166                 nm_release_user_record((NMUserRecord *) element->data);
00167                 element->data = NULL;
00168                 conference->participants =
00169                         g_slist_remove_link(conference->participants, element);
