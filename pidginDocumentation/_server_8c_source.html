00068 static gboolean
00069 expire_last_auto_responses(gpointer data)
00070 {
00071         GSList *tmp, *cur;
00072         struct last_auto_response *lar;
00073 
00074         tmp = last_auto_responses;
00075 
00076         while (tmp) {
00077                 cur = tmp;
00078                 tmp = tmp->next;
00079                 lar = (struct last_auto_response *)cur->data;
00080 
00081                 if ((time(NULL) - lar->sent) > SECS_BEFORE_RESENDING_AUTORESPONSE) {
00082                         last_auto_responses = g_slist_remove(last_auto_responses, lar);
00083                         g_free(lar);
00084                 }
00085         }
00086 
00087         return FALSE; /* do not run again */

00090 static struct last_auto_response *
00091 get_last_auto_response(PurpleConnection *gc, const char *name)
00092 {
00093         GSList *tmp;
00094         struct last_auto_response *lar;
00095 
00096         /* because we're modifying or creating a lar, schedule the
00097          * function to expire them as the pref dictates */
00098         purple_timeout_add_seconds((SECS_BEFORE_RESENDING_AUTORESPONSE + 1), expire_last_auto_responses, NULL);
00099 
00100         tmp = last_auto_responses;
00101 
00102         while (tmp) {
00103                 lar = (struct last_auto_response *)tmp->data;
00104 
00105                 if (gc == lar->gc && !strncmp(name, lar->name, sizeof(lar->name)))
00106                         return lar;
00107 
00108                 tmp = tmp->next;
00109         }
00110 
00111         lar = (struct last_auto_response *)g_new0(struct last_auto_response, 1);
00112         g_snprintf(lar->name, sizeof(lar->name), "%s", name);
00113         lar->gc = gc;
00114         lar->sent = 0;
00115         last_auto_responses = g_slist_prepend(last_auto_responses, lar);
00116 
00117         return lar;
00118 }

00675                         {
00676                                 /*
00677                                  * We don't want to send an autoresponse in response to the other user's
00678                                  * autoresponse.  We do, however, not want to then send one in response to the
00679                                  * _next_ message, so we still set lar->sent to now.
00680                                  */
00681                                 lar->sent = now;
00682 
00683                                 if (!(flags & PURPLE_MESSAGE_AUTO_RESP))
00684                                 {
00685                                         serv_send_im(gc, name, away_msg, PURPLE_MESSAGE_AUTO_RESP);
00686 
00687                                         purple_conv_im_write(PURPLE_CONV_IM(conv), NULL, away_msg,
00688                                                                            PURPLE_MESSAGE_SEND | PURPLE_MESSAGE_AUTO_RESP,
00689                                                                            mtime);
00690                                 }
00691                         }
00692                 }
00693         }
00694 
00695         g_free(name);

00896 void serv_got_chat_in(PurpleConnection *g, int id, const char *who,
00897                                           PurpleMessageFlags flags, const char *message, time_t mtime)
00898 {
00899         GSList *bcs;
00900         PurpleConversation *conv = NULL;
00901         PurpleConvChat *chat = NULL;
00902         char *buffy, *angel;
00903         int plugin_return;
00904 
00905         g_return_if_fail(who != NULL);
00906         g_return_if_fail(message != NULL);
00907 
00908         for (bcs = g->buddy_chats; bcs != NULL; bcs = bcs->next) {
00909                 conv = (PurpleConversation *)bcs->data;
00910 
00911                 chat = PURPLE_CONV_CHAT(conv);
00912 
00913                 if (purple_conv_chat_get_id(chat) == id)
00914                         break;
00915 
00916                 conv = NULL;
00917         }
00918 
00919         if (!conv)
00920                 return;
00921 
00922         /* Did I send the message? */
00923         if (purple_strequal(purple_conv_chat_get_nick(chat),
00924                                 purple_normalize(purple_conversation_get_account(conv), who))) {
00925                 flags |= PURPLE_MESSAGE_SEND;
00926                 flags &= ~PURPLE_MESSAGE_RECV; /* Just in case some prpl sets it! */
00927         } else {
00928                 flags |= PURPLE_MESSAGE_RECV;
00929         }
00930 
00931         /*
00932          * Make copies of the message and the sender in case plugins want
00933          * to free these strings and replace them with a modifed version.
00934          */
00935         buffy = g_strdup(message);
00936         angel = g_strdup(who);
00937 
00938         plugin_return = GPOINTER_TO_INT(
00939                 purple_signal_emit_return_1(purple_conversations_get_handle(),
00940                                                                   "receiving-chat-msg", g->account,
00941                                                                   &angel, &buffy, conv, &flags));
00942 
00943         if (!buffy || !angel || plugin_return) {
00944                 g_free(buffy);
00945                 g_free(angel);
00946                 return;
00947         }
00948 
00949         who = angel;
00950         message = buffy;
00951 
00952         purple_signal_emit(purple_conversations_get_handle(), "received-chat-msg", g->account,
00953                                          who, message, conv, flags);
00954 
00955         purple_conv_chat_write(chat, who, message, flags, mtime);
00956 
00957         g_free(angel);
00958         g_free(buffy);
00959 }
