00099 purple_dnsquery_resolved(PurpleDnsQueryData *query_data, GSList *hosts)
00100 {
00101         purple_debug_info("dnsquery", "IP resolved for %s\n", query_data->hostname);
00102         if (query_data->callback != NULL)
00103                 query_data->callback(hosts, query_data->data, NULL);
00104         else
00105         {
00106                 /*
00107                  * Callback is a required parameter, but it can get set to
00108                  * NULL if we cancel a thread-based DNS lookup.  So we need
00109                  * to free hosts.
00110                  */

00112                 {
00113                         hosts = g_slist_remove(hosts, hosts->data);
00114                         g_free(hosts->data);

00116                 }
00117         }

00259 purple_dnsquery_resolver_run(int child_out, int child_in, gboolean show_debug)
00260 {
00261         dns_params_t dns_params;
00262         const size_t zero = 0;
00263         int rc;
00264 #ifdef HAVE_GETADDRINFO
00265         struct addrinfo hints, *res, *tmp;
00266         char servname[20];
00267 #else
00268         struct sockaddr_in sin;
00269         const size_t addrlen = sizeof(sin);
00270 #endif
00271         char *hostname;
00272 
00273 #ifdef HAVE_SIGNAL_H
00274         purple_restore_default_signal_handlers();
00275         signal(SIGTRAP, trap_gdb_bug);
00276 #endif
00277 
00278         /*
00279          * We resolve 1 host name for each iteration of this
00280          * while loop.
00281          *
00282          * The top half of this reads in the hostname and port
00283          * number from the socket with our parent.  The bottom
00284          * half of this resolves the IP (blocking) and sends
00285          * the result back to our parent, when finished.
00286          */
00287         while (1) {
00288                 fd_set fds;
00289                 struct timeval tv = { .tv_sec = 20, .tv_usec = 0 };
00290                 FD_ZERO(&fds);
00291                 FD_SET(child_in, &fds);
00292                 rc = select(child_in + 1, &fds, NULL, NULL, &tv);
00293                 if (!rc) {
00294                         if (show_debug)
00295                                 printf("dns[%d]: nobody needs me... =(\n", getpid());
00296                         break;
00297                 }
00298                 rc = read(child_in, &dns_params, sizeof(dns_params_t));
00299                 if (rc < 0) {
00300                         fprintf(stderr, "dns[%d]: Error: Could not read dns_params: "
00301                                         "%s\n", getpid(), strerror(errno));
00302                         break;
00303                 }
00304                 if (rc == 0) {
00305                         if (show_debug)
00306                                 printf("dns[%d]: Oops, father has gone, wait for me, wait...!\n", getpid());
00307                         _exit(0);
00308                 }
00309                 if (dns_params.hostname[0] == '\0') {
00310                         fprintf(stderr, "dns[%d]: Error: Parent requested resolution "
00311                                         "of an empty hostname (port = %d)!!!\n", getpid(),
00312                                         dns_params.port);
00313                         _exit(1);
00314                 }
00315 
00316 #ifdef USE_IDN
00317                 if (!dns_str_is_ascii(dns_params.hostname)) {
00318                         rc = purple_network_convert_idn_to_ascii(dns_params.hostname, &hostname);
00319                         if (rc != 0) {
00320                                 write_to_parent(child_out, &rc, sizeof(rc));
00321                                 if (show_debug)
00322                                         fprintf(stderr, "dns[%d] Error: IDN conversion returned "
00323                                                         "%d\n", getpid(), rc);
00324                                 dns_params.hostname[0] = '\0';
00325                                 break;
00326                         }
00327                 } else /* intentional to execute the g_strdup */
00328 #endif
00329                 hostname = g_strdup(dns_params.hostname);
00330 
00331                 /* We have the hostname and port, now resolve the IP */
00332 
00333 #ifdef HAVE_GETADDRINFO
00334                 g_snprintf(servname, sizeof(servname), "%d", dns_params.port);
00335                 memset(&hints, 0, sizeof(hints));
00336 
00337                 /* This is only used to convert a service
00338                  * name to a port number. As we know we are
00339                  * passing a number already, we know this
00340                  * value will not be really used by the C
00341                  * library.
00342                  */
00343                 hints.ai_socktype = SOCK_STREAM;
00344 #ifdef AI_ADDRCONFIG
00345                 hints.ai_flags |= AI_ADDRCONFIG;
00346 #endif /* AI_ADDRCONFIG */
00347                 rc = getaddrinfo(hostname, servname, &hints, &res);
00348                 write_to_parent(child_out, &rc, sizeof(rc));
00349                 if (rc != 0) {
00350                         if (show_debug)
00351                                 printf("dns[%d] Error: getaddrinfo returned %d\n",
00352                                         getpid(), rc);
00353                         dns_params.hostname[0] = '\0';
00354                         g_free(hostname);
00355                         hostname = NULL;
00356                         break;
00357                 }
00358                 tmp = res;
00359                 while (res) {
00360                         size_t ai_addrlen = res->ai_addrlen;
00361                         write_to_parent(child_out, &ai_addrlen, sizeof(ai_addrlen));
00362                         write_to_parent(child_out, res->ai_addr, res->ai_addrlen);
00363                         res = res->ai_next;
00364                 }
00365                 freeaddrinfo(tmp);
00366 #else
00367                 struct hostent *hp;
00368                 if (!(hp = gethostbyname(hostname))) {
00369                         write_to_parent(child_out, &h_errno, sizeof(int));
00370                         close(child_out);
00371                         if (show_debug)
00372                                 printf("DNS Error: %d\n", h_errno);
00373                         _exit(0);
00374                 }
00375                 memset(&sin, 0, sizeof(struct sockaddr_in));
00376                 memcpy(&sin.sin_addr.s_addr, hp->h_addr, hp->h_length);
00377                 sin.sin_family = hp->h_addrtype;
00378 
00379                 sin.sin_port = htons(dns_params.port);
00380                 rc = 0;
00381                 write_to_parent(child_out, &rc, sizeof(rc));
00382                 write_to_parent(child_out, &addrlen, sizeof(addrlen));
00383                 write_to_parent(child_out, &sin, addrlen);
00384 #endif
00385                 write_to_parent(child_out, &zero, sizeof(zero));
00386                 dns_params.hostname[0] = '\0';
00387 
00388                 g_free(hostname);
00389                 hostname = NULL;
00390         }
00391 
00392         close(child_out);
00393         close(child_in);
00394 
00395         _exit(0);
00396 }

00517 static gboolean
00518 send_dns_request_to_child(PurpleDnsQueryData *query_data,
00519                 PurpleDnsQueryResolverProcess *resolver)
00520 {
00521         pid_t pid;
00522         dns_params_t dns_params;
00523         ssize_t rc;
00524 
00525         /* This waitpid might return the child's PID if it has recently
00526          * exited, or it might return an error if it exited "long
00527          * enough" ago that it has already been reaped; in either
00528          * instance, we can't use it. */
00529         pid = waitpid(resolver->dns_pid, NULL, WNOHANG);
00530         if (pid > 0) {
00531                 purple_debug_warning("dns", "DNS child %d no longer exists\n",
00532                                 resolver->dns_pid);
00533                 purple_dnsquery_resolver_destroy(resolver);
00534                 return FALSE;
00535         } else if (pid < 0) {
00536                 purple_debug_warning("dns", "Wait for DNS child %d failed: %s\n",
00537                                 resolver->dns_pid, g_strerror(errno));
00538                 purple_dnsquery_resolver_destroy(resolver);
00539                 return FALSE;
00540         }
00541 
00542         /* Copy the hostname and port into a single data structure */
00543         strncpy(dns_params.hostname, query_data->hostname, sizeof(dns_params.hostname) - 1);
00544         dns_params.hostname[sizeof(dns_params.hostname) - 1] = '\0';
00545         dns_params.port = query_data->port;
00546 
00547         /* Send the data structure to the child */
00548         rc = write(resolver->fd_in, &dns_params, sizeof(dns_params));
00549         if (rc < 0) {
00550                 purple_debug_error("dns", "Unable to write to DNS child %d: %s\n",
00551                                 resolver->dns_pid, g_strerror(errno));
00552                 purple_dnsquery_resolver_destroy(resolver);
00553                 return FALSE;
00554         }
00555         if (rc < sizeof(dns_params)) {
00556                 purple_debug_error("dns", "Tried to write %" G_GSSIZE_FORMAT
00557                                 " bytes to child but only wrote %" G_GSSIZE_FORMAT "\n",
00558                                 sizeof(dns_params), rc);
00559                 purple_dnsquery_resolver_destroy(resolver);
00560                 return FALSE;
00561         }
00562 
00563         purple_debug_info("dns",
00564                         "Successfully sent DNS request to child %d\n",
00565                         resolver->dns_pid);
00566 
00567         query_data->resolver = resolver;
00568 
00569         return TRUE;
00570 }

00576 {
00577         PurpleDnsQueryData *query_data;

00579 
00580         if (queued_requests == NULL)

00729 static gpointer
00730 dns_thread(gpointer data)
00731 {
00732         PurpleDnsQueryData *query_data;
00733 #ifdef HAVE_GETADDRINFO
00734         int rc;
00735         struct addrinfo hints, *res, *tmp;
00736         char servname[20];
00737 #else
00738         struct sockaddr_in sin;
00739         struct hostent *hp;
00740 #endif
00741         char *hostname;
00742 
00743         query_data = data;
00744 
00745 #ifdef USE_IDN
00746         if (!dns_str_is_ascii(query_data->hostname)) {
00747                 rc = purple_network_convert_idn_to_ascii(query_data->hostname, &hostname);
00748                 if (rc != 0) {
00749                         query_data->error_message = g_strdup_printf(_("Error converting %s "
00750                                         "to punycode: %d"), query_data->hostname, rc);
00751                         /* back to main thread */
00752                         purple_timeout_add(0, dns_main_thread_cb, query_data);
00753                         return 0;
00754                 }
00755         } else /* intentional fallthru */
00756 #endif
00757         hostname = g_strdup(query_data->hostname);
00758 
00759 #ifdef HAVE_GETADDRINFO
00760         g_snprintf(servname, sizeof(servname), "%d", query_data->port);
00761         memset(&hints,0,sizeof(hints));
00762 
00763         /*
00764          * This is only used to convert a service
00765          * name to a port number. As we know we are
00766          * passing a number already, we know this
00767          * value will not be really used by the C
00768          * library.
00769          */
00770         hints.ai_socktype = SOCK_STREAM;
00771 #ifdef AI_ADDRCONFIG
00772         hints.ai_flags |= AI_ADDRCONFIG;
00773 #endif /* AI_ADDRCONFIG */
00774         if ((rc = getaddrinfo(hostname, servname, &hints, &res)) == 0) {
00775                 tmp = res;
00776                 while(res) {
00777                         query_data->hosts = g_slist_append(query_data->hosts,
00778                                 GSIZE_TO_POINTER(res->ai_addrlen));
00779                         query_data->hosts = g_slist_append(query_data->hosts,
00780                                 g_memdup(res->ai_addr, res->ai_addrlen));
00781                         res = res->ai_next;
00782                 }
00783                 freeaddrinfo(tmp);
00784         } else {
00785                 query_data->error_message = g_strdup_printf(_("Error resolving %s:\n%s"), query_data->hostname, purple_gai_strerror(rc));
00786         }
00787 #else
00788         if ((hp = gethostbyname(hostname))) {
00789                 memset(&sin, 0, sizeof(struct sockaddr_in));
00790                 memcpy(&sin.sin_addr.s_addr, hp->h_addr, hp->h_length);
00791                 sin.sin_family = hp->h_addrtype;
00792                 sin.sin_port = htons(query_data->port);
00793 
00794                 query_data->hosts = g_slist_append(query_data->hosts,
00795                                 GSIZE_TO_POINTER(sizeof(sin)));
00796                 query_data->hosts = g_slist_append(query_data->hosts,
00797                                 g_memdup(&sin, sizeof(sin)));
00798         } else {
00799                 query_data->error_message = g_strdup_printf(_("Error resolving %s: %d"), query_data->hostname, h_errno);
00800         }
00801 #endif
00802         g_free(hostname);
00803 
00804         /* back to main thread */
00805         purple_timeout_add(0, dns_main_thread_cb, query_data);
00806 
00807         return 0;

00950 void
00951 purple_dnsquery_destroy(PurpleDnsQueryData *query_data)
00952 {
00953         PurpleDnsQueryUiOps *ops = purple_dnsquery_get_ui_ops();
00954 
00955         if (ops && ops->destroy)
00956                 ops->destroy(query_data);
00957 
00958 #if defined(PURPLE_DNSQUERY_USE_FORK)
00959         queued_requests = g_slist_remove(queued_requests, query_data);
00960 
00961         if (query_data->resolver != NULL)
00962                 /*
00963                  * This is only non-NULL when we're cancelling an in-progress
00964                  * query.  Ideally we would tell our resolver child to stop
00965                  * resolving shit and then we would add it back to the
00966                  * free_dns_children linked list.  However, it's hard to tell
00967                  * children stuff, they just don't listen.  So we'll just
00968                  * kill the process and allow a new child to be started if we
00969                  * have more stuff to resolve.
00970                  */
00971                 purple_dnsquery_resolver_destroy(query_data->resolver);
00972 #elif defined _WIN32 /* end PURPLE_DNSQUERY_USE_FORK */
00973         if (query_data->resolver != NULL)
00974         {
00975                 /*
00976                  * It's not really possible to kill a thread.  So instead we
00977                  * just set the callback to NULL and let the DNS lookup
00978                  * finish.
00979                  */
00980                 query_data->callback = NULL;
00981                 return;
00982         }
00983 
00984         while (query_data->hosts != NULL)
00985         {
00986                 /* Discard the length... */
00987                 query_data->hosts = g_slist_remove(query_data->hosts, query_data->hosts->data);
00988                 /* Free the address... */
00989                 g_free(query_data->hosts->data);
00990                 query_data->hosts = g_slist_remove(query_data->hosts, query_data->hosts->data);
00991         }
00992         g_free(query_data->error_message);
00993 #endif /* end _WIN32 */
00994 
00995         if (query_data->timeout > 0)
00996                 purple_timeout_remove(query_data->timeout);
00997 
00998         g_free(query_data->hostname);
00999         g_free(query_data);
