00439         read = recv(conn->fd,
00440                                 &bs->data[bs->offset],

00443 
00444         /* Check if the remote user closed the connection */
00445         if (read == 0)
00446         {
00447                 peer_connection_destroy(conn, OSCAR_DISCONNECT_REMOTE_CLOSED, NULL);
00448                 return;
00449         }
00450 
00451         if (read < 0)
00452         {
00453                 if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00454                         /* No worries */
00455                         return;

00458                                 OSCAR_DISCONNECT_LOST_CONNECTION, g_strerror(errno));
00459                 return;
00460         }
00461 
00462         bs->offset += read;
00463         if (bs->offset < bs->len)
00464                 /* Waiting for more data to arrive */
00465                 return;
00466 
00467         /* We have a complete ODC/OFT frame!  Handle it and continue reading */
00468         byte_stream_rewind(bs);
00469         peer_odc_handle_payload(conn, (const char *)bs->data,
00470                         bs->len, frame->encoding, frame->flags & 0x0001);
00471         g_free(bs->data);
00472         bs->data = NULL;

00474         conn->frame = NULL;
00475 
00476         purple_input_remove(conn->watcher_incoming);
