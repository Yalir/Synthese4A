00100 /* Looks through the list of whiteboard sessions for one that is between
00101  * usernames 'me' and 'who'.  Returns a pointer to a matching whiteboard
00102  * session; if none match, it returns NULL.
00103  */
00104 PurpleWhiteboard *purple_whiteboard_get_session(const PurpleAccount *account, const char *who)
00105 {
00106         PurpleWhiteboard *wb;
00107 
00108         GList *l = wbList;
00109 
00110         /* Look for a whiteboard session between the local user and the remote user
00111          */
00112         while(l != NULL)
00113         {
00114                 wb = l->data;
00115 
00116                 if(wb->account == account && purple_strequal(wb->who, who))
00117                         return wb;
00118 
00119                 l = l->next;
00120         }
00121 
00122         return NULL;
00123 }
