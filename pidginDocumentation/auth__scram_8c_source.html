00076 guchar *jabber_scram_hi(const JabberScramHash *hash, const GString *str,
00077                         GString *salt, guint iterations)
00078 {
00079         PurpleCipherContext *context;
00080         guchar *result;
00081         guint i;
00082         guchar *prev, *tmp;
00083 
00084         g_return_val_if_fail(hash != NULL, NULL);
00085         g_return_val_if_fail(str != NULL && str->len > 0, NULL);
00086         g_return_val_if_fail(salt != NULL && salt->len > 0, NULL);
00087         g_return_val_if_fail(iterations > 0, NULL);
00088 
00089         prev   = g_new0(guint8, hash->size);
00090         tmp    = g_new0(guint8, hash->size);
00091         result = g_new0(guint8, hash->size);
00092 
00093         context = purple_cipher_context_new_by_name("hmac", NULL);
00094 
00095         /* Append INT(1), a four-octet encoding of the integer 1, most significant
00096          * octet first. */
00097         g_string_append_len(salt, "\0\0\0\1", 4);
00098 
00099         /* Compute U0 */
00100         purple_cipher_context_set_option(context, "hash", (gpointer)hash->name);
00101         purple_cipher_context_set_key_with_len(context, (guchar *)str->str, str->len);
00102         purple_cipher_context_append(context, (guchar *)salt->str, salt->len);
00103         purple_cipher_context_digest(context, hash->size, result, NULL);
00104 
00105         memcpy(prev, result, hash->size);
00106 
00107         /* Compute U1...Ui */
00108         for (i = 1; i < iterations; ++i) {
00109                 guint j;
00110                 purple_cipher_context_set_option(context, "hash", (gpointer)hash->name);
00111                 purple_cipher_context_set_key_with_len(context, (guchar *)str->str, str->len);
00112                 purple_cipher_context_append(context, prev, hash->size);
00113                 purple_cipher_context_digest(context, hash->size, tmp, NULL);
00114 
00115                 for (j = 0; j < hash->size; ++j)
00116                         result[j] ^= tmp[j];
00117 
00118                 memcpy(prev, tmp, hash->size);
00119         }
00120 
00121         purple_cipher_context_destroy(context);
00122         g_free(tmp);
00123         g_free(prev);
00124         return result;
00125 }

00160 gboolean
00161 jabber_scram_calc_proofs(JabberScramData *data, GString *salt, guint iterations)
00162 {
00163         guint hash_len = data->hash->size;
00164         guint i;
00165 
00166         GString *pass = g_string_new(data->password);
00167 
00168         guchar *salted_password;
00169         guchar *client_key, *stored_key, *client_signature, *server_key;
00170 
00171         client_key = g_new0(guchar, hash_len);
00172         stored_key = g_new0(guchar, hash_len);
00173         client_signature = g_new0(guchar, hash_len);
00174         server_key = g_new0(guchar, hash_len);
00175 
00176         data->client_proof = g_string_sized_new(hash_len);
00177         data->client_proof->len = hash_len;
00178         data->server_signature = g_string_sized_new(hash_len);
00179         data->server_signature->len = hash_len;
00180 
00181         salted_password = jabber_scram_hi(data->hash, pass, salt, iterations);
00182 
00183         memset(pass->str, 0, pass->allocated_len);
00184         g_string_free(pass, TRUE);
00185 
00186         if (!salted_password)
00187                 return FALSE;
00188 
00189         /* client_key = HMAC(salted_password, "Client Key") */
00190         hmac(data->hash, client_key, salted_password, "Client Key");
00191         /* server_key = HMAC(salted_password, "Server Key") */
00192         hmac(data->hash, server_key, salted_password, "Server Key");
00193         g_free(salted_password);
00194 
00195         /* stored_key = HASH(client_key) */
00196         hash(data->hash, stored_key, client_key);
00197 
00198         /* client_signature = HMAC(stored_key, auth_message) */
00199         hmac(data->hash, client_signature, stored_key, data->auth_message->str);
00200         /* server_signature = HMAC(server_key, auth_message) */
00201         hmac(data->hash, (guchar *)data->server_signature->str, server_key, data->auth_message->str);
00202 
00203         /* client_proof = client_key XOR client_signature */
00204         for (i = 0; i < hash_len; ++i)
00205                 data->client_proof->str[i] = client_key[i] ^ client_signature[i];
00206 
00207         g_free(server_key);
00208         g_free(client_signature);
00209         g_free(stored_key);
00210         g_free(client_key);
00211 
00212         return TRUE;
00213 }

00215 static gboolean
00216 parse_server_step1(JabberScramData *data, const char *challenge,
00217                    gchar **out_nonce, GString **out_salt, guint *out_iterations)
00218 {
00219         char **tokens;
00220         char *token, *decoded, *tmp;
00221         gsize len;
00222         char *nonce = NULL;
00223         GString *salt = NULL;
00224         guint iterations;
00225 
00226         tokens = g_strsplit(challenge, ",", -1);
00227         if (tokens == NULL)
00228                 return FALSE;
00229 
00230         token = tokens[0];
00231         if (token[0] != 'r' || token[1] != '=')
00232                 goto err;
00233 
00234         /* Ensure that the first cnonce_len bytes of the nonce are the original
00235          * cnonce we sent to the server.
00236          */
00237         if (0 != strncmp(data->cnonce, token + 2, strlen(data->cnonce)))
00238                 goto err;
00239 
00240         nonce = g_strdup(token + 2);
00241 
00242         /* The Salt, base64-encoded */
00243         token = tokens[1];
00244         if (token[0] != 's' || token[1] != '=')
00245                 goto err;
00246 
00247         decoded = (gchar *)purple_base64_decode(token + 2, &len);
00248         if (!decoded || *decoded == '\0') {
00249                 g_free(decoded);
00250                 goto err;
00251         }
00252         salt = g_string_new_len(decoded, len);
00253         g_free(decoded);
00254 
00255         /* The iteration count */
00256         token = tokens[2];
00257         if (token[0] != 'i' || token[1] != '=' || token[2] == '\0')
00258                 goto err;
00259 
00260         /* Validate the string */
00261         for (tmp = token + 2; *tmp; ++tmp)
00262                 if (!g_ascii_isdigit(*tmp))
00263                         goto err;
00264 
00265         iterations = strtoul(token + 2, NULL, 10);
00266 
00267         g_strfreev(tokens);
00268         *out_nonce = nonce;
00269         *out_salt = salt;
00270         *out_iterations = iterations;
00271         return TRUE;
00272 
00273 err:
00274         g_free(nonce);
00275         if (salt)
00276                 g_string_free(salt, TRUE);
00277         g_strfreev(tokens);
00278         return FALSE;
00279 }

00521                 *error = g_strdup(_("Invalid challenge from server"));
00522                 g_free(enc_in);

00536                 *error = g_strdup(_("Unexpected response from server"));
00537                 g_free(enc_in);
00538                 return JABBER_SASL_STATE_FAIL;
00539         }
00540 
00541         dec_in = (gchar *)purple_base64_decode(enc_in, &len);
00542         g_free(enc_in);
00543         if (!dec_in || len != strlen(dec_in)) {
00544                 /* Danger afoot; SCRAM shouldn't contain NUL bytes */
00545                 g_free(dec_in);
00546                 *error = g_strdup(_("Malicious challenge from server"));
00547                 return JABBER_SASL_STATE_FAIL;
00548         }
00549 
00550         purple_debug_misc("jabber", "decoded success: %s\n", dec_in);
00551 
00552         if (!jabber_scram_feed_parser(data, dec_in, &dec_out) || dec_out != NULL) {
00553                 g_free(dec_in);
00554                 g_free(dec_out);
00555                 *error = g_strdup(_("Invalid challenge from server"));
00556                 return JABBER_SASL_STATE_FAIL;
00557         }
00558
