00332 void silcpurple_show_public_key(SilcPurple sg,
00333                                 const char *name, SilcPublicKey public_key,
00334                                 GCallback callback, void *context)
00335 {
00336         SilcPublicKeyIdentifier ident;
00337         SilcSILCPublicKey silc_pubkey;
00338         char *fingerprint, *babbleprint;
00339         unsigned char *pk;
00340         SilcUInt32 pk_len, key_len = 0;
00341         GString *s;
00342 
00343         /* We support showing only SILC public keys for now */
00344         if (silc_pkcs_get_type(public_key) != SILC_PKCS_SILC)
00345           return;
00346 
00347         silc_pubkey = silc_pkcs_get_context(SILC_PKCS_SILC, public_key);
00348         ident = &silc_pubkey->identifier;
00349         key_len = silc_pkcs_public_key_get_len(public_key);
00350 
00351         pk = silc_pkcs_public_key_encode(public_key, &pk_len);
00352         if (!pk)
00353           return;
00354         fingerprint = silc_hash_fingerprint(NULL, pk, pk_len);
00355         babbleprint = silc_hash_babbleprint(NULL, pk, pk_len);
00356         if (!fingerprint || !babbleprint)
00357           return;
00358 
00359         s = g_string_new("");
00360         if (ident->realname)
00361                 /* Hint for translators: Please check the tabulator width here and in
00362                    the next strings (short strings: 2 tabs, longer strings 1 tab,
00363                    sum: 3 tabs or 24 characters) */
00364                 g_string_append_printf(s, _("Real Name: \t%s\n"), ident->realname);
00365         if (ident->username)
00366                 g_string_append_printf(s, _("User Name: \t%s\n"), ident->username);
00367         if (ident->email)
00368                 g_string_append_printf(s, _("Email: \t\t%s\n"), ident->email);
00369         if (ident->host)
00370                 g_string_append_printf(s, _("Host Name: \t%s\n"), ident->host);
00371         if (ident->org)
00372                 g_string_append_printf(s, _("Organization: \t%s\n"), ident->org);
00373         if (ident->country)
00374                 g_string_append_printf(s, _("Country: \t%s\n"), ident->country);
00375         g_string_append_printf(s, _("Algorithm: \t%s\n"), silc_pubkey->pkcs->name);
00376         g_string_append_printf(s, _("Key Length: \t%d bits\n"), (int)key_len);
00377         if (ident->version)
00378           g_string_append_printf(s, _("Version: \t%s\n"), ident->version);
00379         g_string_append_printf(s, "\n");
00380         g_string_append_printf(s, _("Public Key Fingerprint:\n%s\n\n"), fingerprint);
00381         g_string_append_printf(s, _("Public Key Babbleprint:\n%s"), babbleprint);
00382 
00383         purple_request_action(sg->gc, _("Public Key Information"),
00384                               _("Public Key Information"),
00385                               s->str, 0, purple_connection_get_account(sg->gc),
00386                               NULL, NULL, context, 1, _("Close"), callback);
00387 
00388         g_string_free(s, TRUE);
00389         silc_free(fingerprint);
00390         silc_free(babbleprint);
00391         silc_free(pk);
00392 }

00646 /* Checks if message has images, and assembles MIME message if it has.
00647    If only one image is present, creates simple MIME image message.  If
00648    there are multiple images and/or text with images multipart MIME

00651 SilcDList silcpurple_image_message(const char *msg, SilcUInt32 *mflags)
00652 {

00654         SilcDList list, parts = NULL;
00655         const char *start, *end, *last;
00656         GData *attribs;
00657         char *type;
00658         gboolean images = FALSE;
00659 
00660         last = msg;
00661         while (last && *last && purple_markup_find_tag("img", last, &start,
00662                                                      &end, &attribs)) {
00663                 PurpleStoredImage *image = NULL;
00664                 const char *id;
00665 
00666                 /* Check if there is text before image */
00667                 if (start - last) {
00668                         char *text, *tmp;
00669                         p = silc_mime_alloc();
00670 
00671                         /* Add content type */
00672                         silc_mime_add_field(p, "Content-Type",
00673                                             "text/plain; charset=utf-8");
00674 
00675                         tmp = g_strndup(last, start - last);
00676                         text = purple_unescape_html(tmp);
00677                         g_free(tmp);
00678 
00679                         /* Add text */
00680                         silc_mime_add_data(p, (const unsigned char *)text, strlen(text));
00681                         g_free(text);
00682 
00683                         if (!parts)
00684                                 parts = silc_dlist_init();
00685                         silc_dlist_add(parts, p);
00686                 }
00687 
00688                 id = g_datalist_get_data(&attribs, "id");
00689                 if (id && (image = purple_imgstore_find_by_id(atoi(id)))) {
00690                         unsigned long imglen = purple_imgstore_get_size(image);
00691                         gconstpointer img = purple_imgstore_get_data(image);
00692 
00693                         p = silc_mime_alloc();
00694 
00695                         /* Add content type */
00696                         type = silcpurple_file2mime(purple_imgstore_get_filename(image));
00697                         if (!type) {
00698                                 g_datalist_clear(&attribs);
00699                                 last = end + 1;
00700                                 continue;
00701                         }
00702                         silc_mime_add_field(p, "Content-Type", type);
00703                         g_free(type);
00704 
00705                         /* Add content transfer encoding */
00706                         silc_mime_add_field(p, "Content-Transfer-Encoding", "binary");
00707 
00708                         /* Add image data */
00709                         silc_mime_add_data(p, img, imglen);
00710 
00711                         if (!parts)
00712                                 parts = silc_dlist_init();
00713                         silc_dlist_add(parts, p);
00714                         images = TRUE;
00715                 }
00716 
00717                 g_datalist_clear(&attribs);
00718 
00719                 /* Continue after tag */
00720                 last = end + 1;
00721         }
00722 
00723         /* Check for text after the image(s) */
00724         if (images && last && *last) {
00725                 char *tmp = purple_unescape_html(last);
00726                 p = silc_mime_alloc();
00727 
00728                 /* Add content type */
00729                 silc_mime_add_field(p, "Content-Type",
00730                                     "text/plain; charset=utf-8");
00731 
00732                 /* Add text */
00733                 silc_mime_add_data(p, (const unsigned char *)tmp, strlen(tmp));
00734                 g_free(tmp);
00735 
00736                 if (!parts)
00737                         parts = silc_dlist_init();
00738                 silc_dlist_add(parts, p);
00739         }
00740 
00741         /* If there weren't any images, don't return anything. */
00742         if (!images) {

00752                 silc_mime_add_field(mime, "MIME-Version", "1.0");

00755                            silc_dlist_count(parts));

00765         }

00768         *mflags |= SILC_MESSAGE_FLAG_DATA;
00769 
00770         /* Encode message. Fragment if it is too large */

00772 
00773         silc_dlist_uninit(parts);
00774 
00775         /* Added multiparts gets freed here */
00776         silc_mime_free(mime);
