00030 {
00031         size_t length;
00032         ByteStream bs;
00033 
00034         purple_debug_info("oscar", "Outgoing peer proxy frame with "

00037                         frame->type, frame->unknown,
00038                         frame->flags, frame->payload.len);
00039 
00040         length = 12 + frame->payload.len;
00041         byte_stream_new(&bs, length);
00042         byte_stream_put16(&bs, length - 2);
00043         byte_stream_put16(&bs, PEER_PROXY_PACKET_VERSION);
00044         byte_stream_put16(&bs, frame->type);
00045         byte_stream_put32(&bs, frame->unknown);
00046         byte_stream_put16(&bs, frame->flags);
00047         byte_stream_putraw(&bs, frame->payload.data, frame->payload.len);
00048 
00049         peer_connection_send(conn, &bs);
00050 
00051         byte_stream_destroy(&bs);
00052 }

00132 static void
00133 peer_proxy_recv_frame(PeerConnection *conn, ProxyFrame *frame)
00134 {
00135         purple_debug_info("oscar", "Incoming peer proxy frame with "
00136                         "type=0x%04hx, unknown=0x%08x, flags=0x%04hx, and "
00137                         "payload length=%" G_GSIZE_FORMAT "\n", frame->type,
00138                         frame->unknown, frame->flags, frame->payload.len);
00139 
00140         if (frame->type == PEER_PROXY_TYPE_CREATED)
00141         {
00142                 /*
00143                  * Read in 2 byte port then 4 byte IP and tell the
00144                  * remote user to connect to it by sending an ICBM.
00145                  */
00146                 guint16 pin;
00147                 int i;
00148                 guint8 ip[4];
00149 
00150                 pin = byte_stream_get16(&frame->payload);
00151                 for (i = 0; i < 4; i++)
00152                         ip[i] = byte_stream_get8(&frame->payload);
00153                 if (conn->type == OSCAR_CAPABILITY_DIRECTIM)
00154                         aim_im_sendch2_odc_requestproxy(conn->od,
00155                                         conn->cookie,
00156                                         conn->bn, ip, pin, ++conn->lastrequestnumber);
00157                 else if (conn->type == OSCAR_CAPABILITY_SENDFILE)
00158                 {
00159                         aim_im_sendch2_sendfile_requestproxy(conn->od,
00160                                         conn->cookie, conn->bn,
00161                                         ip, pin, ++conn->lastrequestnumber,
00162                                         (const gchar *)conn->xferdata.name,
00163                                         conn->xferdata.size, conn->xferdata.totfiles);
00164                 }
00165         }
00166         else if (frame->type == PEER_PROXY_TYPE_READY)
00167         {
00168                 purple_input_remove(conn->watcher_incoming);
00169                 conn->watcher_incoming = 0;
00170 
00171                 peer_connection_finalize_connection(conn);
00172         }
00173         else if (frame->type == PEER_PROXY_TYPE_ERROR)
00174         {
00175                 if (byte_stream_bytes_left(&frame->payload) >= 2)
00176                 {
00177                         guint16 error;
00178                         const char *msg;
00179                         error = byte_stream_get16(&frame->payload);
00180                         if (error == 0x000d)
00181                                 msg = "bad request";
00182                         else if (error == 0x0010)
00183                                 msg = "initial request timed out";
00184                         else if (error == 0x001a)
00185                                 msg ="accept period timed out";
00186                         else
00187                                 msg = "unknown reason";
00188                         purple_debug_info("oscar", "Proxy negotiation failed with "
00189                                         "error 0x%04hx: %s\n", error, msg);
00190                 }
00191                 else
00192                 {
00193                         purple_debug_warning("oscar", "Proxy negotiation failed with "
00194                                         "an unknown error\n");
00195                 }
00196                 peer_connection_trynext(conn);
00197         }
00198         else
00199         {
00200                 purple_debug_warning("oscar", "Unknown peer proxy frame type 0x%04hx.\n",
00201                                 frame->type);
00202         }
00203 }

00208         PeerConnection *conn;

00210         ProxyFrame *frame;
00211 
00212         conn = data;
00213         frame = conn->frame;
00214 
00215         /* Start reading a new proxy frame */
00216         if (frame == NULL)
00217         {
00218                 /* Read the first 12 bytes (frame length and header) */
00219                 read = recv(conn->fd, conn->proxy_header + conn->proxy_header_received,
00220                                 12 - conn->proxy_header_received, 0);
00221 
00222                 /* Check if the proxy server closed the connection */
00223                 if (read == 0)
00224                 {
00225                         purple_debug_info("oscar", "Peer proxy server closed connection\n");
00226                         peer_connection_trynext(conn);
00227                         return;
00228                 }
00229 
00230                 /* If there was an error then close the connection */
00231                 if (read < 0)
00232                 {
00233                         if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00234                                 /* No worries */
00235                                 return;
00236 
00237                         purple_debug_info("oscar", "Lost connection with peer proxy server\n");
00238                         peer_connection_trynext(conn);
00239                         return;
00240                 }
00241 
00242                 conn->lastactivity = time(NULL);
00243 
00244                 /* If we don't even have the first 12 bytes then do nothing */
00245                 conn->proxy_header_received += read;
00246                 if (conn->proxy_header_received < 12)
00247                         return;
00248 
00249                 /* We only support a specific version of the proxy protocol */
00250                 if (aimutil_get16(&conn->proxy_header[2]) != PEER_PROXY_PACKET_VERSION)
00251                 {
00252                         purple_debug_warning("oscar", "Expected peer proxy protocol "
00253                                 "version %u but received version %u.  Closing "
00254                                 "connection.\n", PEER_PROXY_PACKET_VERSION,
00255                                 aimutil_get16(&conn->proxy_header[2]));
00256                         peer_connection_trynext(conn);
00257                         return;
00258                 }
00259 
00260                 /* Initialize a new temporary ProxyFrame for incoming data */
00261                 frame = g_new0(ProxyFrame, 1);
00262                 frame->payload.len = aimutil_get16(&conn->proxy_header[0]) - 10;
00263                 frame->version = aimutil_get16(&conn->proxy_header[2]);
00264                 frame->type = aimutil_get16(&conn->proxy_header[4]);
00265                 frame->unknown = aimutil_get16(&conn->proxy_header[6]);
00266                 frame->flags = aimutil_get16(&conn->proxy_header[10]);
00267                 if (frame->payload.len > 0)
00268                         frame->payload.data = g_new(guint8, frame->payload.len);
00269                 conn->frame = frame;
00270         }
00271 
00272         /* If this frame has a payload then attempt to read it */
00273         if (frame->payload.len - frame->payload.offset > 0)
00274         {
00275                 /* Read data into the temporary buffer until it is complete */
00276                 read = recv(conn->fd,
00277                                         &frame->payload.data[frame->payload.offset],
00278                                         frame->payload.len - frame->payload.offset,
00279                                         0);
00280 
00281                 /* Check if the proxy server closed the connection */
00282                 if (read == 0)
00283                 {
00284                         purple_debug_info("oscar", "Peer proxy server closed connection\n");
00285                         g_free(frame->payload.data);
00286                         g_free(frame);
00287                         conn->frame = NULL;
00288                         peer_connection_trynext(conn);
00289                         return;
00290                 }
00291 
00292                 /* If there was an error then close the connection */
00293                 if (read < 0)
00294                 {
00295                         if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00296                                 /* No worries */
00297                                 return;
00298 
00299                         purple_debug_info("oscar", "Lost connection with peer proxy server\n");
00300                         g_free(frame->payload.data);
00301                         g_free(frame);
00302                         conn->frame = NULL;
00303                         peer_connection_trynext(conn);
00304                         return;
00305                 }
00306 
00307                 frame->payload.offset += read;
00308         }
00309 
00310         conn->lastactivity = time(NULL);
00311         if (frame->payload.offset < frame->payload.len)
00312                 /* Waiting for more data to arrive */
00313                 return;
00314 
00315         /* We have a complete proxy frame!  Handle it and continue reading */
00316         conn->frame = NULL;
00317         byte_stream_rewind(&frame->payload);
00318         peer_proxy_recv_frame(conn, frame);
00319 
00320         g_free(frame->payload.data);
00321         g_free(frame);
00322 
00323         conn->proxy_header_received = 0;
00324 }
