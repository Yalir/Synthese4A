00932 void msn_emoticon_msg(MsnCmdProc *cmdproc, MsnMessage *msg)
00933 {
00934         MsnSession *session;
00935         MsnSlpLink *slplink;
00936         MsnSwitchBoard *swboard;
00937         MsnObject *obj;
00938         char **tokens;
00939         char *smile, *body_str;
00940         const char *body, *who, *sha1;
00941         guint tok;
00942         size_t body_len;
00943 
00944         PurpleConversation *conv;
00945 
00946         session = cmdproc->servconn->session;
00947 
00948         if (!purple_account_get_bool(session->account, "custom_smileys", TRUE))
00949                 return;
00950 
00951         swboard = cmdproc->data;
00952         conv = swboard->conv;
00953 
00954         body = msn_message_get_bin_data(msg, &body_len);
00955         if (!body || !body_len)
00956                 return;
00957         body_str = g_strndup(body, body_len);
00958 
00959         /* MSN Messenger 7 may send more than one MSNObject in a single message...
00960          * Maybe 10 tokens is a reasonable max value. */
00961         tokens = g_strsplit(body_str, "\t", 10);
00962 
00963         g_free(body_str);
00964 
00965         for (tok = 0; tok < 9; tok += 2) {
00966                 if (tokens[tok] == NULL || tokens[tok + 1] == NULL) {
00967                         break;
00968                 }
00969 
00970                 smile = tokens[tok];
00971                 obj = msn_object_new_from_string(purple_url_decode(tokens[tok + 1]));
00972 
00973                 if (obj == NULL)
00974                         break;
00975 
00976                 who = msn_object_get_creator(obj);
00977                 sha1 = msn_object_get_sha1(obj);
00978 
00979                 slplink = msn_session_get_slplink(session, who);
00980                 if (slplink->swboard != swboard) {
00981                         if (slplink->swboard != NULL)
00982                                 /*
00983                                  * Apparently we're using a different switchboard now or
00984                                  * something?  I don't know if this is normal, but it
00985                                  * definitely happens.  So make sure the old switchboard
00986                                  * doesn't still have a reference to us.
00987                                  */
00988                                 slplink->swboard->slplinks = g_list_remove(slplink->swboard->slplinks, slplink);
00989                         slplink->swboard = swboard;
00990                         slplink->swboard->slplinks = g_list_prepend(slplink->swboard->slplinks, slplink);
00991                 }
00992 
00993                 /* If the conversation doesn't exist then this is a custom smiley
00994                  * used in the first message in a MSN conversation: we need to create
00995                  * the conversation now, otherwise the custom smiley won't be shown.
00996                  * This happens because every GtkIMHtml has its own smiley tree: if
00997                  * the conversation doesn't exist then we cannot associate the new
00998                  * smiley with its GtkIMHtml widget. */
00999                 if (!conv) {
01000                         conv = purple_conversation_new(PURPLE_CONV_TYPE_IM, session->account, who);
01001                 }
01002 
01003                 if (purple_conv_custom_smiley_add(conv, smile, "sha1", sha1, TRUE)) {
01004                         msn_slplink_request_object(slplink, smile, got_emoticon, NULL, obj);
01005                 }
01006 
01007                 msn_object_destroy(obj);
01008                 obj =   NULL;
01009                 who =   NULL;
01010                 sha1 = NULL;
01011         }
01012         g_strfreev(tokens);
01013 }
