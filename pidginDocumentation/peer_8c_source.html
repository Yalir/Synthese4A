00290 void
00291 peer_connection_recv_cb(gpointer data, gint source, PurpleInputCondition cond)
00292 {
00293         PeerConnection *conn;
00294         gssize read;
00295 
00296         conn = data;
00297 
00298         /* Start reading a new ODC/OFT frame */
00299         if (conn->buffer_incoming.data == NULL)
00300         {
00301                 /* Read the first 6 bytes (magic string and frame length) */
00302                 read = recv(conn->fd, conn->header + conn->header_received,
00303                                 6 - conn->header_received, 0);
00304 
00305                 /* Check if the remote user closed the connection */
00306                 if (read == 0)
00307                 {
00308                         peer_connection_destroy(conn, OSCAR_DISCONNECT_REMOTE_CLOSED, NULL);
00309                         return;
00310                 }
00311 
00312                 /* If there was an error then close the connection */
00313                 if (read < 0)
00314                 {
00315                         if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00316                                 /* No worries */
00317                                 return;
00318 
00319                         peer_connection_destroy(conn,
00320                                         OSCAR_DISCONNECT_LOST_CONNECTION, g_strerror(errno));
00321                         return;
00322                 }
00323 
00324                 conn->lastactivity = time(NULL);
00325 
00326                 /* If we don't even have the first 6 bytes then do nothing */
00327                 conn->header_received += read;
00328                 if (conn->header_received < 6)
00329                         return;
00330 
00331                 /* All ODC/OFT frames must start with a magic string */
00332                 if (memcmp(conn->magic, conn->header, 4))
00333                 {
00334                         purple_debug_warning("oscar", "Expecting magic string to "
00335                                 "be %c%c%c%c but received magic string %c%c%c%c.  "
00336                                 "Closing connection.\n",
00337                                 conn->magic[0], conn->magic[1], conn->magic[2],
00338                                 conn->magic[3], conn->header[0], conn->header[1],
00339                                 conn->header[2], conn->header[3]);
00340                         peer_connection_destroy(conn, OSCAR_DISCONNECT_INVALID_DATA, NULL);
00341                         return;
00342                 }
00343 
00344                 /* Initialize a new temporary ByteStream for incoming data */
00345                 conn->buffer_incoming.len = aimutil_get16(&conn->header[4]) - 6;
00346                 conn->buffer_incoming.data = g_new(guint8, conn->buffer_incoming.len);
00347                 conn->buffer_incoming.offset = 0;
00348         }
00349 
00350         /* Read data into the temporary buffer until it is complete */
00351         read = recv(conn->fd,
00352                                 &conn->buffer_incoming.data[conn->buffer_incoming.offset],
00353                                 conn->buffer_incoming.len - conn->buffer_incoming.offset,
00354                                 0);
00355 
00356         /* Check if the remote user closed the connection */
00357         if (read == 0)
00358         {
00359                 peer_connection_destroy(conn, OSCAR_DISCONNECT_REMOTE_CLOSED, NULL);
00360                 return;
00361         }
00362 
00363         if (read < 0)
00364         {
00365                 if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00366                         /* No worries */
00367                         return;
00368 
00369                 peer_connection_destroy(conn,
00370                                 OSCAR_DISCONNECT_LOST_CONNECTION, g_strerror(errno));
00371                 return;
00372         }
00373 
00374         conn->lastactivity = time(NULL);
00375         conn->buffer_incoming.offset += read;
00376         if (conn->buffer_incoming.offset < conn->buffer_incoming.len)
00377                 /* Waiting for more data to arrive */
00378                 return;
00379 
00380         /* We have a complete ODC/OFT frame!  Handle it and continue reading */
00381         byte_stream_rewind(&conn->buffer_incoming);
00382         if (conn->type == OSCAR_CAPABILITY_DIRECTIM)
00383         {
00384                 peer_odc_recv_frame(conn, &conn->buffer_incoming);
00385         }
00386         else if (conn->type == OSCAR_CAPABILITY_SENDFILE)
00387         {
00388                 peer_oft_recv_frame(conn, &conn->buffer_incoming);
00389         }
00390 
00391         g_free(conn->buffer_incoming.data);
00392         conn->buffer_incoming.data = NULL;
00393 
00394         conn->header_received = 0;
00395 }

00405 static void
00406 send_cb(gpointer data, gint source, PurpleInputCondition cond)
00407 {
00408         PeerConnection *conn;
00409         gsize writelen;
00410         gssize wrotelen;
00411 
00412         conn = data;
00413         writelen = purple_circ_buffer_get_max_read(conn->buffer_outgoing);
00414 
00415         if (writelen == 0)
00416         {
00417                 purple_input_remove(conn->watcher_outgoing);
00418                 conn->watcher_outgoing = 0;
00419                 /*
00420                  * The buffer is currently empty, so reset the current input
00421                  * and output positions to the start of the buffer.  We do
00422                  * this so that the next chunk of data that we put into the
00423                  * buffer can be read back out of the buffer in one fell swoop.
00424                  * Otherwise it gets fragmented and we have to read from the
00425                  * second half of the buffer than go back and read the rest of
00426                  * the chunk from the first half.
00427                  *
00428                  * We're using TCP, which is a stream based protocol, so this
00429                  * isn't supposed to matter.  However, experience has shown
00430                  * that at least the proxy file transfer code in AIM 6.1.41.2
00431                  * requires that the entire OFT frame arrive all at once.  If
00432                  * the frame is fragmented then AIM freaks out and aborts the
00433                  * file transfer.  Somebody should teach those guys how to
00434                  * write good TCP code.
00435                  */
00436                 conn->buffer_outgoing->inptr = conn->buffer_outgoing->buffer;
00437                 conn->buffer_outgoing->outptr = conn->buffer_outgoing->buffer;
00438                 return;
00439         }
00440 
00441         wrotelen = send(conn->fd, conn->buffer_outgoing->outptr, writelen, 0);
00442         if (wrotelen <= 0)
00443         {
00444                 if (wrotelen < 0 && ((errno == EAGAIN) || (errno == EWOULDBLOCK)))
00445                         /* No worries */
00446                         return;
00447 
00448                 if (conn->ready)
00449                 {
00450                         purple_input_remove(conn->watcher_outgoing);
00451                         conn->watcher_outgoing = 0;
00452                         close(conn->fd);
00453                         conn->fd = -1;
00454                         peer_connection_schedule_destroy(conn,
00455                                         OSCAR_DISCONNECT_LOST_CONNECTION, NULL);
00456                 }
00457                 else
00458                 {
00459                         /*
00460                          * This could happen when unable to send a negotiation
00461                          * frame to a peer proxy server.
00462                          */
00463                         peer_connection_trynext(conn);
00464                 }
00465                 return;
00466         }
00467 
00468         purple_circ_buffer_mark_read(conn->buffer_outgoing, wrotelen);
00469         conn->lastactivity = time(NULL);
00470 }

00477 void
00478 peer_connection_send(PeerConnection *conn, ByteStream *bs)
00479 {
00480         /* Add everything to our outgoing buffer */
00481         purple_circ_buffer_append(conn->buffer_outgoing, bs->data, bs->len);
00482 
00483         /* If we haven't already started writing stuff, then start the cycle */
00484         if ((conn->watcher_outgoing == 0) && (conn->fd >= 0))
00485         {
00486                 conn->watcher_outgoing = purple_input_add(conn->fd,
00487                                 PURPLE_INPUT_WRITE, send_cb, conn);
00488                 send_cb(conn, conn->fd, 0);
00489         }
00490 }

00500 void
00501 peer_connection_finalize_connection(PeerConnection *conn)
00502 {
00503         conn->watcher_incoming = purple_input_add(conn->fd,
00504                         PURPLE_INPUT_READ, peer_connection_recv_cb, conn);
00505 
00506         if (conn->type == OSCAR_CAPABILITY_DIRECTIM)
00507         {
00508                 /*
00509                  * If we are connecting to them then send our cookie so they
00510                  * can verify who we are.  Note: This doesn't seem to be
00511                  * necessary, but it also doesn't seem to hurt.
00512                  */
00513                 if (!(conn->flags & PEER_CONNECTION_FLAG_IS_INCOMING))
00514                         peer_odc_send_cookie(conn);
00515         }
00516         else if (conn->type == OSCAR_CAPABILITY_SENDFILE)
00517         {
00518                 if (purple_xfer_get_type(conn->xfer) == PURPLE_XFER_SEND)
00519                 {
00520                         peer_oft_send_prompt(conn);
00521                 }
00522         }
00523 
00524         /*
00525          * Tell the remote user that we're connected (which may also imply
00526          * that we've accepted their request).
00527          */
00528         if (!(conn->flags & PEER_CONNECTION_FLAG_IS_INCOMING))
00529                 aim_im_sendch2_connected(conn);
00530 }

00602 void
00603 peer_connection_listen_cb(gpointer data, gint source, PurpleInputCondition cond)
00604 {
00605         PeerConnection *conn;
00606         OscarData *od;
00607         PurpleConnection *gc;
00608         struct sockaddr addr;
00609         socklen_t addrlen = sizeof(addr);
00610         int flags;
00611 
00612         conn = data;
00613         od = conn->od;
00614         gc = od->gc;
00615 
00616         purple_debug_info("oscar", "Accepting connection on listener socket.\n");
00617 
00618         conn->fd = accept(conn->listenerfd, &addr, &addrlen);
00619         if (conn->fd < 0)
00620         {
00621                 if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
00622                         /* No connection yet--no worries */
00623                         /* TODO: Hmm, but they SHOULD be connected if we're here, right? */
00624                         return;
00625 
00626                 peer_connection_trynext(conn);
00627                 return;
00628         }
00629 
00630         if ((addr.sa_family != PF_INET) && (addr.sa_family != PF_INET6))
00631         {
00632                 /* Invalid connection type?!  Continue waiting. */
00633                 close(conn->fd);
00634                 return;
00635         }
00636 
00637         flags = fcntl(conn->fd, F_GETFL);
00638         fcntl(conn->fd, F_SETFL, flags | O_NONBLOCK);
00639 #ifndef _WIN32
00640         fcntl(conn->fd, F_SETFD, FD_CLOEXEC);
00641 #endif
00642 
00643         purple_input_remove(conn->watcher_incoming);
00644 
00645         peer_connection_finalize_connection(conn);
00646 }

00656         OscarData *od;

00783 void
00784 peer_connection_trynext(PeerConnection *conn)
00785 {
00786         PurpleAccount *account;
00787 
00788         account = purple_connection_get_account(conn->od->gc);
00789 
00790         /*
00791          * Close any remnants of a previous failed connection attempt.
00792          */
00793         peer_connection_close(conn);
00794 
00795         /*
00796          * 1. Attempt to connect to the remote user using their verifiedip and clientip.
00797          *    We try these at the same time and use whichever succeeds first, so we don't
00798          *    have to wait for a timeout.
00799          */
00800         if (!(conn->flags & PEER_CONNECTION_FLAG_TRIED_DIRECT) &&
00801                 (conn->verifiedip != NULL) && (conn->port != 0) && (!conn->use_proxy))
00802         {
00803                 conn->flags |= PEER_CONNECTION_FLAG_TRIED_DIRECT;
00804 
00805                 if (conn->type == OSCAR_CAPABILITY_DIRECTIM)
00806                 {
00807                         gchar *tmp;
00808                         PurpleConversation *conv;
00809                         tmp = g_strdup_printf(_("Attempting to connect to %s:%hu."),
00810                                         conn->verifiedip, conn->port);
00811                         conv = purple_conversation_new(PURPLE_CONV_TYPE_IM, account, conn->bn);
00812                         purple_conversation_write(conv, NULL, tmp,
00813                                         PURPLE_MESSAGE_SYSTEM, time(NULL));
00814                         g_free(tmp);
00815                 }
00816 
00817                 conn->verified_connect_data = purple_proxy_connect(NULL, account,
00818                                 conn->verifiedip, conn->port,
00819                                 peer_connection_verified_established_cb, conn);
00820 
00821                 if ((conn->verifiedip == NULL) ||
00822                         strcmp(conn->verifiedip, conn->clientip))
00823                 {
00824                         conn->client_connect_data = purple_proxy_connect(NULL, account,
00825                                         conn->clientip, conn->port,
00826                                         peer_connection_client_established_cb, conn);
00827                 }
00828 
00829                 if ((conn->verified_connect_data != NULL) ||
00830                         (conn->client_connect_data != NULL))
00831                 {
00832                         /* Connecting... */
00833                         conn->connect_timeout_timer = purple_timeout_add_seconds(5,
00834                                         peer_connection_tooktoolong, conn);
00835                         return;
00836                 }
00837         }
00838 
00839         /*
00840          * 2. Attempt to have the remote user connect to us (using both
00841          *    our verifiedip and our clientip).
00842          */
00843         if (!(conn->flags & PEER_CONNECTION_FLAG_TRIED_INCOMING) &&
00844                 (!conn->use_proxy))
00845         {
00846                 conn->flags |= PEER_CONNECTION_FLAG_TRIED_INCOMING;
00847 
00848                 /*
00849                  * Remote user is connecting to us, so we'll need to verify
00850                  * that the user who connected is our friend.
00851                  */
00852                 conn->flags |= PEER_CONNECTION_FLAG_IS_INCOMING;
00853 
00854                 conn->listen_data = purple_network_listen_range(5190, 5290, SOCK_STREAM,
00855                                 peer_connection_establish_listener_cb, conn);
00856                 if (conn->listen_data != NULL)
00857                 {
00858                         /* Opening listener socket... */
00859                         return;
00860                 }
00861         }
00862 
00863         /*
00864          * 3. Attempt to have both users connect to an intermediate proxy
00865          *    server.
00866          */
00867         if (!(conn->flags & PEER_CONNECTION_FLAG_TRIED_PROXY))
00868         {
00869                 conn->flags |= PEER_CONNECTION_FLAG_TRIED_PROXY;
00870 
00871                 /*
00872                  * If we initiate the proxy connection, then the remote user
00873                  * could be anyone, so we need to verify that the user who
00874                  * connected is our friend.
00875                  */
00876                 if (!conn->use_proxy)
00877                         conn->flags |= PEER_CONNECTION_FLAG_IS_INCOMING;
00878 
00879                 if (conn->type == OSCAR_CAPABILITY_DIRECTIM)
00880                 {
00881                         gchar *tmp;
00882                         PurpleConversation *conv;
00883                         tmp = g_strdup(_("Attempting to connect via proxy server."));
00884                         conv = purple_conversation_new(PURPLE_CONV_TYPE_IM, account, conn->bn);
00885                         purple_conversation_write(conv, NULL, tmp,
00886                                         PURPLE_MESSAGE_SYSTEM, time(NULL));
00887                         g_free(tmp);
00888                 }
00889 
00890                 conn->verified_connect_data = purple_proxy_connect(NULL, account,
00891                                 (conn->proxyip != NULL)
00892                                         ? conn->proxyip
00893                                         : (conn->od->icq ? ICQ_PEER_PROXY_SERVER : AIM_PEER_PROXY_SERVER),
00894                                 PEER_PROXY_PORT,
00895                                 peer_proxy_connection_established_cb, conn);
00896                 if (conn->verified_connect_data != NULL)
00897                 {
00898                         /* Connecting... */
00899                         return;
00900                 }
00901         }
00902 
00903         /* Give up! */
00904         peer_connection_destroy(conn, OSCAR_DISCONNECT_COULD_NOT_CONNECT, NULL);
00905 }

00910 void
00911 peer_connection_propose(OscarData *od, guint64 type, const char *bn)
00912 {
00913         PeerConnection *conn;
00914 
00915         if (type == OSCAR_CAPABILITY_DIRECTIM)
00916         {
00917                 conn = peer_connection_find_by_type(od, bn, type);
00918                 if (conn != NULL)
00919                 {
00920                         if (conn->ready)
00921                         {
00922                                 PurpleAccount *account;
00923                                 PurpleConversation *conv;
00924 
00925                                 purple_debug_info("oscar", "Already have a direct IM "
00926                                                 "session with %s.\n", bn);
00927                                 account = purple_connection_get_account(od->gc);
00928                                 conv = purple_find_conversation_with_account(PURPLE_CONV_TYPE_IM,
00929                                                 bn, account);
00930                                 if (conv != NULL)
00931                                         purple_conversation_present(conv);
00932                                 return;
00933                         }
00934 
00935                         /* Cancel the old connection and try again */
00936                         peer_connection_destroy(conn, OSCAR_DISCONNECT_RETRYING, NULL);
00937                 }
00938         }
00939 
00940         conn = peer_connection_new(od, type, bn);
00941         conn->flags |= PEER_CONNECTION_FLAG_INITIATED_BY_ME;
00942         conn->flags |= PEER_CONNECTION_FLAG_APPROVED;
00943         aim_icbm_makecookie(conn->cookie);
00944
