00326                         break;

00338 static Encoding
00339 get_encoding (FILE *df)
00340 {
00341         gboolean old_kde = FALSE;
00342         char     buf [BUFSIZ];
00343         gboolean all_valid_utf8 = TRUE;
00344 
00345         while (my_fgets (buf, sizeof (buf), df) != NULL) {
00346                 if (strncmp (PURPLE_DESKTOP_ITEM_ENCODING,
00347                              buf,
00348                              strlen (PURPLE_DESKTOP_ITEM_ENCODING)) == 0) {
00349                         char *p = &buf[strlen (PURPLE_DESKTOP_ITEM_ENCODING)];
00350                         if (*p == ' ')
00351                                 p++;
00352                         if (*p != '=')
00353                                 continue;
00354                         p++;
00355                         if (*p == ' ')
00356                                 p++;
00357                         if (purple_strequal (p, "UTF-8")) {
00358                                 return ENCODING_UTF8;
00359                         } else if (purple_strequal (p, "Legacy-Mixed")) {
00360                                 return ENCODING_LEGACY_MIXED;
00361                         } else {
00362                                 /* According to the spec we're not supposed
00363                                  * to read a file like this */
00364                                 return ENCODING_UNKNOWN;
00365                         }
00366                 } else if (purple_strequal ("[KDE Desktop Entry]", buf)) {
00367                         old_kde = TRUE;
00368                         /* don't break yet, we still want to support
00369                          * Encoding even here */
00370                 }
00371                 if (all_valid_utf8 && ! g_utf8_validate (buf, -1, NULL))
00372                         all_valid_utf8 = FALSE;
00373         }
00374 
00375         if (old_kde)
00376                 return ENCODING_LEGACY_MIXED;
00377 
00378         /* A dilemma, new KDE files are in UTF-8 but have no Encoding
00379          * info, at this time we really can't tell.  The best thing to
00380          * do right now is to just assume UTF-8 if the whole file
00381          * validates as utf8 I suppose */
00382 
00383         if (all_valid_utf8)
00384                 return ENCODING_UTF8;
00385         else
00386                 return ENCODING_LEGACY_MIXED;
00387 }

00483 get_encoding_from_locale (const char *locale)
00484 {
00485         char lang[3];
00486         const char *encoding;
00487         static GHashTable *encodings = NULL;
00488 
00489         if (locale == NULL)
00490                 return NULL;
00491 
00492         /* if locale includes encoding, use it */
00493         encoding = strchr (locale, '.');
00494         if (encoding != NULL) {
00495                 return encoding+1;
00496         }
00497

00499                 encodings = init_encodings ();

00502         encoding = g_hash_table_lookup (encodings, locale);

00504                 return encoding;
00505 
00506         /* Try just the language */
00507         strncpy (lang, locale, 2);
00508         lang[2] = '\0';

00549 static char *
00550 decode_string (const char *value, Encoding encoding, const char *locale)
00551 {
00552         char *retval = NULL;
00553 
00554         /* if legacy mixed, then convert */
00555         if (locale != NULL && encoding == ENCODING_LEGACY_MIXED) {
00556                 const char *char_encoding = get_encoding_from_locale (locale);
00557                 char *utf8_string;
00558                 if (char_encoding == NULL)
00559                         return NULL;
00560                 if (purple_strequal (char_encoding, "ASCII")) {
00561                         return decode_string_and_dup (value);
00562                 }
00563                 utf8_string = g_convert (value, -1, "UTF-8", char_encoding,
00564                                         NULL, NULL, NULL);
00565                 if (utf8_string == NULL)
00566                         return NULL;
00567                 retval = decode_string_and_dup (utf8_string);
00568                 g_free (utf8_string);
00569                 return retval;
00570         /* if utf8, then validate */
00571         } else if (locale != NULL && encoding == ENCODING_UTF8) {
00572                 if ( ! g_utf8_validate (value, -1, NULL))
00573                         /* invalid utf8, ignore this key */
00574                         return NULL;
00575                 return decode_string_and_dup (value);
00576         } else {
00577                 /* Meaning this is not a localized string */
00578                 return decode_string_and_dup (value);
00579         }

00930         encoding = get_encoding (df);

00932                 fclose(df);
00933                 /* spec says, don't read this file */
00934                 printf ("Unknown encoding of .desktop file");
00935 
00936                 return NULL;
00937         }
00938 
00939         /* Rewind since get_encoding goes through the file */
00940         if (fseek(df, 0L, SEEK_SET)) {
00941                 fclose(df);
00942                 /* spec says, don't read this file */
00943                 printf ("fseek() error on .desktop file");
00944                 return NULL;
00945         }
00946 
00947         item = _purple_desktop_item_new ();

00949 
00950         /* Note: location and mtime are filled in by the new_from_file
00951          * function since it has those values */
00952 
00953 #define PURPLE_DESKTOP_ITEM_OVERFLOW (next == &CharBuffer [sizeof(CharBuffer)-1])
00954 
00955         state = FirstBrace;
00956         while ((c = getc (df)) != EOF) {
00957                 if (c == '\r')          /* Ignore Carriage Return */
00958                         continue;
00959

00961 
00962                 case OnSecHeader:
00963                         if (c == ']' || PURPLE_DESKTOP_ITEM_OVERFLOW) {
00964                                 *next = '\0';

00966

00970                                         cur_section->keys = g_list_reverse
00971                                                 (cur_section->keys);
00972                                 }

00974                                         /* Main section */

00976                                         old_kde = TRUE;
00977                                 } else if (purple_strequal(CharBuffer, "Desktop Entry")) {
00978                                         /* Main section */
00979                                         cur_section = NULL;
00980                                 } else {
00981                                         cur_section = g_new0 (Section, 1);
00982                                         cur_section->name =
00983                                                 g_strdup (CharBuffer);
00984                                         cur_section->keys = NULL;
00985                                         item->sections = g_list_prepend
00986                                                 (item->sections, cur_section);
00987                                 }
00988                                 state = IgnoreToEOL;
00989                         } else if (c == '[') {
00990                                 /* FIXME: probably error out instead of ignoring this */
00991                         } else {
00992                                 *next++ = c;
00993                         }
00994                         break;
00995 
00996                 case IgnoreToEOL:
00997                 case IgnoreToEOLFirst:
00998                         if (c == '\n'){
00999                                 if (state == IgnoreToEOLFirst)
01000                                         state = FirstBrace;
01001                                 else
01002                                         state = KeyDef;
01003                                 next = CharBuffer;
01004                         }
01005                         break;
01006 
01007                 case FirstBrace:
01008                 case KeyDef:
01009                 case KeyDefOnKey:
01010                         if (c == '#') {
01011                                 if (state == FirstBrace)
01012                                         state = IgnoreToEOLFirst;
01013                                 else
01014                                         state = IgnoreToEOL;
01015                                 break;
01016                         }

01019                                 state = OnSecHeader;

01023                                 break;

01026                         if (state == FirstBrace)
01027                                 break;

01029                         if ((c == ' ' && state != KeyDefOnKey) || c == '\t')
01030                                 break;
01031 
01032                         if (c == '\n' || PURPLE_DESKTOP_ITEM_OVERFLOW) { /* Abort Definition */
01033                                 next = CharBuffer;
01034                                 state = KeyDef;
01035                                 break;
01036                         }
01037 
01038                         if (c == '=' || PURPLE_DESKTOP_ITEM_OVERFLOW){
01039                                 *next = '\0';
01040 
01041                                 g_free (key);
01042                                 key = g_strdup (CharBuffer);
01043                                 state = KeyValue;
01044                                 next = CharBuffer;
01045                         } else {
01046                                 *next++ = c;
01047                                 state = KeyDefOnKey;
01048                         }
01049                         break;
01050 
01051                 case KeyValue:
01052                         if (PURPLE_DESKTOP_ITEM_OVERFLOW || c == '\n'){
01053                                 *next = '\0';
01054 
01055                                 insert_key (item, cur_section, encoding,
01056                                             key, CharBuffer, old_kde,
01057                                             no_translations);

01060                                 key = NULL;
01061 
01062                                 state = (c == '\n') ? KeyDef : IgnoreToEOL;
01063                                 next = CharBuffer;
01064                         } else {
01065                                 *next++ = c;
01066                         }
01067                         break;
01068 
01069                 } /* switch */
01070 
01071         } /* while ((c = getc_unlocked (f)) != EOF) */
01072         if (c == EOF && state == KeyValue) {
01073                 *next = '\0';
01074 
01075                 insert_key (item, cur_section, encoding,
01076                             key, CharBuffer, old_kde,
01077                             no_translations);

01081         }
01082 
01083 #undef PURPLE_DESKTOP_ITEM_OVERFLOW
01084 
01085         /* keys were inserted in reverse */
01086         if (cur_section != NULL &&
01087             cur_section->keys != NULL) {
01088                 cur_section->keys = g_list_reverse (cur_section->keys);
01089         }
