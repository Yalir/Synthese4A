00907 void jabber_presence_parse(JabberStream *js, xmlnode *packet)
00908 {
00909         const char *type;
00910         JabberBuddyResource *jbr = NULL;
00911         gboolean signal_return, ret;
00912         JabberPresence presence;
00913         xmlnode *child;
00914 
00915         memset(&presence, 0, sizeof(presence));
00916         /* defaults */
00917         presence.state = JABBER_BUDDY_STATE_UNKNOWN;
00918         presence.sent = time(NULL);
00919         /* interesting values */
00920         presence.from = xmlnode_get_attrib(packet, "from");
00921         presence.to   = xmlnode_get_attrib(packet, "to");
00922         type = xmlnode_get_attrib(packet, "type");
00923         presence.type = str_to_presence_type(type);
00924 
00925         presence.jb = jabber_buddy_find(js, presence.from, TRUE);
00926         g_return_if_fail(presence.jb != NULL);
00927 
00928         presence.jid_from = jabber_id_new(presence.from);
00929         if (presence.jid_from == NULL) {
00930                 purple_debug_error("jabber", "Ignoring presence with malformed 'from' "
00931                                    "JID: %s\n", presence.from);
00932                 return;
00933         }
00934 
00935         signal_return = GPOINTER_TO_INT(purple_signal_emit_return_1(purple_connection_get_prpl(js->gc),
00936                         "jabber-receiving-presence", js->gc, type, presence.from, packet));
00937         if (signal_return) {
00938                 goto out;
00939         }
00940 
00941         if (presence.jid_from->node)
00942                 presence.chat = jabber_chat_find(js, presence.jid_from->node,
00943                                                  presence.jid_from->domain);
00944         if(presence.jb->error_msg) {
00945                 g_free(presence.jb->error_msg);
00946                 presence.jb->error_msg = NULL;
00947         }
00948 
00949         if (presence.type == JABBER_PRESENCE_AVAILABLE) {
00950                 presence.state = JABBER_BUDDY_STATE_ONLINE;
00951         } else if (presence.type == JABBER_PRESENCE_ERROR) {
00952                 /* TODO: Is this handled properly?  Should it be treated as per-jbr? */
00953                 char *msg = jabber_parse_error(js, packet, NULL);
00954                 presence.state = JABBER_BUDDY_STATE_ERROR;
00955                 presence.jb->error_msg = msg ? msg : g_strdup(_("Unknown Error in presence"));
00956         } else if (presence.type == JABBER_PRESENCE_SUBSCRIBE) {
00957                 /* TODO: Move to handle_subscribe() (so nick is extracted by the
00958                  * PresenceHandler */
00959                 struct _jabber_add_permit *jap = g_new0(struct _jabber_add_permit, 1);
00960                 gboolean onlist = FALSE;
00961                 PurpleAccount *account;
00962                 PurpleBuddy *buddy;
00963                 xmlnode *nick;
00964 
00965                 account = purple_connection_get_account(js->gc);
00966                 buddy = purple_find_buddy(account, presence.from);
00967                 nick = xmlnode_get_child_with_namespace(packet, "nick", "http://jabber.org/protocol/nick");
00968                 if (nick)
00969                         presence.nickname = xmlnode_get_data(nick);
00970 
00971                 if (buddy) {
00972                         if ((presence.jb->subscription & (JABBER_SUB_TO | JABBER_SUB_PENDING)))
00973                                 onlist = TRUE;
00974                 }
00975 
00976                 jap->gc = js->gc;
00977                 jap->who = g_strdup(presence.from);
00978                 jap->js = js;
00979 
00980                 purple_account_request_authorization(account, presence.from, NULL, presence.nickname,
00981                                 NULL, onlist, authorize_add_cb, deny_add_cb, jap);
00982 
00983                 goto out;
00984         } else if (presence.type == JABBER_PRESENCE_SUBSCRIBED) {
00985                 /* This case (someone has approved our subscribe request) is handled
00986                  * by the roster push the server sends along with this.
00987                  */
00988                 goto out;
00989         } else if (presence.type == JABBER_PRESENCE_UNSUBSCRIBE) {
00990                 /* XXX I'm not sure this is the right way to handle this, it
00991                  * might be better to add "unsubscribe" to the presence status
00992                  * if lower down, but I'm not sure. */
00993                 /* they are unsubscribing from our presence, we don't care */
00994                 /* Well, maybe just a little, we might want/need to start
00995                  * acknowledging this (and the others) at some point. */
00996                 goto out;
00997         } else if (presence.type == JABBER_PRESENCE_PROBE) {
00998                 purple_debug_warning("jabber", "Ignoring presence probe\n");
00999                 goto out;
01000         } else if (presence.type == JABBER_PRESENCE_UNAVAILABLE) {
01001                 presence.state = JABBER_BUDDY_STATE_UNAVAILABLE;
01002         } else if (presence.type == JABBER_PRESENCE_UNSUBSCRIBED) {
01003                 presence.state = JABBER_BUDDY_STATE_UNKNOWN;
01004         } else {
01005                 purple_debug_warning("jabber", "Ignoring presence with invalid type "
01006                                      "'%s'\n", type);
01007                 goto out;
01008         }
01009 
01010         for (child = packet->child; child; child = child->next) {
01011                 const char *xmlns;
01012                 char *key;
01013                 JabberPresenceHandler *pih;
01014                 if (child->type != XMLNODE_TYPE_TAG)
01015                         continue;
01016 
01017                 xmlns = xmlnode_get_namespace(child);
01018                 key = g_strdup_printf("%s %s", child->name, xmlns ? xmlns : "");
01019                 pih = g_hash_table_lookup(presence_handlers, key);
01020                 g_free(key);
01021                 if (pih)
01022                         pih(js, &presence, child);
01023         }
01024 
01025         if (presence.delayed && presence.idle) {
01026                 /* Delayed and idle, so update idle time */
01027                 presence.idle = presence.idle + (time(NULL) - presence.sent);
01028         }
01029 
01030         /* TODO: Handle tracking jb(r) here? */
01031 
01032         if (presence.chat)
01033                 ret = handle_presence_chat(js, &presence, packet);
01034         else
01035                 ret = handle_presence_contact(js, &presence);
01036         if (!ret)
01037                 goto out;
01038 
01039         if (presence.caps && presence.type == JABBER_PRESENCE_AVAILABLE) {
01040                 /* handle Entity Capabilities (XEP-0115) */
01041                 const char *node = xmlnode_get_attrib(presence.caps, "node");
01042                 const char *ver  = xmlnode_get_attrib(presence.caps, "ver");
01043                 const char *hash = xmlnode_get_attrib(presence.caps, "hash");
01044                 const char *ext  = xmlnode_get_attrib(presence.caps, "ext");
01045 
01046                 /* v1.3 uses: node, ver, and optionally ext.
01047                  * v1.5 uses: node, ver, and hash. */
01048                 if (node && *node && ver && *ver) {
01049                         gchar **exts = ext && *ext ? g_strsplit(ext, " ", -1) : NULL;
01050                         jbr = jabber_buddy_find_resource(presence.jb, presence.jid_from->resource);
01051 
01052                         /* Look it up if we don't already have all this information */
01053                         if (!jbr || !jbr->caps.info ||
01054                                         !g_str_equal(node, jbr->caps.info->tuple.node) ||
01055                                         !g_str_equal(ver, jbr->caps.info->tuple.ver) ||
01056                                         !purple_strequal(hash, jbr->caps.info->tuple.hash) ||
01057                                         !jabber_caps_exts_known(jbr->caps.info, (gchar **)exts)) {
01058                                 JabberPresenceCapabilities *userdata = g_new0(JabberPresenceCapabilities, 1);
01059                                 userdata->js = js;
01060                                 userdata->jb = presence.jb;
01061                                 userdata->from = g_strdup(presence.from);
01062                                 jabber_caps_get_info(js, presence.from, node, ver, hash, exts,
01063                                     (jabber_caps_get_info_cb)jabber_presence_set_capabilities,
01064                                     userdata);
01065                         } else {
01066                                 if (exts)
01067                                         g_strfreev(exts);
01068                         }
01069                 }
01070         }
01071 
01072 out:
01073         while (presence.chat_info.codes)
01074                 presence.chat_info.codes =
01075                         g_slist_delete_link(presence.chat_info.codes,
01076                                             presence.chat_info.codes);
01077 
01078         g_free(presence.status);
01079         g_free(presence.vcard_avatar_hash);
01080         g_free(presence.nickname);
01081         jabber_id_free(presence.jid_from);
01082 }

01224 {
