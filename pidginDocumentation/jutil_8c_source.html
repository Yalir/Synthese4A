00073                         domain_len = slash - (at + 1);

00319 {
00320         const char *at = NULL;
00321         const char *slash = NULL;
00322         const char *c;
00323         gboolean needs_validation = FALSE;
00324 #if 0
00325         gboolean node_is_required = FALSE;
00326 #endif
00327 #ifndef USE_IDN
00328         char *node = NULL;
00329         char *domain;
00330 #endif
00331         JabberID *jid;
00332 
00333         if (!str)
00334                 return NULL;
00335 
00336         for (c = str; *c != '\0'; c++)
00337         {
00338                 switch (*c) {
00339                         case '@':
00340                                 if (!slash) {
00341                                         if (at) {
00342                                                 /* Multiple @'s in the node/domain portion, not a valid JID! */
00343                                                 return NULL;
00344                                         }
00345                                         if (c == str) {
00346                                                 /* JIDs cannot start with @ */
00347                                                 return NULL;
00348                                         }
00349                                         if (c[1] == '\0') {
00350                                                 /* JIDs cannot end with @ */
00351                                                 return NULL;
00352                                         }
00353                                         at = c;
00354                                 }
00355                                 break;
00356 
00357                         case '/':
00358                                 if (!slash) {
00359                                         if (c == str) {
00360                                                 /* JIDs cannot start with / */
00361                                                 return NULL;
00362                                         }
00363                                         if (c[1] == '\0' && !allow_terminating_slash) {
00364                                                 /* JIDs cannot end with / */
00365                                                 return NULL;
00366                                         }
00367                                         slash = c;
00368                                 }
00369                                 break;
00370 
00371                         default:
00372                                 /* characters allowed everywhere */
00373                                 if ((*c >= 'a' && *c <= 'z')
00374                                                 || (*c >= '0' && *c <= '9')
00375                                                 || (*c >= 'A' && *c <= 'Z')
00376                                                 || *c == '.' || *c == '-')
00377                                         /* We're good */
00378                                         break;
00379 
00380 #if 0
00381                                 if (slash != NULL) {
00382                                         /* characters allowed only in the resource */
00383                                         if (implement_me)
00384                                                 /* We're good */
00385                                                 break;
00386                                 }
00387 
00388                                 /* characters allowed only in the node */
00389                                 if (implement_me) {
00390                                         /*
00391                                          * Ok, this character is valid, but only if it's a part
00392                                          * of the node and not the domain.  But we don't know
00393                                          * if "c" is a part of the node or the domain until after
00394                                          * we've found the @.  So set a flag for now and check
00395                                          * that we found an @ later.
00396                                          */
00397                                         node_is_required = TRUE;
00398                                         break;
00399                                 }
00400 #endif
00401 
00402                                 /*
00403                                  * Hmm, this character is a bit more exotic.  Better fall
00404                                  * back to using the more expensive UTF-8 compliant
00405                                  * stringprep functions.
00406                                  */
00407                                 needs_validation = TRUE;
00408                                 break;
00409                 }
00410         }
00411 
00412 #if 0
00413         if (node_is_required && at == NULL)
00414                 /* Found invalid characters in the domain */
00415                 return NULL;
00416 #endif
00417 
00418         if (!needs_validation) {
00419                 /* JID is made of only ASCII characters--just lowercase and return */
00420                 jid = g_new0(JabberID, 1);
00421 
00422                 if (at) {
00423                         jid->node = g_ascii_strdown(str, at - str);
00424                         if (slash) {
00425                                 jid->domain = g_ascii_strdown(at + 1, slash - (at + 1));
00426                                 if (*(slash + 1))
00427                                         jid->resource = g_strdup(slash + 1);
00428                         } else {
00429                                 jid->domain = g_ascii_strdown(at + 1, -1);
00430                         }
00431                 } else {
00432                         if (slash) {
00433                                 jid->domain = g_ascii_strdown(str, slash - str);
00434                                 if (*(slash + 1))
00435                                         jid->resource = g_strdup(slash + 1);
00436                         } else {
00437                                 jid->domain = g_ascii_strdown(str, -1);
00438                         }
00439                 }
00440                 return jid;
00441         }
00442 
00443         /*
00444          * If we get here, there are some non-ASCII chars in the string, so
00445          * we'll need to validate it, normalize, and finally do a full jabber

00447          */
00448 
00449         if (!g_utf8_validate(str, -1, NULL))
00450                 return NULL;
00451 
00452 #ifdef USE_IDN
00453         return jabber_idn_validate(str, at, slash, c /* points to the null */);
00454 #else /* USE_IDN */

00456         jid = g_new0(JabberID, 1);
00457 
00458         /* normalization */
00459         if(at) {
00460                 node = g_utf8_casefold(str, at-str);
00461                 if(slash) {
00462                         domain = g_utf8_casefold(at+1, slash-(at+1));
00463                         if (*(slash + 1))
00464                                 jid->resource = g_utf8_normalize(slash+1, -1, G_NORMALIZE_NFKC);

00466                         domain = g_utf8_casefold(at+1, -1);
00467                 }
00468         } else {
00469                 if(slash) {
00470                         domain = g_utf8_casefold(str, slash-str);
00471                         if (*(slash + 1))
00472                                 jid->resource = g_utf8_normalize(slash+1, -1, G_NORMALIZE_NFKC);
00473                 } else {
00474                         domain = g_utf8_casefold(str, -1);
00475                 }
00476         }
00477 
00478         if (node) {
00479                 jid->node = g_utf8_normalize(node, -1, G_NORMALIZE_NFKC);
00480                 g_free(node);
00481         }
00482 
00483         if (domain) {
00484                 jid->domain = g_utf8_normalize(domain, -1, G_NORMALIZE_NFKC);

00486         }
00487

00494         }
00495 
00496         return jid;
