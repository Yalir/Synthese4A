00034 static gboolean
00035 parse_from_attrib_and_find_buddy(BonjourJabberConversation *bconv, int nb_attributes, const xmlChar **attributes) {
00036         int i;
00037 
00038         /* If the "from" attribute is specified, attach it to the conversation. */
00039         for(i=0; i < nb_attributes * 5; i+=5) {
00040                 if(!xmlStrcmp(attributes[i], (xmlChar*) "from")) {
00041                         int len = attributes[i+4] - attributes[i+3];
00042                         bconv->buddy_name = g_strndup((char *)attributes[i+3], len);
00043                         bonjour_jabber_conv_match_by_name(bconv);
00044 
00045                         return (bconv->pb != NULL);
00046                 }
00047         }
00048 
00049         return FALSE;
00050 }

00052 static void
00053 bonjour_parser_element_start_libxml(void *user_data,
00054                                    const xmlChar *element_name, const xmlChar *prefix, const xmlChar *namespace,
00055                                    int nb_namespaces, const xmlChar **namespaces,
00056                                    int nb_attributes, int nb_defaulted, const xmlChar **attributes)
00057 {
00058         BonjourJabberConversation *bconv = user_data;
00059 
00060         xmlnode *node;
00061         int i;
00062 
00063         g_return_if_fail(element_name != NULL);
00064 
00065         if(!xmlStrcmp(element_name, (xmlChar*) "stream")) {
00066                 if(!bconv->recv_stream_start) {
00067                         bconv->recv_stream_start = TRUE;
00068 
00069                         if (bconv->pb == NULL)
00070                                 parse_from_attrib_and_find_buddy(bconv, nb_attributes, attributes);
00071 
00072                         bonjour_jabber_stream_started(bconv);
00073                 }
00074         } else {
00075 
00076                 /* If we haven't yet attached a buddy and this isn't "<stream:features />",
00077                  * try to get a "from" attribute as a last resort to match our buddy. */
00078                 if(bconv->pb == NULL
00079                                 && !(prefix && !xmlStrcmp(prefix, (xmlChar*) "stream")
00080                                         && !xmlStrcmp(element_name, (xmlChar*) "features"))
00081                                 && !parse_from_attrib_and_find_buddy(bconv, nb_attributes, attributes))
00082                         /* We've run out of options for finding who the conversation is from
00083                            using explicitly specified stuff; see if we can make a good match
00084                            by using the IP */
00085                         bonjour_jabber_conv_match_by_ip(bconv);
00086 
00087                 if(bconv->current)
00088                         node = xmlnode_new_child(bconv->current, (const char*) element_name);
00089                 else
00090                         node = xmlnode_new((const char*) element_name);
00091                 xmlnode_set_namespace(node, (const char*) namespace);
00092 
00093                 for(i=0; i < nb_attributes * 5; i+=5) {
00094                         const char *name = (const char *)attributes[i];
00095                         const char *prefix = (const char *)attributes[i+1];
00096                         const char *attrib_ns = (const char *)attributes[i+2];
00097                         char *txt;
00098                         int attrib_len = attributes[i+4] - attributes[i+3];
00099                         char *attrib = g_malloc(attrib_len + 1);
00100 
00101                         memcpy(attrib, attributes[i+3], attrib_len);
00102                         attrib[attrib_len] = '\0';
00103 
00104                         txt = attrib;
00105                         attrib = purple_unescape_text(txt);
00106                         g_free(txt);
00107                         xmlnode_set_attrib_full(node, name, attrib_ns, prefix, attrib);
00108                         g_free(attrib);
00109                 }
00110 
00111                 bconv->current = node;
00112         }
00113 }

00115 static void
00116 bonjour_parser_element_end_libxml(void *user_data, const xmlChar *element_name,
00117                                  const xmlChar *prefix, const xmlChar *namespace)
00118 {
00119         BonjourJabberConversation *bconv = user_data;
00120 
00121         if(!bconv->current) {
00122                 /* We don't keep a reference to the start stream xmlnode,
00123                  * so we have to check for it here to close the conversation */
00124                 if(!xmlStrcmp(element_name, (xmlChar*) "stream"))
00125                         /* Asynchronously close the conversation to prevent bonjour_parser_setup()
00126                          * being called from within this context */
00127                         async_bonjour_jabber_close_conversation(bconv);
00128                 return;
00129         }
00130 
00131         if(bconv->current->parent) {
00132                 if(!xmlStrcmp((xmlChar*) bconv->current->name, element_name))
00133                         bconv->current = bconv->current->parent;
00134         } else {
00135                 xmlnode *packet = bconv->current;
00136                 bconv->current = NULL;
00137                 bonjour_jabber_process_packet(bconv->pb, packet);
00138                 xmlnode_free(packet);
00139         }
00140 }

00203 void
00204 bonjour_parser_setup(BonjourJabberConversation *bconv)
00205 {
00206 
00207         /* This seems backwards, but it makes sense. The libxml code creates
00208          * the parser context when you try to use it (this way, it can figure
00209          * out the encoding at creation time. So, setting up the parser is
00210          * just a matter of destroying any current parser. */
00211         if (bconv->context) {
00212                 xmlParseChunk(bconv->context, NULL,0,1);
00213                 xmlFreeParserCtxt(bconv->context);
00214                 bconv->context = NULL;
00215         }
00216 }
