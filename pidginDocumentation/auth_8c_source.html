00167                 JabberSaslMech *possible = l->data;

00171                         js->auth_mech = possible;

00175                 /* Can we find this mechanism in the server's list? */

00189                 purple_connection_error_reason(js->gc,
00190                                 PURPLE_CONNECTION_ERROR_AUTHENTICATION_IMPOSSIBLE,

00196         if (state == JABBER_SASL_STATE_FAIL) {
00197                 purple_connection_error_reason(js->gc,
00198                                 PURPLE_CONNECTION_ERROR_AUTHENTICATION_IMPOSSIBLE,
00199                                 msg ? msg : _("Unknown Error"));

00355                 jabber_register_start(js);
00356                 return;

00360          * IQ Auth doesn't have support for resource binding, so we need to pick a
00361          * default resource so it will work properly.  jabberd14 throws an error and

00364         if (!js->user->resource || *js->user->resource == '\0') {
00365                 g_free(js->user->resource);

00376                 purple_account_request_password(account, G_CALLBACK(auth_old_pass_cb), G_CALLBACK(auth_no_pass_cb), js->gc);
