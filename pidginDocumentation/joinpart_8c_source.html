00079 static gboolean should_hide_notice(PurpleConversation *conv, const char *name,
00080                                    GHashTable *users)
00081 {
00082         PurpleConvChat *chat;
00083         int threshold;
00084         struct joinpart_key key;
00085         time_t *last_said;
00086 
00087         g_return_val_if_fail(conv != NULL, FALSE);
00088         g_return_val_if_fail(purple_conversation_get_type(conv) == PURPLE_CONV_TYPE_CHAT, FALSE);
00089 
00090         /* If the room is small, don't bother. */
00091         chat = PURPLE_CONV_CHAT(conv);
00092         threshold = purple_prefs_get_int(THRESHOLD_PREF);
00093         if (g_list_length(purple_conv_chat_get_users(chat)) < threshold)
00094                 return FALSE;
00095 
00096         if (!purple_prefs_get_bool(HIDE_BUDDIES_PREF) &&
00097             purple_find_buddy(purple_conversation_get_account(conv), name))
00098                 return FALSE;
00099 
00100         /* Only show the notice if the user has spoken recently. */
00101         key.conv = conv;
00102         key.user = (gchar *)name;
00103         last_said = g_hash_table_lookup(users, &key);
00104         if (last_said != NULL)
00105         {
00106                 int delay = purple_prefs_get_int(DELAY_PREF);
00107                 if (delay > 0 && (*last_said + (delay * 60)) >= time(NULL))
00108                         return FALSE;
00109         }
00110 
00111         return TRUE;
00112 }

00178         void *conv_handle;
00179         GHashTable *users;

00182 
00183         users = g_hash_table_new_full((GHashFunc)joinpart_key_hash,

00190                             PURPLE_CALLBACK(chat_buddy_joining_cb), users);
00191         purple_signal_connect(conv_handle, "chat-buddy-leaving", plugin,
00192                             PURPLE_CALLBACK(chat_buddy_leaving_cb), users);
00193         purple_signal_connect(conv_handle, "received-chat-msg", plugin,
00194                             PURPLE_CALLBACK(received_chat_msg_cb), users);
00195 
00196         /* Cleanup every 5 minutes */
00197         id = purple_timeout_add_seconds(60 * 5, (GSourceFunc)clean_users_hash, users);
00198 
00199         data = g_new(gpointer, 2);
00200         data[0] = users;
00201         data[1] = GUINT_TO_POINTER(id);
00202         plugin->extra = data;
