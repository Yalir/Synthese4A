00028 /* When you ask other people for authorization */
00029 void
00030 oscar_auth_sendrequest(PurpleConnection *gc, const char *bname, const char *msg)
00031 {
00032         OscarData *od;
00033         PurpleAccount *account;
00034         PurpleBuddy *buddy;
00035         PurpleGroup *group;
00036         const char *gname;
00037 
00038         od = purple_connection_get_protocol_data(gc);
00039         account = purple_connection_get_account(gc);
00040         buddy = purple_find_buddy(account, bname);
00041         if (buddy != NULL)
00042                 group = purple_buddy_get_group(buddy);
00043         else
00044                 group = NULL;
00045 
00046         if (group != NULL)
00047         {
00048                 gname = purple_group_get_name(group);
00049                 purple_debug_info("oscar", "ssi: adding buddy %s to group %s\n",
00050                                    bname, gname);
00051                 aim_ssi_sendauthrequest(od, bname, msg ? msg : _("Please authorize me so I can add you to my buddy list."));
00052                 if (!aim_ssi_itemlist_finditem(od->ssi.local, gname, bname, AIM_SSI_TYPE_BUDDY))
00053                 {
00054                         aim_ssi_addbuddy(od, bname, gname, NULL, purple_buddy_get_alias_only(buddy), NULL, NULL, TRUE);
00055 
00056                         /* Mobile users should always be online */
00057                         if (bname[0] == '+') {
00058                                 purple_prpl_got_user_status(account,
00059                                                 purple_buddy_get_name(buddy),
00060                                                 OSCAR_STATUS_ID_AVAILABLE, NULL);
00061                                 purple_prpl_got_user_status(account,
00062                                                 purple_buddy_get_name(buddy),
00063                                                 OSCAR_STATUS_ID_MOBILE, NULL);
00064                         }
00065                 }
00066         }
00067 }
