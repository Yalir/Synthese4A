00021 /*
00022  * Family 0x0008 - Popups.
00023  *
00024  * Popups are just what it sounds like.  They're a way for the server to
00025  * open up an informative box on the client's screen.

00030 /*
00031  * This is all there is to it.

00036 static int
00037 parsepopup(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00038 {
00039         aim_rxcallback_t userfunc;
00040         GSList *tlvlist;
00041         int ret = 0;
00042         char *msg, *url;
00043         guint16 width, height, delay;
00044 
00045         tlvlist = aim_tlvlist_read(bs);
00046 
00047         msg = aim_tlv_getstr(tlvlist, 0x0001, 1);
00048         url = aim_tlv_getstr(tlvlist, 0x0002, 1);
00049         width = aim_tlv_get16(tlvlist, 0x0003, 1);
00050         height = aim_tlv_get16(tlvlist, 0x0004, 1);
00051         delay = aim_tlv_get16(tlvlist, 0x0005, 1);
00052 
00053         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00054                 ret = userfunc(od, conn, frame, msg, url, width, height, delay);
00055 
00056         aim_tlvlist_free(tlvlist);
00057         g_free(msg);
00058         g_free(url);
00059 
00060         return ret;
00061 }

00063 static int
00064 snachandler(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00065 {
00066         if (snac->subtype == 0x0002)
00067                 return parsepopup(od, conn, mod, frame, snac, bs);
00068 
00069         return 0;
00070 }
