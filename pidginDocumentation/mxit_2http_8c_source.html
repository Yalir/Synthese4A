00058 /*------------------------------------------------------------------------
00059  * Write the request to the HTTP server.
00060  *
00061  *  @param fd                   The file descriptor
00062  *  @param pktdata              The packet data
00063  *  @param pktlen               The length of the packet data
00064  *  @return                             Return -1 on error, otherwise 0
00065  */
00066 static int mxit_http_raw_write( int fd, const char* pktdata, int pktlen )
00067 {
00068         int             written;
00069         int             res;
00070 
00071         written = 0;
00072         while ( written < pktlen ) {
00073                 res = write( fd, &pktdata[written], pktlen - written );
00074                 if ( res <= 0 ) {
00075                         /* error on socket */
00076                         if ( errno == EAGAIN )
00077                                 continue;
00078 
00079                         purple_debug_error( MXIT_PLUGIN_ID, "Error while writing packet to HTTP server (%i)\n", res );
00080                         return -1;
00081                 }
00082                 written += res;
00083         }
00084 
00085         return 0;
00086 }

00264 /*------------------------------------------------------------------------
00265  * Callback invoked once the connection has been established to the HTTP server,
00266  * or on connection failure.
00267  *
00268  *  @param user_data            The MXit session object
00269  *  @param source                       The file-descriptor associated with the connection
00270  *  @param error_message        Message explaining why the connection failed
00271  */
00272 static void mxit_cb_http_connect( gpointer user_data, gint source, const gchar* error_message )
00273 {
00274         struct http_request*    req     = (struct http_request*) user_data;
00275 
00276         purple_debug_info( MXIT_PLUGIN_ID, "mxit_cb_http_connect\n" );
00277 
00278         /* source is the file descriptor of the new connection */
00279         if ( source < 0 ) {
00280                 purple_debug_info( MXIT_PLUGIN_ID, "mxit_cb_http_connect failed: %s\n", error_message );
00281                 purple_connection_error( req->session->con, _( "Unable to connect to the MXit HTTP server. Please check your server settings." ) );
00282                 return;
00283         }
00284 
00285         /* we now have an open and active TCP connection to the mxit server */
00286         req->session->fd = source;
00287 
00288         /* reset the receive buffer */
00289         req->session->rx_state = RX_STATE_RLEN;
00290         req->session->rx_lbuf[0] = '\0';
00291         req->session->rx_i = 0;
00292         req->session->rx_res = 0;
00293 
00294         /* start listening on the open connection for messages from the server (reference: "libpurple/eventloop.h") */
00295         req->session->http_handler = purple_input_add( req->session->fd, PURPLE_INPUT_READ, mxit_cb_http_read, req->session );
00296 
00297         /* actually send the request to the HTTP server */
00298         mxit_http_raw_write( req->session->fd, req->data, req->datalen );
00299 
00300         /* free up resources */
00301         free_http_request( req );
00302         req = NULL;
00303 }
