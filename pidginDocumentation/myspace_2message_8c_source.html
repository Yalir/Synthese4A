01003 MsimMessage *
01004 msim_parse(const gchar *raw)
01005 {
01006         MsimMessage *msg;
01007         gchar *token;
01008         gchar **tokens;
01009         gchar *key;
01010         gchar *value;
01011         int i;
01012 
01013         g_return_val_if_fail(raw != NULL, NULL);
01014 
01015         purple_debug_info("msim", "msim_parse: got <%s>\n", raw);
01016 
01017         key = NULL;
01018 
01019         /* All messages begin with a \. */
01020         if (raw[0] != '\\' || raw[1] == 0) {
01021                 purple_debug_info("msim", "msim_parse: incomplete/bad string, "
01022                                 "missing initial backslash: <%s>\n", raw);
01023                 /* XXX: Should we try to recover, and read to first backslash? */
01024 
01025                 return NULL;
01026         }
01027 
01028         msg = msim_msg_new(FALSE);
01029 
01030         for (tokens = g_strsplit(raw + 1, "\\", 0), i = 0;
01031                         (token = tokens[i]);
01032                         i++) {
01033 #ifdef MSIM_DEBUG_PARSE
01034                 purple_debug_info("msim", "tok=<%s>, i%2=%d\n", token, i % 2);
01035 #endif
01036                 if (i % 2) {
01037                         /* Odd-numbered ordinal is a value. */
01038 
01039                         value = token;
01040 
01041                         /* Incoming protocol messages get tagged as MSIM_TYPE_RAW, which
01042                          * represents an untyped piece of data. msim_msg_get_* will
01043                          * convert to appropriate types for caller, and handle unescaping if needed. */
01044                         msg = msim_msg_append_dynamic_name(msg, g_strdup(key), MSIM_TYPE_RAW, g_strdup(value));
01045 #ifdef MSIM_DEBUG_PARSE
01046                         purple_debug_info("msim", "insert string: |%s|=|%s|\n", key, value);
01047 #endif
01048                 } else {
01049                         /* Even numbered indexes are key names. */
01050                         key = token;
01051                 }
01052         }
01053         g_strfreev(tokens);
01054 
01055         return msg;
01056 }

01338 static gboolean
01339 msim_msg_get_binary_from_element(MsimMessageElement *elem, gchar **binary_data, gsize *binary_length)
01340 {
01341         GString *gs;
01342 
01343         g_return_val_if_fail(elem != NULL, FALSE);
01344 
01345         switch (elem->type) {
01346                 case MSIM_TYPE_RAW:
01347                          /* Incoming messages are tagged with MSIM_TYPE_RAW, and
01348                          * converted appropriately. They can still be "strings", just they won't
01349                          * be tagged as MSIM_TYPE_STRING (as MSIM_TYPE_STRING is intended to be used
01350                          * by msimprpl code for things like instant messages - stuff that should be
01351                          * escaped if needed). DWIM.
01352                          */
01353 
01354                         /* Previously, incoming messages were stored as MSIM_TYPE_STRING.
01355                          * This was fine for integers and strings, since they can easily be
01356                          * converted in msim_get_*, as desirable. However, it does not work
01357                          * well for binary strings. Consider:
01358                          *
01359                          * If incoming base64'd elements were tagged as MSIM_TYPE_STRING.
01360                          * msim_msg_get_binary() sees MSIM_TYPE_STRING, base64 decodes, returns.
01361                          * everything is fine.
01362                          * But then, msim_send() is called on the incoming message, which has
01363                          * a base64'd MSIM_TYPE_STRING that really is encoded binary. The values
01364                          * will be escaped since strings are escaped, and / becomes /2; no good.
01365                          *
01366                          */
01367                         *binary_data = (gchar *)purple_base64_decode((const gchar *)elem->data, binary_length);
01368                         return ((*binary_data) != NULL);
01369 
01370                 case MSIM_TYPE_BINARY:
01371                         gs = (GString *)elem->data;
01372 
01373                         /* Duplicate data, so caller can g_free() it. */
01374                         *binary_data = g_memdup(gs->str, gs->len);
01375                         *binary_length = gs->len;
01376 
01377                         return TRUE;
01378 
01379 
01380                         /* Rejected because if it isn't already a GString, have to g_new0 it and
01381                          * then caller has to ALSO free the GString!
01382                          *
01383                          * return (GString *)elem->data; */
01384 
01385                 default:
01386                         purple_debug_info("msim", "msim_msg_get_binary: unhandled type %d for key %s\n",
01387                                         elem->type, elem->name ? elem->name : "(NULL)");
01388                         return FALSE;
01389         }
01390 }
