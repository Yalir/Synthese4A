01865 static gboolean
01866 create_stream(PurpleMediaBackendFs2 *self,
01867                 const gchar *sess_id, const gchar *who,
01868                 PurpleMediaSessionType type, gboolean initiator,
01869                 const gchar *transmitter,
01870                 guint num_params, GParameter *params)
01871 {
01872         PurpleMediaBackendFs2Private *priv =
01873                         PURPLE_MEDIA_BACKEND_FS2_GET_PRIVATE(self);
01874         GError *err = NULL;
01875         FsStream *fsstream = NULL;
01876         const gchar *stun_ip = purple_network_get_stun_ip();
01877         const gchar *turn_ip = purple_network_get_turn_ip();
01878         guint _num_params = num_params;
01879         GParameter *_params;

01881                         session_type_to_fs_stream_direction(type);
01882         PurpleMediaBackendFs2Session *session;
01883         PurpleMediaBackendFs2Stream *stream;
01884         FsParticipant *participant;
01885         /* check if the prpl has already specified a relay-info
01886           we need to do this to allow them to override when using non-standard
01887           TURN modes, like Google f.ex. */
01888         gboolean got_turn_from_prpl = FALSE;
01889         int i;
01890 
01891         session = get_session(self, sess_id);
01892 
01893         if (session == NULL) {
01894                 purple_debug_error("backend-fs2",
01895                                 "Couldn't find session to create stream.\n");
01896                 return FALSE;
01897         }
01898 
01899         participant = get_participant(self, who);
01900 
01901         if (participant == NULL) {
01902                 purple_debug_error("backend-fs2", "Couldn't find "
01903                                 "participant to create stream.\n");
01904                 return FALSE;
01905         }
01906 
01907 #ifndef HAVE_FARSIGHT

01910                         (type_direction & FS_DIRECTION_RECV), &err);

01914                         purple_debug_error("backend-fs2",
01915                                         "Error creating stream: %s\n",

01917                                         err->message : "NULL");

01919                 } else
01920                         purple_debug_error("backend-fs2",
01921                                         "Error creating stream\n");
01922                 return FALSE;
01923         }
01924 #endif
01925 
01926         for (i = 0 ; i < num_params ; i++) {
01927                 if (purple_strequal(params[i].name, "relay-info")) {
01928                         got_turn_from_prpl = TRUE;
01929                         break;
01930                 }
01931         }
01932

01935

01938         g_value_init(&_params[_num_params].value, G_TYPE_BOOLEAN);
01939         g_value_set_boolean(&_params[_num_params].value, initiator);
01940         ++_num_params;

01944                         "Setting stun-ip on new stream: %s\n", stun_ip);
01945

01947                 g_value_init(&_params[_num_params].value, G_TYPE_STRING);
01948                 g_value_set_string(&_params[_num_params].value, stun_ip);
01949                 ++_num_params;
01950         }

01955                 const gchar *username = purple_prefs_get_string(
01956                                 "/purple/network/turn_username");
01957                 const gchar *password = purple_prefs_get_string(
01958                                 "/purple/network/turn_password");
01959 
01960                 /* UDP */
01961                 port = purple_prefs_get_int("/purple/network/turn_port");

01964                                 password, "udp");
01965                 }
01966                 
01967                 /* TCP */
01968                 port = purple_prefs_get_int("/purple/network/turn_port_tcp");
01969                 if (port > 0) {
01970                         relay_info = append_relay_info(relay_info, turn_ip, port, username,
01971                                 password, "tcp");
01972                 }
01973 
01974                 /* TURN over SSL is only supported by libnice for Google's "psuedo" SSL mode
01975                         at this time */
01976 
01977                 purple_debug_info("backend-fs2",
01978                         "Setting relay-info on new stream\n");
01979                 _params[_num_params].name = "relay-info";
01980                 g_value_init(&_params[_num_params].value,
01981                         G_TYPE_VALUE_ARRAY);
01982                 g_value_set_boxed(&_params[_num_params].value,
01983                         relay_info);
01984                 g_value_array_free(relay_info);
01985                 _num_params++;
01986         }
01987 
01988 #ifdef HAVE_FARSIGHT
01989         fsstream = fs_session_new_stream(session->session, participant,
01990                         initiator == TRUE ? type_direction :
01991                         (type_direction & FS_DIRECTION_RECV), transmitter,
01992                         _num_params, _params, &err);
01993         g_free(_params);
01994 
01995         if (fsstream == NULL) {
01996                 if (err) {
01997                         purple_debug_error("backend-fs2",
01998                                         "Error creating stream: %s\n",
01999                                         err && err->message ?
02000                                         err->message : "NULL");
02001                         g_error_free(err);
02002                 } else
02003                         purple_debug_error("backend-fs2",
02004                                         "Error creating stream\n");
02005                 return FALSE;
02006         }
02007 #else
02008         if (!fs_stream_set_transmitter(fsstream, transmitter,
02009                         _params, _num_params, &err)) {
02010                 purple_debug_error("backend-fs2",
02011                                 "Could not set transmitter %s: %s.\n",
02012                                 transmitter, err->message);
02013                 g_clear_error(&err);
02014                 g_free(_params);
02015                 return FALSE;
02016         }
02017         g_free(_params);
02018 #endif
02019 
02020         stream = g_new0(PurpleMediaBackendFs2Stream, 1);
02021         stream->participant = g_strdup(who);
02022         stream->session = session;
02023         stream->stream = fsstream;

02025         stream->supports_add = !strcmp(transmitter, "nice");
02026 #endif
02027
