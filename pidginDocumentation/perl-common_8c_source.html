00135   2003/02/06: execute_perl modified by Mark Doliner <mark@kingant.net>
00136                 Pass parameters by pushing them onto the stack rather than
00137                 passing an array of strings.  This way, perl scripts can
00138                 modify the parameters and we can get the changed values
00139                 and then shoot ourselves.  I mean, uh, use them.
00140 
00141   2001/06/14: execute_perl replaced by Martin Persson <mep@passagen.se>
00142                 previous use of perl_eval leaked memory, replaced with
00143                 a version that uses perl_call instead
00144 
00145   30/11/2002: execute_perl modified by Eric Timme <timothy@voidnet.com>
00146                 args changed to char** so that we can have preparsed
00147                 arguments again, and many headaches ensued! This essentially
00148                 means we replaced one hacked method with a messier hacked
00149                 method out of perceived necessity. Formerly execute_perl
00150                 required a single char_ptr, and it would insert it into an
00151                 array of character pointers and NULL terminate the new array.
00152                 Now we have to pass in pre-terminated character pointer arrays
00153                 to accomodate functions that want to pass in multiple arguments.
00154 
00155                 Previously arguments were preparsed because an argument list
00156                 was constructed in the form 'arg one','arg two' and was
00157                 executed via a call like &funcname(arglist) (see .59.x), so
00158                 the arglist was magically pre-parsed because of the method.
00159                 With Martin Persson's change to perl_call we now need to
00160                 use a null terminated list of character pointers for arguments
00161                 if we wish them to be parsed. Lacking a better way to allow
00162                 for both single arguments and many I created a NULL terminated
00163                 array in every function that called execute_perl and passed
00164                 that list into the function.  In the former version a single
00165                 character pointer was passed in, and was placed into an array
00166                 of character pointers with two elements, with a NULL element
00167                 tacked onto the back, but this method no longer seemed prudent.
00168 
00169                 Enhancements in the future might be to get rid of pre-declaring
00170                 the array sizes?  I am not comfortable enough with this
00171                 subject to attempt it myself and hope it to stand the test
00172                 of time.
00173 */

00175 execute_perl(const char *function, int argc, char **args)
00176 {
00177         int count = 0, i, ret_value = 1;
00178         SV *sv_args[argc];
00179         dSP;
00180         PERL_SET_CONTEXT(my_perl);
00181         /*
00182          * Set up the perl environment, push arguments onto the
00183          * perl stack, then call the given function
00184          */
00185         SPAGAIN;
00186         ENTER;
00187         SAVETMPS;
00188         PUSHMARK(sp);
00189 
00190         for (i = 0; i < argc; i++) {
00191                 if (args[i]) {
00192                         sv_args[i] = sv_2mortal(newSVpv(args[i], 0));
00193                         XPUSHs(sv_args[i]);
00194                 }
00195         }
00196 
00197         PUTBACK;
00198         PERL_SET_CONTEXT(my_perl);
00199         count = call_pv(function, G_EVAL | G_SCALAR);
00200         SPAGAIN;
00201 
00202         /*
00203          * Check for "die," make sure we have 1 argument, and set our
00204          * return value.
00205          */
00206         if (SvTRUE(ERRSV)) {
00207                 purple_debug(PURPLE_DEBUG_ERROR, "perl",
00208                                    "Perl function %s exited abnormally: %s\n",
00209                                    function, SvPVutf8_nolen(ERRSV));
00210                 (void)POPs;
00211         } else if (count != 1) {
00212                 /*
00213                  * This should NEVER happen.  G_SCALAR ensures that we WILL
00214                  * have 1 parameter.
00215                  */
00216                 purple_debug(PURPLE_DEBUG_ERROR, "perl",
00217                                    "Perl error from %s: expected 1 return value, "
00218                                    "but got %d\n", function, count);
00219         } else
00220                 ret_value = POPi;
00221 
00222         /* Check for changed arguments */
00223         for (i = 0; i < argc; i++) {
00224                 if (args[i] && strcmp(args[i], SvPVX(sv_args[i]))) {
00225                         /*
00226                          * Shizzel.  So the perl script changed one of the parameters,
00227                          * and we want this change to affect the original parameters.
00228                          * args[i] is just a temporary little list of pointers.  We don't
00229                          * want to free args[i] here because the new parameter doesn't
00230                          * overwrite the data that args[i] points to.  That is done by
00231                          * the function that called execute_perl.  I'm not explaining this
00232                          * very well.  See, it's aggregate...  Oh, but if 2 perl scripts
00233                          * both modify the data, _that's_ a memleak.  This is really kind
00234                          * of hackish.  I should fix it.  Look how long this comment is.
00235                          * Holy crap.
00236                          */
00237                         args[i] = g_strdup(SvPVutf8_nolen(sv_args[i]));
00238                 }
00239         }
00240 
00241         PUTBACK;
