00042 /*
00043  * Subtype 0x0003 - Rights.
00044  *
00045  */
00046 static int
00047 rights(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00048 {
00049         aim_rxcallback_t userfunc;
00050         GSList *tlvlist;
00051         guint16 maxbuddies = 0, maxwatchers = 0;
00052         int ret = 0;
00053 
00054         /*
00055          * TLVs follow
00056          */
00057         tlvlist = aim_tlvlist_read(bs);
00058 
00059         /*
00060          * TLV type 0x0001: Maximum number of buddies.
00061          */
00062         if (aim_tlv_gettlv(tlvlist, 0x0001, 1))
00063                 maxbuddies = aim_tlv_get16(tlvlist, 0x0001, 1);
00064 
00065         /*
00066          * TLV type 0x0002: Maximum number of watchers.
00067          *
00068          * Watchers are other users who have you on their buddy
00069          * list.  (This is called the "reverse list" by a certain
00070          * other IM protocol.)
00071          *
00072          */
00073         if (aim_tlv_gettlv(tlvlist, 0x0002, 1))
00074                 maxwatchers = aim_tlv_get16(tlvlist, 0x0002, 1);
00075 
00076         /*
00077          * TLV type 0x0003: Unknown.
00078          *
00079          * ICQ only?
00080          */
00081 
00082         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00083                 ret = userfunc(od, conn, frame, maxbuddies, maxwatchers);
00084 
00085         aim_tlvlist_free(tlvlist);
00086 
00087         return ret;
00088 }

00090 /*
00091  * Subtypes 0x000b (SNAC_SUBTYPE_BUDDY_ONCOMING) and 0x000c (SNAC_SUBTYPE_BUDDY_OFFGOING) - Change in buddy status
00092  *
00093  * Oncoming Buddy notifications contain a subset of the
00094  * user information structure.  It's close enough to run
00095  * through aim_info_extract() however.
00096  *
00097  * Although the offgoing notification contains no information,
00098  * it is still in a format parsable by aim_info_extract().
00099  *
00100  */
00101 static int
00102 buddychange(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00103 {
00104         int ret = 0;
00105         aim_userinfo_t userinfo;
00106         aim_rxcallback_t userfunc;
00107 
00108         aim_info_extract(od, bs, &userinfo);
00109 
00110         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00111                 ret = userfunc(od, conn, frame, &userinfo);
00112 
00113         if (snac->subtype == SNAC_SUBTYPE_BUDDY_ONCOMING &&
00114             userinfo.capabilities & OSCAR_CAPABILITY_XTRAZ) {
00115                 PurpleAccount *account = purple_connection_get_account(od->gc);
00116                 PurpleBuddy *buddy = purple_find_buddy(account, userinfo.bn);
00117 
00118                 if (buddy) {
00119                         PurplePresence *presence = purple_buddy_get_presence(buddy);
00120 
00121                         if (purple_presence_is_status_primitive_active(presence, PURPLE_STATUS_MOOD))
00122                                 icq_im_xstatus_request(od, userinfo.bn);
00123                 }
00124         }
00125         aim_info_free(&userinfo);
00126 
00127         return ret;
00128 }

00130 static int
00131 snachandler(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00132 {
00133         if (snac->subtype == SNAC_SUBTYPE_BUDDY_RIGHTSINFO)
00134                 return rights(od, conn, mod, frame, snac, bs);
00135         else if ((snac->subtype == SNAC_SUBTYPE_BUDDY_ONCOMING) || (snac->subtype == SNAC_SUBTYPE_BUDDY_OFFGOING))
00136                 return buddychange(od, conn, mod, frame, snac, bs);
00137 
00138         return 0;
00139 }
