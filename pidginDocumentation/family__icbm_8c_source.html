00021 /*
00022  * Family 0x0004 - Routines for sending/receiving Instant Messages.
00023  *
00024  * Note the term ICBM (Inter-Client Basic Message) which blankets
00025  * all types of generically routed through-server messages.  Within
00026  * the ICBM types (family 4), a channel is defined.  Each channel
00027  * represents a different type of message.  Channel 1 is used for
00028  * what would commonly be called an "instant message".  Channel 2
00029  * is used for negotiating "rendezvous".  These transactions end in
00030  * something more complex happening, such as a chat invitation, or
00031  * a file transfer.  Channel 3 is used for chat messages (not in
00032  * the same family as these channels).  Channel 4 is used for
00033  * various ICQ messages.  Examples are normal messages, URLs, and
00034  * old-style authorization.
00035  *
00036  * In addition to the channel, every ICBM contains a cookie.  For
00037  * standard IMs, these are only used for error messages.  However,
00038  * the more complex rendezvous messages make suitably more complex
00039  * use of this field.
00040  *
00041  * TODO: Split this up into an im.c file an an icbm.c file.  It
00042  *       will be beautiful, you'll see.
00043  *
00044  *       Make sure flap_connection_findbygroup is used by all functions.
00045  */

00158                            "Message error with bn %s and reason %hu and errcode %hu\n",

00163         if ((xfer = oscar_find_xfer_by_cookie(od->file_transfers, bn))) {
00164                 purple_xfer_cancel_remote(xfer);

00175                 buf = g_strdup_printf(_("Unable to send message: %s"), reason_str);
00176 
00177         if (!purple_conv_present_error(bn, purple_connection_get_account(gc), buf)) {
00178                 g_free(buf);
00179                 if (errcode != 0 && errcode < errcodereasonlen)
00180                         buf = g_strdup_printf(_("Unable to send message to %s: %s (%s)"),
00181                                               bn ? bn : "(unknown)", reason_str,
00182                                               _(errcodereason[errcode]));
00183                 else
00184                         buf = g_strdup_printf(_("Unable to send message to %s: %s"),
00185                                               bn ? bn : "(unknown)", reason_str);
00186                 purple_notify_error(od->gc, NULL, buf, reason_str);
00187         }
00188         g_free(buf);
00189 
00190         g_free(snac2->data);
00191         g_free(snac2);
00192

00295 int aim_im_sendch1_ext(OscarData *od, struct aim_sendimext_args *args)
00296 {
00297         FlapConnection *conn;
00298         aim_snacid_t snacid;
00299         ByteStream data;
00300         guchar cookie[8];
00301         int msgtlvlen;
00302 
00303         if (!od || !(conn = flap_connection_findbygroup(od, SNAC_FAMILY_ICBM)))
00304                 return -EINVAL;
00305 
00306         if (!args)
00307                 return -EINVAL;
00308 
00309         if (!args->msg || (args->msglen <= 0))
00310                 return -EINVAL;
00311 
00312         if (args->msglen > MAXMSGLEN)
00313                 return -E2BIG;
00314 
00315         /* Painfully calculate the size of the message TLV */
00316         msgtlvlen = 1 + 1; /* 0501 */
00317         msgtlvlen += 2 + args->featureslen;
00318         msgtlvlen += 2 /* 0101 */ + 2 /* block len */;
00319         msgtlvlen += 4 /* charset */ + args->msglen;
00320 
00321         byte_stream_new(&data, msgtlvlen + 128);
00322 
00323         /* Generate an ICBM cookie */
00324         aim_icbm_makecookie(cookie);
00325 
00326         /* ICBM header */
00327         aim_im_puticbm(&data, cookie, 0x0001, args->destbn);
00328 
00329         /* Message TLV (type 0x0002) */
00330         byte_stream_put16(&data, 0x0002);
00331         byte_stream_put16(&data, msgtlvlen);
00332 
00333         /* Features TLV (type 0x0501) */
00334         byte_stream_put16(&data, 0x0501);
00335         byte_stream_put16(&data, args->featureslen);
00336         byte_stream_putraw(&data, args->features, args->featureslen);
00337 
00338         /* Insert message text in a TLV (type 0x0101) */
00339         byte_stream_put16(&data, 0x0101);
00340 
00341         /* Message block length */
00342         byte_stream_put16(&data, args->msglen + 0x04);
00343 
00344         /* Character set */
00345         byte_stream_put16(&data, args->charset);
00346         /* Character subset -- we always use 0 here */
00347         byte_stream_put16(&data, 0x0);
00348 
00349         /* Message.  Not terminated */
00350         byte_stream_putraw(&data, (guchar *)args->msg, args->msglen);
00351 
00352         /* Set the Autoresponse flag */
00353         if (args->flags & AIM_IMFLAGS_AWAY) {
00354                 byte_stream_put16(&data, 0x0004);
00355                 byte_stream_put16(&data, 0x0000);
00356         } else {
00357                 /* Set the Request Acknowledge flag */
00358                 byte_stream_put16(&data, 0x0003);
00359                 byte_stream_put16(&data, 0x0000);
00360 
00361                 if (args->flags & AIM_IMFLAGS_OFFLINE) {
00362                         /* Allow this message to be queued as an offline message */
00363                         byte_stream_put16(&data, 0x0006);
00364                         byte_stream_put16(&data, 0x0000);
00365                 }
00366         }
00367 
00368         /*
00369          * Set the I HAVE A REALLY PURTY ICON flag.
00370          * XXX - This should really only be sent on initial
00371          * IMs and when you change your icon.
00372          */
00373         if (args->flags & AIM_IMFLAGS_HASICON) {
00374                 byte_stream_put16(&data, 0x0008);
00375                 byte_stream_put16(&data, 0x000c);
00376                 byte_stream_put32(&data, args->iconlen);
00377                 byte_stream_put16(&data, 0x0001);
00378                 byte_stream_put16(&data, args->iconsum);
00379                 byte_stream_put32(&data, args->iconstamp);
00380         }
00381 
00382         /*
00383          * Set the Buddy Icon Requested flag.
00384          * XXX - Every time?  Surely not...
00385          */
00386         if (args->flags & AIM_IMFLAGS_BUDDYREQ) {
00387                 byte_stream_put16(&data, 0x0009);
00388                 byte_stream_put16(&data, 0x0000);
00389         }
00390 
00391         /* XXX - should be optional */
00392         snacid = aim_cachesnac(od, SNAC_FAMILY_ICBM, 0x0006, 0x0000, args->destbn, strlen(args->destbn)+1);
00393 
00394         flap_connection_send_snac(od, conn, SNAC_FAMILY_ICBM, 0x0006, snacid, &data);
00395         byte_stream_destroy(&data);
00396 
00397         /* clean out SNACs over 60sec old */
00398         aim_cleansnacs(od, 60);
00399 
00400         return 0;
00401 }

00937 static int
00938 incomingim_ch1(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, guint16 channel, aim_userinfo_t *userinfo, ByteStream *bs, guint8 *cookie)
00939 {
00940         guint16 type, length;
00941         aim_rxcallback_t userfunc;
00942         int ret = 0;
00943         struct aim_incomingim_ch1_args args;
00944         unsigned int endpos;
00945 
00946         memset(&args, 0, sizeof(args));
00947 
00948         /*
00949          * This used to be done using tlvchains.  For performance reasons,
00950          * I've changed it to process the TLVs in-place.  This avoids lots
00951          * of per-IM memory allocations.
00952          */
00953         while (byte_stream_bytes_left(bs) >= 4)
00954         {
00955                 type = byte_stream_get16(bs);
00956                 length = byte_stream_get16(bs);
00957 
00958                 if (length > byte_stream_bytes_left(bs))
00959                 {
00960                         purple_debug_misc("oscar", "Received an IM containing an invalid message part from %s.  They are probably trying to do something malicious.\n", userinfo->bn);
00961                         break;
00962                 }
00963 
00964                 endpos = byte_stream_curpos(bs) + length;
00965 
00966                 if (type == 0x0002) { /* Message Block */
00967                         ByteStream tlv02;
00968                         byte_stream_init(&tlv02, bs->data + bs->offset, length);
00969                         incomingim_ch1_parsemsg(od, userinfo, &tlv02, &args);
00970                 } else if (type == 0x0003) { /* Server Ack Requested */
00971                         args.icbmflags |= AIM_IMFLAGS_ACK;
00972                 } else if (type == 0x0004) { /* Message is Auto Response */
00973                         args.icbmflags |= AIM_IMFLAGS_AWAY;
00974                 } else if (type == 0x0006) { /* Message was received offline. */
00975                         /*
00976                          * This flag is set on incoming offline messages for both
00977                          * AIM and ICQ accounts.
00978                          */
00979                         args.icbmflags |= AIM_IMFLAGS_OFFLINE;
00980                 } else if (type == 0x0008) { /* I-HAVE-A-REALLY-PURTY-ICON Flag */
00981                         args.iconlen = byte_stream_get32(bs);
00982                         byte_stream_get16(bs); /* 0x0001 */
00983                         args.iconsum = byte_stream_get16(bs);
00984                         args.iconstamp = byte_stream_get32(bs);
00985 
00986                         /*
00987                          * This looks to be a client bug.  MacAIM 4.3 will
00988                          * send this tag, but with all zero values, in the
00989                          * first message of a conversation. This makes no
00990                          * sense whatsoever, so I'm going to say its a bug.
00991                          *
00992                          * You really shouldn't advertise a zero-length icon
00993                          * anyway.
00994                          *
00995                          */
00996                         if (args.iconlen)
00997                                 args.icbmflags |= AIM_IMFLAGS_HASICON;
00998                 } else if (type == 0x0009) {
00999                         args.icbmflags |= AIM_IMFLAGS_BUDDYREQ;
01000                 } else if (type == 0x000b) { /* Non-direct connect typing notification */
01001                         args.icbmflags |= AIM_IMFLAGS_TYPINGNOT;
01002                 } else if (type == 0x0016) {
01003                         /*
01004                          * UTC timestamp for when the message was sent.  Only
01005                          * provided for offline messages.
01006                          */
01007                         args.timestamp = byte_stream_get32(bs);
01008                 }
01009 
01010                 /*
01011                  * This is here to protect ourselves from ourselves.  That
01012                  * is, if something above doesn't completely parse its value
01013                  * section, or, worse, overparses it, this will set the
01014                  * stream where it needs to be in order to land on the next
01015                  * TLV when the loop continues.
01016                  *
01017                  */
01018                 byte_stream_setpos(bs, endpos);
01019         }
01020 
01021 
01022         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
01023                 ret = userfunc(od, conn, frame, channel, userinfo, &args);
01024 
01025         g_free(args.msg);
01026         return ret;
01027 }
