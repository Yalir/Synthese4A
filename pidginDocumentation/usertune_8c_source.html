00033 static void jabber_tune_cb(JabberStream *js, const char *from, xmlnode *items) {
00034         /* it doesn't make sense to have more than one item here, so let's just pick the first one */
00035         xmlnode *item = xmlnode_get_child(items, "item");
00036         JabberBuddy *buddy = jabber_buddy_find(js, from, FALSE);
00037         xmlnode *tuneinfo, *tune;
00038         PurpleJabberTuneInfo tuneinfodata;
00039         JabberBuddyResource *resource;
00040         gboolean valid = FALSE;
00041 
00042         /* ignore the tune of people not on our buddy list */
00043         if (!buddy || !item)
00044                 return;
00045 
00046         tuneinfodata.artist = NULL;
00047         tuneinfodata.title = NULL;
00048         tuneinfodata.album = NULL;
00049         tuneinfodata.track = NULL;
00050         tuneinfodata.time = -1;
00051         tuneinfodata.url = NULL;
00052 
00053         tune = xmlnode_get_child_with_namespace(item, "tune", "http://jabber.org/protocol/tune");
00054         if (!tune)
00055                 return;
00056         resource = jabber_buddy_find_resource(buddy, NULL);
00057         if(!resource)
00058                 return; /* huh? */
00059         for (tuneinfo = tune->child; tuneinfo; tuneinfo = tuneinfo->next) {
00060                 if (tuneinfo->type == XMLNODE_TYPE_TAG) {
00061                         if (!strcmp(tuneinfo->name, "artist")) {
00062                                 if (tuneinfodata.artist == NULL) /* only pick the first one */
00063                                         tuneinfodata.artist = xmlnode_get_data(tuneinfo);
00064                                 valid = TRUE;
00065                         } else if (!strcmp(tuneinfo->name, "length")) {
00066                                 if (tuneinfodata.time == -1) {
00067                                         char *length = xmlnode_get_data(tuneinfo);
00068                                         if (length)
00069                                                 tuneinfodata.time = strtol(length, NULL, 10);
00070                                         g_free(length);
00071                                         if (tuneinfodata.time > 0)
00072                                                 valid = TRUE;
00073                                 }
00074                         } else if (!strcmp(tuneinfo->name, "source")) {
00075                                 if (tuneinfodata.album == NULL) /* only pick the first one */
00076                                         tuneinfodata.album = xmlnode_get_data(tuneinfo);
00077                                 valid = TRUE;
00078                         } else if (!strcmp(tuneinfo->name, "title")) {
00079                                 if (tuneinfodata.title == NULL) /* only pick the first one */
00080                                         tuneinfodata.title = xmlnode_get_data(tuneinfo);
00081                                 valid = TRUE;
00082                         } else if (!strcmp(tuneinfo->name, "track")) {
00083                                 if (tuneinfodata.track == NULL) /* only pick the first one */
00084                                         tuneinfodata.track = xmlnode_get_data(tuneinfo);
00085                                 valid = TRUE;
00086                         } else if (!strcmp(tuneinfo->name, "uri")) {
00087                                 if (tuneinfodata.url == NULL) /* only pick the first one */
00088                                         tuneinfodata.url = xmlnode_get_data(tuneinfo);
00089                                 valid = TRUE;
00090                         }
00091                 }
00092         }
00093 
00094         if (valid) {
00095                 purple_prpl_got_user_status(js->gc->account, from, "tune",
00096                                 PURPLE_TUNE_ARTIST, tuneinfodata.artist,
00097                                 PURPLE_TUNE_TITLE, tuneinfodata.title,
00098                                 PURPLE_TUNE_ALBUM, tuneinfodata.album,
00099                                 PURPLE_TUNE_TRACK, tuneinfodata.track,
00100                                 PURPLE_TUNE_TIME, tuneinfodata.time,
00101                                 PURPLE_TUNE_URL, tuneinfodata.url, NULL);
00102         } else {
00103                 purple_prpl_got_user_status_deactive(js->gc->account, from, "tune");
00104         }
00105 
00106         g_free(tuneinfodata.artist);
00107         g_free(tuneinfodata.title);
00108         g_free(tuneinfodata.album);
00109         g_free(tuneinfodata.track);
00110         g_free(tuneinfodata.url);
00111 }

00118 void jabber_tune_set(PurpleConnection *gc, const PurpleJabberTuneInfo *tuneinfo) {
00119         xmlnode *publish, *tunenode;
00120         JabberStream *js = gc->proto_data;
00121 
00122         publish = xmlnode_new("publish");
00123         xmlnode_set_attrib(publish,"node","http://jabber.org/protocol/tune");
00124         tunenode = xmlnode_new_child(xmlnode_new_child(publish, "item"), "tune");
00125         xmlnode_set_namespace(tunenode, "http://jabber.org/protocol/tune");
00126 
00127         if(tuneinfo) {
00128                 if(tuneinfo->artist && tuneinfo->artist[0] != '\0')
00129                         xmlnode_insert_data(xmlnode_new_child(tunenode, "artist"),tuneinfo->artist,-1);
00130                 if(tuneinfo->title && tuneinfo->title[0] != '\0')
00131                         xmlnode_insert_data(xmlnode_new_child(tunenode, "title"),tuneinfo->title,-1);
00132                 if(tuneinfo->album && tuneinfo->album[0] != '\0')
00133                         xmlnode_insert_data(xmlnode_new_child(tunenode, "source"),tuneinfo->album,-1);
00134                 if(tuneinfo->url && tuneinfo->url[0] != '\0')
00135                         xmlnode_insert_data(xmlnode_new_child(tunenode, "uri"),tuneinfo->url,-1);
00136                 if(tuneinfo->time > 0) {
00137                         char *length = g_strdup_printf("%d", tuneinfo->time);
00138                         xmlnode_insert_data(xmlnode_new_child(tunenode, "length"),length,-1);
00139                         g_free(length);
00140                 }
00141                 if(tuneinfo->track && tuneinfo->track[0] != '\0')
00142                         xmlnode_insert_data(xmlnode_new_child(tunenode, "track"),tuneinfo->track,-1);
00143         }
00144 
00145         jabber_pep_publish(js, publish);
00146         /* publish is freed by jabber_pep_publish -> jabber_iq_send -> jabber_iq_free
00147            (yay for well-defined memory management rules) */
00148 }
