00188 static PurpleAccount *find_acct(const char *prpl, const char *acct_id)
00189 {
00190         PurpleAccount *acct = NULL;
00191 
00192         /* If we have a specific acct, use it */
00193         if (acct_id) {
00194                 acct = purple_accounts_find(acct_id, prpl);
00195                 if (acct && !purple_account_is_connected(acct))
00196                         acct = NULL;
00197         } else { /* Otherwise find an active account for the protocol */
00198                 GList *l = purple_accounts_get_all();
00199                 while (l) {
00200                         if (!strcmp(prpl, purple_account_get_protocol_id(l->data))
00201                                         && purple_account_is_connected(l->data)) {
00202                                 acct = l->data;
00203                                 break;
00204                         }
00205                         l = l->next;
00206                 }
00207         }
00208 
00209         return acct;
00210 }

00212 static gboolean xmpp_uri_handler(const char *proto, const char *user, GHashTable *params)
00213 {
00214         char *acct_id = params ? g_hash_table_lookup(params, "account") : NULL;
00215         PurpleAccount *acct;
00216 
00217         if (g_ascii_strcasecmp(proto, "xmpp"))
00218                 return FALSE;
00219 
00220         acct = find_acct(purple_plugin_get_id(my_protocol), acct_id);
00221 
00222         if (!acct)
00223                 return FALSE;
00224 
00225         /* xmpp:romeo@montague.net?message;subject=Test%20Message;body=Here%27s%20a%20test%20message */
00226         /* params is NULL if the URI has no '?' (or anything after it) */
00227         if (!params || g_hash_table_lookup_extended(params, "message", NULL, NULL)) {
00228                 char *body = g_hash_table_lookup(params, "body");
00229                 if (user && *user) {
00230                         PurpleConversation *conv =
00231                                         purple_conversation_new(PURPLE_CONV_TYPE_IM, acct, user);
00232                         purple_conversation_present(conv);
00233                         if (body && *body)
00234                                 purple_conv_send_confirm(conv, body);
00235                 }
00236         } else if (g_hash_table_lookup_extended(params, "roster", NULL, NULL)) {
00237                 char *name = g_hash_table_lookup(params, "name");
00238                 if (user && *user)
00239                         purple_blist_request_add_buddy(acct, user, NULL, name);
00240         } else if (g_hash_table_lookup_extended(params, "join", NULL, NULL)) {
00241                 PurpleConnection *gc = purple_account_get_connection(acct);
00242                 if (user && *user) {
00243                         GHashTable *params = jabber_chat_info_defaults(gc, user);
00244                         jabber_chat_join(gc, params);
00245                 }
00246                 return TRUE;
00247         }
00248 
00249         return FALSE;
00250 }
