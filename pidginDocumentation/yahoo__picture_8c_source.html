00511 static int yahoo_buddy_icon_calculate_checksum(const guchar *data, gsize len)
00512 {
00513         /* This code is borrowed from Kopete, which seems to be managing to calculate
00514            checksums in such a manner that Yahoo!'s servers are happy */
00515 
00516         const guchar *p = data;
00517         int checksum = 0, g, i = len;
00518 
00519         while(i--) {
00520                 checksum = (checksum << 4) + *p++;
00521 
00522                 if((g = (checksum & 0xf0000000)) != 0)
00523                         checksum ^= g >> 23;
00524 
00525                 checksum &= ~g;
00526         }
00527 
00528         purple_debug_misc("yahoo", "Calculated buddy icon checksum: %d\n", checksum);
00529 
00530         return checksum;
00531 }
