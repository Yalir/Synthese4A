00182 static JabberSaslState
00183 jabber_auth_start_cyrus(JabberStream *js, xmlnode **reply, char **error)
00184 {
00185         PurpleAccount *account;
00186         const char *clientout = NULL;
00187         char *enc_out;
00188         unsigned coutlen = 0;
00189         sasl_security_properties_t secprops;
00190         gboolean again;
00191         gboolean plaintext = TRUE;
00192 
00193         /* Set up security properties and options */
00194         secprops.min_ssf = 0;
00195         secprops.security_flags = SASL_SEC_NOANONYMOUS;
00196 
00197         account = purple_connection_get_account(js->gc);
00198 
00199         if (!jabber_stream_is_ssl(js)) {
00200                 secprops.max_ssf = -1;
00201                 secprops.maxbufsize = 4096;
00202                 plaintext = purple_account_get_bool(account, "auth_plain_in_clear", FALSE);
00203                 if (!plaintext)
00204                         secprops.security_flags |= SASL_SEC_NOPLAINTEXT;
00205         } else {
00206                 secprops.max_ssf = 0;
00207                 secprops.maxbufsize = 0;
00208                 plaintext = TRUE;
00209         }
00210         secprops.property_names = 0;
00211         secprops.property_values = 0;
00212 
00213         do {
00214                 again = FALSE;
00215 
00216                 js->sasl_state = sasl_client_new("xmpp", js->serverFQDN, NULL, NULL, js->sasl_cb, 0, &js->sasl);
00217                 if (js->sasl_state==SASL_OK) {
00218                         sasl_setprop(js->sasl, SASL_SEC_PROPS, &secprops);
00219                         purple_debug_info("sasl", "Mechs found: %s\n", js->sasl_mechs->str);
00220                         js->sasl_state = sasl_client_start(js->sasl, js->sasl_mechs->str, NULL, &clientout, &coutlen, &js->current_mech);
00221                 }
00222                 switch (js->sasl_state) {
00223                         /* Success */
00224                         case SASL_OK:
00225                         case SASL_CONTINUE:
00226                                 break;
00227                         case SASL_NOMECH:
00228                                 /* No mechanisms have offered to help */
00229 
00230                                 /* Firstly, if we don't have a password try
00231                                  * to get one
00232                                  */
00233 
00234                                 if (!purple_account_get_password(account)) {
00235                                         purple_account_request_password(account, G_CALLBACK(auth_pass_cb), G_CALLBACK(auth_no_pass_cb), js->gc);
00236                                         return JABBER_SASL_STATE_CONTINUE;
00237 
00238                                 /* If we've got a password, but aren't sending
00239                                  * it in plaintext, see if we can turn on
00240                                  * plaintext auth
00241                                  */
00242                                 /* XXX Should we just check for PLAIN/LOGIN being offered mechanisms? */
00243                                 } else if (!plaintext) {
00244                                         char *msg = g_strdup_printf(_("%s may require plaintext authentication over an unencrypted connection.  Allow this and continue authentication?"),
00245                                                         purple_account_get_username(account));
00246                                         purple_request_yes_no(js->gc, _("Plaintext Authentication"),
00247                                                         _("Plaintext Authentication"),
00248                                                         msg,
00249                                                         1, account, NULL, NULL, account,
00250                                                         allow_cyrus_plaintext_auth,
00251                                                         disallow_plaintext_auth);
00252                                         g_free(msg);
00253                                         return JABBER_SASL_STATE_CONTINUE;
00254 
00255                                 } else
00256                                         js->auth_fail_count++;
00257 
00258                                 if (js->auth_fail_count == 1 &&
00259                                         (js->sasl_mechs->str && g_str_equal(js->sasl_mechs->str, "GSSAPI"))) {
00260                                         /* If we tried GSSAPI first, it failed, and it was the only method we had to try, try jabber:iq:auth
00261                                          * for compatibility with iChat 10.5 Server and other jabberd based servers.
00262                                          *
00263                                          * iChat Server 10.5 and certain other corporate servers offer SASL GSSAPI by default, which is often
00264                                          * not configured on the client side, and expects a fallback to jabber:iq:auth when it (predictably) fails.
00265                                          *
00266                                          * Note: xep-0078 points out that using jabber:iq:auth after a sasl failure is wrong. However,
00267                                          * I believe this refers to actual authentication failure, not a simple lack of concordant mechanisms.
00268                                          * Doing otherwise means that simply compiling with SASL support renders the client unable to connect to servers
00269                                          * which would connect without issue otherwise. -evands
00270                                          */
00271                                         js->auth_mech = NULL;
00272                                         jabber_auth_start_old(js);
00273                                         return JABBER_SASL_STATE_CONTINUE;
00274                                 }
00275 
00276                                 break;
00277 
00278                                 /* Fatal errors. Give up and go home */
00279                         case SASL_BADPARAM:
00280                         case SASL_NOMEM:
00281                                 *error = g_strdup(_("SASL authentication failed"));
00282                                 break;
00283 
00284                                 /* For everything else, fail the mechanism and try again */
00285                         default:
00286                                 purple_debug_info("sasl", "sasl_state is %d, failing the mech and trying again\n", js->sasl_state);
00287 
00288                                 js->auth_fail_count++;
00289 
00290                                 /*
00291                                  * DAA: is this right?
00292                                  * The manpage says that "mech" will contain the chosen mechanism on success.
00293                                  * Presumably, if we get here that isn't the case and we shouldn't try again?
00294                                  * I suspect that this never happens.
00295                                  */
00296                                 /*
00297                                  * SXW: Yes, this is right. What this handles is the situation where a
00298                                  * mechanism, say GSSAPI, is tried. If that mechanism fails, it may be
00299                                  * due to mechanism specific issues, so we want to try one of the other
00300                                  * supported mechanisms. This code handles that case
00301                                  */
00302                                 if (js->current_mech && *js->current_mech) {
00303                                         char *pos;
00304                                         if ((pos = strstr(js->sasl_mechs->str, js->current_mech))) {
00305                                                 g_string_erase(js->sasl_mechs, pos-js->sasl_mechs->str, strlen(js->current_mech));
00306                                         }
00307                                         /* Remove space which separated this mech from the next */
00308                                         if ((js->sasl_mechs->str)[0] == ' ') {
00309                                                 g_string_erase(js->sasl_mechs, 0, 1);
00310                                         }
00311                                         again = TRUE;
00312                                 }
00313 
00314                                 sasl_dispose(&js->sasl);
00315                 }
00316         } while (again);
00317 
00318         if (js->sasl_state == SASL_CONTINUE || js->sasl_state == SASL_OK) {
00319                 xmlnode *auth = xmlnode_new("auth");
00320                 xmlnode_set_namespace(auth, NS_XMPP_SASL);
00321                 xmlnode_set_attrib(auth, "mechanism", js->current_mech);
00322 
00323                 xmlnode_set_attrib(auth, "xmlns:ga", "http://www.google.com/talk/protocol/auth");
00324                 xmlnode_set_attrib(auth, "ga:client-uses-full-bind-result", "true");
00325 
00326                 if (clientout) {
00327                         if (coutlen == 0) {
00328                                 xmlnode_insert_data(auth, "=", -1);
00329                         } else {
00330                                 enc_out = purple_base64_encode((unsigned char*)clientout, coutlen);
00331                                 xmlnode_insert_data(auth, enc_out, -1);
00332                                 g_free(enc_out);
00333                         }
00334                 }
00335 
00336                 *reply = auth;
00337                 return JABBER_SASL_STATE_CONTINUE;
00338         } else {
00339                 return JABBER_SASL_STATE_FAIL;
00340         }
00341 }

00394 static JabberSaslState
00395 jabber_cyrus_start(JabberStream *js, xmlnode *mechanisms,
00396                    xmlnode **reply, char **error)
00397 {
00398         xmlnode *mechnode;
00399         JabberSaslState ret;
00400 
00401         js->sasl_mechs = g_string_new("");
00402 
00403         for(mechnode = xmlnode_get_child(mechanisms, "mechanism"); mechnode;
00404                         mechnode = xmlnode_get_next_twin(mechnode))
00405         {
00406                 char *mech_name = xmlnode_get_data(mechnode);
00407 
00408                 /* Ignore blank mechanisms and EXTERNAL.  External isn't
00409                  * supported, and Cyrus SASL's mechanism returns
00410                  * SASL_NOMECH when the caller (us) doesn't configure it.
00411                  * Except SASL_NOMECH is supposed to mean "no concordant
00412                  * mechanisms"...  Easiest just to blacklist it (for now).
00413                  */
00414                 if (!mech_name || !*mech_name ||
00415                                 g_str_equal(mech_name, "EXTERNAL")) {
00416                         g_free(mech_name);
00417                         continue;
00418                 }
00419 
00420                 g_string_append(js->sasl_mechs, mech_name);
00421                 g_string_append_c(js->sasl_mechs, ' ');
00422                 g_free(mech_name);
00423         }
00424 
00425         /* Strip off the trailing ' ' */
00426         if (js->sasl_mechs->len > 1)
00427                 g_string_truncate(js->sasl_mechs, js->sasl_mechs->len - 1);
00428 
00429         jabber_sasl_build_callbacks(js);
00430         ret = jabber_auth_start_cyrus(js, reply, error);
00431 
00432         /*
00433          * Triggered if no overlap between server and client
00434          * supported mechanisms.
00435          */
00436         if (ret == JABBER_SASL_STATE_FAIL && *error == NULL)
00437                 *error = g_strdup(_("Server does not use any supported authentication method"));
00438 
00439         return ret;
00440 }

00443 jabber_cyrus_handle_challenge(JabberStream *js, xmlnode *packet,
00444                               xmlnode **reply, char **error)
00445 {
00446         char *enc_in = xmlnode_get_data(packet);
00447         unsigned char *dec_in;
00448         char *enc_out;
00449         const char *c_out;
00450         unsigned int clen;
00451         gsize declen;
00452 
00453         dec_in = purple_base64_decode(enc_in, &declen);
00454 
00455         js->sasl_state = sasl_client_step(js->sasl, (char*)dec_in, declen,

00461                                 sasl_errdetail(js->sasl));
00462                 purple_debug_error("jabber", "Error is %d : %s\n",
00463                                 js->sasl_state, sasl_errdetail(js->sasl));
00464                 *error = tmp;
00465                 return JABBER_SASL_STATE_FAIL;
00466         } else {
00467                 xmlnode *response = xmlnode_new("response");
00468                 xmlnode_set_namespace(response, NS_XMPP_SASL);
00469                 if (clen > 0) {
00470                         /* Cyrus SASL 2.1.22 appears to contain code to add the charset
00471                          * to the response for DIGEST-MD5 but there is no possibility
00472                          * it will be executed.
00473                          *
00474                          * My reading of the digestmd5 plugin indicates the username and
00475                          * realm are always encoded in UTF-8 (they seem to be the values
00476                          * we pass in), so we need to ensure charset=utf-8 is set.
00477                          */
00478                         if (!purple_strequal(js->current_mech, "DIGEST-MD5") ||
00479                                         strstr(c_out, ",charset="))
00480                                 /* If we're not using DIGEST-MD5 or Cyrus SASL is fixed */
00481                                 enc_out = purple_base64_encode((unsigned char*)c_out, clen);
00482                         else {
00483                                 char *tmp = g_strdup_printf("%s,charset=utf-8", c_out);
00484                                 enc_out = purple_base64_encode((unsigned char*)tmp, clen + 14);
00485                                 g_free(tmp);
00486                         }
00487 
00488                         xmlnode_insert_data(response, enc_out, -1);
00489                         g_free(enc_out);
00490                 }
00491 
00492                 *reply = response;
00493                 return JABBER_SASL_STATE_CONTINUE;
00494         }
00495 }

00497 static JabberSaslState
00498 jabber_cyrus_handle_success(JabberStream *js, xmlnode *packet,
00499                             char **error)
00500 {
00501         const void *x;
00502 
00503         /* The SASL docs say that if the client hasn't returned OK yet, we
00504          * should try one more round against it
00505          */
00506         if (js->sasl_state != SASL_OK) {
00507                 char *enc_in = xmlnode_get_data(packet);
00508                 unsigned char *dec_in = NULL;
00509                 const char *c_out;
00510                 unsigned int clen;
00511                 gsize declen = 0;
00512 
00513                 if(enc_in != NULL)
00514                         dec_in = purple_base64_decode(enc_in, &declen);
00515 
00516                 js->sasl_state = sasl_client_step(js->sasl, (char*)dec_in, declen, NULL, &c_out, &clen);
00517 
00518                 g_free(enc_in);
00519                 g_free(dec_in);
00520 
00521                 if (js->sasl_state != SASL_OK) {
00522                         /* This happens when the server sends back jibberish
00523                          * in the "additional data with success" case.
00524                          * Seen with Wildfire 3.0.1.
00525                          */
00526                         *error = g_strdup(_("Invalid response from server"));
00527                         return JABBER_SASL_STATE_FAIL;
00528                 }
00529         }
00530 
00531         /* If we've negotiated a security layer, we need to enable it */
00532         if (js->sasl) {
00533                 sasl_getprop(js->sasl, SASL_SSF, &x);
00534                 if (*(int *)x > 0) {
00535                         sasl_getprop(js->sasl, SASL_MAXOUTBUF, &x);
00536                         js->sasl_maxbuf = *(int *)x;
00537                 }
00538         }
00539 
00540         return JABBER_SASL_STATE_OK;
00541 }

00543 static JabberSaslState
00544 jabber_cyrus_handle_failure(JabberStream *js, xmlnode *packet,
00545                             xmlnode **reply, char **error)
00546 {
00547         if (js->auth_fail_count++ < 5) {
00548                 if (js->current_mech && *js->current_mech) {
00549                         char *pos;
00550                         if ((pos = strstr(js->sasl_mechs->str, js->current_mech))) {
00551                                 g_string_erase(js->sasl_mechs, pos-js->sasl_mechs->str, strlen(js->current_mech));
00552                         }
00553                         /* Remove space which separated this mech from the next */
00554                         if ((js->sasl_mechs->str)[0] == ' ') {
00555                                 g_string_erase(js->sasl_mechs, 0, 1);
00556                         }
00557                 }
00558                 if (*js->sasl_mechs->str) {
00559                         /* If we have remaining mechs to try, do so */
00560                         sasl_dispose(&js->sasl);
00561 
00562                         return jabber_auth_start_cyrus(js, reply, error);
00563 
00564                 } else if ((js->auth_fail_count == 1) &&
00565                                    (js->current_mech && g_str_equal(js->current_mech, "GSSAPI"))) {
00566                         /* If we tried GSSAPI first, it failed, and it was the only method we had to try, try jabber:iq:auth
00567                          * for compatibility with iChat 10.5 Server and other jabberd based servers.
00568                          *
00569                          * iChat Server 10.5 and certain other corporate servers offer SASL GSSAPI by default, which is often
00570                          * not configured on the client side, and expects a fallback to jabber:iq:auth when it (predictably) fails.
00571                          *
00572                          * Note: xep-0078 points out that using jabber:iq:auth after a sasl failure is wrong. However,
00573                          * I believe this refers to actual authentication failure, not a simple lack of concordant mechanisms.
00574                          * Doing otherwise means that simply compiling with SASL support renders the client unable to connect to servers
00575                          * which would connect without issue otherwise. -evands
00576                          */
00577                         sasl_dispose(&js->sasl);
00578                         js->sasl = NULL;
00579                         js->auth_mech = NULL;
00580                         jabber_auth_start_old(js);
00581                         return JABBER_SASL_STATE_CONTINUE;
00582                 }
00583         }
00584 
00585         /* Nothing to send */
00586         return JABBER_SASL_STATE_FAIL;
00587 }
