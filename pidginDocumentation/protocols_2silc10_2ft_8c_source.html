00026 /* This implements the secure file transfer protocol (SFTP) using the SILC
00027    SFTP library implementation.  The API we use from the SILC Toolkit is the
00028    SILC Client file transfer API, as it provides a simple file transfer we
00029    need in this case.  We could use the SILC SFTP API directly, but it would
00030    be an overkill since we'd effectively re-implement the file transfer what
00031    the SILC Client's file transfer API already provides.
00032 
00033    From Purple we do NOT use the FT API to do the transfer as it is very limiting.
00034    In fact it does not suite to file transfers like SFTP at all.  For example,
00035    it assumes that read operations are synchronous what they are not in SFTP.
00036    It also assumes that the file transfer socket is to be handled by the Purple
00037    eventloop, and this naturally is something we don't want to do in case of
00038    SILC Toolkit.  The FT API suites well to purely stream based file transfers
00039    like HTTP GET and similar.
00040 
00041    For this reason, we directly access the Purple GKT FT API and hack the FT
00042    API to merely provide the user interface experience and all the magic
00043    is done in the SILC Toolkit.  Ie. we update the statistics information in
00044    the FT API for user interface, and that's it.  A bit dirty but until the
00045    FT API gets better this is the way to go.  Good thing that FT API allowed
00046    us to do this. */

00060 static void
00061 silcpurple_ftp_monitor(SilcClient client,
00062                      SilcClientConnection conn,
00063                      SilcClientMonitorStatus status,
00064                      SilcClientFileError error,
00065                      SilcUInt64 offset,
00066                      SilcUInt64 filesize,
00067                      SilcClientEntry client_entry,
00068                      SilcUInt32 session_id,
00069                      const char *filepath,
00070                      void *context)
00071 {
00072         SilcPurpleXfer xfer = context;
00073         PurpleConnection *gc = xfer->sg->gc;
00074         char tmp[256];
00075 
00076         if (status == SILC_CLIENT_FILE_MONITOR_CLOSED) {
00077                 purple_xfer_unref(xfer->xfer);
00078                 silc_free(xfer);
00079                 return;
00080         }
00081 
00082         if (status == SILC_CLIENT_FILE_MONITOR_KEY_AGREEMENT)
00083                 return;
00084 
00085         if (status == SILC_CLIENT_FILE_MONITOR_ERROR) {
00086                 if (error == SILC_CLIENT_FILE_NO_SUCH_FILE) {
00087                         g_snprintf(tmp, sizeof(tmp), "No such file %s",
00088                                    filepath ? filepath : "[N/A]");
00089                         purple_notify_error(gc, _("Secure File Transfer"),
00090                                           _("Error during file transfer"), tmp);
00091                 } else if (error == SILC_CLIENT_FILE_PERMISSION_DENIED) {
00092                         purple_notify_error(gc, _("Secure File Transfer"),
00093                                           _("Error during file transfer"),
00094                                           _("Permission denied"));
00095                 } else if (error == SILC_CLIENT_FILE_KEY_AGREEMENT_FAILED) {
00096                         purple_notify_error(gc, _("Secure File Transfer"),
00097                                           _("Error during file transfer"),
00098                                           _("Key agreement failed"));
00099                 } else if (error == SILC_CLIENT_FILE_UNKNOWN_SESSION) {
00100                         purple_notify_error(gc, _("Secure File Transfer"),
00101                                           _("Error during file transfer"),
00102                                           _("File transfer session does not exist"));
00103                 } else {
00104                         purple_notify_error(gc, _("Secure File Transfer"),
00105                                           _("Error during file transfer"), NULL);
00106                 }
00107                 silc_client_file_close(client, conn, session_id);
00108                 purple_xfer_unref(xfer->xfer);
00109                 silc_free(xfer);
00110                 return;
00111         }
00112 
00113         /* Update file transfer UI */
00114         if (!offset && filesize)
00115                 purple_xfer_set_size(xfer->xfer, filesize);
00116         if (offset && filesize) {
00117                 xfer->xfer->bytes_sent = offset;
00118                 xfer->xfer->bytes_remaining = filesize - offset;
00119         }
00120         purple_xfer_update_progress(xfer->xfer);
00121 
00122         if (status == SILC_CLIENT_FILE_MONITOR_SEND ||
00123             status == SILC_CLIENT_FILE_MONITOR_RECEIVE) {
00124                 if (offset == filesize) {
00125                         /* Download finished */
00126                         purple_xfer_set_completed(xfer->xfer, TRUE);
00127                         silc_client_file_close(client, conn, session_id);
00128                 }
00129         }
00130 }

00188         SilcClientFileError status;

00199         switch (status) {
00200         case SILC_CLIENT_FILE_OK:

00205                 purple_notify_error(gc, _("Secure File Transfer"),

00210                 purple_notify_error(gc, _("Secure File Transfer"),

00215                 purple_notify_error(gc, _("Secure File Transfer"),

00221                 purple_notify_error(gc, _("Secure File Transfer"),
00222                                   _("Could not start the file transfer"), NULL);
00223                 break;

00286 static void
00287 silcpurple_ftp_send(PurpleXfer *x)
00288 {
00289         SilcPurpleXfer xfer = x->data;
00290         const char *name;
00291         char *local_ip = NULL, *remote_ip = NULL;
00292         gboolean local = TRUE;
00293 
00294         name = purple_xfer_get_local_filename(x);
00295 
00296         /* Do the same magic what we do with key agreement (see silcpurple_buddy.c)
00297            to see if we are behind NAT. */
00298         if (silc_net_check_local_by_sock(xfer->sg->conn->sock->sock,
00299                                          NULL, &local_ip)) {
00300                 /* Check if the IP is private */
00301                 if (silcpurple_ip_is_private(local_ip)) {
00302                         local = FALSE;
00303                         /* Local IP is private, resolve the remote server IP to see whether
00304                            we are talking to Internet or just on LAN. */
00305                         if (silc_net_check_host_by_sock(xfer->sg->conn->sock->sock, NULL,
00306                                                         &remote_ip))
00307                                 if (silcpurple_ip_is_private(remote_ip))
00308                                         /* We assume we are in LAN.  Let's provide the connection point. */
00309                                         local = TRUE;
00310                 }
00311         }
00312 
00313         if (local && !local_ip)
00314                 local_ip = silc_net_localip();
00315 
00316         /* Send the file */
00317         silc_client_file_send(xfer->sg->client, xfer->sg->conn,
00318                               silcpurple_ftp_monitor, xfer,
00319                               local_ip, 0, !local, xfer->client_entry,
00320                               name, &xfer->session_id);
00321 
00322         silc_free(local_ip);
00323         silc_free(remote_ip);
00324 }
