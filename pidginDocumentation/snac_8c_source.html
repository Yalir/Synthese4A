00021 /*
00022  *
00023  * Various SNAC-related dodads...
00024  *
00025  * outstanding_snacs is a list of aim_snac_t structs.  A SNAC should be added
00026  * whenever a new SNAC is sent and it should remain in the list until the
00027  * response for it has been received.
00028  *
00029  * cleansnacs() should be called periodically by the client in order
00030  * to facilitate the aging out of unreplied-to SNACs. This can and does
00031  * happen, so it should be handled.
00032  *
00033  */

00052         aim_snac_t snac;

00054         snac.id = od->snacid_next++;

00067 /*
00068  * Clones the passed snac structure and caches it in the
00069  * list/hash.
00070  */
00071 aim_snacid_t aim_newsnac(OscarData *od, aim_snac_t *newsnac)
00072 {
00073         aim_snac_t *snac;
00074         int index;
00075 
00076         if (!newsnac)
00077                 return 0;
00078 
00079         snac = g_memdup(newsnac, sizeof(aim_snac_t));
00080         snac->issuetime = time(NULL);
00081 
00082         index = snac->id % FAIM_SNAC_HASH_SIZE;
00083 
00084         snac->next = (aim_snac_t *)od->snac_hash[index];
00085         od->snac_hash[index] = (void *)snac;
00086 
00087         return snac->id;
00088 }

00090 /*
00091  * Finds a snac structure with the passed SNAC ID,
00092  * removes it from the list/hash, and returns a pointer to it.
00093  *
00094  * The returned structure must be freed by the caller.
00095  *
00096  */
00097 aim_snac_t *aim_remsnac(OscarData *od, aim_snacid_t id)
00098 {
00099         aim_snac_t *cur, **prev;
00100         int index;
00101 
00102         index = id % FAIM_SNAC_HASH_SIZE;
00103 
00104         for (prev = (aim_snac_t **)&od->snac_hash[index]; (cur = *prev); ) {
00105                 if (cur->id == id) {
00106                         *prev = cur->next;
00107                         if (cur->flags & AIM_SNACFLAGS_DESTRUCTOR) {
00108                                 g_free(cur->data);
00109                                 cur->data = NULL;
00110                         }
00111                         return cur;
00112                 } else
00113                         prev = &cur->next;
00114         }
00115 
00116         return cur;
00117 }

00119 /*
00120  * This is for cleaning up old SNACs that either don't get replies or
00121  * a reply was never received for.  Garbage collection. Plain and simple.
00122  *
00123  * maxage is the _minimum_ age in seconds to keep SNACs.
00124  *
00125  */
00126 void aim_cleansnacs(OscarData *od, int maxage)
00127 {
00128         int i;
00129 
00130         for (i = 0; i < FAIM_SNAC_HASH_SIZE; i++) {
00131                 aim_snac_t *cur, **prev;
00132                 time_t curtime;
00133 
00134                 if (!od->snac_hash[i])
00135                         continue;
00136 
00137                 curtime = time(NULL); /* done here in case we waited for the lock */
00138 
00139                 for (prev = (aim_snac_t **)&od->snac_hash[i]; (cur = *prev); ) {
00140                         if ((curtime - cur->issuetime) > maxage) {
00141 
00142                                 *prev = cur->next;
00143 
00144                                 g_free(cur->data);
00145                                 g_free(cur);
00146                         } else
00147                                 prev = &cur->next;
00148                 }
00149         }
00150 
00151         return;
00152 }
