00057 void
00058 aim_genericreq_l(OscarData *od, FlapConnection *conn, guint16 family, guint16 subtype, guint32 *longdata)
00059 {
00060         ByteStream bs;
00061         aim_snacid_t snacid;
00062 
00063         if (!longdata)
00064         {
00065                 aim_genericreq_n(od, conn, family, subtype);
00066                 return;
00067         }
00068 
00069         byte_stream_new(&bs, 4);
00070 
00071         snacid = aim_cachesnac(od, family, subtype, 0x0000, NULL, 0);
00072 
00073         byte_stream_put32(&bs, *longdata);
00074 
00075         flap_connection_send_snac(od, conn, family, subtype, snacid, &bs);
00076 
00077         byte_stream_destroy(&bs);
00078 }

00080 /*
00081  * Should be generic enough to handle the errors for all groups.

00083  */
00084 static int
00085 generror(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00086 {
00087         int ret = 0;
00088         int error = 0;
00089         aim_rxcallback_t userfunc;
00090         aim_snac_t *snac2;
00091 
00092         snac2 = aim_remsnac(od, snac->id);
00093 
00094         if (byte_stream_bytes_left(bs))
00095                 error = byte_stream_get16(bs);
00096 
00097         if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00098                 ret = userfunc(od, conn, frame, error, snac2 ? snac2->data : NULL);
00099 
00100         if (snac2) {
00101                 g_free(snac2->data);
00102                 g_free(snac2);
00103         }
00104 
00105         return ret;
00106 }

00108 static int
00109 snachandler(OscarData *od, FlapConnection *conn, aim_module_t *mod, FlapFrame *frame, aim_modsnac_t *snac, ByteStream *bs)
00110 {
00111         if (snac->subtype == 0x0001)
00112                 return generror(od, conn, mod, frame, snac, bs);
00113         else if ((snac->family == 0xffff) && (snac->subtype == 0xffff)) {
00114                 aim_rxcallback_t userfunc;
00115 
00116                 if ((userfunc = aim_callhandler(od, snac->family, snac->subtype)))
00117                         return userfunc(od, conn, frame);
00118         }
00119 
00120         return 0;
00121 }
