00024 encoding_multi_convert_to_utf8(const gchar *text, gssize textlen, const gchar *encodings, GError **error, gboolean fallback)
00025 {
00026         gchar *utf8 = NULL;
00027         const gchar *begin = encodings;
00028         const gchar *end = NULL;
00029         gchar *curr_encoding = NULL; /* allocated buffer for encoding name */
00030         const gchar *curr_encoding_ro = NULL; /* read-only encoding name */
00031 
00032         if (!encodings) {
00033                 purple_debug_error("oscar", "encodings is NULL");
00034                 return NULL;
00035         }
00036 
00037         for (;;)
00038         {
00039                 /* extract next encoding */
00040                 end = strchr(begin, ',');
00041                 if (!end) {
00042                         curr_encoding_ro = begin;
00043                 }       else { /* allocate buffer for encoding */
00044                         curr_encoding = g_strndup(begin, end - begin);
00045                         if (!curr_encoding) {
00046                                 purple_debug_error("oscar", "Error allocating memory for encoding");
00047                                 break;
00048                         }
00049                         curr_encoding_ro = curr_encoding;
00050                 }
00051 
00052                 if (!g_ascii_strcasecmp(curr_encoding_ro, "utf-8") && g_utf8_validate(text, textlen, NULL)) {
00053                         break;
00054                 }
00055 
00056                 utf8 = g_convert(text, textlen, "UTF-8", curr_encoding_ro, NULL, NULL, NULL);
00057 
00058                 if (!end) /* last occurence. do not free curr_encoding: buffer was'nt allocated */
00059                         break;
00060 
00061                 g_free(curr_encoding); /* free allocated buffer for encoding here */
00062 
00063                 if (utf8) /* text was successfully converted */
00064                         break;
00065 
00066                 begin = end + 1;
00067         }
00068 
00069         if (!utf8 && fallback)
00070         { /* "begin" points to last encoding */
00071                 utf8 = g_convert_with_fallback(text, textlen, "UTF-8", begin, "?", NULL, NULL, error);
00072         }
00073 
00074         return utf8;
00075 }

00104 {
00105         gchar *utf8 = NULL;
00106         const gchar *glib_encoding = NULL;
00107         gchar *extracted_encoding = encoding_extract(encoding);
00108 
00109         if (extracted_encoding == NULL || *extracted_encoding == '\0') {
00110                 purple_debug_info("oscar", "Empty encoding, assuming UTF-8\n");
00111         } else if (!g_ascii_strcasecmp(extracted_encoding, "iso-8859-1")) {
00112                 glib_encoding = "iso-8859-1";
00113         } else if (!g_ascii_strcasecmp(extracted_encoding, "ISO-8859-1-Windows-3.1-Latin-1") || !g_ascii_strcasecmp(extracted_encoding, "us-ascii")) {
00114                 glib_encoding = "Windows-1252";
00115         } else if (!g_ascii_strcasecmp(extracted_encoding, "unicode-2-0")) {
00116                 glib_encoding = "UTF-16BE";
00117         } else if (g_ascii_strcasecmp(extracted_encoding, "utf-8")) {
00118                 glib_encoding = extracted_encoding;
00119         }
00120 
00121         if (glib_encoding != NULL) {
00122                 utf8 = encoding_multi_convert_to_utf8(text, textlen, glib_encoding, NULL, FALSE);
00123         }
00124 
00125         /*
00126          * If utf8 is still NULL then either the encoding is utf-8 or
00127          * we have been unable to convert the text to utf-8 from the encoding
00128          * that was specified.  So we check if the text is valid utf-8 then
00129          * just copy it.
00130          */
00131         if (utf8 == NULL) {
00132                 if (textlen != 0 && *text != '\0' && !g_utf8_validate(text, textlen, NULL))
00133                         utf8 = g_strdup(_("(There was an error receiving this message.  The buddy you are speaking with is probably using a different encoding than expected.  If you know what encoding he is using, you can specify it in the advanced account options for your AIM/ICQ account.)"));
00134                 else
00135                         utf8 = g_strndup(text, textlen);
00136         }
00137 
00138         g_free(extracted_encoding);
00139         return utf8;

00198 
00199         if ((datalen == 0) || (data == NULL))
00200                 return NULL;
00201

00203                 charsetstr1 = "UTF-16BE";

00206                 if ((sourcebn != NULL) && oscar_util_valid_name_icq(sourcebn))
00207                         charsetstr1 = purple_account_get_string(account, "encoding", OSCAR_DEFAULT_CUSTOM_ENCODING);
00208                 else
00209                         charsetstr1 = "ISO-8859-1";
00210                 charsetstr2 = "UTF-8";
00211         } else if (charset == AIM_CHARSET_ASCII) {
00212                 /* Should just be "ASCII" */
00213                 charsetstr1 = "ASCII";
00214                 charsetstr2 = purple_account_get_string(account, "encoding", OSCAR_DEFAULT_CUSTOM_ENCODING);
00215         } else if (charset == 0x000d) {
00216                 /* iChat sending unicode over a Direct IM connection = UTF-8 */
00217                 /* Mobile AIM client on multiple devices (including Blackberry Tour, Nokia 3100, and LG VX6000) = ISO-8859-1 */
00218                 charsetstr1 = "UTF-8";
00219                 charsetstr2 = "ISO-8859-1";
00220                 charsetstr3 = purple_account_get_string(account, "encoding", OSCAR_DEFAULT_CUSTOM_ENCODING);
00221         } else {
00222                 /* Unknown, hope for valid UTF-8... */
00223                 charsetstr1 = "UTF-8";
00224                 charsetstr2 = purple_account_get_string(account, "encoding", OSCAR_DEFAULT_CUSTOM_ENCODING);
00225         }
00226 
00227         purple_debug_info("oscar", "Parsing IM, charset=0x%04hx, datalen=%" G_GSIZE_FORMAT ", choice1=%s, choice2=%s, choice3=%s\n",
00228                                           charset, datalen, charsetstr1, charsetstr2, (charsetstr3 ? charsetstr3 : ""));
00229 
00230         ret = oscar_convert_to_utf8(data, datalen, charsetstr1, FALSE);
00231         if (ret == NULL) {
00232                 if (charsetstr3 != NULL) {
00233                         /* Try charsetstr2 without allowing substitutions, then fall through to charsetstr3 if needed */
00234                         ret = oscar_convert_to_utf8(data, datalen, charsetstr2, FALSE);
00235                         if (ret == NULL)
00236                                 ret = oscar_convert_to_utf8(data, datalen, charsetstr3, TRUE);
00237                 } else {
00238                         /* Try charsetstr2, allowing substitutions */
00239                         ret = oscar_convert_to_utf8(data, datalen, charsetstr2, TRUE);
00240                 }
00241         }
00242         if (ret == NULL) {
00243                 char *str, *salvage, *tmp;
00244 
00245                 str = g_malloc(datalen + 1);
00246                 strncpy(str, data, datalen);
00247                 str[datalen] = '\0';
00248                 salvage = purple_utf8_salvage(str);
00249                 tmp = g_strdup_printf(_("(There was an error receiving this message.  Either you and %s have different encodings selected, or %s has a buggy client.)"),
00250                                           sourcebn, sourcebn);
00251                 ret = g_strdup_printf("%s %s", salvage, tmp);
00252                 g_free(tmp);
00253                 g_free(str);
00254                 g_free(salvage);
00255         }
00256 
00257         return ret;
00258 }
