01421                                      const char *id, const char *alias, const char *message, gboolean on_list,
01422                                      PurpleAccountRequestAuthorizationCb auth_cb, PurpleAccountRequestAuthorizationCb deny_cb, void *user_data)
01423 {

01434                         purple_signal_emit_return_1(purple_accounts_get_handle(),

01439                         auth_cb(user_data);

02902 void
02903 purple_accounts_delete(PurpleAccount *account)
02904 {
02905         PurpleBlistNode *gnode, *cnode, *bnode;
02906         GList *iter;
02907 
02908         g_return_if_fail(account != NULL);
02909 
02910         /*
02911          * Disable the account before blowing it out of the water.
02912          * Conceptually it probably makes more sense to disable the
02913          * account for all UIs rather than the just the current UI,
02914          * but it doesn't really matter.
02915          */
02916         purple_account_set_enabled(account, purple_core_get_ui(), FALSE);
02917 
02918         purple_notify_close_with_handle(account);
02919         purple_request_close_with_handle(account);
02920 
02921         purple_accounts_remove(account);
02922 
02923         /* Remove this account's buddies */
02924         for (gnode = purple_blist_get_root();
02925              gnode != NULL;
02926                  gnode = purple_blist_node_get_sibling_next(gnode))
02927         {
02928                 if (!PURPLE_BLIST_NODE_IS_GROUP(gnode))
02929                         continue;
02930 
02931                 cnode = purple_blist_node_get_first_child(gnode);
02932                 while (cnode) {
02933                         PurpleBlistNode *cnode_next = purple_blist_node_get_sibling_next(cnode);
02934 
02935                         if(PURPLE_BLIST_NODE_IS_CONTACT(cnode)) {
02936                                 bnode = purple_blist_node_get_first_child(cnode);
02937                                 while (bnode) {
02938                                         PurpleBlistNode *bnode_next = purple_blist_node_get_sibling_next(bnode);
02939 
02940                                         if (PURPLE_BLIST_NODE_IS_BUDDY(bnode)) {
02941                                                 PurpleBuddy *b = (PurpleBuddy *)bnode;
02942 
02943                                                 if (purple_buddy_get_account(b) == account)
02944                                                         purple_blist_remove_buddy(b);
02945                                         }
02946                                         bnode = bnode_next;
02947                                 }
02948                         } else if (PURPLE_BLIST_NODE_IS_CHAT(cnode)) {
02949                                 PurpleChat *c = (PurpleChat *)cnode;
02950 
02951                                 if (purple_chat_get_account(c) == account)
02952                                         purple_blist_remove_chat(c);
02953                         }
02954                         cnode = cnode_next;
02955                 }
02956         }
02957 
02958         /* Remove any open conversation for this account */
02959         for (iter = purple_get_conversations(); iter; ) {
02960                 PurpleConversation *conv = iter->data;
02961                 iter = iter->next;
02962                 if (purple_conversation_get_account(conv) == account)
02963                         purple_conversation_destroy(conv);
02964         }
02965 
02966         /* Remove this account's pounces */
02967         purple_pounce_destroy_all_by_account(account);
02968 
02969         /* This will cause the deletion of an old buddy icon. */
02970         purple_buddy_icons_set_account_icon(account, NULL, 0);
02971 
02972         purple_account_destroy(account);
02973 }
