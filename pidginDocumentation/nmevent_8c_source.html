00176                         if (!autoreply) {

00184                                                    "Message without RTF is %s\n", nortf);

00187                                 nm_event_set_text(event, nortf);
00188 
00189                         } else {
00190 
00191                                 /* Store the event data */
00192                                 nm_event_set_text(event, msg);
00193                         }
00194                 }
00195         }
00196 
00197         /* Check to see if we already know about the conference */
00198         conference = nm_conference_list_find(user, guid);
00199         if (conference) {
00200

00202                 nm_event_set_conference(event, conference);
00203 
00204                 /* Add a reference to the user record in our event object */
00205                 user_record = nm_find_user_record(user, nm_event_get_source(event));
00206                 if (user_record) {
00207                         nm_event_set_user_record(event, user_record);
00208                 }
00209 
00210         } else {
00211

00213                 conference = nm_create_conference(guid);

00215 
00216                 /* Add a reference to the conference in the event */
00217                 nm_event_set_conference(event, conference);
00218 
00219                 /* Add new conference to the conference list */
00220                 nm_conference_list_add(user, conference);
00221 
00222                 /* Check to see if we have details for the event source yet */

00224                 if (user_record) {
00225 
00226                         /* We do so add the user record as a recipient of the conference */
00227                         nm_conference_add_participant(conference, user_record);
00228 
00229                         /* Add a reference to the user record in our event object */
00230                         nm_event_set_user_record(event, user_record);
00231 
00232                 } else {
00233

00235                         rc = nm_send_get_details(user, nm_event_get_source(event),
00236                                                                          _got_user_for_conference, event);
00237                         if (rc == NM_OK)
00238                                 rc = -1;                /* Not done processing the event yet! */
00239                 }
00240 
00241                 nm_release_conference(conference);
00242         }
00243 
00244         if (msg)
00245                 g_free(msg);
00246 
00247         if (nortf)
00248                 g_free(nortf);

00826                 return NMERR_BAD_PARM;

00829                 return NMERR_PROTOCOL;

00851                                 rc = handle_status_change(user, event);

00864                                 rc = handle_typing(user, event);

00873                                 break;
