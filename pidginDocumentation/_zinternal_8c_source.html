00069 /* Find or insert uid in the old uids buffer.  The buffer is a sorted
00070  * circular queue.  We make the assumption that most packets arrive in
00071  * order, so we can usually search for a uid or insert it into the buffer
00072  * by looking back just a few entries from the end.  Since this code is
00073  * only executed by the client, the implementation isn't microoptimized. */
00074 static int find_or_insert_uid(uid, kind)
00075     ZUnique_Id_t *uid;
00076     ZNotice_Kind_t kind;
00077 {
00078     static struct _filter {
00079         ZUnique_Id_t    uid;
00080         ZNotice_Kind_t  kind;
00081         time_t          t;
00082     } *buffer;
00083     static long size;
00084     static long start;
00085     static long num;
00086 
00087     time_t now;
00088     struct _filter *new;
00089     long i, j, new_size;
00090     int result;
00091 
00092     /* Initialize the uid buffer if it hasn't been done already. */
00093     if (!buffer) {
00094         size = Z_INITFILTERSIZE;
00095         buffer = (struct _filter *) malloc(size * sizeof(*buffer));
00096         if (!buffer)
00097             return 0;
00098     }
00099 
00100     /* Age the uid buffer, discarding any uids older than the clock skew. */
00101     time(&now);
00102     while (num && (now - buffer[start % size].t) > CLOCK_SKEW)
00103         start++, num--;
00104     start %= size;
00105 
00106     /* Make room for a new uid, since we'll probably have to insert one. */
00107     if (num == size) {
00108         new_size = size * 2 + 2;
00109         new = (struct _filter *) malloc(new_size * sizeof(*new));
00110         if (!new)
00111             return 0;
00112         for (i = 0; i < num; i++)
00113             new[i] = buffer[(start + i) % size];
00114         free(buffer);
00115         buffer = new;
00116         size = new_size;
00117         start = 0;
00118     }
00119 
00120     /* Search for this uid in the buffer, starting from the end. */
00121     for (i = start + num - 1; i >= start; i--) {
00122         result = memcmp(uid, &buffer[i % size].uid, sizeof(*uid));
00123         if (result == 0 && buffer[i % size].kind == kind)
00124             return 1;
00125         if (result > 0)
00126             break;
00127     }
00128 
00129     /* We didn't find it; insert the uid into the buffer after i. */
00130     i++;
00131     for (j = start + num; j > i; j--)
00132         buffer[j % size] = buffer[(j - 1) % size];
00133     buffer[i % size].uid = *uid;
00134     buffer[i % size].kind = kind;
00135     buffer[i % size].t = now;
00136     num++;
00137 
00138     return 0;
00139 }
