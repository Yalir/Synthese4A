00038 static gboolean
00039 purple_sound_play_required(const PurpleAccount *account)
00040 {
00041         gint pref_status = purple_prefs_get_int("/purple/sound/while_status");
00042 
00043         if (pref_status == 3)
00044         {
00045                 /* Play sounds: Always */
00046                 return TRUE;
00047         }
00048 
00049         if (account != NULL)
00050         {
00051                 PurpleStatus *status = purple_account_get_active_status(account);
00052 
00053                 if (purple_status_is_online(status))
00054                 {
00055                         gboolean available = purple_status_is_available(status);
00056                         return (( available && pref_status == STATUS_AVAILABLE) ||
00057                                 (!available && pref_status == STATUS_AWAY));
00058                 }
00059         }
00060 
00061         /* We get here a couple of ways.  Either the request has been OK'ed
00062          * by purple_sound_play_event() and we're here because the UI has
00063          * called purple_sound_play_file(), or we're here for something
00064          * not related to an account (like testing a sound). */
00065         return TRUE;
00066 }

00078 void
00079 purple_sound_play_event(PurpleSoundEventID event, const PurpleAccount *account)
00080 {
00081         if (!purple_sound_play_required(account))
00082                 return;
00083 
00084         if (time(NULL) - last_played[event] < 2)
00085                 return;
00086         last_played[event] = time(NULL);
00087 
00088         if(sound_ui_ops && sound_ui_ops->play_event) {
00089                 int plugin_return;
00090 
00091                 plugin_return = GPOINTER_TO_INT(purple_signal_emit_return_1(
00092                         purple_sounds_get_handle(), "playing-sound-event",
00093                         event, account));
00094 
00095                 if (plugin_return)
00096                         return;
00097                 else
00098                         sound_ui_ops->play_event(event);
00099         }
00100 }
