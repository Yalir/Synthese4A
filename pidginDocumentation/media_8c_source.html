00652 purple_media_end(PurpleMedia *media,
00653                 const gchar *session_id, const gchar *participant)
00654 {
00655 #ifdef USE_VV
00656         GList *iter, *sessions = NULL, *participants = NULL;
00657 
00658         g_return_if_fail(PURPLE_IS_MEDIA(media));
00659 
00660         iter = purple_media_get_streams(media, session_id, participant);
00661 
00662         /* Free matching streams */
00663         for (; iter; iter = g_list_delete_link(iter, iter)) {
00664                 PurpleMediaStream *stream = iter->data;
00665 
00666                 g_signal_emit(media, purple_media_signals[STATE_CHANGED],
00667                                 0, PURPLE_MEDIA_STATE_END,
00668                                 stream->session->id, stream->participant);
00669 
00670                 media->priv->streams =
00671                                 g_list_remove(media->priv->streams, stream);
00672 
00673                 if (g_list_find(sessions, stream->session) == NULL)
00674                         sessions = g_list_prepend(sessions, stream->session);
00675 
00676                 if (g_list_find_custom(participants, stream->participant,
00677                                 (GCompareFunc)strcmp) == NULL)
00678                         participants = g_list_prepend(participants,
00679                                         g_strdup(stream->participant));
00680 
00681                 purple_media_stream_free(stream);
00682         }
00683 
00684         iter = media->priv->streams;
00685 
00686         /* Reduce to list of sessions to remove */
00687         for (; iter; iter = g_list_next(iter)) {
00688                 PurpleMediaStream *stream = iter->data;
00689 
00690                 sessions = g_list_remove(sessions, stream->session);
00691         }
00692 
00693         /* Free sessions with no streams left */
00694         for (; sessions; sessions = g_list_delete_link(sessions, sessions)) {
00695                 PurpleMediaSession *session = sessions->data;
00696 
00697                 g_signal_emit(media, purple_media_signals[STATE_CHANGED],
00698                                 0, PURPLE_MEDIA_STATE_END,
00699                                 session->id, NULL);
00700 
00701                 g_hash_table_remove(media->priv->sessions, session->id);
00702                 purple_media_session_free(session);
00703         }
00704 
00705         iter = media->priv->streams;
00706 
00707         /* Reduce to list of participants to remove */
00708         for (; iter; iter = g_list_next(iter)) {
00709                 PurpleMediaStream *stream = iter->data;
00710                 GList *tmp;
00711 
00712                 tmp = g_list_find_custom(participants,
00713                                 stream->participant, (GCompareFunc)strcmp);
00714 
00715                 if (tmp != NULL) {
00716                         g_free(tmp->data);
00717                         participants = g_list_delete_link(participants, tmp);
00718                 }
00719         }
00720 
00721         /* Remove participants with no streams left (just emit the signal) */
00722         for (; participants; participants =
00723                         g_list_delete_link(participants, participants)) {
00724                 gchar *participant = participants->data;
00725                 GList *link = g_list_find_custom(media->priv->participants,
00726                                 participant, (GCompareFunc)strcmp);
00727 
00728                 g_signal_emit(media, purple_media_signals[STATE_CHANGED],
00729                                 0, PURPLE_MEDIA_STATE_END,
00730                                 NULL, participant);
00731 
00732                 if (link != NULL) {
00733                         g_free(link->data);
00734                         media->priv->participants = g_list_delete_link(
00735                                         media->priv->participants, link);
00736                 }
00737 
00738                 g_free(participant);
00739         }
00740 
00741         /* Free the conference if no sessions left */
00742         if (media->priv->sessions != NULL &&
00743                         g_hash_table_size(media->priv->sessions) == 0) {
00744                 g_signal_emit(media, purple_media_signals[STATE_CHANGED],
00745                                 0, PURPLE_MEDIA_STATE_END,
00746                                 NULL, NULL);
00747                 g_object_unref(media);
00748                 return;
00749         }
00750 #endif

00755                 const gchar *session_id, const gchar *participant,

00759         g_return_if_fail(PURPLE_IS_MEDIA(media));

00771                                 g_list_delete_link(streams, streams)) {

00794                                 g_list_delete_link(sessions, sessions)) {
00795                         PurpleMediaSession *session = sessions->data;
00796

00798                                 g_signal_emit(media, purple_media_signals[
00799                                                 STREAM_INFO], 0,

00806                                 participants, participants)) {
00807                         gchar *participant = participants->data;

00810                                 g_signal_emit(media, purple_media_signals[
00811                                                 STREAM_INFO], 0,

00819                 if (purple_media_accepted(media, NULL, NULL))
00820                         g_signal_emit(media,
00821                                         purple_media_signals[STREAM_INFO],
00822                                         0, PURPLE_MEDIA_INFO_ACCEPT,
00823                                         NULL, NULL, local);
00824

00826         } else if (type == PURPLE_MEDIA_INFO_HANGUP ||
00827                         type == PURPLE_MEDIA_INFO_REJECT) {
00828                 GList *streams;
00829

00831 
00832                 streams = purple_media_get_streams(media,
00833                                 session_id, participant);
00834 
00835                 /* Emit for stream */

00837                                 g_list_delete_link(streams, streams)) {
00838                         PurpleMediaStream *stream = streams->data;
00839 
00840                         g_signal_emit(media,
00841                                         purple_media_signals[STREAM_INFO],
00842                                         0, type, stream->session->id,
00843                                         stream->participant, local);
00844                 }
00845 
00846                 if (session_id != NULL && participant != NULL) {
00847                         /* Everything that needs to be emitted has been */
00848                 } else if (session_id == NULL && participant == NULL) {
00849                         /* Emit for everything in the conference */
00850                         GList *sessions = NULL;
00851                         GList *participants = media->priv->participants;
00852 
00853                         if (media->priv->sessions != NULL)
00854                                 sessions = g_hash_table_get_values(
00855                                         media->priv->sessions);
00856 
00857                         /* Emit for sessions */
00858                         for (; sessions; sessions = g_list_delete_link(
00859                                         sessions, sessions)) {
00860                                 PurpleMediaSession *session = sessions->data;
00861 
00862                                 g_signal_emit(media, purple_media_signals[
00863                                                 STREAM_INFO], 0, type,
00864                                                 session->id, NULL, local);
00865                         }
00866 
00867                         /* Emit for participants */
00868                         for (; participants; participants =
00869                                         g_list_next(participants)) {
00870                                 gchar *participant = participants->data;
00871 
00872                                 g_signal_emit(media, purple_media_signals[
00873                                                 STREAM_INFO], 0, type,
00874                                                 NULL, participant, local);
00875                         }
00876 
00877                         /* Emit for conference */
00878                         g_signal_emit(media,
00879                                         purple_media_signals[STREAM_INFO],
00880                                         0, type, NULL, NULL, local);
00881                 } else if (session_id != NULL) {
00882                         /* Emit just the specific session */
00883                         PurpleMediaSession *session =
00884                                         purple_media_get_session(
00885                                         media, session_id);
00886 
00887                         if (session == NULL) {
00888                                 purple_debug_warning("media",
00889                                                 "Couldn't find session"
00890                                                 " to hangup/reject.\n");
00891                         } else {
00892                                 g_signal_emit(media, purple_media_signals[
00893                                                 STREAM_INFO], 0, type,
00894                                                 session->id, NULL, local);
00895                         }
00896                 } else if (participant != NULL) {
00897                         /* Emit just the specific participant */
00898                         if (!g_list_find_custom(media->priv->participants,
00899                                         participant, (GCompareFunc)strcmp)) {
00900                                 purple_debug_warning("media",
00901                                                 "Couldn't find participant"
00902                                                 " to hangup/reject.\n");
00903                         } else {
00904                                 g_signal_emit(media, purple_media_signals[
00905                                                 STREAM_INFO], 0, type, NULL,
00906                                                 participant, local);
00907                         }
00908                 }
00909 
00910                 purple_media_end(media, session_id, participant);
00911                 return;
00912         }
00913 
00914         g_signal_emit(media, purple_media_signals[STREAM_INFO],
00915                         0, type, session_id, participant, local);
00916 #endif
