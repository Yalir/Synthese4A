00042 const char* splash_current(struct MXitSession* session)

00079 /*------------------------------------------------------------------------

00081  *
00082  *  @param session              The MXit session object
00083  */
00084 void splash_remove(struct MXitSession* session)
00085 {
00086         const char* splashId = NULL;
00087         char* filename;
00088 
00089         /* Get current splash ID */
00090         splashId = splash_current(session);
00091

00093                 purple_debug_info(MXIT_PLUGIN_ID, "Removing splashId: '%s'\n", splashId);

00096                 filename = g_strdup_printf("%s/mxit/%s.png", purple_user_dir(), splashId);
00097                 g_unlink(filename);
00098                 g_free(filename);
00099 
00100                 /* Clear current splash ID from settings */
00101                 purple_account_set_string(session->acc, MXIT_CONFIG_SPLASHID, "");
00102                 purple_account_set_bool(session->acc, MXIT_CONFIG_SPLASHCLICK, FALSE);
00103         }
00104 }

00107 /*------------------------------------------------------------------------
00108  * Save a new splash-screen for later display.
00109  *
00110  *  @param session              The MXit session object
00111  *  @param splashID             The ID of the splash-screen
00112  *  @param data                 Splash-screen image data (PNG format)
00113  *  @param datalen              Splash-screen image data size

00115 void splash_update(struct MXitSession* session, const char* splashId, const char* data, int datalen, gboolean clickable)
00116 {
00117         char* dir;
00118         char* filename;
00119 
00120         /* Remove the current splash-screen */
00121         splash_remove(session);
00122 
00123         /* Save the new splash image */
00124         dir = g_strdup_printf("%s" G_DIR_SEPARATOR_S "mxit",  purple_user_dir());
00125         purple_build_dir(dir, S_IRUSR | S_IWUSR | S_IXUSR);             /* ensure directory exists */
00126 
00127         filename = g_strdup_printf("%s" G_DIR_SEPARATOR_S "%s.png", dir, purple_escape_filename(splashId));
00128         if (purple_util_write_data_to_file_absolute(filename, data, datalen)) {
00129                 /* Store new splash-screen ID to settings */
00130                 purple_account_set_string(session->acc, MXIT_CONFIG_SPLASHID, splashId);
00131                 purple_account_set_bool(session->acc, MXIT_CONFIG_SPLASHCLICK, clickable );
00132         }
00133 
00134         g_free(dir);
00135         g_free(filename);

00139 /*------------------------------------------------------------------------

00141  *
00142  *  @param gc                   The connection object

00145 static void splash_click_ok(PurpleConnection* gc, PurpleRequestFields* fields)
00146 {
00147         struct MXitSession*     session = (struct MXitSession*) gc->proto_data;
00148         const char* splashId;
00149 
00150         /* Get current splash ID */
00151         splashId = splash_current(session);

00166 void splash_display(struct MXitSession* session)

00168         const char* splashId = NULL;

00179         purple_debug_info(MXIT_PLUGIN_ID, "Display Splash: '%s'\n", splashId);

00181         /* Load splash-screen image from file */

00183         if (g_file_get_contents(filename, &imgdata, &imglen, NULL)) {
00184                 char buf[128];
00185 
00186                 /* Add splash-image to imagestore */
00187                 imgid = purple_imgstore_add_with_id(g_memdup(imgdata, imglen), imglen, NULL);
00188 
00189                 /* Generate and display message */
00190                 g_snprintf(buf, sizeof(buf), "<img id=\"%d\">", imgid);
00191 
00192                 /* Open a request-type popup to display the image */
00193                 {
00194                         PurpleRequestFields*            fields;
00195                         PurpleRequestFieldGroup*        group;
00196                         PurpleRequestField*                     field;
00197 
00198                         fields = purple_request_fields_new();
00199                         group = purple_request_field_group_new(NULL);
00200                         purple_request_fields_add_group(fields, group);
00201 
00202                         field = purple_request_field_image_new("splash", "", imgdata, imglen);          /* add splash image */
00203                         purple_request_field_group_add_field(group, field);
00204 
00205                         if (splash_clickable(session)) {
00206                                 purple_request_fields(session->con, _("MXit Advertising"), NULL, NULL, fields,
00207                                         _("More Information"), G_CALLBACK(splash_click_ok), _("Close"), NULL, session->acc, NULL, NULL, session->con);
00208                         }
00209                         else {
00210                                 purple_request_fields(session->con, _("MXit Advertising"), NULL, NULL, fields,
00211                                         _("Continue"), G_CALLBACK(splash_click_ok), _("Close"), NULL, session->acc, NULL, NULL, session->con);
00212                         }
00213                 }
00214 
00215                 /* Release reference to image */
00216                 purple_imgstore_unref_by_id(imgid);
00217 
00218                 g_free(imgdata);
00219         }
00220 
00221         g_free(filename);
