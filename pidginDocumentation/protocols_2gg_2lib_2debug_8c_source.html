00184 
00185                 for (j = 0; j < 16; j++) {
00186                         unsigned char ch;

00189                                 ch = buf[i + j];

00192                                         ch = '.';
00193                         } else {
00194                                 ch = ' ';
00195                         }
00196 
00197                         line[ofs++] = ch;
