00044 static void grow_circ_buffer(PurpleCircBuffer *buf, gsize len) {
00045         int in_offset = 0, out_offset = 0;
00046         int start_buflen;
00047 
00048         g_return_if_fail(buf != NULL);
00049 
00050         start_buflen = buf->buflen;
00051 
00052         while ((buf->buflen - buf->bufused) < len)
00053                 buf->buflen += buf->growsize;
00054 
00055         if (buf->inptr != NULL) {
00056                 in_offset = buf->inptr - buf->buffer;
00057                 out_offset = buf->outptr - buf->buffer;
00058         }
00059         buf->buffer = g_realloc(buf->buffer, buf->buflen);
00060 
00061         /* adjust the fill and remove pointer locations */
00062         if (buf->inptr == NULL) {
00063                 buf->inptr = buf->outptr = buf->buffer;
00064         } else {
00065                 buf->inptr = buf->buffer + in_offset;
00066                 buf->outptr = buf->buffer + out_offset;
00067         }
00068 
00069         /* If the fill pointer is wrapped to before the remove
00070          * pointer, we need to shift the data */
00071         if (in_offset < out_offset
00072                         || (in_offset == out_offset && buf->bufused > 0)) {
00073                 int shift_n = MIN(buf->buflen - start_buflen,
00074                         in_offset);
00075                 memcpy(buf->buffer + start_buflen, buf->buffer,
00076                         shift_n);
00077 
00078                 /* If we couldn't fit the wrapped read buffer
00079                  * at the end */
00080                 if (shift_n < in_offset) {
00081                         memmove(buf->buffer,
00082                                 buf->buffer + shift_n,
00083                                 in_offset - shift_n);
00084                         buf->inptr = buf->buffer +
00085                                 (in_offset - shift_n);
00086                 } else {
00087                         buf->inptr = buf->buffer +
00088                                 start_buflen + in_offset;
00089                 }
00090         }
00091 }

00093 void purple_circ_buffer_append(PurpleCircBuffer *buf, gconstpointer src, gsize len) {
00094 
00095         int len_stored;
00096 
00097         g_return_if_fail(buf != NULL);
00098 
00099         /* Grow the buffer, if necessary */
00100         if ((buf->buflen - buf->bufused) < len)
00101                 grow_circ_buffer(buf, len);
00102 
00103         /* If there is not enough room to copy all of src before hitting
00104          * the end of the buffer then we will need to do two copies.
00105          * One copy from inptr to the end of the buffer, and the
00106          * second copy from the start of the buffer to the end of src. */
00107         if (buf->inptr >= buf->outptr)
00108                 len_stored = MIN(len, buf->buflen
00109                         - (buf->inptr - buf->buffer));
00110         else
00111                 len_stored = len;
00112 
00113         if (len_stored > 0)
00114                 memcpy(buf->inptr, src, len_stored);
00115 
00116         if (len_stored < len) {
00117                 memcpy(buf->buffer, (char*)src + len_stored, len - len_stored);
00118                 buf->inptr = buf->buffer + (len - len_stored);
00119         } else {
00120                 buf->inptr += len_stored;
00121         }
00122 
00123         buf->bufused += len;
00124 }
