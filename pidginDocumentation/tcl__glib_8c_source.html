00123 static void tcl_set_timer(Tcl_Time *timePtr)
00124 {
00125         guint interval;
00126 
00127         if (tcl_timer_pending)
00128                 g_source_remove(tcl_timer);

00155 {
00156         struct tcl_file_handler *tfh = g_new0(struct tcl_file_handler, 1);
00157         GIOChannel *channel;
00158         GIOCondition cond = 0;
00159 
00160         if (g_hash_table_lookup(tcl_file_handlers, GINT_TO_POINTER(fd)))
00161             tcl_delete_file_handler(fd);
00162 
00163         if (mask & TCL_READABLE)
00164                 cond |= G_IO_IN;
00165         if (mask & TCL_WRITABLE)
00166                 cond |= G_IO_OUT;
00167         if (mask & TCL_EXCEPTION)
00168                 cond |= G_IO_ERR|G_IO_HUP|G_IO_NVAL;
00169 
00170         tfh->fd = fd;
00171         tfh->mask = mask;
00172         tfh->proc = proc;

00175         channel = g_io_channel_unix_new(fd);

00179         g_hash_table_insert(tcl_file_handlers, GINT_TO_POINTER(fd), tfh);

00206 static gboolean tcl_file_callback(GIOChannel *source, GIOCondition condition, gpointer data)
00207 {
00208         struct tcl_file_handler *tfh = data;
00209         struct tcl_file_event *fev;
00210         int mask = 0;
00211 
00212         if (condition & G_IO_IN)
00213                 mask |= TCL_READABLE;
00214         if (condition & G_IO_OUT)
00215                 mask |= TCL_WRITABLE;
00216         if (condition & (G_IO_ERR|G_IO_HUP|G_IO_NVAL))
00217                 mask |= TCL_EXCEPTION;
00218 
00219         if (!(tfh->mask & (mask & ~tfh->pending)))
00220                 return TRUE;

00235 {
00236         struct tcl_file_handler *tfh;
00237         struct tcl_file_event *fev = (struct tcl_file_event *)event;
00238         int mask;
00239 
00240         if (!(flags & TCL_FILE_EVENTS)) {
00241                 return 0;

00244         tfh = g_hash_table_lookup(tcl_file_handlers, GINT_TO_POINTER(fev->fd));
00245         if (tfh == NULL)
00246                 return 1;
00247 
00248         mask = tfh->mask & tfh->pending;
