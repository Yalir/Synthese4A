00504 char *
00505 purple_notify_user_info_get_text_with_newline(PurpleNotifyUserInfo *user_info, const char *newline)
00506 {
00507         GList *l;
00508         GString *text;
00509 
00510         text = g_string_new("");
00511 
00512         for (l = user_info->user_info_entries; l != NULL; l = l->next) {
00513                 PurpleNotifyUserInfoEntry *user_info_entry = l->data;
00514                 /* Add a newline before a section header */
00515                 if (user_info_entry->type == PURPLE_NOTIFY_USER_INFO_ENTRY_SECTION_HEADER)
00516                         g_string_append(text, newline);
00517 
00518                 /* Handle the label/value pair itself */
00519                 /* XXX Todo: Use a larger size for a section header? */
00520                 if (user_info_entry->label)
00521                         g_string_append_printf(text, "<b>%s</b>", user_info_entry->label);
00522                 if (user_info_entry->label && user_info_entry->value)
00523                         g_string_append(text, ": ");
00524                 if (user_info_entry->value)
00525                         g_string_append(text, user_info_entry->value);
00526 
00527                 /* Display a section break as a horizontal line */
00528                 if (user_info_entry->type == PURPLE_NOTIFY_USER_INFO_ENTRY_SECTION_BREAK)
00529                         g_string_append(text, "<HR>");
00530 
00531                 /* Don't insert a new line before or after a section break; <HR> does that for us */
00532                 if ((user_info_entry->type != PURPLE_NOTIFY_USER_INFO_ENTRY_SECTION_BREAK) &&
00533                         (l->next && ((((PurpleNotifyUserInfoEntry *)(l->next->data))->type != PURPLE_NOTIFY_USER_INFO_ENTRY_SECTION_BREAK))))
00534                         g_string_append(text, newline);
00535 
00536                 /* Add an extra newline after a section header */
00537                 if (user_info_entry->type == PURPLE_NOTIFY_USER_INFO_ENTRY_SECTION_HEADER)
00538                         g_string_append(text, newline);
00539         }
00540 
00541         return g_string_free(text, FALSE);
00542 }
