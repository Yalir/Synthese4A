00057 /*------------------------------------------------------------------------
00058  * Return the list of media capabilities this contact supports.
00059  *
00060  *  @param account              The MXit account object
00061  *  @param who                  The username of the contact.
00062  *  @return                             The media capabilities supported
00063  */
00064 PurpleMediaCaps mxit_media_caps(PurpleAccount *account, const char *who)
00065 {
00066         struct MXitSession*     session = purple_account_get_connection(account)->proto_data;
00067         PurpleBuddy*            buddy;
00068         struct contact*         contact;
00069         PurpleMediaCaps         capa    = PURPLE_MEDIA_CAPS_NONE;
00070 
00071         purple_debug_info(MXIT_PLUGIN_ID, "mxit_media_caps: buddy '%s'\n", who);
00072 
00073         /* We need to have a voice/video server */
00074         if (strlen(session->voip_server) == 0)
00075                 return PURPLE_MEDIA_CAPS_NONE;
00076 
00077         /* find the buddy information for this contact (reference: "libpurple/blist.h") */
00078         buddy = purple_find_buddy(account, who);
00079         if (!buddy) {
00080                 purple_debug_warning(MXIT_PLUGIN_ID, "mxit_media_caps: unable to find the buddy '%s'\n", who);
00081                 return PURPLE_MEDIA_CAPS_NONE;
00082         }
00083 
00084         contact = purple_buddy_get_protocol_data(buddy);
00085         if (!contact)
00086                 return PURPLE_MEDIA_CAPS_NONE;
00087 
00088         /* can only communicate with MXit users */
00089         if (contact->type != MXIT_TYPE_MXIT)
00090                 return PURPLE_MEDIA_CAPS_NONE;
00091 
00092         /* and only with contacts in the 'Both' subscription state */
00093         if (contact->subtype != MXIT_SUBTYPE_BOTH)
00094                 return PURPLE_MEDIA_CAPS_NONE;
00095 
00096         /* and only when they're online */
00097         if (contact->presence == MXIT_PRESENCE_OFFLINE)
00098                 return MXIT_PRESENCE_OFFLINE;
00099 
00100         /* they support voice-only */
00101         if (contact->capabilities & MXIT_PFLAG_VOICE)
00102                 capa |= PURPLE_MEDIA_CAPS_AUDIO;
00103 
00104         /* they support voice-and-video */
00105         if (contact->capabilities & MXIT_PFLAG_VIDEO)
00106                 capa |= (PURPLE_MEDIA_CAPS_AUDIO | PURPLE_MEDIA_CAPS_VIDEO | PURPLE_MEDIA_CAPS_AUDIO_VIDEO);
00107 
00108         return capa;
00109 }

00165 /*------------------------------------------------------------------------
00166  * Initiate a voice/video session with a contact.
00167  *
00168  *  @param account              The MXit account object
00169  *  @param who                  The username of the contact.
00170  *  @param type                 The type of media session to initiate
00171  *  @return                             TRUE if session was initiated
00172  */
00173 gboolean mxit_media_initiate(PurpleAccount *account, const char *who, PurpleMediaSessionType type)
00174 {
00175         gchar* transmitter = "rawudp";
00176         PurpleMedia* media = NULL;
00177 
00178         purple_debug_info(MXIT_PLUGIN_ID, "mxit_media_initiate: buddy '%s'\n", who);
00179 
00180         media = purple_media_manager_create_media(
00181                 purple_media_manager_get(),
00182                 account,
00183                 "fsrtpconference",
00184                 who,
00185                 TRUE
00186         );
00187 
00188         if (!media) {
00189                 purple_debug_info(MXIT_PLUGIN_ID, "mxit_media_initiate: could not create media session\n");
00190                 return FALSE;
00191         }
00192 
00193         /* attach callbacks */
00194         g_signal_connect(G_OBJECT(media), "candidates-prepared", G_CALLBACK(mxit_candidates_prepared_cb), NULL);
00195         g_signal_connect(G_OBJECT(media), "stream-info", G_CALLBACK(mxit_stream_info_cb), NULL);
00196         g_signal_connect(G_OBJECT(media), "state-changed", G_CALLBACK(mxit_state_changed_cb), NULL);
00197 
00198         /* initiate audio session */
00199         if ((type & PURPLE_MEDIA_AUDIO) && 
00200                         (!purple_media_add_stream(media, "audio", who, PURPLE_MEDIA_AUDIO, TRUE, transmitter, 0, NULL))) {
00201                 purple_media_end(media, NULL, NULL);
00202                 return FALSE;
00203         }
00204 
00205         /* initiate video session */
00206         if ((type & PURPLE_MEDIA_VIDEO) &&
00207                         (!purple_media_add_stream(media, "video", who, PURPLE_MEDIA_VIDEO, TRUE, transmitter, 0, NULL))) {
00208                 purple_media_end(media, NULL, NULL);
00209                 return FALSE;
00210         }
00211 
00212         return TRUE;
00213 }
