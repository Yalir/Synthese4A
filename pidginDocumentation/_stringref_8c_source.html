00065 PurpleStringref *purple_stringref_new(const char *value)
00066 {
00067         PurpleStringref *newref;
00068         size_t len;
00069 
00070         if (value == NULL)
00071                 return NULL;
00072 
00073         len = strlen(value);
00074 
00075         newref = g_malloc(sizeof(PurpleStringref) + len);
00076         /* g_strlcpy() takes the size of the buffer, including the NUL.
00077            strlen() returns the length of the string, without the NUL. */
00078         g_strlcpy(newref->value, value, len + 1);
00079         newref->ref = 1;
00080 
00081         return newref;
