00278 static unsigned __stdcall
00279 select_thread (void *parameter)
00280 {
00281   GIOWin32Channel *channel = parameter;
00282   fd_set read_fds, write_fds, except_fds;
00283   GSList *tmp;
00284   int n;
00285   char buffer[8];
00286 
00287   g_io_channel_ref ((GIOChannel *)channel);
00288 
00289   if (channel->debug)
00290     g_print ("select_thread %#x: start fd:%d data_avail:%#x data_avail_noticed:%#x\n",
00291              channel->thread_id,
00292              channel->fd,
00293              (guint) channel->data_avail_event,
00294              (guint) channel->data_avail_noticed_event);
00295 
00296   channel->rdp = channel->wrp = 0;
00297   channel->running = TRUE;
00298 
00299   SetEvent (channel->space_avail_event);
00300 
00301   while (channel->running)
00302     {
00303       FD_ZERO (&read_fds);
00304       FD_ZERO (&write_fds);
00305       FD_ZERO (&except_fds);
00306       FD_SET (channel->reset_recv, &read_fds);
00307 
00308       LOCK (channel->mutex);
00309       tmp = channel->watches;
00310       while (tmp)
00311         {
00312           GIOWin32Watch *watch = (GIOWin32Watch *)tmp->data;
00313 
00314           if (watch->condition & (G_IO_IN | G_IO_HUP))
00315             FD_SET (channel->fd, &read_fds);
00316           if (watch->condition & G_IO_OUT)
00317             FD_SET (channel->fd, &write_fds);
00318           if (watch->condition & G_IO_ERR)
00319             FD_SET (channel->fd, &except_fds);
00320 
00321           tmp = tmp->next;
00322         }
00323       UNLOCK (channel->mutex);
00324 
00325       if (channel->debug)
00326         g_print ("select_thread %#x: calling select() for%s%s%s\n",
00327                  channel->thread_id,
00328                  (FD_ISSET (channel->fd, &read_fds) ? " IN" : ""),
00329                  (FD_ISSET (channel->fd, &write_fds) ? " OUT" : ""),
00330                  (FD_ISSET (channel->fd, &except_fds) ? " ERR" : ""));
00331 
00332       n = select (1, &read_fds, &write_fds, &except_fds, NULL);
00333 
00334       LOCK (channel->mutex);
00335       if (channel->needs_close)
00336         {
00337           UNLOCK (channel->mutex);
00338           break;
00339         }
00340       UNLOCK (channel->mutex);
00341 
00342       if (n == SOCKET_ERROR)
00343         {
00344           if (channel->debug)
00345             g_print ("select_thread %#x: select returned SOCKET_ERROR\n",
00346                      channel->thread_id);
00347           break;
00348         }
00349 
00350     if (FD_ISSET (channel->reset_recv, &read_fds))
00351     {
00352       if (channel->debug)
00353         g_print ("select_thread %#x: re-looping\n",
00354             channel->thread_id);
00355       recv (channel->reset_recv,  (char *)&buffer, (int) sizeof (buffer), 0);
00356       continue;
00357     }
00358 
00359     if (channel->debug)
00360       g_print ("select_thread %#x: got%s%s%s\n",
00361                channel->thread_id,
00362                (FD_ISSET (channel->fd, &read_fds) ? " IN" : ""),
00363                (FD_ISSET (channel->fd, &write_fds) ? " OUT" : ""),
00364                (FD_ISSET (channel->fd, &except_fds) ? " ERR" : ""));
00365 
00366     if (FD_ISSET (channel->fd, &read_fds))
00367       channel->revents |= G_IO_IN;
00368     if (FD_ISSET (channel->fd, &write_fds))
00369       channel->revents |= G_IO_OUT;
00370     if (FD_ISSET (channel->fd, &except_fds))
00371       channel->revents |= G_IO_ERR;
00372 
00373     if (channel->debug)
00374       g_print ("select_thread %#x: resetting data_avail_noticed, setting data_avail\n",
00375                channel->thread_id);
00376 
00377     LOCK (channel->mutex);
00378     ResetEvent (channel->data_avail_noticed_event);
00379     SetEvent (channel->data_avail_event);
00380     if (channel->needs_close)
00381       {
00382         UNLOCK (channel->mutex);
00383         break;
00384       }
00385     UNLOCK (channel->mutex);
00386 
00387     if (channel->debug)
00388       g_print ("select_thread %#x: waiting for data_avail_noticed\n",
00389         channel->thread_id);
00390 
00391     WaitForSingleObject (channel->data_avail_noticed_event, INFINITE);
00392     if (channel->debug)
00393       g_print ("select_thread %#x: got data_avail_noticed\n",
00394                  channel->thread_id);
00395     }
00396 
00397   LOCK (channel->mutex);
00398   channel->running = FALSE;
00399   if (channel->debug)
00400     g_print ("select_thread %#x: got error, setting data_avail\n",
00401              channel->thread_id);
00402   SetEvent (channel->data_avail_event);
00403   UNLOCK (channel->mutex);
00404   g_io_channel_unref ((GIOChannel *)channel);
00405 
00406   /* No need to call _endthreadex(), the actual thread starter routine
00407    * in MSVCRT (see crt/src/threadex.c:_threadstartex) calls
00408    * _endthreadex() for us.
00409    */
00410 
00411   return 0;
00412 }

00414 static gboolean
00415 g_io_win32_prepare (GSource *source,
00416                     gint    *timeout)
00417 {
00418   GIOWin32Watch *watch = (GIOWin32Watch *)source;
00419   GIOCondition buffer_condition = g_io_channel_get_buffer_condition (watch->channel);
00420   GIOWin32Channel *channel = (GIOWin32Channel *)watch->channel;
00421 
00422   *timeout = -1;
00423 
00424   if (channel->debug)
00425     g_print ("g_io_win32_prepare: for thread %#x buffer_condition:%#x\n"
00426              "  watch->pollfd.events:%#x watch->pollfd.revents:%#x channel->revents:%#x\n",
00427              channel->thread_id, buffer_condition,
00428              watch->pollfd.events, watch->pollfd.revents, channel->revents);
00429 
00430   if (channel->type == G_IO_WIN32_FILE_DESC)
00431     {
00432       LOCK (channel->mutex);
00433       if (channel->running && channel->wrp == channel->rdp)
00434         {
00435           if (channel->debug)
00436             g_print ("g_io_win32_prepare: for thread %#x, setting channel->revents = 0\n",
00437                      channel->thread_id);
00438           channel->revents = 0;
00439         }
00440       UNLOCK (channel->mutex);
00441     }
00442   else if (channel->type == G_IO_WIN32_SOCKET)
00443     {
00444       LOCK (channel->mutex);
00445       channel->revents = 0;
00446       if (channel->debug)
00447         g_print ("g_io_win32_prepare: for thread %#x, setting data_avail_noticed\n",
00448                  channel->thread_id);
00449       SetEvent (channel->data_avail_noticed_event);
00450       if (channel->debug)
00451         g_print ("g_io_win32_prepare: thread %#x, there.\n",
00452                  channel->thread_id);
00453       UNLOCK (channel->mutex);
00454     }
00455 
00456   return ((watch->condition & buffer_condition) == watch->condition);
00457 }
