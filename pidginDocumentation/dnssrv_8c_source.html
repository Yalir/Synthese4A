00142 {
00143         GList *cur;
00144         size_t runningtotal;
00145         int r;
00146 
00147         runningtotal = 0;

00149 
00150         while (cur) {
00151                 PurpleSrvResponseContainer *container = cur->data;
00152                 runningtotal += container->response->weight;
00153                 container->sum = runningtotal;
00154                 cur = cur->next;
00155         }
00156 
00157         /*
00158          * If the running total is greater than 0, pick a number between
00159          * 1 and the runningtotal inclusive. (This is not precisely what
00160          * the RFC algorithm describes, but we wish to deal with integers
00161          * and avoid floats.  This is functionally equivalent.)
00162          * If running total is 0, then choose r = 0.
00163          */

00165         cur = list;
00166         while (r > ((PurpleSrvResponseContainer *)cur->data)->sum) {
00167                 cur = cur->next;
00168         }
00169 
00170         /* Set the return parameter and remove cur from the list */
00171         *container_ptr =  cur->data;

00181 {
00182         int i;
00183         GList *cur, *container_list = NULL;
00184         PurpleSrvResponseContainer *container;

00193                 container->response = cur->data;
00194                 container_list = g_list_prepend(container_list, container);
00195         }
00196         container_list = g_list_reverse(container_list);

00763                 cb(NULL, 0, extradata);

00788         {

00847                 query_data->error_message = g_strdup("System missing DNS API (Requires W2K+)\n");

00852                         g_error_free(err);

00860                 query_data->handle = purple_timeout_add(0, res_main_thread_cb, query_data);
00861

01029 purple_srv_query_resolved(PurpleSrvTxtQueryData *query_data, GList *records)
01030 {
01031         GList *l;
01032         PurpleSrvResponse *records_array;
01033         int i = 0, length;
01034 
01035         g_return_if_fail(records != NULL);
01036 
01037         if (query_data->cb.srv == NULL) {
01038                 purple_srv_txt_query_destroy(query_data);
01039 
01040                 while (records) {
01041                         g_free(records->data);
01042                         records = g_list_delete_link(records, records);
01043                 }
01044                 return;

01046 
01047         records = purple_srv_sort(records);
01048         length = g_list_length(records);
01049 
01050         purple_debug_info("dnssrv", "SRV records resolved for %s, count: %d\n",
01051                                     query_data->query, length);
01052 
01053         records_array = g_new(PurpleSrvResponse, length);
01054         for (l = records; l; l = l->next, i++) {
01055                 records_array[i] = *(PurpleSrvResponse *)l->data;

01057 
01058         query_data->cb.srv(records_array, length, query_data->extradata);
01059 
01060         purple_srv_txt_query_destroy(query_data);
01061 
01062         while (records) {
