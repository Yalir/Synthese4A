00076 static
00077 int error(OscarData *od, aim_modsnac_t *error_snac, ByteStream *bs)
00078 {
00079         aim_snac_t *original_snac = aim_remsnac(od, error_snac->id);
00080         guint16 *request_type;
00081         GSList *original_info_ptr;
00082         struct aim_icq_info *original_info;
00083         guint16 reason;
00084         gchar *uin;
00085 
00086         if (!original_snac || (original_snac->family != SNAC_FAMILY_ICQ) || !original_snac->data) {
00087                 purple_debug_misc("oscar", "icq: the original snac for the error packet was not found");
00088                 g_free(original_snac);
00089                 return 0;
00090         }
00091 
00092         request_type = original_snac->data;
00093         original_info_ptr = g_slist_find_custom(od->icq_info, &original_snac->id, compare_icq_infos);
00094 
00095         if (!original_info_ptr) {
00096                 purple_debug_misc("oscar", "icq: the request info for the error packet was not found");
00097                 g_free(original_snac);
00098                 return 0;
00099         }
00100 
00101         original_info = original_info_ptr->data;
00102 
00103         reason = byte_stream_get16(bs);
00104         uin = g_strdup_printf("%u", original_info->uin);
00105         switch (*request_type) {
00106                 case AIM_ICQ_INFO_REQUEST:
00107                         oscar_user_info_display_error(od, reason, uin);
00108                         break;
00109                 case AIM_ICQ_ALIAS_REQUEST:
00110                         /* Couldn't retrieve an alias for the buddy requesting authorization; have to make do with UIN only. */
00111                         if (original_info->for_auth_request)
00112                                 oscar_auth_recvrequest(od->gc, uin, NULL, original_info->auth_request_reason);
00113                         break;
00114                 default:
00115                         purple_debug_misc("oscar", "icq: got an error packet with unknown request type %u", *request_type);
00116                         break;
00117         }
00118 
00119         aim_icq_freeinfo(original_info);
00120         od->icq_info = g_slist_remove(od->icq_info, original_info_ptr);
00121         g_free(original_snac->data);
00122         g_free(original_snac);
00123         return 1;
00124 }

00312 /*
00313  * Send an SMS message.  This is the non-US way.  The US-way is to IM
00314  * their cell phone number (+19195551234).
00315  *
00316  * We basically construct and send an XML message.  The format is:
00317  * <icq_sms_message>
00318  *   <destination>full_phone_without_leading_+</destination>
00319  *   <text>message</text>
00320  *   <codepage>1252</codepage>
00321  *   <senders_UIN>self_uin</senders_UIN>
00322  *   <senders_name>self_name</senders_name>
00323  *   <delivery_receipt>Yes|No</delivery_receipt>
00324  *   <time>Wkd, DD Mmm YYYY HH:MM:SS TMZ</time>
00325  * </icq_sms_message>
00326  *
00327  * Yeah hi Peter, whaaaat's happening.  If there's any way to use
00328  * a codepage other than 1252 that would be great.  Thaaaanks.
00329  */

00331 {
00332         FlapConnection *conn;

00334         ByteStream bs;
00335         aim_snacid_t snacid;

00338         const char *timestr, *username;
00339         time_t t;
00340         struct tm *tm;
00341         gchar *stripped;
00342 
00343         if (!od || !(conn = flap_connection_findbygroup(od, SNAC_FAMILY_ICQ)))
00344                 return -EINVAL;
00345 
00346         if (!name || !msg || !alias)
00347                 return -EINVAL;
00348 
00349         account = purple_connection_get_account(od->gc);
00350         username = purple_account_get_username(account);
00351 
00352         time(&t);

00374 
00375         byte_stream_new(&bs, 4 + bslen);
00376 
00377         snacid = aim_cachesnac(od, SNAC_FAMILY_ICQ, 0x0002, 0x0000, NULL, 0);
