00281         JabberDataRequestData *request_data = (JabberDataRequestData *) data;
00282         gpointer userdata = request_data->userdata;

00285         JabberDataRequestCallback *cb = request_data->cb;
00286 
00287         xmlnode *data_element = xmlnode_get_child(packet, "data");
00288         xmlnode *item_not_found = xmlnode_get_child(packet, "item-not-found");
00289 
00290         /* did we get a data element as result? */
00291         if (data_element && type == JABBER_IQ_RESULT) {
00292                 JabberData *data = jabber_data_create_from_xml(data_element);
00293 
00294                 if (data && !ephemeral) {
00295                         jabber_data_associate_remote(js, from, data);
00296                 }
00297                 cb(data, alt, userdata);
00298         } else if (item_not_found) {
00299                 purple_debug_info("jabber",
00300                         "Responder didn't recognize requested data\n");
00301                 cb(NULL, alt, userdata);
00302         } else {
00303                 purple_debug_warning("jabber", "Unknown response to data request\n");
00304                 cb(NULL, alt, userdata);
00305         }
00306

00420                 xmlnode_insert_child(result->node, item_not_found);

00422                 result = jabber_iq_new(js, JABBER_IQ_RESULT);
00423                 if (who)
00424                         xmlnode_set_attrib(result->node, "to", who);
00425                 xmlnode_set_attrib(result->node, "id", id);

00427                                                          jabber_data_get_xml_definition(data));
00428                 /* if the data object is temporary, destroy it and remove the references

00430                 if (data->ephemeral) {
00431                         g_hash_table_remove(local_data_by_cid, cid);
00432                 }
